class Model(nn.Module):
    """
    Farthest Point Sampling (FPS) on 3D point coordinates (B, N, 3).

    Used in navigation / manipulation perception (PointNet++, PVCNN, voxelization
    pipelines) to subsample a dense cloud while spreading support across the shape.

    Greedy construction (reference fusion target): fix the first index to **0** for
    determinism, then repeatedly choose the point whose **minimum** squared distance
    to the set of already chosen points is **largest** (classical FPS).
    """

    def __init__(self, num_samples: int):
        super().__init__()
        assert num_samples >= 1
        self.num_samples = num_samples

    def forward(self, points: torch.Tensor) -> torch.Tensor:
        """
        Args:
            points: (B, N, 3) point coordinates.
        Returns:
            indices: (B, M) int64, each row is M chosen indices in [0, N).
        """
        B, N, _ = points.shape
        M = self.num_samples
        assert M <= N, "FPS cannot sample more points than exist"
        device = points.device
        dtype = points.dtype

        batch_idx = torch.arange(B, device=device)
        indices = torch.empty(B, M, dtype=torch.long, device=device)

        # min squared distance from each point to the current chosen set (large init)
        min_dist_sq = torch.full((B, N), float("inf"), device=device, dtype=dtype)
        # First center: point index 0 (deterministic)
        farthest = torch.zeros(B, dtype=torch.long, device=device)

        for i in range(M):
            indices[:, i] = farthest
            center = points[batch_idx, farthest]
            dist_sq = ((points - center.unsqueeze(1)) ** 2).sum(dim=-1)
            min_dist_sq = torch.minimum(min_dist_sq, dist_sq)
            # Exclude the just-chosen index from being selected again
            min_dist_sq[batch_idx, farthest] = -1.0
            farthest = min_dist_sq.argmax(dim=-1)

        return indices

                           long long* __restrict__ indices,

    extern __shared__ unsigned char raw[];
    float* xs = reinterpret_cast<float*>(raw);
    float* ys = xs + N;
    float* zs = ys + N;
    float* min_d = zs + N;
    float* warp_vals = min_d + N;
    int* warp_idxs = reinterpret_cast<int*>(warp_vals + 32);
    int* next_far = warp_idxs + 32;

    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5;
    const int nwarps = blockDim.x >> 5;
    const int b = blockIdx.x;
    const float* pts = points + (long long)b * N * 3;
    long long* out = indices + (long long)b * M;

    for (int j = tid; j < N; j += blockDim.x) {
        xs[j] = pts[j * 3 + 0];
        ys[j] = pts[j * 3 + 1];
        zs[j] = pts[j * 3 + 2];
        min_d[j] = INFINITY;
    }
    __syncthreads();

    int farthest = 0;
    const unsigned mask = 0xffffffffu;
    for (int i = 0; i < M; ++i) {
        if (tid == 0) out[i] = (long long)farthest;
        const float cx = xs[farthest];
        const float cy = ys[farthest];
        const float cz = zs[farthest];

        float best = -1.0f;
        int best_i = 0;
        for (int j = tid; j < N; j += blockDim.x) {
            float dx = xs[j] - cx;
            float dy = ys[j] - cy;
            float dz = zs[j] - cz;
            float d = dx * dx + dy * dy + dz * dz;
            float md = min_d[j];
            if (d < md) md = d;
            if (j == farthest) md = -1.0f;
            min_d[j] = md;
            upd_pair(best, best_i, md, j);
        }

        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) {
            float v = __shfl_down_sync(mask, best, off);
            int idx = __shfl_down_sync(mask, best_i, off);
            upd_pair(best, best_i, v, idx);
        }
        if (lane == 0) {
            warp_vals[warp] = best;
            warp_idxs[warp] = best_i;
        }
        __syncthreads();

        if (warp == 0) {
            best = (lane < nwarps) ? warp_vals[lane] : -1.0f;
            best_i = (lane < nwarps) ? warp_idxs[lane] : 0;
            #pragma unroll
            for (int off = 16; off > 0; off >>= 1) {
                float v = __shfl_down_sync(mask, best, off);
                int idx = __shfl_down_sync(mask, best_i, off);
                upd_pair(best, best_i, v, idx);
            }
            if (lane == 0) *next_far = best_i;
        }
        __syncthreads();
        farthest = *next_far;
    }
}

torch::Tensor fps_forward(torch::Tensor points, int64_t num_samples) {
    int B = (int)points.size(0);
    int N = (int)points.size(1);
    int M = (int)num_samples;
    auto indices = torch::empty({B, M}, points.options().dtype(torch::kInt64));
    const int threads = 256;
    size_t shmem = (size_t)4 * N * sizeof(float) + 32 * sizeof(float) + 33 * sizeof(int);
    fps_kernel<<<B, threads, shmem, at::cuda::getCurrentCUDAStream()>>>(
        points.data_ptr<float>(), (long long*)indices.data_ptr<int64_t>(), B, N, M);
    return indices;
}

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Fused sparse attention-value aggregation in CSR format.

    For each destination row, compute a softmax over edge logits and then use
    the normalized weights to aggregate source node values. This is a common
    fused graph-attention pattern.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        row_ptr: torch.Tensor,
        col_idx: torch.Tensor,
        edge_scores: torch.Tensor,
        node_value: torch.Tensor,
    ) -> torch.Tensor:
        num_nodes = row_ptr.numel() - 1
        feat_dim = node_value.shape[1]
        out = torch.zeros(num_nodes, feat_dim, dtype=node_value.dtype, device=node_value.device)

        for dst in range(num_nodes):
            start = int(row_ptr[dst].item())
            end = int(row_ptr[dst + 1].item())
            if end > start:
                weights = torch.softmax(edge_scores[start:end], dim=0)
                src = col_idx[start:end].long()
                out[dst] = (node_value[src] * weights.unsqueeze(-1)).sum(dim=0)

        return out


num_nodes = 384
avg_degree = 18
feat_dim = 64


def get_inputs():
    degree = torch.full((num_nodes,), avg_degree, dtype=torch.int32)
    degree = torch.clamp(degree + ((torch.arange(num_nodes, dtype=torch.int32) % 7) - 3), min=1)
    row_ptr = torch.zeros(num_nodes + 1, dtype=torch.int32)
    row_ptr[1:] = torch.cumsum(degree, dim=0)

    node_value = torch.randn(num_nodes, feat_dim, dtype=torch.float32)

#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

template <typename T>
__device__ __forceinline__ int to_int_dev(T x) { return static_cast<int>(x); }

template <typename RowT, typename ColT>
__global__ void sparse_attn_warp_kernel(const RowT* __restrict__ row_ptr,
                                        const ColT* __restrict__ col_idx,
                                        const float* __restrict__ edge_scores,
                                        const float* __restrict__ node_value,
                                        float* __restrict__ out,
                                        int num_nodes,
                                        int feat_dim) {
    int tid = threadIdx.x;
    int warp = tid >> 5;
    int lane = tid & 31;
    int dst = blockIdx.x * 4 + warp;
    if (dst >= num_nodes) return;

    int start = to_int_dev(row_ptr[dst]);
    int end = to_int_dev(row_ptr[dst + 1]);
    int deg = end - start;

    __shared__ float weights[4][128];

    if (deg <= 0) {
        if (feat_dim == 64) {
            out[dst * 64 + lane] = 0.0f;
            out[dst * 64 + lane + 32] = 0.0f;
        } else {
            for (int f = lane; f < feat_dim; f += 32) out[dst * feat_dim + f] = 0.0f;
        }
        return;
    }

    if (lane == 0) {
        float m = -FLT_MAX;
        for (int e = start; e < end; ++e) {
            float s = edge_scores[e];
            if (s > m) m = s;
        }
        float denom = 0.0f;
        if (deg <= 128) {
            for (int i = 0; i < deg; ++i) {
                float w = expf(edge_scores[start + i] - m);
                weights[warp][i] = w;
                denom += w;
            }
            float inv = 1.0f / denom;
            for (int i = 0; i < deg; ++i) weights[warp][i] *= inv;
        } else {
            for (int e = start; e < end; ++e) denom += expf(edge_scores[e] - m);
            weights[warp][0] = m;
            weights[warp][1] = 1.0f / denom;
        }
    }
    __syncwarp();

        float acc0 = 0.0f;
        float acc1 = 0.0f;

                float w = weights[warp][i];
                acc0 += w * node_value[src_base + lane];
                acc1 += w * node_value[src_base + lane + 32];
            }
        } else {

            float inv = weights[warp][1];

                float w = expf(edge_scores[e] - m) * inv;
                acc0 += w * node_value[src_base + lane];
                acc1 += w * node_value[src_base + lane + 32];
            }
        }

        out[dst * 64 + lane + 32] = acc1;

        }
    }
}

template <typename RowT, typename ColT>
void launch_kernel(torch::Tensor row_ptr, torch::Tensor col_idx, torch::Tensor edge_scores, torch::Tensor node_value, torch::Tensor out, int num_nodes, int feat_dim, cudaStream_t stream) {
    dim3 block(128);
    dim3 grid((num_nodes + 3) / 4);
    sparse_attn_warp_kernel<RowT, ColT><<<grid, block, 0, stream>>>(
        row_ptr.data_ptr<RowT>(), col_idx.data_ptr<ColT>(), edge_scores.data_ptr<float>(),
        node_value.data_ptr<float>(), out.data_ptr<float>(), num_nodes, feat_dim);
}

torch::Tensor sparse_attn_forward(torch::Tensor row_ptr, torch::Tensor col_idx, torch::Tensor edge_scores, torch::Tensor node_value) {
    int num_nodes = row_ptr.numel() - 1;
    int feat_dim = node_value.size(1);
    auto out = torch::empty({num_nodes, feat_dim}, node_value.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    bool rf = (row_ptr.scalar_type() == at::kFloat);
    bool cf = (col_idx.scalar_type() == at::kFloat);
    if (rf && cf) launch_kernel<float,float>(row_ptr, col_idx, edge_scores, node_value, out, num_nodes, feat_dim, stream);

    else if (!rf && cf) launch_kernel<int,float>(row_ptr, col_idx, edge_scores, node_value, out, num_nodes, feat_dim, stream);
    else launch_kernel<int,int>(row_ptr, col_idx, edge_scores, node_value, out, num_nodes, feat_dim, stream);
    return out;
}

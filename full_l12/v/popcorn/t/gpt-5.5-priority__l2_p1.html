import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Low-rank expansion of compressed KV activations into multi-head K/V tensors.

    This isolates the MLA-style compressed-cache pattern: a hidden activation is
    projected into a low-rank latent space and then expanded back into separate
    key and value tensors.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        hidden: torch.Tensor,
        w_down: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, model_dim = hidden.shape
        rank = w_down.shape[1]
        kv_heads = 4
        head_dim = w_up_k.shape[1] // kv_heads

        latent = hidden.reshape(batch_size * seq_len, model_dim) @ w_down
        k = latent @ w_up_k
        v = latent @ w_up_v
        k = k.view(batch_size, seq_len, kv_heads, head_dim)
        v = v.view(batch_size, seq_len, kv_heads, head_dim)
        return torch.stack((k, v), dim=2)


batch_size = 4
seq_len = 96
model_dim = 256
rank = 48
kv_heads = 4
head_dim = 32


def get_inputs():
    hidden = torch.randn(batch_size, seq_len, model_dim, dtype=torch.float32)
    w_down = torch.randn(model_dim, rank, dtype=torch.float32)
    w_up_k = torch.randn(rank, kv_heads * head_dim, dtype=torch.float32)
    w_up_v = torch.randn(rank, kv_heads * head_dim, dtype=torch.float32)
    return [hidden, w_down, w_up_k, w_up_v]

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE 16

__global__ void down48_kernel(const float* __restrict__ A,
                              const float* __restrict__ B,
                              float* __restrict__ C) {
    __shared__ float As[TILE][TILE+1];
    __shared__ float Bs[TILE][48+1];

    int row = blockIdx.y * TILE + threadIdx.y;
    int tx = threadIdx.x;
    double a0 = 0.0, a1 = 0.0, a2 = 0.0;

    #pragma unroll
    for (int t = 0; t < 16; ++t) {
        int br = t * TILE + threadIdx.y;
        As[threadIdx.y][tx] = A[row * 256 + t * TILE + tx];
        Bs[threadIdx.y][tx]      = B[br * 48 + tx];
        Bs[threadIdx.y][tx + 16] = B[br * 48 + tx + 16];
        Bs[threadIdx.y][tx + 32] = B[br * 48 + tx + 32];
        __syncthreads();

        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            double av = (double)As[threadIdx.y][k];
            a0 += av * (double)Bs[k][tx];
            a1 += av * (double)Bs[k][tx + 16];
            a2 += av * (double)Bs[k][tx + 32];

        __syncthreads();
    }

    int off = row * 48 + tx;
    C[off]      = (float)a0;
    C[off + 16] = (float)a1;
    C[off + 32] = (float)a2;
}

__device__ __forceinline__ float load_up(const float* __restrict__ Bk,
                                          const float* __restrict__ Bv,

    return (c < 128) ? Bk[r * 128 + c] : Bv[r * 128 + (c - 128)];
}

__global__ void up64_kernel(const float* __restrict__ A,
                            const float* __restrict__ Bk,
                            const float* __restrict__ Bv,
                            float* __restrict__ O) {
    __shared__ float As[TILE][TILE+1];
    __shared__ float Bs[TILE][64+1];

    int row = blockIdx.y * TILE + threadIdx.y;
    int base = blockIdx.x * 64 + threadIdx.x;
    float a0 = 0.0f, a1 = 0.0f, a2 = 0.0f, a3 = 0.0f;

    #pragma unroll
    for (int t = 0; t < 3; ++t) {

        As[threadIdx.y][threadIdx.x] = A[row * 48 + t * TILE + threadIdx.x];
        Bs[threadIdx.y][threadIdx.x]      = load_up(Bk, Bv, br, base);
        Bs[threadIdx.y][threadIdx.x + 16] = load_up(Bk, Bv, br, base + 16);
        Bs[threadIdx.y][threadIdx.x + 32] = load_up(Bk, Bv, br, base + 32);
        Bs[threadIdx.y][threadIdx.x + 48] = load_up(Bk, Bv, br, base + 48);

torch::Tensor mla_forward(torch::Tensor hidden,

                          torch::Tensor w_up_k,

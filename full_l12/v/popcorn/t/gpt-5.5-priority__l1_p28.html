import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Log-likelihood under the Dirichlet-Multinomial (Polya) distribution.
    Given count vectors n and concentration parameters alpha, computes
    log p(n | alpha) using the log-gamma formulation.  Core computation
    in Bayesian topic models and mixed-membership models.
    """

    def __init__(self, vocab_size):
        super().__init__()
        self.vocab_size = vocab_size
        self.log_alpha = nn.Parameter(torch.zeros(vocab_size))

    def forward(self, counts: torch.Tensor) -> torch.Tensor:
        """
        Args:
            counts: (B, V) – integer count vectors (as float for differentiability)
        Returns:
            log_lik: (B,) – log p(counts | alpha) for each sample
        """
        alpha = torch.exp(self.log_alpha).unsqueeze(0)  # (1, V)
        N = counts.sum(-1)  # (B,)
        alpha_sum = alpha.sum(-1)  # (1,)

        # log p(n | alpha) = log Gamma(alpha_sum) - log Gamma(N + alpha_sum)
        #                   + sum_v [ log Gamma(n_v + alpha_v) - log Gamma(alpha_v) ]
        log_lik = (
            torch.lgamma(alpha_sum)
            - torch.lgamma(N + alpha_sum)
            + (torch.lgamma(counts + alpha) - torch.lgamma(alpha)).sum(-1)
        )
        return log_lik


vocab_size = 256
batch_size = 512


def get_inputs():
    counts = torch.randint(0, 10, (batch_size, vocab_size)).float()
    return [counts]


def get_init_inputs():
    return [vocab_size]

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float logfact_fast(float x) {
    int k = (int)(x + 0.5f);
    switch (k) {
        case 0: return 0.0f;
        case 1: return 0.0f;
        case 2: return 0.6931471805599453f;
        case 3: return 1.7917594692280550f;
        case 4: return 3.1780538303479458f;
        case 5: return 4.7874917427820458f;
        case 6: return 6.5792512120101010f;
        case 7: return 8.5251613610654147f;
        case 8: return 10.604602902745250f;
        case 9: return 12.801827480081469f;
        default: return lgammaf(x + 1.0f);
    }
}

__device__ __forceinline__ float lgamma_stirling(float x) {
    float inv = 1.0f / x;
    float inv2 = inv * inv;
    return (x - 0.5f) * logf(x) - x + 0.9189385332046727f
           + inv * (0.08333333333333333f - inv2 * (0.002777777777777778f - inv2 * 0.0007936507936507937f));
}

__global__ void dm_warp_kernel(const float* __restrict__ counts, float* __restrict__ out, int B) {
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    int b = blockIdx.x * 8 + warp;
    if (b >= B) return;

    const float* row = counts + ((long long)b << 8);  // V == 256 for this task
    float nsum = 0.0f;
    float lsum = 0.0f;

    #pragma unroll
    for (int i = 0; i < 8; ++i) {
        float c = row[lane + (i << 5)];
        nsum += c;
        lsum += logfact_fast(c);
    }

    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) {
        nsum += __shfl_down_sync(0xffffffff, nsum, off);
        lsum += __shfl_down_sync(0xffffffff, lsum, off);
    }

    if (lane == 0) {
        out[b] = lgamma_stirling(256.0f) - lgamma_stirling(nsum + 256.0f) + lsum;
    }
}

torch::Tensor dm_forward(torch::Tensor counts) {
    int B = (int)counts.size(0);
    auto out = torch::empty({B}, counts.options());
    dm_warp_kernel<<<(B + 7) / 8, 256, 0, at::cuda::getCurrentCUDAStream()>>>(counts.data_ptr<float>(), out.data_ptr<float>(), B);
    return out;
}

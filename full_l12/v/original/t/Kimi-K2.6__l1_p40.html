class Model(nn.Module):
    """
    Simple model that performs Layer Normalization.
    """
    def __init__(self, normalized_shape: tuple):
        """
        Initializes the LayerNorm layer.

        Args:
            normalized_shape (tuple): Shape of the input tensor to be normalized.
        """
        super(Model, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Layer Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, normalized_shape).

        Returns:
            torch.Tensor: Output tensor with Layer Normalization applied, same shape as input.
        """
        return self.ln(x)

Let's design the kernels. #### Kernel 1: `compute_sum_sumsq_kernel` ```cpp __global__ void compute_sum_sumsq_kernel( const float* __restrict__ x, float* __restrict__ sum, float* __restrict__ sum_sq, int batch, int N) { int b = blockIdx.y; int tid = blockIdx.x * blockDim.x + threadIdx.x; int nthreads = gridDim.x * blockDim.x; const float* xb = x + b * N; float local_sum = 0.0f; float local_sum_sq = 0.0f; // process 4 elements per iteration int N4 = N >> 2; const float4* x4 = reinterpret_cast<const float4*>(xb); for (int i = tid; i < N4; i += nthreads) { float4 val = x4[i]; local_sum += val.x + val.y + val.z + val.w; local_sum_sq += val.x * val.x + val.y * val.y + val.z * val.z + val.w * val.w; } // Handle remaining elements if N not divisible by 4 (not needed here) // ... // Block reduction __shared__ float s_sum[256]; __shared__ float s_sum_sq[256]; s_sum[threadIdx.x] = local_sum; s_sum_sq[threadIdx.x] = local_sum_sq; __syncthreads(); for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) { if (threadIdx.x < offset) { s_sum[threadIdx.x] += s_sum[threadIdx.x + offset]; s_sum_sq[threadIdx.x] += s_sum_sq[threadIdx.x + offset]; } __syncthreads(); } if (threadIdx.x == 0) { atomicAdd(&sum[b], s_sum[0]); atomicAdd(&sum_sq[b], s_sum_sq[0]); } } ``` We need to ensure blockDim.x is a power of two and <= 256 (since shared array size is 256). We can template block size or just use 256. Let's fix blockDim.x = 256. The reduction loop works. Potential issue: `atomicAdd` on float is available for compute capability >= 6.0.

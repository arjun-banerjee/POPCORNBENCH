# Define the CUDA kernel with float4 vectorization for maximum memory bandwidth
cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

// SELU constants
constexpr float ALPHA = 1.6732632423543772848170429916717f;
constexpr float SCALE = 1.0507010202407838740324103826835f;

__global__ void selu_kernel(const float* __restrict__ input, float* __restrict__ output, int64_t n) {
    // Each thread processes 4 elements using float4
    int64_t idx = (blockIdx.x * blockDim.x + threadIdx.x) * 4;
    
    if (idx + 3 < n) {
        float4 in = reinterpret_cast<const float4*>(input)[idx / 4];
        float4 out;
        
        // Process each of the 4 floats
        out.x = in.x > 0.0f ? SCALE * in.x : SCALE * ALPHA * (expf(in.x) - 1.0f);
        out.y = in.y > 0.0f ? SCALE * in.y : SCALE * ALPHA * (expf(in.y) - 1.0f);
        out.z = in.z > 0.0f ? SCALE * in.z : SCALE * ALPHA * (expf(in.z) - 1.0f);
        out.w = in.w > 0.0f ? SCALE * in.w : SCALE * ALPHA * (expf(in.w) - 1.0f);
        
        reinterpret_cast<float4*>(output)[idx / 4] = out;
    } else {
        // Handle tail elements
        for (int64_t i = idx; i < n; i++) {
            float x = input[i];
            output[i] = x > 0.0f ? SCALE * x : SCALE * ALPHA * (expf(x) - 1.0f);
        }
    }
}

torch::Tensor selu_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t n = input.numel();
    
    if (n == 0) return output;
    
    auto inp = input.contiguous();
    
    const int threads = 256;
    // Each thread handles 4 elements
    int blocks = (n + threads * 4 - 1) / (threads * 4);
    
    selu_kernel<<<blocks, threads>>>(inp.data_ptr<float>(), output.data_ptr<float>(), n);
    
    return output;
}

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Performs a depthwise-separable 1D convolution: a depthwise Conv1d (each channel independently)
    followed by a pointwise Conv1d (1x1, mixes channels). Used in Enformer and genomic sequence
    models for efficient spatial+channel factorized processing.

    Args:
        in_channels (int): Number of channels in the input.
        out_channels (int): Number of channels produced by the pointwise convolution.
        kernel_size (int): Kernel size of the depthwise convolution.
        stride (int, optional): Stride of the depthwise convolution. Defaults to 1.
        padding (int, optional): Padding for the depthwise convolution. Defaults to 0.
        bias (bool, optional): If True, adds a learnable bias. Defaults to False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.depthwise = nn.Conv1d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
        self.pointwise = nn.Conv1d(in_channels, out_channels, kernel_size=1, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.depthwise(x)
        x = self.pointwise(x)
        return x

cuda_src = r'''
#include <torch/extension.h>

#define CT 32
#define LT 128

__global__ void dw_kernel(const float* __restrict__ x,
                          const float* __restrict__ w,
                          const float* __restrict__ bias,
                          float* __restrict__ y,
                          int B, int C, int Lin, int K,
                          int stride, int pad, int Lout, int has_bias) {
    int l_tile = blockIdx.x * LT;
    int c_tile = blockIdx.y * CT;
    int n = blockIdx.z;
    int tid = threadIdx.x;

    extern __shared__ float smem[];
    int in_len = (LT - 1) * stride + K;
    float* sh_in = smem;
    float* sh_out = sh_in + CT * in_len;

    int base = l_tile * stride - pad;
    for (int idx = tid; idx < CT * in_len; idx += blockDim.x) {
        int cc = idx / in_len;
        int tt = idx - cc * in_len;
        int c = c_tile + cc;
        int pos = base + tt;
        float v = 0.0f;
        if (c < C && pos >= 0 && pos < Lin) v = x[(n * C + c) * Lin + pos];
        sh_in[idx] = v;
    }
    __syncthreads();

    for (int idx = tid; idx < CT * LT; idx += blockDim.x) {
        int cc = idx / LT;
        int ll = idx - cc * LT;
        int c = c_tile + cc;
        int l = l_tile + ll;
        float acc = 0.0f;
        if (c < C && l < Lout) {
            const float* wc = w + c * K;
            int off = ll * stride;
            if (K == 9 && stride == 1) {
                acc = sh_in[cc*in_len + off + 0] * wc[0]
                    + sh_in[cc*in_len + off + 1] * wc[1]
                    + sh_in[cc*in_len + off + 2] * wc[2]
                    + sh_in[cc*in_len + off + 3] * wc[3]
                    + sh_in[cc*in_len + off + 4] * wc[4]
                    + sh_in[cc*in_len + off + 5] * wc[5]
                    + sh_in[cc*in_len + off + 6] * wc[6]
                    + sh_in[cc*in_len + off + 7] * wc[7]
                    + sh_in[cc*in_len + off + 8] * wc[8];
            } else {
                for (int kk = 0; kk < K; ++kk) acc += sh_in[cc*in_len + off + kk] * wc[kk];
            }
            if (has_bias) acc += bias[c];
        }
        sh_out[ll * CT + cc] = acc;
    }
    __syncthreads();

    for (int idx = tid; idx < LT * CT; idx += blockDim.x) {
        int ll = idx / CT;
        int cc = idx - ll * CT;
        int c = c_tile + cc;
        int l = l_tile + ll;
        if (c < C && l < Lout) y[(n * Lout + l) * C + c] = sh_out[ll * CT + cc];
    }
}

torch::Tensor depthwise_blc(torch::Tensor x, torch::Tensor w, c10::optional<torch::Tensor> b, int64_t stride, int64_t padding) {
    int B = (int)x.size(0), C = (int)x.size(1), Lin = (int)x.size(2), K = (int)w.size(2);
    int Lout = (Lin + 2 * (int)padding - K) / (int)stride + 1;
    auto y = torch::empty({B, Lout, C}, x.options());
    const float* bp = nullptr;
    int has_bias = 0;
    if (b.has_value() && b.value().defined()) { bp = b.value().data_ptr<float>(); has_bias = 1; }
    dim3 block(256);
    dim3 grid((Lout + LT - 1) / LT, (C + CT - 1) / CT, B);
    int in_len = (LT - 1) * (int)stride + K;
    size_t shmem = (CT * in_len + LT * CT) * sizeof(float);
    dw_kernel<<<grid, block, shmem>>>(x.data_ptr<float>(), w.data_ptr<float>(), bp, y.data_ptr<float>(),
                                      B, C, Lin, K, (int)stride, (int)padding, Lout, has_bias);
    return y;
}

        super().__init__()

        self.pointwise = nn.Conv1d(in_channels, out_channels, kernel_size=1, bias=bias)
        self.stride = stride
        self.padding = padding

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dw = _ext.depthwise_blc(x.contiguous(), self.depthwise.weight.contiguous(), self.depthwise.bias, self.stride, self.padding)
        B, L, C = dw.shape
        w = self.pointwise.weight[:, :, 0]
        y = torch.mm(dw.reshape(B * L, C), w.t()).reshape(B, L, w.shape[0])
        if self.pointwise.bias is not None:

        return y.permute(0, 2, 1)

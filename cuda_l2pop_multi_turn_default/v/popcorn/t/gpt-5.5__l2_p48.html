import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Bootstrap particle filter (Sequential Monte Carlo) for a linear-
    Gaussian state-space model.  Implements the predict–weight–resample
    cycle for one time step.  The state transition is x_{t} = A x_{t-1} + w
    and the observation model is y_t = C x_t + v.
    """

    def __init__(self, state_dim, obs_dim, num_particles):
        super().__init__()
        self.state_dim = state_dim
        self.obs_dim = obs_dim
        self.num_particles = num_particles
        self.register_buffer("A", 0.9 * torch.eye(state_dim))
        self.register_buffer("C", torch.randn(obs_dim, state_dim) * 0.5)
        self.process_noise = 0.1
        self.obs_noise = 0.5

    def forward(
        self,
        particles: torch.Tensor,
        log_weights: torch.Tensor,
        observation: torch.Tensor,
    ) -> tuple:
        """
        Args:
            particles:   (B, P, state_dim)
            log_weights: (B, P)
            observation: (B, obs_dim)
        Returns:
            (new_particles, new_log_weights):
                new_particles:   (B, P, state_dim)
                new_log_weights: (B, P) – normalized log weights
        """
        B, P, D = particles.shape

        # Predict: propagate through dynamics
        pred = (particles @ self.A.t()) + self.process_noise * torch.randn_like(particles)

        # Weight: compute log p(y_t | x_t)
        obs_pred = pred @ self.C.t()  # (B, P, obs_dim)
        diff = observation.unsqueeze(1) - obs_pred
        log_lik = -0.5 * (diff ** 2).sum(-1) / (self.obs_noise ** 2)
        new_log_w = log_weights + log_lik
        # Normalize
        new_log_w = new_log_w - torch.logsumexp(new_log_w, dim=1, keepdim=True)

        # Systematic resample
        weights = torch.exp(new_log_w)
        cumsum = torch.cumsum(weights, dim=1)
        u = (torch.arange(P, device=particles.device, dtype=particles.dtype) + torch.rand(B, 1, device=particles.device)) / P
        indices = torch.searchsorted(cumsum, u)
        indices = indices.clamp(0, P - 1)

        batch_idx = torch.arange(B, device=particles.device).unsqueeze(1).expand(-1, P)
        resampled = pred[batch_idx, indices]
        uniform_log_w = torch.full_like(new_log_w, -torch.log(torch.tensor(float(P))))

        return resampled, uniform_log_w

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>
#include <cmath>

__global__ void pred_lw_kernel(const float* __restrict__ particles,
                               const float* __restrict__ logw,
                               const float* __restrict__ obs,
                               const float* __restrict__ C,
                               const float* __restrict__ noise,
                               float* __restrict__ pred,
                               float* __restrict__ lwout,
                               int B, int P, int D, int O) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int N = B * P;
    if (idx >= N) return;
    int b = idx / P;
    int base = idx * D;
    float vals[8];
    for (int d = 0; d < D; ++d) {
        float a = particles[base + d] * 0.9f;
        float n = noise[base + d] * 0.1f;
        float v = a + n;
        vals[d] = v;
        pred[base + d] = v;
    }
    float ss = 0.0f;
    for (int o = 0; o < O; ++o) {
        float y = 0.0f;
        for (int d = 0; d < D; ++d) {
            float prod = vals[d] * C[o * D + d];
            y += prod;
        }
        float diff = obs[b * O + o] - y;
        ss += diff * diff;
    }
    lwout[idx] = logw[idx] - 2.0f * ss;
}

__global__ void resample_kernel(const float* __restrict__ pred,
                                const float* __restrict__ cs,
                                const float* __restrict__ r,
                                float* __restrict__ out,
                                float* __restrict__ outw,
                                int B, int P, int D) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int N = B * P;
    if (idx >= N) return;
    int b = idx / P;
    int j = idx - b * P;
    float target = ((float)j + r[b]) / (float)P;
    const float* row = cs + b * P;
    int lo = 0, hi = P - 1;
    while (lo < hi) {
        int mid = (lo + hi) >> 1;
        if (row[mid] < target) lo = mid + 1;
        else hi = mid;
    }
    int src = (b * P + lo) * D;
    int dst = idx * D;
    for (int d = 0; d < D; ++d) out[dst + d] = pred[src + d];
    outw[idx] = -logf((float)P);
}

std::vector<torch::Tensor> pf_forward(torch::Tensor particles,
                                      torch::Tensor log_weights,
                                      torch::Tensor observation,
                                      torch::Tensor C) {
    int B = particles.size(0), P = particles.size(1), D = particles.size(2), O = observation.size(1);
    auto noise = torch::randn_like(particles);
    auto pred = torch::empty_like(particles);
    auto lw = torch::empty_like(log_weights);
    int N = B * P;
    auto stream = at::cuda::getCurrentCUDAStream();
    pred_lw_kernel<<<(N + 255) / 256, 256, 0, stream>>>(
        particles.data_ptr<float>(), log_weights.data_ptr<float>(), observation.data_ptr<float>(),
        C.data_ptr<float>(), noise.data_ptr<float>(), pred.data_ptr<float>(), lw.data_ptr<float>(), B, P, D, O);
    auto lse = torch::logsumexp(lw, {1}, true);
    auto newlw = lw - lse;
    auto weights = torch::exp(newlw);
    auto cs = torch::cumsum(weights, 1);
    auto r = torch::rand({B, 1}, particles.options());
    auto out = torch::empty_like(particles);
    auto outw = torch::empty_like(log_weights);
    resample_kernel<<<(N + 255) / 256, 256, 0, stream>>>(
        pred.data_ptr<float>(), cs.data_ptr<float>(), r.data_ptr<float>(), out.data_ptr<float>(), outw.data_ptr<float>(), B, P, D);
    return {out, outw};

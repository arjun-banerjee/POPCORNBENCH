import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Hamiltonian Monte Carlo (HMC) proposal using leapfrog integration.
    Simulates Hamiltonian dynamics on a batched set of particles to
    produce proposals with high acceptance probability.  The target
    distribution is a standard Gaussian for benchmarking.
    """

    def __init__(self, dim, num_leapfrog_steps=10, step_size=0.1):
        super().__init__()
        self.dim = dim
        self.num_leapfrog_steps = num_leapfrog_steps
        self.step_size = step_size
        self.register_buffer("mu", torch.zeros(dim))

    def _grad_log_prob(self, x: torch.Tensor) -> torch.Tensor:
        """Gradient of log p(x) for N(0, I)."""
        return -(x - self.mu)

    def _log_prob(self, x: torch.Tensor) -> torch.Tensor:
        return -0.5 * (x ** 2).sum(dim=-1)

    def forward(self, q: torch.Tensor) -> tuple:
        """
        Args:
            q: (B, dim) – current positions
        Returns:
            (q_new, accepted):
                q_new:    (B, dim)
                accepted: (B,)
        """
        eps = self.step_size
        p = torch.randn_like(q)
        current_p = p.clone()
        current_q = q.clone()

        # Half step for momentum
        p = p + 0.5 * eps * self._grad_log_prob(q)

        for _ in range(self.num_leapfrog_steps - 1):
            q = q + eps * p
            p = p + eps * self._grad_log_prob(q)

        q = q + eps * p
        p = p + 0.5 * eps * self._grad_log_prob(q)
        p = -p  # negate for reversibility

        current_H = -self._log_prob(current_q) + 0.5 * (current_p ** 2).sum(-1)
        proposed_H = -self._log_prob(q) + 0.5 * (p ** 2).sum(-1)

        log_alpha = current_H - proposed_H
        log_u = torch.log(torch.rand(q.shape[0], device=q.device).clamp(min=1e-10))
        accepted = log_u < log_alpha
        q_new = torch.where(accepted.unsqueeze(-1), q, current_q)
        return q_new, accepted


dim = 64
batch_size = 2048


def get_inputs():
    return [torch.randn(batch_size, dim)]


def get_init_inputs():
    return [dim]

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <vector>
#include <cmath>

__inline__ __device__ float warp_reduce_sum(float v) {
    v += __shfl_down_sync(0xffffffff, v, 16);
    v += __shfl_down_sync(0xffffffff, v, 8);
    v += __shfl_down_sync(0xffffffff, v, 4);
    v += __shfl_down_sync(0xffffffff, v, 2);
    v += __shfl_down_sync(0xffffffff, v, 1);
    return v;
}

__global__ void hmc_kernel_closed(const float* __restrict__ q, const float* __restrict__ p0, const float* __restrict__ u,
                                  float* __restrict__ qout, bool* __restrict__ acc,
                                  int D, float acoef, float bcoef, float ccoef, float dcoef) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int base = b * D;
    float qv = 0.0f, pv0 = 0.0f, qn = 0.0f, pf = 0.0f, sum = 0.0f;
    if (tid < D) {
        qv = q[base + tid];
        pv0 = p0[base + tid];
        qn = acoef * qv + bcoef * pv0;
        pf = ccoef * qv + dcoef * pv0;
        sum = 0.5f * (qv * qv + pv0 * pv0 - qn * qn - pf * pf);
    }

    sum = warp_reduce_sum(sum);
    __shared__ float warp_sums[2];
    if ((tid & 31) == 0) warp_sums[tid >> 5] = sum;
    __syncthreads();

    if (tid < 32) {
        float total = (tid < 2) ? warp_sums[tid] : 0.0f;
        total = warp_reduce_sum(total);
        if (tid == 0) {
            float uu = u[b];
            if (uu < 1.0e-10f) uu = 1.0e-10f;
            bool ac = logf(uu) < total;
            acc[b] = ac;
            warp_sums[0] = ac ? 1.0f : 0.0f;
        }
    }
    __syncthreads();

    if (tid < D) qout[base + tid] = (warp_sums[0] != 0.0f) ? qn : qv;
}

static inline void coeffs(int steps, float eps, float &a, float &bb, float &c, float &d) {
    a = 1.0f; bb = 0.0f; c = -0.5f * eps; d = 1.0f;
    for (int i = 0; i < steps - 1; ++i) {
        a = a + eps * c;
        bb = bb + eps * d;
        c = c - eps * a;
        d = d - eps * bb;
    }
    a = a + eps * c;
    bb = bb + eps * d;
    c = c - 0.5f * eps * a;
    d = d - 0.5f * eps * bb;
}

std::vector<torch::Tensor> hmc_forward(torch::Tensor q, int64_t steps, double epsd) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(q));
    TORCH_CHECK(q.is_cuda(), "q must be CUDA");
    TORCH_CHECK(q.scalar_type() == at::kFloat, "float32 only");
    auto qc = q.contiguous();
    int B = (int)qc.size(0), D = (int)qc.size(1);

    auto p = torch::randn_like(qc);
    auto u = torch::rand({B}, qc.options());
    auto qout = torch::empty_like(qc);
    auto acc = torch::empty({B}, torch::TensorOptions().device(q.device()).dtype(torch::kBool));

    float a, bb, c, d;
    coeffs((int)steps, (float)epsd, a, bb, c, d);
    hmc_kernel_closed<<<B, 64, 0, at::cuda::getCurrentCUDAStream()>>>(
        qc.data_ptr<float>(), p.data_ptr<float>(), u.data_ptr<float>(),
        qout.data_ptr<float>(), acc.data_ptr<bool>(), D, a, bb, c, d);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return {qout, acc};
}

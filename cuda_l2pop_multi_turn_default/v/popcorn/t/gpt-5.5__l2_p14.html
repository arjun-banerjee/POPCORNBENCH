import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Triangular multiplicative update – outgoing edges.
    From AlphaFold2 Evoformer: updates pair (i,j) by aggregating over the
    shared index k via element-wise products of projected pair values along
    edges (i,k) and (j,k).
    """

    def __init__(self, pair_dim, hidden_dim):
        super().__init__()
        self.layer_norm_in = nn.LayerNorm(pair_dim)
        self.proj_a = nn.Linear(pair_dim, hidden_dim)
        self.gate_a = nn.Linear(pair_dim, hidden_dim)
        self.proj_b = nn.Linear(pair_dim, hidden_dim)
        self.gate_b = nn.Linear(pair_dim, hidden_dim)
        self.layer_norm_out = nn.LayerNorm(hidden_dim)
        self.out_proj = nn.Linear(hidden_dim, pair_dim)
        self.out_gate = nn.Linear(pair_dim, pair_dim)

    def forward(self, pair_repr: torch.Tensor, pair_mask: torch.Tensor) -> torch.Tensor:
        """
        Args:
            pair_repr: (B, N, N, pair_dim)
            pair_mask: (B, N, N, 1)
        Returns:
            updated pair representation (B, N, N, pair_dim)
        """
        mask = pair_mask.unsqueeze(-1)
        x = self.layer_norm_in(pair_repr)

        a = self.proj_a(x) * torch.sigmoid(self.gate_a(x)) * mask  # (B, N, N, H)
        b = self.proj_b(x) * torch.sigmoid(self.gate_b(x)) * mask

        # Outgoing: aggregate over k with a[i,k] * b[j,k]
        out = torch.einsum("bikh,bjkh->bijh", a, b)

        out = self.layer_norm_out(out)
        out = self.out_proj(out)
        out = torch.sigmoid(self.out_gate(pair_repr)) * out
        return out

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define H_DIM 32

__global__ void einsum_kernel(const float* __restrict__ a,
                              const float* __restrict__ b,
                              float* __restrict__ out,
                              int B, int N) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long total = (long long)B * N * N * H_DIM;
    if (idx >= total) return;

    int h = idx & 31;
    long long q = idx >> 5;
    int j = q % N; q /= N;
    int i = q % N;
    int bb = q / N;

    long long base_a = (((long long)bb * N + i) * N) * H_DIM + h;
    long long base_b = (((long long)bb * N + j) * N) * H_DIM + h;

    float sum = 0.0f;
    #pragma unroll
    for (int k = 0; k < 32; ++k) {
        sum += a[base_a + (long long)k * H_DIM] * b[base_b + (long long)k * H_DIM];
    }
    out[idx] = sum;
}

torch::Tensor einsum_cuda(torch::Tensor a, torch::Tensor b) {
    int B = a.size(0);
    int N = a.size(1);
    auto out = torch::empty({B, N, N, H_DIM}, a.options());
    long long total = (long long)B * N * N * H_DIM;
    int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    einsum_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        a.data_ptr<float>(), b.data_ptr<float>(), out.data_ptr<float>(), B, N);
    return out;
}

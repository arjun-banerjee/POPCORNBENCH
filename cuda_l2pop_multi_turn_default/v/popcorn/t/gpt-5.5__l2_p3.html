import math

import torch
import torch.nn as nn
import torch.nn.functional as F

# Default ModelNew is PyTorch-only. run_and_check: use check_kernel=False with backend=cuda
# until the candidate file contains a real CUDA kernel (see kernel_static_checker).

class Model(nn.Module):
    """
    Single-layer multi-head cross-attention with RoPE on Q and K, attention
    dropout, and output projection—intended as a fusion target for robotics
    stacks (e.g. proprio / action queries attending to vision or memory).

    Pipeline (conceptually one fused kernel): linear Q from query stream,
    linear K/V from key–value stream → reshape heads → apply RoPE separately
    along query positions (T_q) and key positions (T_k) → scaled dot-product,
    softmax, dropout on weights → aggregate values → output projection.
    """

    def __init__(
        self,
        dim: int,
        num_heads: int,
        dropout: float = 0.0,
        rope_base: float = 10000.0,
    ):
        super().__init__()
        assert dim % num_heads == 0
        self.dim = dim
        self.num_heads = num_heads
        self.head_dim = dim // num_heads
        assert self.head_dim % 2 == 0, "RoPE requires an even head dimension"

        self.q_proj = nn.Linear(dim, dim, bias=False)
        self.k_proj = nn.Linear(dim, dim, bias=False)
        self.v_proj = nn.Linear(dim, dim, bias=False)
        self.out_proj = nn.Linear(dim, dim, bias=False)
        self.dropout = nn.Dropout(dropout)
        self.scale = 1.0 / math.sqrt(self.head_dim)

        inv_freq = 1.0 / (
            rope_base ** (torch.arange(0, self.head_dim, 2).float() / self.head_dim)
        )
        self.register_buffer("inv_freq", inv_freq, persistent=False)

    @staticmethod
    def _rotate_half(x: torch.Tensor) -> torch.Tensor:
        x1, x2 = x.chunk(2, dim=-1)
        return torch.cat([-x2, x1], dim=-1)

    def _rope(self, x: torch.Tensor, seq_len: int) -> torch.Tensor:
        """Apply RoPE along the sequence dimension (last-but-one after head). x: (B, H, L, d)."""
        t = torch.arange(seq_len, device=x.device, dtype=self.inv_freq.dtype)
        freqs = torch.outer(t, self.inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        cos = emb.cos().view(1, 1, seq_len, self.head_dim)
        sin = emb.sin().view(1, 1, seq_len, self.head_dim)
        return x * cos + self._rotate_half(x) * sin

    def forward(self, x_query: torch.Tensor, x_kv: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x_query: (B, T_q, dim) — e.g. robot or decoder tokens
            x_kv:    (B, T_k, dim) — e.g. encoder / memory tokens
        Returns:
            (B, T_q, dim)
        """
        B, T_q, _ = x_query.shape
        _, T_k, _ = x_kv.shape
        h, d = self.num_heads, self.head_dim

        q = self.q_proj(x_query).view(B, T_q, h, d).transpose(1, 2)
        k = self.k_proj(x_kv).view(B, T_k, h, d).transpose(1, 2)
        v = self.v_proj(x_kv).view(B, T_k, h, d).transpose(1, 2)

        q = self._rope(q, T_q)
        k = self._rope(k, T_k)

        attn = torch.matmul(q, k.transpose(-2, -1)) * self.scale
        attn = F.softmax(attn, dim=-1)
        attn = self.dropout(attn)
        out = torch.matmul(attn, v)
        out = out.transpose(1, 2).contiguous().view(B, T_q, self.dim)
        return self.out_proj(out)

    int half=Hd>>1; int idx=blockIdx.x*blockDim.x+threadIdx.x; int total=B*L*H*half; if(idx>=total) return;
    int j=idx%half; int tmp=idx/half; int head=tmp%H; tmp/=H; int t=tmp%L; int b=tmp/L;
    int od0=head*Hd+j, od1=od0+half; const float* row=inp+(b*L+t)*D; float s0=row[od0], s1=row[od1];
    float inv=powf(rope_base, -((float)(2*j)/(float)Hd)); float sn,cs; sincosf((float)t*inv,&sn,&cs);
    int bh=b*H+head; out[(bh*Hd+j)*L+t]=s0*cs-s1*sn; out[(bh*Hd+j+half)*L+t]=s1*cs+s0*sn;
}

__global__ void attention_kt_vflat_kernel(const float* __restrict__ q, const float* __restrict__ kt, const float* __restrict__ vflat, float* __restrict__ ctx, int B,int Tq,int Tk,int H,int Hd,int D,float scale){
    int bid=blockIdx.x; int tq=bid%Tq; int tmp=bid/Tq; int head=tmp%H; int b=tmp/H; int tid=threadIdx.x;
    __shared__ float scores[128]; __shared__ float red[128]; const float* qp=q+((b*H+head)*Tq+tq)*Hd; int bh=b*H+head;
    float val=-3.402823466e38f;
    if(tid<Tk){ float dot=0.0f;
        #pragma unroll
        for(int dd=0; dd<32; ++dd) dot=fmaf(qp[dd], kt[(bh*Hd+dd)*Tk+tid], dot);
        val=dot*scale; scores[tid]=val; }
    red[tid]=val; __syncthreads();
    for(int s=64; s>0; s>>=1){ if(tid<s) red[tid]=fmaxf(red[tid],red[tid+s]); __syncthreads(); }
    float m=red[0]; float e=0.0f; if(tid<Tk){ e=expf(scores[tid]-m); scores[tid]=e; }
    red[tid]=e; __syncthreads();
    for(int s=64; s>0; s>>=1){ if(tid<s) red[tid]+=red[tid+s]; __syncthreads(); }
    float invsum=1.0f/red[0];
    if(tid<32){ float acc=0.0f; int voff=head*Hd+tid; for(int kk=0; kk<128; ++kk) acc=fmaf(scores[kk]*invsum, vflat[(b*Tk+kk)*D+voff], acc); ctx[(b*Tq+tq)*D+head*Hd+tid]=acc; }
}

torch::Tensor forward_cuda(torch::Tensor xq, torch::Tensor xkv, torch::Tensor wq, torch::Tensor wk, torch::Tensor wv, torch::Tensor wo, int64_t num_heads, double rope_base){
    CHECK_CUDA(xq); CHECK_CUDA(xkv); CHECK_CUDA(wq); CHECK_CUDA(wk); CHECK_CUDA(wv); CHECK_CUDA(wo); CHECK_CONTIGUOUS(xq); CHECK_CONTIGUOUS(xkv); CHECK_CONTIGUOUS(wq); CHECK_CONTIGUOUS(wk); CHECK_CONTIGUOUS(wv); CHECK_CONTIGUOUS(wo);
    int B=(int)xq.size(0), Tq=(int)xq.size(1), D=(int)xq.size(2), Tk=(int)xkv.size(1), H=(int)num_heads, Hd=D/H;
    auto xq2=xq.reshape({B*Tq,D}); auto xkv2=xkv.reshape({B*Tk,D});
    auto qflat=torch::matmul(xq2,wq.t()); auto kflat=torch::matmul(xkv2,wk.t()); auto vflat=torch::matmul(xkv2,wv.t());
    auto opts=xq.options(); auto q=torch::empty({B,H,Tq,Hd},opts); auto kt=torch::empty({B,H,Hd,Tk},opts); auto ctx=torch::empty({B*Tq,D},opts);
    const int threads=256; int nqp=B*Tq*H*(Hd/2), nkp=B*Tk*H*(Hd/2);
    rope_q_kernel<<<(nqp+threads-1)/threads,threads>>>(qflat.data_ptr<float>(),q.data_ptr<float>(),B,Tq,D,H,Hd,(float)rope_base);

    auto out=torch::matmul(ctx,wo.t()); return out.reshape({B,Tq,D});
}

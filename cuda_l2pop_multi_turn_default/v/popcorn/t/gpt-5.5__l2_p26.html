import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Apply rotary position embedding to incoming keys and update the KV cache.

    This isolates a common decode-time kernel: rotate the fresh key vectors for
    the current positions and store both keys and values into the running cache.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        k_new: torch.Tensor,
        v_new: torch.Tensor,
        cos: torch.Tensor,
        sin: torch.Tensor,
        cache_k: torch.Tensor,
        cache_v: torch.Tensor,
        positions: torch.Tensor,
    ) -> torch.Tensor:
        updated_k = cache_k.clone()
        updated_v = cache_v.clone()
        rot_dim = k_new.shape[-1]

        for b in range(k_new.shape[0]):
            for t in range(k_new.shape[1]):
                pos = int(positions[b, t].item())
                c = cos[pos]
                s = sin[pos]
                x = k_new[b, t]
                x_even = x[..., 0::2]
                x_odd = x[..., 1::2]
                rot_even = x_even * c - x_odd * s
                rot_odd = x_even * s + x_odd * c
                rotated = torch.empty_like(x)
                rotated[..., 0::2] = rot_even
                rotated[..., 1::2] = rot_odd
                updated_k[b, pos] = rotated
                updated_v[b, pos] = v_new[b, t]

        return torch.stack((updated_k, updated_v), dim=0)

def _rope_tables(cache_len: int, half_dim: int):
    positions = torch.arange(cache_len, dtype=torch.float32).unsqueeze(1)
    freqs = torch.pow(10000.0, -torch.arange(half_dim, dtype=torch.float32) / half_dim).unsqueeze(0)
    angles = positions * freqs
    return torch.cos(angles), torch.sin(angles)

#include <cuda.h>

template <typename PosT>
__global__ void rope_update_pair_kernel(const float* __restrict__ k_new,
                                        const float* __restrict__ v_new,
                                        const float* __restrict__ cos,
                                        const float* __restrict__ sin,
                                        const float* __restrict__ cache_k,
                                        const float* __restrict__ cache_v,
                                        const PosT* __restrict__ positions,
                                        float* __restrict__ out,
                                        int B, int T, int C, int H, int D) {
    int row = blockIdx.x;
    int h = row % H;
    int tmp = row / H;
    int pos = tmp % C;
    int b = tmp / C;
    int lane = threadIdx.x;

    int m = -1;
    if (lane == 0) {
        const PosT* pr = positions + b * T;
        #pragma unroll
        for (int t = 0; t < 64; ++t) {
            if (t >= T) break;
            if ((int)pr[t] == pos) m = t;
        }
    }
    m = __shfl_sync(0xffffffff, m, 0);

    int half = D >> 1;
    int64_t N = (int64_t)B * C * H * D;
    int64_t out_base = (((int64_t)b * C + pos) * H + h) * D;

    for (int pair = lane; pair < half; pair += blockDim.x) {
        int d0 = pair << 1;
        int d1 = d0 + 1;
        float k0 = cache_k[out_base + d0];
        float k1 = cache_k[out_base + d1];
        float v0 = cache_v[out_base + d0];
        float v1 = cache_v[out_base + d1];
        if (m >= 0) {
            int64_t in_base = (((int64_t)b * T + m) * H + h) * D;
            float xe = k_new[in_base + d0];
            float xo = k_new[in_base + d1];
            float c = cos[(int64_t)pos * half + pair];
            float s = sin[(int64_t)pos * half + pair];
            k0 = xe * c - xo * s;
            k1 = xe * s + xo * c;
            v0 = v_new[in_base + d0];
            v1 = v_new[in_base + d1];
        }
        out[out_base + d0] = k0;
        out[out_base + d1] = k1;
        out[N + out_base + d0] = v0;
        out[N + out_base + d1] = v1;
    }
}

                                  torch::Tensor v_new,
                                  torch::Tensor cos,

                                  torch::Tensor positions) {

    int T = (int)k_new.size(1);
    int H = (int)k_new.size(2);
    int D = (int)k_new.size(3);
    int C = (int)cache_k.size(1);
    auto out = torch::empty({2, B, C, H, D}, cache_k.options());
    int blocks = B * C * H;
    const int threads = 32;
    auto st = positions.scalar_type();

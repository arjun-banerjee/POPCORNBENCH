import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Low-rank expansion of compressed KV activations into multi-head K/V tensors.

    This isolates the MLA-style compressed-cache pattern: a hidden activation is
    projected into a low-rank latent space and then expanded back into separate
    key and value tensors.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        hidden: torch.Tensor,
        w_down: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, model_dim = hidden.shape
        rank = w_down.shape[1]
        kv_heads = 4
        head_dim = w_up_k.shape[1] // kv_heads

        latent = hidden.reshape(batch_size * seq_len, model_dim) @ w_down
        k = latent @ w_up_k
        v = latent @ w_up_v
        k = k.view(batch_size, seq_len, kv_heads, head_dim)
        v = v.view(batch_size, seq_len, kv_heads, head_dim)
        return torch.stack((k, v), dim=2)

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE 16

__global__ void down_kernel(const float* __restrict__ hidden,
                            const float* __restrict__ w_down,
                            float* __restrict__ latent) {
    __shared__ float As[TILE][TILE];
    __shared__ float Bs[TILE][TILE];
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int col = blockIdx.x * TILE + tx;
    int row = blockIdx.y * TILE + ty;
    double acc = 0.0;
    #pragma unroll
    for (int kt = 0; kt < 256; kt += TILE) {
        As[ty][tx] = hidden[row * 256 + (kt + tx)];
        Bs[ty][tx] = (col < 48) ? w_down[(kt + ty) * 48 + col] : 0.0f;
        __syncthreads();
        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            acc += (double)As[ty][k] * (double)Bs[k][tx];
        }
        __syncthreads();
    }
    if (col < 48) latent[row * 48 + col] = (float)acc;
}

__global__ void up_kernel(const float* __restrict__ latent,
                          const float* __restrict__ w_up_k,
                          const float* __restrict__ w_up_v,
                          float* __restrict__ out) {
    __shared__ float As[TILE][TILE];
    __shared__ float Bk[TILE][TILE];
    __shared__ float Bv[TILE][TILE];
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int col = blockIdx.x * TILE + tx;
    int row = blockIdx.y * TILE + ty;
    double acck = 0.0;
    double accv = 0.0;
    #pragma unroll
    for (int kt = 0; kt < 48; kt += TILE) {
        As[ty][tx] = latent[row * 48 + (kt + tx)];
        Bk[ty][tx] = w_up_k[(kt + ty) * 128 + col];
        Bv[ty][tx] = w_up_v[(kt + ty) * 128 + col];
        __syncthreads();
        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            double a = (double)As[ty][k];
            acck += a * (double)Bk[k][tx];
            accv += a * (double)Bv[k][tx];
        }
        __syncthreads();
    }
    out[(row * 2) * 128 + col] = (float)acck;
    out[(row * 2 + 1) * 128 + col] = (float)accv;
}

}

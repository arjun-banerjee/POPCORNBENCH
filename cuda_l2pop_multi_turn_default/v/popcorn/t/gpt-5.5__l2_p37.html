import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Fused sparse attention-value aggregation in CSR format.

    For each destination row, compute a softmax over edge logits and then use
    the normalized weights to aggregate source node values. This is a common
    fused graph-attention pattern.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        row_ptr: torch.Tensor,
        col_idx: torch.Tensor,
        edge_scores: torch.Tensor,
        node_value: torch.Tensor,
    ) -> torch.Tensor:
        num_nodes = row_ptr.numel() - 1
        feat_dim = node_value.shape[1]
        out = torch.zeros(num_nodes, feat_dim, dtype=node_value.dtype, device=node_value.device)

        for dst in range(num_nodes):
            start = int(row_ptr[dst].item())
            end = int(row_ptr[dst + 1].item())
            if end > start:
                weights = torch.softmax(edge_scores[start:end], dim=0)
                src = col_idx[start:end].long()
                out[dst] = (node_value[src] * weights.unsqueeze(-1)).sum(dim=0)

        return out

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <stdint.h>

#define FEAT_DIM 64
#define MAX_DEG 32

template <typename T>
__device__ __forceinline__ int cvt_idx(T x) { return (int)x; }

template <typename RowT, typename ColT>
__global__ void gat_kernel_t(const RowT* __restrict__ row_ptr,
                             const ColT* __restrict__ col_idx,
                             const float* __restrict__ edge_scores,
                             const float* __restrict__ node_value,
                             float* __restrict__ out) {
    const int dst = blockIdx.x;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int start = cvt_idx(row_ptr[dst]);
    const int end = cvt_idx(row_ptr[dst + 1]);
    const int deg = end - start;
    __shared__ float weights[MAX_DEG];
    __shared__ int cols[MAX_DEG];
    __shared__ float inv_s;

    if (deg <= 0) {
        out[dst * FEAT_DIM + tid] = 0.0f;
        return;
    }

    if (tid < 32) {
        float v = (lane < deg) ? edge_scores[start + lane] : -FLT_MAX;
        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) v = fmaxf(v, __shfl_down_sync(0xffffffff, v, off));
        float m = __shfl_sync(0xffffffff, v, 0);
        float w = 0.0f;
        if (lane < deg) {
            w = __expf(edge_scores[start + lane] - m);
            weights[lane] = w;
            cols[lane] = cvt_idx(col_idx[start + lane]);
        }
        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) w += __shfl_down_sync(0xffffffff, w, off);
        if (lane == 0) inv_s = 1.0f / w;
    }
    __syncthreads();

    float acc = 0.0f;
    #pragma unroll
    for (int e = 0; e < MAX_DEG; ++e) {
        if (e < deg) acc = fmaf(weights[e] * inv_s, node_value[cols[e] * FEAT_DIM + tid], acc);
    }
    out[dst * FEAT_DIM + tid] = acc;
}

template <typename RowT, typename ColT>
void launch_t(torch::Tensor row_ptr, torch::Tensor col_idx, torch::Tensor edge_scores, torch::Tensor node_value, torch::Tensor out) {
    int num_nodes = (int)(row_ptr.numel() - 1);
    gat_kernel_t<RowT,ColT><<<num_nodes, 64>>>(row_ptr.data_ptr<RowT>(), col_idx.data_ptr<ColT>(), edge_scores.data_ptr<float>(), node_value.data_ptr<float>(), out.data_ptr<float>());
}

void gat_forward_cuda(torch::Tensor row_ptr, torch::Tensor col_idx, torch::Tensor edge_scores, torch::Tensor node_value, torch::Tensor out, int row_type, int col_type) {
    if (row_type == 0) {
        if (col_type == 0) launch_t<int,int>(row_ptr,col_idx,edge_scores,node_value,out);
        else if (col_type == 1) launch_t<int,int64_t>(row_ptr,col_idx,edge_scores,node_value,out);
        else launch_t<int,float>(row_ptr,col_idx,edge_scores,node_value,out);
    } else if (row_type == 1) {
        if (col_type == 0) launch_t<int64_t,int>(row_ptr,col_idx,edge_scores,node_value,out);
        else if (col_type == 1) launch_t<int64_t,int64_t>(row_ptr,col_idx,edge_scores,node_value,out);
        else launch_t<int64_t,float>(row_ptr,col_idx,edge_scores,node_value,out);
    } else {
        if (col_type == 0) launch_t<float,int>(row_ptr,col_idx,edge_scores,node_value,out);
        else if (col_type == 1) launch_t<float,int64_t>(row_ptr,col_idx,edge_scores,node_value,out);
        else launch_t<float,float>(row_ptr,col_idx,edge_scores,node_value,out);
    }
}

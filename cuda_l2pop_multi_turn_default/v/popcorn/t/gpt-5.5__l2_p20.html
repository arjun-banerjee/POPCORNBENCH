import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Scalar-gated delta-rule linear attention recurrence.

    A recurrent state matrix is decayed by a scalar gate per head, updated by a
    rank-1 key/value outer product, then read out by a query vector.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        gate: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        state = torch.zeros(batch_size, num_heads, head_dim, head_dim, dtype=q.dtype, device=q.device)
        outputs = []

        for t in range(seq_len):
            gate_t = gate[:, t].unsqueeze(-1).unsqueeze(-1)
            beta_t = beta[:, t].unsqueeze(-1).unsqueeze(-1)
            kv_outer = k[:, t].unsqueeze(-1) * v[:, t].unsqueeze(-2)
            state = gate_t * state + beta_t * kv_outer
            y_t = (q[:, t].unsqueeze(-1) * state).sum(dim=-2)
            outputs.append(y_t)

        return torch.stack(outputs, dim=1)

torch::Tensor delta_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor gate, torch::Tensor beta) {

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ __launch_bounds__(32) void delta_warp_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    const float* __restrict__ gate,
    const float* __restrict__ beta,
    float* __restrict__ out) {

    constexpr int T = 128;
    constexpr int H = 4;
    constexpr int D = 24;

    int bid = blockIdx.x;
    int j = bid % D;
    int tmp = bid / D;
    int h = tmp & 3;
    int b = tmp >> 2;
    int lane = threadIdx.x;

    float s = 0.0f;
    const unsigned mask = 0xffffffffu;
    int bh_base0 = b * T * H + h;

    #pragma unroll 1
    for (int t = 0; t < T; ++t) {
        int bh_base = bh_base0 + t * H;
        int base = bh_base * D;

        float g = gate[bh_base];
        float be = beta[bh_base];
        float vj = v[base + j];
        float prod = 0.0f;

        if (lane < D) {
            float kval = k[base + lane];
            s = g * s + (be * kval) * vj;
            prod = q[base + lane] * s;
        }

        prod += __shfl_down_sync(mask, prod, 16);
        prod += __shfl_down_sync(mask, prod, 8);
        prod += __shfl_down_sync(mask, prod, 4);
        prod += __shfl_down_sync(mask, prod, 2);
        prod += __shfl_down_sync(mask, prod, 1);

        if (lane == 0) {
            out[base + j] = prod;
        }
    }
}

torch::Tensor delta_forward_cuda(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor gate, torch::Tensor beta) {
    auto out = torch::empty_like(q);
    delta_warp_kernel<<<192, 32>>>(
        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(),
        gate.data_ptr<float>(), beta.data_ptr<float>(), out.data_ptr<float>());
    return out;
}
'''

def _fake_quant_symmetric_int8(x: torch.Tensor, scale: torch.Tensor) -> torch.Tensor:
    """Symmetric int8 fake-quant: round(x/scale) clamped to [-127, 127], times scale."""
    q = torch.clamp(torch.round(x / scale), -127.0, 127.0)
    return q * scale


class Model(nn.Module):
    """
    **QuantVLA selective layout fusion** (reference target for a fused CUDA/Triton kernel).

    Vision–language–action stacks often use **token-wise mixed precision**: some
    timesteps or modalities tolerate aggressive quantization while others stay in
    full precision. This module fuses in **one forward**:

      1. **Layout routing** — ``sigmoid(layout_logits)`` yields a per-token mix
         coefficient in ``(0, 1)`` (quant vs FP residual).
      2. **Quantitative path** — global symmetric **int8-style fake quantization**
         (scale from the tensor max, ``/ 127``).
      3. **Selective blend** — ``α · x_quant + (1 - α) · x``.
      4. **Head** — linear projection on the mixed activations.

    This matches a “quantitative selective layout” block: the layout logits (from a
    router, modality tags, or calibration) choose how strongly each position uses the
    quantized layout vs the identity layout.
    """

    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim
        self.out_proj = nn.Linear(dim, dim, bias=True)

    def forward(self, x: torch.Tensor, layout_logits: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (B, T, D) VLA token activations (fused modalities).
            layout_logits: (B, T) unbounded logits; larger → more mass on quantized path.
        Returns:
            (B, T, D) updated activations.
        """
        B, T, D = x.shape
        assert D == self.dim
        assert layout_logits.shape == (B, T)

        alpha = torch.sigmoid(layout_logits).unsqueeze(-1)
        with torch.no_grad():
            amax = x.detach().abs().max().clamp(min=1e-8)
        scale = amax / 127.0
        x_q = _fake_quant_symmetric_int8(x, scale)
        mixed = alpha * x_q + (1.0 - alpha) * x
        return self.out_proj(mixed)

from torch.utils.cpp_extension import load_inline

_cuda = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cooperative_groups.h>
#include <math.h>

namespace cg = cooperative_groups;

#define BM 8
#define BN 32
#define BK 32

__global__ void coop_fused_kernel(const float* __restrict__ x,
                                  const float* __restrict__ logits,
                                  const float* __restrict__ weight,
                                  const float* __restrict__ bias,
                                  float* __restrict__ partial,
                                  float* __restrict__ out,
                                  int M, int D, int n) {
    cg::grid_group grid = cg::this_grid();
    extern __shared__ float red[];
    __shared__ float As[BM][BK + 1];
    __shared__ float Bs[BK][BN + 1];

    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int tid = ty * BN + tx;
    const int nthreads = blockDim.x * blockDim.y;
    const int bid = blockIdx.y * gridDim.x + blockIdx.x;
    const int total_blocks = gridDim.x * gridDim.y;

    // Distributed global absmax reduction over x.  For the benchmark shape this
    // is exactly one element per thread, and the following grid sync makes the
    // scale available to the GEMM part without a second kernel launch.
    float m = 0.0f;
    for (int i = bid * nthreads + tid; i < n; i += total_blocks * nthreads) {
        m = fmaxf(m, fabsf(x[i]));
    }
    red[tid] = m;
    __syncthreads();
    for (int stride = nthreads >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) red[tid] = fmaxf(red[tid], red[tid + stride]);
        __syncthreads();
    }
    if (tid == 0) partial[bid] = red[0];

    grid.sync();

    if (bid == 0) {
        float mm = 0.0f;
        for (int i = tid; i < total_blocks; i += nthreads) {
            mm = fmaxf(mm, partial[i]);
        }
        red[tid] = mm;
        __syncthreads();
        for (int stride = nthreads >> 1; stride > 0; stride >>= 1) {
            if (tid < stride) red[tid] = fmaxf(red[tid], red[tid + stride]);
            __syncthreads();
        }
        if (tid == 0) partial[0] = fmaxf(red[0], 1.0e-8f) * (1.0f / 127.0f);
    }

    grid.sync();

    const float scale = partial[0];
    const float inv_scale = 1.0f / scale;
    const int row = blockIdx.y * BM + ty;
    const int col = blockIdx.x * BN + tx;
    float acc = 0.0f;

    for (int k0 = 0; k0 < D; k0 += BK) {
        // Mixed activation tile: fake quantize, sigmoid route, and blend.
        int ar = tid >> 5;
        int ak = tid & 31;
        int grow = blockIdx.y * BM + ar;
        int gk = k0 + ak;
        float av = 0.0f;
        if (grow < M && gk < D) {
            float xv = x[grow * D + gk];
            float q = nearbyintf(xv * inv_scale);
            q = fminf(127.0f, fmaxf(-127.0f, q));
            float a = 1.0f / (1.0f + expf(-logits[grow]));
            av = fmaf(a, q * scale - xv, xv);
        }
        As[ar][ak] = av;

        // Weight tile, using PyTorch Linear layout: output = input @ weight.T + bias.
        #pragma unroll
        for (int e = 0; e < 4; ++e) {
            int idx = tid + e * (BM * BN);
            int bk = idx >> 5;
            int bc = idx & 31;
            int wk = k0 + bk;
            int wc = blockIdx.x * BN + bc;
            float w = 0.0f;
            if (wk < D && wc < D) w = weight[wc * D + wk];
            Bs[bk][bc] = w;
        }
        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            acc = fmaf(As[ty][kk], Bs[kk][tx], acc);
        }
        __syncthreads();
    }

    if (row < M && col < D) {
        out[row * D + col] = acc + bias[col];
    }
}

torch::Tensor quant_layout_linear_forward(torch::Tensor x,
                                          torch::Tensor layout_logits,
                                          torch::Tensor weight,
                                          torch::Tensor bias) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(x));
    x = x.contiguous();
    layout_logits = layout_logits.contiguous();
    weight = weight.contiguous();
    bias = bias.contiguous();

    const int64_t B = x.size(0);
    const int64_t T = x.size(1);
    const int64_t D64 = x.size(2);
    const int M = (int)(B * T);
    const int D = (int)D64;

    auto out = torch::empty({B, T, D64}, x.options());
    dim3 block(BN, BM);
    dim3 grid((D + BN - 1) / BN, (M + BM - 1) / BM);
    int total_blocks = (int)(grid.x * grid.y);
    auto partial = torch::empty({total_blocks}, x.options());
    int n = M * D;

    const float* xp = x.data_ptr<float>();
    const float* lp = layout_logits.data_ptr<float>();
    const float* wp = weight.data_ptr<float>();
    const float* bp = bias.data_ptr<float>();
    float* pp = partial.data_ptr<float>();
    float* op = out.data_ptr<float>();

    void* args[] = {(void*)&xp, (void*)&lp, (void*)&wp, (void*)&bp,
                    (void*)&pp, (void*)&op, (void*)&M, (void*)&D, (void*)&n};
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cudaLaunchCooperativeKernel((void*)coop_fused_kernel, grid, block, args,
                                256 * sizeof(float), stream);
    return out;
}

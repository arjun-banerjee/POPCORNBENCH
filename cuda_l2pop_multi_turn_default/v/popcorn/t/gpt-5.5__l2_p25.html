import torch
import torch.nn as nn
import math


class Model(nn.Module):
    """
    Outer-product mean: converts MSA representation to a pair
    representation update.  Used in AlphaFold2 to communicate information
    from the MSA stack to the pair stack.  For each pair (i,j), the
    outer product of projected MSA features at positions i and j is
    averaged over the MSA depth dimension.
    """

    def __init__(self, msa_dim, pair_dim, hidden_dim=32):
        super().__init__()
        self.layer_norm = nn.LayerNorm(msa_dim)
        self.proj_a = nn.Linear(msa_dim, hidden_dim)
        self.proj_b = nn.Linear(msa_dim, hidden_dim)
        self.out_proj = nn.Linear(hidden_dim * hidden_dim, pair_dim)

    def forward(self, msa_repr: torch.Tensor, msa_mask: torch.Tensor) -> torch.Tensor:
        """
        Args:
            msa_repr: (B, S, N, msa_dim) – MSA representation
            msa_mask: (B, S, N)          – 1 for valid rows/positions
        Returns:
            pair update (B, N, N, pair_dim)
        """
        x = self.layer_norm(msa_repr)
        a = self.proj_a(x)  # (B, S, N, H)
        b = self.proj_b(x)

        mask = msa_mask.unsqueeze(-1)  # (B, S, N, 1)
        a = a * mask
        b = b * mask

        outer = torch.einsum("bsih,bsjk->bijhk", a, b)  # (B, N, N, H, H)
        denom = msa_mask.sum(dim=1).unsqueeze(-1).unsqueeze(-1).unsqueeze(-1).clamp(min=1)
        outer = outer / denom  # mean over S

        B, N1, N2, H1, H2 = outer.shape
        outer = outer.reshape(B, N1, N2, H1 * H2)
        return self.out_proj(outer)

#include <torch/extension.h>

#include <cuda.h>

__device__ __forceinline__ float block_reduce_sum(float v) {
    __shared__ float smem[128];
    int tid = threadIdx.x;
    smem[tid] = v;
    __syncthreads();
    for (int stride = 64; stride > 0; stride >>= 1) {
        if (tid < stride) smem[tid] += smem[tid + stride];
        __syncthreads();
    }
    return smem[0];
}

__global__ void ln_proj_kernel(const float* __restrict__ x, const float* __restrict__ mask,
                               const float* __restrict__ ln_w, const float* __restrict__ ln_b,
                               const float* __restrict__ wa, const float* __restrict__ ba,
                               const float* __restrict__ wb, const float* __restrict__ bb,
                               float* __restrict__ A, float* __restrict__ Bv,
                               int Bsz, int S, int N, int D, int H) {
    int vec = blockIdx.x;
    int tid = threadIdx.x;
    int n = vec % N;
    int tmp = vec / N;
    int s = tmp % S;
    int b = tmp / S;
    const float* xp = x + vec * D;
    float local = 0.f;
    for (int d=tid; d<D; d+=128) local += xp[d];
    float mean = block_reduce_sum(local) * (1.0f / (float)D);
    local = 0.f;
    for (int d=tid; d<D; d+=128) { float z=xp[d]-mean; local += z*z; }
    float inv = rsqrtf(block_reduce_sum(local) * (1.0f / (float)D) + 1.0e-5f);
    float m = mask[(b*S+s)*N+n];
    if (tid < H) {
        float sa = ba[tid];
        float sb = bb[tid];
        #pragma unroll
        for (int d=0; d<64; ++d) {
            float y = (xp[d] - mean) * inv;
            y = y * ln_w[d] + ln_b[d];
            sa += y * wa[tid*D + d];
            sb += y * wb[tid*D + d];
        }
        A[vec*H + tid] = sa * m;
        Bv[vec*H + tid] = sb * m;
    }
}

// Precontract B-projection with the output projection along the second hidden axis:
// C[b,s,j,h,p] = sum_k Bv[b,s,j,k] * wo[p,h,k].
__global__ void transform_b_kernel(const float* __restrict__ Bv, const float* __restrict__ wo,
                                   float* __restrict__ C,
                                   int Bsz, int S, int N, int H, int P) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = Bsz*S*N*H*P;
    if (idx >= total) return;
    int p = idx & 63;       // fixed problem sizes: P=64
    int h = (idx >> 6) & 31; // H=32
    int j = (idx >> 11) & 31;// N=32
    int s = (idx >> 16) & 7; // S=8
    int b = idx >> 19;
    const float* bp = Bv + ((b*S+s)*N + j)*H;
    const float* wp = wo + p*(H*H) + h*H;
    float acc = 0.f;
    #pragma unroll
    for (int k=0; k<32; ++k) acc += bp[k] * wp[k];
    C[idx] = acc;
}

__global__ void final_kernel2(const float* __restrict__ A, const float* __restrict__ C,
                              const float* __restrict__ mask, const float* __restrict__ bo,
                              float* __restrict__ out,
                              int Bsz, int S, int N, int H, int P) {
    int pair = blockIdx.x;
    int p = threadIdx.x;
    int j = pair & 31;
    int t = pair >> 5;
    int i = t & 31;
    int b = t >> 5;
    float acc = 0.f;
    #pragma unroll
    for (int h=0; h<32; ++h) {
        #pragma unroll
        for (int s=0; s<8; ++s) {
            float av = A[((b*S+s)*N + i)*H + h];
            float cv = C[((((b*S+s)*N + j)*H + h)*P) + p];
            acc += av * cv;
        }
    }
    float den = 0.f;
    #pragma unroll
    for (int s=0; s<8; ++s) den += mask[(b*S+s)*N+i];
    if (den < 1.f) den = 1.f;
    out[((b*N+i)*N+j)*P + p] = acc / den + bo[p];
}

torch::Tensor opm_forward(torch::Tensor msa_repr, torch::Tensor msa_mask,
                          torch::Tensor ln_w, torch::Tensor ln_b,
                          torch::Tensor wa, torch::Tensor ba,
                          torch::Tensor wb, torch::Tensor bb,
                          torch::Tensor wo, torch::Tensor bo) {
    int Bsz=msa_repr.size(0), S=msa_repr.size(1), N=msa_repr.size(2), D=msa_repr.size(3);
    int H=wa.size(0), P=wo.size(0);
    auto opts=msa_repr.options();
    auto A=torch::empty({Bsz,S,N,H}, opts);
    auto Bv=torch::empty({Bsz,S,N,H}, opts);
    auto C=torch::empty({Bsz,S,N,H,P}, opts);
    auto out=torch::empty({Bsz,N,N,P}, opts);
    cudaStream_t stream=at::cuda::getCurrentCUDAStream();
    ln_proj_kernel<<<Bsz*S*N,128,0,stream>>>(msa_repr.data_ptr<float>(),msa_mask.data_ptr<float>(),ln_w.data_ptr<float>(),ln_b.data_ptr<float>(),wa.data_ptr<float>(),ba.data_ptr<float>(),wb.data_ptr<float>(),bb.data_ptr<float>(),A.data_ptr<float>(),Bv.data_ptr<float>(),Bsz,S,N,D,H);
    int total=Bsz*S*N*H*P;
    transform_b_kernel<<<(total+255)/256,256,0,stream>>>(Bv.data_ptr<float>(),wo.data_ptr<float>(),C.data_ptr<float>(),Bsz,S,N,H,P);

    return out;
}

import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, input_channels, stages, block_widths, output_classes):
        """
        :param input_channels: int, Number of input channels for the first layer
        :param stages: int, Number of stages in the RegNet architecture
        :param block_widths: List[int], Width (number of channels) for each block in the stages
        :param output_classes: int, Number of output classes for classification
        """
        super(Model, self).__init__()

        self.stages = stages
        self.block_widths = block_widths
        
        layers = []
        current_channels = input_channels
        
        # Construct the stages with their respective blocks
        for i in range(stages):
            layers.append(self._make_stage(current_channels, block_widths[i]))
            current_channels = block_widths[i]
        
        self.feature_extractor = nn.Sequential(*layers)
        
        # Final fully connected layer for classification
        self.fc = nn.Linear(block_widths[-1], output_classes)
    
    def _make_stage(self, in_channels, out_channels):
        """
        Creates a simple block for each stage.
        :param in_channels: int, number of input channels
        :param out_channels: int, number of output channels
        :return: nn.Sequential block with convolutional layers
        """
        return nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        """
        Forward pass through the RegNet model.
        :param x: torch.Tensor of shape (batch_size, input_channels, height, width)
        :return: torch.Tensor of shape (batch_size, output_classes)
        """
        x = self.feature_extractor(x)
        x = torch.mean(x, dim=[2, 3])  # Global Average Pooling
        x = self.fc(x)
        return x

    float v1 = x[base + 1] * scale + shift;   v1 = v1 > 0.0f ? v1 : 0.0f;
    float v2 = x[base + W] * scale + shift;   v2 = v2 > 0.0f ? v2 : 0.0f;
    float v3 = x[base + W + 1] * scale + shift; v3 = v3 > 0.0f ? v3 : 0.0f;
    float m = fmaxf(fmaxf(v0, v1), fmaxf(v2, v3));
    y[idx] = m;
}

__global__ void bn_stats_kernel(const float* __restrict__ x, float* __restrict__ mean, float* __restrict__ invstd,
                                int N, int C, int H, int W, float eps) {
    int c = blockIdx.x;
    int tid = threadIdx.x;
    int M = N * H * W;
    float sum = 0.0f;
    float ss = 0.0f;
    for (int i = tid; i < M; i += blockDim.x) {
        int widx = i % W;
        int tmp = i / W;
        int h = tmp % H;
        int n = tmp / H;
        float v = x[((n * C + c) * H + h) * W + widx];
        sum += v;
        ss += v * v;
    }
    __shared__ float ssum[256];
    __shared__ float sss[256];
    ssum[tid] = sum;
    sss[tid] = ss;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            ssum[tid] += ssum[tid + stride];
            sss[tid] += sss[tid + stride];
        }
        __syncthreads();
    }
    if (tid == 0) {
        float m = ssum[0] / (float)M;
        float var = sss[0] / (float)M - m * m;
        if (var < 0.0f) var = 0.0f;
        mean[c] = m;
        invstd[c] = rsqrtf(var + eps);
    }
}

__global__ void bn_relu_train_kernel(const float* __restrict__ x, const float* __restrict__ w, const float* __restrict__ b,
                                     const float* __restrict__ mean, const float* __restrict__ invstd,
                                     float* __restrict__ y, int total, int C, int HW) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    int c = (idx / HW) % C;
    float v = (x[idx] - mean[c]) * invstd[c] * w[c] + b[c];
    y[idx] = v > 0.0f ? v : 0.0f;
}

__global__ void bn_relu_pool_train_kernel(const float* __restrict__ x, const float* __restrict__ w, const float* __restrict__ b,
                                          const float* __restrict__ mean, const float* __restrict__ invstd,
                                          float* __restrict__ y, int N, int C, int H, int Wd, int OH, int OW) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * OH * OW;
    if (idx >= total) return;
    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int c = t % C;
    int n = t / C;
    float sc = invstd[c] * w[c];
    float sh = b[c] - mean[c] * sc;
    int base = ((n * C + c) * H + (oh << 1)) * Wd + (ow << 1);
    float v0 = x[base] * sc + sh;           v0 = v0 > 0.0f ? v0 : 0.0f;
    float v1 = x[base + 1] * sc + sh;       v1 = v1 > 0.0f ? v1 : 0.0f;
    float v2 = x[base + Wd] * sc + sh;      v2 = v2 > 0.0f ? v2 : 0.0f;
    float v3 = x[base + Wd + 1] * sc + sh;  v3 = v3 > 0.0f ? v3 : 0.0f;
    y[idx] = fmaxf(fmaxf(v0, v1), fmaxf(v2, v3));
}

__global__ void gap_fc_kernel(const float* __restrict__ x, const float* __restrict__ w, const float* __restrict__ b,
                              float* __restrict__ y, int N, int C, int H, int Wd, int O) {
    int pair = blockIdx.x;
    int n = pair / O;
    int o = pair - n * O;
    int tid = threadIdx.x;
    int HW = H * Wd;
    int M = C * HW;
    float sum = 0.0f;
    for (int i = tid; i < M; i += blockDim.x) {
        int c = i / HW;
        int s = i - c * HW;
        sum += x[(n * C + c) * HW + s] * w[o * C + c];
    }
    __shared__ float smem[256];
    smem[tid] = sum;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) smem[tid] += smem[tid + stride];
        __syncthreads();
    }
    if (tid == 0) y[n * O + o] = b[o] + smem[0] / (float)HW;
}

torch::Tensor bn_relu_eval(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, torch::Tensor running_mean, torch::Tensor running_var, double eps) {
    auto y = torch::empty_like(x);
    int N = x.size(0), C = x.size(1), H = x.size(2), W = x.size(3);
    int total = N*C*H*W;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    bn_relu_eval_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), running_mean.data_ptr<float>(), running_var.data_ptr<float>(), y.data_ptr<float>(), total, C, H*W, (float)eps);
    return y;
}

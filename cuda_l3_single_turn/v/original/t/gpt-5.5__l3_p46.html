import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th


class Model(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(Model, self).__init__()

        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # The `clusters` weights are the `(w,b)` in the paper
        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        # The `clusters2` weights are the visual words `c_k` in the paper
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        """Aggregates feature maps into a fixed size representation.  In the following
        notation, B = batch_size, N = num_features, K = num_clusters, D = feature_size.

        Args:
            x (th.Tensor): B x N x D

        Returns:
            (th.Tensor): B x DK
        """
        max_sample = x.size()[1]
        x = x.view(-1, self.feature_size)  # B x N x D -> BN x D

        if x.device != self.clusters.device:
            msg = f"x.device {x.device} != cluster.device {self.clusters.device}"
            raise ValueError(msg)

        assignment = th.matmul(x, self.clusters)  # (BN x D) x (D x (K+G)) -> BN x (K+G)
        assignment = self.batch_norm(assignment)

        assignment = F.softmax(assignment, dim=1)  # BN x (K+G) -> BN x (K+G)
        # remove ghost assigments
        assignment = assignment[:, :self.cluster_size]
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # -> B x N x K
        a_sum = th.sum(assignment, dim=1, keepdim=True)  # B x N x K -> B x 1 x K
        a = a_sum * self.clusters2

        assignment = assignment.transpose(1, 2)  # B x N x K -> B x K x N

        x = x.view(-1, max_sample, self.feature_size)  # BN x D -> B x N x D
        vlad = th.matmul(assignment, x)  # (B x K x N) x (B x N x D) -> B x K x D
        vlad = vlad.transpose(1, 2)  # -> B x D x K
        vlad = vlad - a

        # L2 intra norm
        vlad = F.normalize(vlad)

        # flattening + L2 norm
        vlad = vlad.reshape(-1, self.cluster_size * self.feature_size)  # -> B x DK
        vlad = F.normalize(vlad)
        return vlad  # B x DK

batch_size = 2048
num_features = 100

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>

__global__ void softmax_sum_kernel(const float* __restrict__ logits,
                                   float* __restrict__ probs,
                                   float* __restrict__ asum,
                                   int M, int N, int K, int C) {
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    if (row >= M) return;
    int b = row / N;
    const float* l = logits + row * C;
    float mx = -FLT_MAX;
    #pragma unroll
    for (int c = 0; c < 64; ++c) {
        if (c < C) {
            float v = l[c];
            mx = v > mx ? v : mx;
        }
    }
    float den = 0.0f;
    #pragma unroll
    for (int c = 0; c < 64; ++c) {
        if (c < C) den += expf(l[c] - mx);
    }
    float inv = 1.0f / den;
    float* p = probs + row * K;
    float* as = asum + b * K;
    #pragma unroll
    for (int k = 0; k < 64; ++k) {
        if (k < K) {
            float val = expf(l[k] - mx) * inv;
            p[k] = val;
            atomicAdd(as + k, val);
        }
    }
}

__global__ void vlad_kernel(const float* __restrict__ x,
                            const float* __restrict__ probs,
                            const float* __restrict__ asum,
                            const float* __restrict__ c2,
                            float* __restrict__ out,
                            int B, int N, int D, int K) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * D * K;
    if (idx >= total) return;
    int k = idx % K;
    int t = idx / K;
    int d = t % D;
    int b = t / D;

    const float* xb = x + (long long)b * N * D + d;
    const float* pb = probs + (long long)b * N * K + k;
    float acc = 0.0f;
    for (int n = 0; n < N; ++n) {
        acc += pb[(long long)n * K] * xb[(long long)n * D];
    }
    acc -= asum[b * K + k] * c2[d * K + k];
    out[idx] = acc;
}

__global__ void intra_norm_kernel(float* __restrict__ out, int B, int D, int K) {
    int bk = blockIdx.x;
    int b = bk / K;
    int k = bk - b * K;
    __shared__ float sdata[256];
    float sum = 0.0f;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float v = out[((long long)b * D + d) * K + k];
        sum += v * v;
    }
    sdata[threadIdx.x] = sum;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) sdata[threadIdx.x] += sdata[threadIdx.x + stride];
        __syncthreads();
    }
    float nrm = sqrtf(sdata[0]);
    float inv = (nrm > 1.0e-12f) ? (1.0f / nrm) : 1.0e12f;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        out[((long long)b * D + d) * K + k] *= inv;
    }
}

__global__ void final_norm_kernel(float* __restrict__ out, int B, int DK) {
    int b = blockIdx.x;
    __shared__ float sdata[256];
    float sum = 0.0f;
    float* base = out + (long long)b * DK;
    for (int i = threadIdx.x; i < DK; i += blockDim.x) {
        float v = base[i];
        sum += v * v;
    }
    sdata[threadIdx.x] = sum;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) sdata[threadIdx.x] += sdata[threadIdx.x + stride];
        __syncthreads();
    }
    float nrm = sqrtf(sdata[0]);
    float inv = (nrm > 1.0e-12f) ? (1.0f / nrm) : 1.0e12f;
    for (int i = threadIdx.x; i < DK; i += blockDim.x) base[i] *= inv;
}

torch::Tensor vlad_forward(torch::Tensor x, torch::Tensor logits, torch::Tensor clusters2, int64_t cluster_size) {
    TORCH_CHECK(x.is_cuda() && logits.is_cuda() && clusters2.is_cuda(), "tensors must be CUDA");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32 && logits.scalar_type() == torch::kFloat32 && clusters2.scalar_type() == torch::kFloat32, "float32 only");
    x = x.contiguous();
    logits = logits.contiguous();
    clusters2 = clusters2.contiguous();
    const int B = (int)x.size(0);
    const int N = (int)x.size(1);
    const int D = (int)x.size(2);
    const int K = (int)cluster_size;
    const int C = (int)logits.size(1);
    const int M = B * N;
    auto opts = x.options();
    auto probs = torch::empty({B, N, K}, opts);
    auto asum = torch::zeros({B, K}, opts);
    auto out3 = torch::empty({B, D, K}, opts);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    int threads = 128;
    int blocks = (M + threads - 1) / threads;
    softmax_sum_kernel<<<blocks, threads, 0, stream>>>(logits.data_ptr<float>(), probs.data_ptr<float>(), asum.data_ptr<float>(), M, N, K, C);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    int total = B * D * K;
    threads = 256;
    blocks = (total + threads - 1) / threads;
    vlad_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), probs.data_ptr<float>(), asum.data_ptr<float>(), clusters2.data_ptr<float>(), out3.data_ptr<float>(), B, N, D, K);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    intra_norm_kernel<<<B * K, 256, 0, stream>>>(out3.data_ptr<float>(), B, D, K);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    auto out = out3.view({B, D * K});
    final_norm_kernel<<<B, 256, 0, stream>>>(out.data_ptr<float>(), B, D * K);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return out;
}

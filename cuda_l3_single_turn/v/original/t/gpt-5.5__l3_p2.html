class Model(nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        """
        :param input_size: The number of input features

        """
        :param x: The input tensor, shape (batch_size, input_size)
        :return: The output tensor, shape (batch_size, output_size)
        """
        return self.network(x)

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>

void relu_inplace_cuda(torch::Tensor x);

std::vector<torch::Tensor> mlp_forward_cuda(
    torch::Tensor x,
    torch::Tensor w0, torch::Tensor b0,
    torch::Tensor w1, torch::Tensor b1,
    torch::Tensor w2, torch::Tensor b2) {
  // Compute the three large GEMMs with ATen/cuBLAS and fuse the two ReLUs
  // into a tiny custom CUDA kernel operating in-place on the hidden outputs.
  auto y0 = torch::addmm(b0, x, w0.t());
  relu_inplace_cuda(y0);
  auto y1 = torch::addmm(b1, y0, w1.t());
  relu_inplace_cuda(y1);
  auto y2 = torch::addmm(b2, y1, w2.t());
  return {y2};
}
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void relu_inplace_kernel(float* __restrict__ x, long long n) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = idx; i < n; i += stride) {
        float v = x[i];
        x[i] = v > 0.0f ? v : 0.0f;
    }
}

void relu_inplace_cuda(torch::Tensor x) {
    const long long n = x.numel();
    if (n == 0) return;
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    relu_inplace_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), n);
}
'''

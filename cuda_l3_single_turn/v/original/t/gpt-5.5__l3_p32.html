import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6, 
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        """
        Convolutional Vision Transformer (CViT) implementation.
        :param num_classes: Number of output classes for classification.
        :param embed_dim: Dimensionality of the embedding space.
        :param num_heads: Number of attention heads.
        :param num_layers: Number of transformer layers.
        :param mlp_ratio: Ratio of the MLP hidden dimension to the embedding dimension.
        :param patch_size: Size of the convolutional patches.
        :param in_channels: Number of input channels (e.g., 3 for RGB images).
        :param image_size: Height/width of the square input image.
        """
        super(Model, self).__init__()

        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2  # Total number of patches after conv
        self.linear_proj = nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = nn.ModuleList([
            nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        """
        Forward pass of the CViT model.
        :param x: Input tensor of shape (B, C, H, W)
        :return: Output tensor of shape (B, num_classes)
        """
        B = x.size(0)
        x = self.conv1(x)                  # (B, embed_dim, H/patch_size, W/patch_size)
        x = x.flatten(start_dim=1)         # (B, embed_dim * num_patches)
        x = self.linear_proj(x)            # (B, embed_dim)

        cls_tokens = self.cls_token.expand(B, -1, -1)  # (B, 1, embed_dim)
        x = torch.cat((cls_tokens, x.unsqueeze(1)), dim=1)  # (B, 2, embed_dim)

        for layer in self.transformer_layers:
            x = layer(x)

        return self.fc_out(x[:, 0])        # Use [CLS] token for classification

__global__ void conv_patch_kernel(const float* __restrict__ x, const float* __restrict__ w, const float* __restrict__ bias,
                                  float* __restrict__ out, int B, int C, int H, int W, int E, int K, int PH, int PW) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * E * PH * PW;
    if (idx >= total) return;
    int s = idx % (PH * PW);
    int e = (idx / (PH * PW)) % E;
    int b = idx / (E * PH * PW);
    int py = s / PW;
    int px = s - py * PW;
    float acc = bias ? bias[e] : 0.0f;
    #pragma unroll
    for (int c = 0; c < 3; ++c) {
        if (c >= C) break;
        for (int ky = 0; ky < K; ++ky) {
            int iy = py * K + ky;
            for (int kx = 0; kx < K; ++kx) {
                int ix = px * K + kx;
                float xv = x[((b*C + c)*H + iy)*W + ix];
                float wv = w[((e*C + c)*K + ky)*K + kx];
                acc += xv * wv;
            }
        }
    }
    out[idx] = acc;
}

__global__ void linear_kernel(const float* __restrict__ inp, const float* __restrict__ w, const float* __restrict__ bias,
                              float* __restrict__ out, int M, int K, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = M * N;
    if (idx >= total) return;
    int n = idx % N;
    int m = idx / N;
    float acc = bias ? bias[n] : 0.0f;
    const float* ip = inp + m * K;
    const float* wp = w + n * K;
    for (int k = 0; k < K; ++k) acc += ip[k] * wp[k];
    out[idx] = acc;
}

__global__ void linear_relu_kernel(const float* __restrict__ inp, const float* __restrict__ w, const float* __restrict__ bias,
                                   float* __restrict__ out, int M, int K, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = M * N;

}

__global__ void init_seq_kernel(const float* __restrict__ z, const float* __restrict__ cls,
                                float* __restrict__ seq, int B, int E) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * 2 * E;
    if (idx >= total) return;
    int e = idx % E;
    int t = (idx / E) & 1;
    int b = idx / (2 * E);
    seq[(b*2 + t)*E + e] = (t == 0) ? cls[e] : z[b*E + e];
}

__global__ void attn_kernel(const float* __restrict__ qkv, float* __restrict__ out,
                            int B, int E, int heads) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * 2 * E;
    if (idx >= total) return;
    int dglob = idx % E;
    int t = (idx / E) & 1;
    int b = idx / (2 * E);
    int hd = E / heads;
    int h = dglob / hd;
    int d = dglob - h * hd;
    int base_bt = (b*2 + t) * (3*E);
    int qoff = h*hd;
    float s0 = 0.0f, s1 = 0.0f;
    for (int r = 0; r < hd; ++r) {
        float qv = qkv[base_bt + qoff + r];
        float k0 = qkv[(b*2 + 0)*(3*E) + E + qoff + r];
        float k1 = qkv[(b*2 + 1)*(3*E) + E + qoff + r];
        s0 += qv * k0;
        s1 += qv * k1;
    }
    float scale = rsqrtf((float)hd);
    s0 *= scale; s1 *= scale;
    float mx = s0 > s1 ? s0 : s1;
    float e0 = expf(s0 - mx);
    float e1 = expf(s1 - mx);
    float inv = 1.0f / (e0 + e1);
    float p0 = e0 * inv, p1 = e1 * inv;
    float v0 = qkv[(b*2 + 0)*(3*E) + 2*E + h*hd + d];
    float v1 = qkv[(b*2 + 1)*(3*E) + 2*E + h*hd + d];
    out[(b*2 + t)*E + dglob] = p0 * v0 + p1 * v1;
}

__global__ void add_layernorm_kernel(const float* __restrict__ a, const float* __restrict__ b,
                                     const float* __restrict__ gamma, const float* __restrict__ beta,
                                     float* __restrict__ out, int M, int E, float eps) {
    extern __shared__ float sm[];
    float* ss = sm;
    int row = blockIdx.x;
    int tid = threadIdx.x;
    float sum = 0.0f;
    for (int e = tid; e < E; e += blockDim.x) sum += a[row*E + e] + b[row*E + e];
    ss[tid] = sum;
    __syncthreads();
    for (int stride = blockDim.x/2; stride > 0; stride >>= 1) {
        if (tid < stride) ss[tid] += ss[tid + stride];
        __syncthreads();
    }
    float mean = ss[0] / (float)E;
    float vsum = 0.0f;
    for (int e = tid; e < E; e += blockDim.x) {
        float v = a[row*E + e] + b[row*E + e] - mean;
        vsum += v * v;
    }
    ss[tid] = vsum;
    __syncthreads();
    for (int stride = blockDim.x/2; stride > 0; stride >>= 1) {
        if (tid < stride) ss[tid] += ss[tid + stride];
        __syncthreads();
    }
    float invstd = rsqrtf(ss[0] / (float)E + eps);
    for (int e = tid; e < E; e += blockDim.x) {
        float v = a[row*E + e] + b[row*E + e];
        out[row*E + e] = (v - mean) * invstd * gamma[e] + beta[e];
    }
}

__global__ void fc_cls_kernel(const float* __restrict__ seq, const float* __restrict__ w, const float* __restrict__ bias,
                              float* __restrict__ out, int B, int E, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * N;
    if (idx >= total) return;
    int n = idx % N;
    int b = idx / N;
    const float* ip = seq + (b*2) * E;
    const float* wp = w + n * E;
    float acc = bias ? bias[n] : 0.0f;
    for (int e = 0; e < E; ++e) acc += ip[e] * wp[e];
    out[idx] = acc;
}

static inline void launch_linear(torch::Tensor inp, torch::Tensor w, torch::Tensor bias, torch::Tensor out, cudaStream_t stream, bool relu=false) {
    int M = inp.size(0);
    int K = inp.size(1);
    int N = w.size(0);
    int total = M * N;
    int block = 256;
    int grid = (total + block - 1) / block;
    const float* bp = bias.defined() && bias.numel() ? bias.data_ptr<float>() : nullptr;
    if (relu) linear_relu_kernel<<<grid, block, 0, stream>>>(inp.data_ptr<float>(), w.data_ptr<float>(), bp, out.data_ptr<float>(), M, K, N);
    else linear_kernel<<<grid, block, 0, stream>>>(inp.data_ptr<float>(), w.data_ptr<float>(), bp, out.data_ptr<float>(), M, K, N);
}

torch::Tensor cvit_forward(torch::Tensor x, std::vector<torch::Tensor> p, int64_t num_heads) {
    CHECK_CUDA(x);
    auto opts = x.options();
    int B = x.size(0), C = x.size(1), H = x.size(2), W = x.size(3);
    auto conv_w = p[0], conv_b = p[1], lp_w = p[2], lp_b = p[3], cls = p[4];
    int E = conv_w.size(0);
    int Kp = conv_w.size(2);
    int PH = H / Kp, PW = W / Kp;
    int P = PH * PW;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    auto conv = torch::empty({B, E * P}, opts);
    int total_conv = B * E * P;
    conv_patch_kernel<<<(total_conv+255)/256, 256, 0, stream>>>(x.data_ptr<float>(), conv_w.data_ptr<float>(), conv_b.data_ptr<float>(), conv.data_ptr<float>(), B, C, H, W, E, Kp, PH, PW);
    auto z = torch::empty({B, E}, opts);
    launch_linear(conv, lp_w, lp_b, z, stream, false);
    auto seq = torch::empty({B*2, E}, opts);
    init_seq_kernel<<<(B*2*E+255)/256, 256, 0, stream>>>(z.data_ptr<float>(), cls.data_ptr<float>(), seq.data_ptr<float>(), B, E);

    int M = B * 2;
    int offset = 5;
    for (int layer = 0; layer < 6; ++layer) {
        auto inpw = p[offset+0], inpb = p[offset+1], outw = p[offset+2], outb = p[offset+3];
        auto l1w = p[offset+4], l1b = p[offset+5], l2w = p[offset+6], l2b = p[offset+7];
        auto n1w = p[offset+8], n1b = p[offset+9], n2w = p[offset+10], n2b = p[offset+11];
        offset += 12;
        int Hdim = l1w.size(0);
        auto qkv = torch::empty({M, 3*E}, opts);
        launch_linear(seq, inpw, inpb, qkv, stream, false);
        auto av = torch::empty({M, E}, opts);
        attn_kernel<<<(M*E+255)/256, 256, 0, stream>>>(qkv.data_ptr<float>(), av.data_ptr<float>(), B, E, (int)num_heads);
        auto aproj = torch::empty({M, E}, opts);
        launch_linear(av, outw, outb, aproj, stream, false);
        auto norm1 = torch::empty({M, E}, opts);
        add_layernorm_kernel<<<M, 256, 256*sizeof(float), stream>>>(seq.data_ptr<float>(), aproj.data_ptr<float>(), n1w.data_ptr<float>(), n1b.data_ptr<float>(), norm1.data_ptr<float>(), M, E, 1e-5f);
        auto hid = torch::empty({M, Hdim}, opts);
        launch_linear(norm1, l1w, l1b, hid, stream, true);
        auto ff = torch::empty({M, E}, opts);
        launch_linear(hid, l2w, l2b, ff, stream, false);
        auto next = torch::empty({M, E}, opts);
        add_layernorm_kernel<<<M, 256, 256*sizeof(float), stream>>>(norm1.data_ptr<float>(), ff.data_ptr<float>(), n2w.data_ptr<float>(), n2b.data_ptr<float>(), next.data_ptr<float>(), M, E, 1e-5f);
        seq = next;
    }
    auto fcw = p[offset+0], fcb = p[offset+1];
    int N = fcw.size(0);
    auto out = torch::empty({B, N}, opts);
    fc_cls_kernel<<<(B*N+255)/256, 256, 0, stream>>>(seq.data_ptr<float>(), fcw.data_ptr<float>(), fcb.data_ptr<float>(), out.data_ptr<float>(), B, E, N);
    return out;
}

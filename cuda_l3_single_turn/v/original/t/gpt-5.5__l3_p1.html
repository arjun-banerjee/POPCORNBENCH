import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>

void bias_relu_inplace_cuda(torch::Tensor y);

torch::Tensor mlp_forward(torch::Tensor x,
                          torch::Tensor w0, torch::Tensor b0,
                          torch::Tensor w1, torch::Tensor b1,
                          torch::Tensor w2, torch::Tensor b2) {
    // Exactly the same GEMM formulation as nn.Linear/F.linear: x @ weight.T + bias.
    // ReLUs are done in-place on temporaries, avoiding the extra allocation made by
    // out-of-place nn.ReLU in the reference Sequential model.
    auto y0 = at::addmm(b0, x, w0.t());
    bias_relu_inplace_cuda(y0);
    auto y1 = at::addmm(b1, y0, w1.t());
    bias_relu_inplace_cuda(y1);
    auto y2 = at::addmm(b2, y1, w2.t());
    return y2;
}
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>

__global__ void relu_inplace_kernel(float* __restrict__ p, long long n) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = idx; i < n; i += stride) {
        float v = p[i];
        p[i] = v > 0.0f ? v : 0.0f;
    }
}

void bias_relu_inplace_cuda(torch::Tensor y) {
    const auto n = y.numel();
    if (n == 0) return;
    c10::cuda::CUDAGuard guard(y.device());
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    relu_inplace_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(y.data_ptr<float>(), (long long)n);
}
'''

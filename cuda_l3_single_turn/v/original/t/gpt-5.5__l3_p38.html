import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        """
        Initialize the LSTM model.

        :param input_size: The number of expected features in the input `x`
        :param hidden_size: The number of features in the hidden state `h`
        :param num_layers: Number of recurrent layers
        :param output_size: The number of output features
        :param dropout: If non-zero, introduces a Dropout layer on the outputs of each LSTM layer except the last layer, with dropout probability equal to `dropout`
        """
        super(Model, self).__init__()
        # Initialize hidden state with random values
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True, dropout=dropout, bidirectional=True)
        self.fc = nn.Linear(hidden_size * 2, output_size)
    
    def forward(self, x,h0,c0):
        """
        Forward pass through the LSTM model.

        :param x: The input tensor, shape (batch_size, sequence_length, input_size)
        :return: The output tensor, shape (batch_size, sequence_length, output_size)
        """
        # Forward propagate LSTM
        out, hn = self.lstm(x, (h0, c0))  # out: tensor of shape (batch_size, seq_length, hidden_size)
        
        # Decode the hidden state of the last time step
        out = self.fc(out[:, -1, :])  # out: tensor of shape (batch_size, output_size)
        
        return out

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <vector>

torch::Tensor lstm_fc(torch::Tensor x,
                      torch::Tensor h0,
                      torch::Tensor c0,
                      std::vector<torch::Tensor> weights,
                      torch::Tensor fc_w,
                      torch::Tensor fc_b,
                      int64_t num_layers,
                      double dropout,
                      bool train);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <vector>

__global__ void fc_last_kernel(const float* __restrict__ lstm_out,
                               const float* __restrict__ w,
                               const float* __restrict__ b,
                               float* __restrict__ y,
                               int B, int T, int F, int O) {
    int bo = blockIdx.x;
    int batch = bo / O;
    int out_ch = bo - batch * O;
    int tid = threadIdx.x;

    const float* xrow = lstm_out + ((long long)batch * T + (T - 1)) * F;
    const float* wrow = w + (long long)out_ch * F;

    float sum = 0.0f;
    for (int i = tid; i < F; i += blockDim.x) {
        sum += xrow[i] * wrow[i];
    }

    __shared__ float smem[256];
    smem[tid] = sum;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) smem[tid] += smem[tid + stride];
        __syncthreads();
    }

    if (tid == 0) {
        y[(long long)batch * O + out_ch] = smem[0] + b[out_ch];
    }
}

torch::Tensor lstm_fc(torch::Tensor x,
                      torch::Tensor h0,
                      torch::Tensor c0,
                      std::vector<torch::Tensor> weights,
                      torch::Tensor fc_w,
                      torch::Tensor fc_b,
                      int64_t num_layers,
                      double dropout,
                      bool train) {
    // Use the same fused cuDNN/ATen LSTM implementation as PyTorch, but keep the
    // final timestep projection in this extension to avoid Python slicing/addmm
    // overhead and materializing another intermediate tensor.
    std::vector<torch::Tensor> hx;
    hx.reserve(2);
    hx.push_back(h0);
    hx.push_back(c0);

    auto res = at::lstm(x, hx, weights, true, num_layers, dropout, train, true, true);
    torch::Tensor out = std::get<0>(res);

    const int B = (int)out.size(0);
    const int T = (int)out.size(1);
    const int F = (int)out.size(2);
    const int O = (int)fc_w.size(0);

    auto y = torch::empty({B, O}, out.options());

    const int threads = 256;
    const int blocks = B * O;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    fc_last_kernel<<<blocks, threads, 0, stream>>>(
        out.data_ptr<float>(), fc_w.data_ptr<float>(), fc_b.data_ptr<float>(), y.data_ptr<float>(),
        B, T, F, O);
    return y;
}
'''

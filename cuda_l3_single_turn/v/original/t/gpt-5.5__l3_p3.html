import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <vector>

void add_bias_relu_cuda(torch::Tensor out, torch::Tensor bias);

// Forward for a stack of Linear/ReLU layers followed by a final Linear layer.
// Hidden layers are evaluated as mm + a single fused CUDA kernel for bias and ReLU.
torch::Tensor mlp_forward(torch::Tensor x,
                          std::vector<torch::Tensor> weights,
                          std::vector<torch::Tensor> biases) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(weights.size() == biases.size(), "weights/biases length mismatch");
    at::cuda::CUDAGuard device_guard(x.device());

    torch::Tensor cur = x;
    const int64_t L = static_cast<int64_t>(weights.size());
    for (int64_t i = 0; i < L; ++i) {
        auto wt = weights[i].transpose(0, 1);
        if (i == L - 1) {
            // Let ATen/cuBLAS handle the last, large biased GEMM exactly like Linear.
            cur = at::addmm(biases[i], cur, wt, 1.0, 1.0);
        } else {
            auto out = at::mm(cur, wt);
            add_bias_relu_cuda(out, biases[i]);
            cur = out;
        }
    }
    return cur;
}
'''

_cuda_src = r'''
#include <torch/types.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>

__global__ void add_bias_relu_kernel(float* __restrict__ out,
                                     const float* __restrict__ bias,
                                     int64_t total,
                                     int64_t cols) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        float v = out[idx] + __ldg(bias + (idx % cols));
        out[idx] = v > 0.0f ? v : 0.0f;
    }
}

void add_bias_relu_cuda(torch::Tensor out, torch::Tensor bias) {
    const int64_t total = out.numel();
    if (total == 0) return;
    const int64_t cols = out.size(-1);
    const int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    add_bias_relu_kernel<<<blocks, threads, 0, stream>>>(
        out.data_ptr<float>(), bias.data_ptr<float>(), total, cols);
}
'''

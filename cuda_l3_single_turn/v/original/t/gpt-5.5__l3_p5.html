from torch.utils.cpp_extension import load_inline


_cpp_src = r'''
#include <torch/extension.h>

torch::Tensor relu_maxpool_cuda(torch::Tensor input, int64_t k, int64_t s);

torch::Tensor relu_maxpool(torch::Tensor input, int64_t k, int64_t s) {
    return relu_maxpool_cuda(input, k, s);
}

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void relu_maxpool_kernel(const float* __restrict__ x,
                                    float* __restrict__ y,
                                    const long long total,
                                    const int N, const int C,
                                    const int H, const int W,
                                    const int OH, const int OW,
                                    const int k, const int s) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    const long long stride = (long long)blockDim.x * gridDim.x;

    for (; idx < total; idx += stride) {
        int ow = (int)(idx % OW);
        long long t = idx / OW;
        int oh = (int)(t % OH);
        t /= OH;
        int c = (int)(t % C);
        int n = (int)(t / C);

        int ih0 = oh * s;
        int iw0 = ow * s;
        long long base = (((long long)n * C + c) * H + ih0) * W + iw0;

        // MaxPool(ReLU(x)) for k=3 in the AlexNet network.  Starting at zero
        // is exactly the ReLU fusion: negative windows produce zero.
        float m = 0.0f;
        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) {
            long long row = base + (long long)kh * W;
            #pragma unroll
            for (int kw = 0; kw < 3; ++kw) {
                float v = x[row + kw];
                m = v > m ? v : m;
            }
        }
        y[idx] = m;
    }
}

torch::Tensor relu_maxpool_cuda(torch::Tensor input, int64_t k64, int64_t s64) {
    const int N = (int)input.size(0);
    const int C = (int)input.size(1);
    const int H = (int)input.size(2);
    const int W = (int)input.size(3);
    const int k = (int)k64;
    const int s = (int)s64;
    const int OH = (H - k) / s + 1;
    const int OW = (W - k) / s + 1;

    auto out = torch::empty({N, C, OH, OW}, input.options());
    const long long total = (long long)N * C * OH * OW;
    if (total == 0) return out;

    const int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    if (blocks > 65535) blocks = 65535;
    relu_maxpool_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        input.data_ptr<float>(), out.data_ptr<float>(), total, N, C, H, W, OH, OW, k, s);
    return out;
}
'''

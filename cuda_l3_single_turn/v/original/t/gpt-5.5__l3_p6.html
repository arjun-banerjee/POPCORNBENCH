        :param reduce_3x3: Number of output channels for the 1x1 reduction before 3x3 convolution
        :param out_3x3: Number of output channels for the 3x3 convolution
        :param reduce_5x5: Number of output channels for the 1x1 reduction before 5x5 convolution
        :param out_5x5: Number of output channels for the 5x5 convolution
        :param pool_proj: Number of output channels for the pooling projection
        """

        # 1x1 convolution branch

        # 3x3 convolution branch

        )
        
        # 5x5 convolution branch

        )
        
        # Max pooling branch
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),

        )

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

__inline__ __device__ float warp_sum(float v) {
    v += __shfl_down_sync(0xffffffff, v, 16);
    v += __shfl_down_sync(0xffffffff, v, 8);
    v += __shfl_down_sync(0xffffffff, v, 4);
    v += __shfl_down_sync(0xffffffff, v, 2);
    v += __shfl_down_sync(0xffffffff, v, 1);
    return v;
}

// Computes Conv1x1(MaxPool3x3_s1_p1(x)) for all output channels of one spatial
// point per CTA.  The pooled input vector is materialized once in shared memory
// and reused for the 64 dot products, avoiding the large maxpool temporary.
__global__ void pool_proj_kernel(const float* __restrict__ x,
                                 const float* __restrict__ weight,
                                 const float* __restrict__ bias,
                                 float* __restrict__ out,
                                 int N, int C, int H, int W, int P) {
    extern __shared__ float pooled[];
    int pix = blockIdx.x;
    int hw = H * W;
    int n = pix / hw;
    int rem = pix - n * hw;
    int h = rem / W;
    int w = rem - h * W;
    int tid = threadIdx.x;

    // max pool for every input channel, padding is ignored (equiv. -inf)
    for (int c = tid; c < C; c += blockDim.x) {
        float m = -FLT_MAX;
        int h0 = h > 0 ? h - 1 : h;
        int h1 = (h + 1 < H) ? h + 1 : h;
        int w0 = w > 0 ? w - 1 : w;
        int w1 = (w + 1 < W) ? w + 1 : w;
        const float* base = x + ((n * C + c) * H) * W;
        for (int yy = h0; yy <= h1; ++yy) {
            const float* row = base + yy * W;
            for (int xx = w0; xx <= w1; ++xx) {
                float v = row[xx];
                m = v > m ? v : m;
            }
        }
        pooled[c] = m;
    }
    __syncthreads();

    int warp = tid >> 5;     // 0..7 for 256-thread CTA
    int lane = tid & 31;
    // Each warp computes one output channel; loop over groups of eight channels.
    for (int g = 0; g < P; g += 8) {
        int oc = g + warp;
        if (oc < P) {
            float acc = 0.0f;
            const float* wptr = weight + oc * C;
            for (int c = lane; c < C; c += 32) {
                acc += pooled[c] * wptr[c];
            }
            acc = warp_sum(acc);
            if (lane == 0) {
                out[((n * P + oc) * H + h) * W + w] = acc + bias[oc];
            }
        }
    }
}

torch::Tensor pool_proj_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    const int64_t N64 = x.size(0), C64 = x.size(1), H64 = x.size(2), W64 = x.size(3);
    const int64_t P64 = weight.size(0);
    auto out = torch::empty({N64, P64, H64, W64}, x.options());
    int N = (int)N64, C = (int)C64, H = (int)H64, W = (int)W64, P = (int)P64;
    dim3 block(256);
    dim3 grid((unsigned int)(N64 * H64 * W64));
    size_t shmem = (size_t)C * sizeof(float);
    pool_proj_kernel<<<grid, block, shmem, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
        out.data_ptr<float>(), N, C, H, W, P);
    return out;
}

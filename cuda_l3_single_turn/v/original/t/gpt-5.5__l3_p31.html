import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        """
        Attention Block using Multihead Self-Attention.
        :param embed_dim: Embedding dimension (the number of channels)
        :param num_heads: Number of attention heads
        """
        super(Model, self).__init__()
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        """
        Forward pass of the AttentionBlock.
        :param x: Input tensor of shape (B, C, H, W)
        :return: Output tensor of the same shape (B, C, H, W)
        """
        B, C, H, W = x.shape
        x = x.view(B, C, H * W).permute(2, 0, 1)  # (seq_len, batch_size, embed_dim)
        attn_output, _ = self.attn(x, x, x)
        x = self.norm(attn_output + x)  # (seq_len, batch_size, embed_dim)
        x = x.permute(1, 2, 0).view(B, C, H, W)
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAException.h>

__global__ void ln_residual_kernel(const float* __restrict__ proj,
                                   const float* __restrict__ residual,
                                   const float* __restrict__ gamma,
                                   const float* __restrict__ beta,
                                   float* __restrict__ out,
                                   int rows, int C, int HW) {
    int row = blockIdx.x;
    int tid = threadIdx.x;

    __shared__ float ssum[256];
    __shared__ float ssq[256];

    float sum = 0.0f;
    float sq = 0.0f;
    int base = row * C;

    for (int c = tid; c < C; c += blockDim.x) {
        float v = proj[base + c] + residual[base + c];
        sum += v;
        sq += v * v;
    }
    ssum[tid] = sum;
    ssq[tid] = sq;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            ssum[tid] += ssum[tid + stride];
            ssq[tid] += ssq[tid + stride];
        }
        __syncthreads();
    }

    float mean = ssum[0] / (float)C;
    float var = ssq[0] / (float)C - mean * mean;
    float inv = rsqrtf(var + 1.0e-5f);

    int n = row % HW;
    int b = row / HW;
    int out_b_base = b * C * HW;

    for (int c = tid; c < C; c += blockDim.x) {
        float v = proj[base + c] + residual[base + c];
        float y = (v - mean) * inv;
        y = y * gamma[c] + beta[c];
        out[out_b_base + c * HW + n] = y;
    }
}

class ModelNew(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(ModelNew, self).__init__()
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads

    def forward(self, x):
        B, C, H, W = x.shape
        L = H * W

        # Equivalent to the reference's (L,B,C) self-attention, but laid out batch-first
        # so PyTorch's fused scaled_dot_product_attention path can be used without
        # materializing/returning the enormous attention-weight tensor.
        residual = x.flatten(2).transpose(1, 2).contiguous()  # (B,L,C)

        qkv = F.linear(residual, self.attn.in_proj_weight, self.attn.in_proj_bias)
        qkv = qkv.view(B, L, 3, self.num_heads, self.head_dim)
        q = qkv[:, :, 0, :, :].permute(0, 2, 1, 3)
        k = qkv[:, :, 1, :, :].permute(0, 2, 1, 3)
        v = qkv[:, :, 2, :, :].permute(0, 2, 1, 3)

        attn = F.scaled_dot_product_attention(q, k, v, attn_mask=None, dropout_p=0.0, is_causal=False)
        attn = attn.permute(0, 2, 1, 3).reshape(B, L, C)
        proj = F.linear(attn, self.attn.out_proj.weight, self.attn.out_proj.bias)

        return _ext.fused_ln_residual(proj, residual, self.norm.weight, self.norm.bias, H, W)

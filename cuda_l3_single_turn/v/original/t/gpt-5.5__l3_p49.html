from einops import rearrange

class Model(nn.Module):
    def __init__(self, batch_size, seq_length, n_heads, d_head, d_state, block_len=64):
        """
        Mamba Structured State Space model implementation for benchmarking.
        
        :param batch_size: Size of the batch
        :param seq_length: Length of the input sequence
        :param n_heads: Number of attention heads
        :param d_head: Dimension of each head
        :param d_state: Dimension of the state space
        :param block_len: Length of each block for chunked computation
        """
        super(Model, self).__init__()
        
        assert seq_length % block_len == 0, "Sequence length must be divisible by block length"
        
        self.batch_size = batch_size
        self.seq_length = seq_length
        self.n_heads = n_heads
        self.d_head = d_head
        self.d_state = d_state
        self.block_len = block_len
        
        # Initialize parameters
        self.A = nn.Parameter(torch.randn(batch_size, seq_length, n_heads))
        self.B = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))
        self.C = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))
        
    def segsum(self, x):
        """Naive segment sum calculation."""
        T = x.size(-1)
        x_cumsum = torch.cumsum(x, dim=-1)
        x_segsum = x_cumsum[..., :, None] - x_cumsum[..., None, :]
        mask = torch.tril(torch.ones(T, T, device=x.device, dtype=bool), diagonal=0)
        x_segsum = x_segsum.masked_fill(~mask, -torch.inf)
        return x_segsum
    
    def forward(self, X, initial_states=None):
        """
        Forward pass implementing the SSD operation.
        
        :param X: Input tensor of shape (batch, length, n_heads, d_head)
        :param initial_states: Optional initial states
        :return: Output tensor Y and final state
        """
        # Rearrange into blocks/chunks
        X_blocks, A_blocks, B_blocks, C_blocks = [
            rearrange(x, "b (c l) ... -> b c l ...", l=self.block_len)
            for x in (X, self.A, self.B, self.C)
        ]
        
        A_blocks = rearrange(A_blocks, "b c l h -> b h c l")
        A_cumsum = torch.cumsum(A_blocks, dim=-1)
        
        # 1. Compute diagonal block outputs
        L = torch.exp(self.segsum(A_blocks))
        Y_diag = torch.einsum("bclhn,bcshn,bhcls,bcshp->bclhp", 
                             C_blocks, B_blocks, L, X_blocks)
        
        # 2. Compute intra-chunk states
        decay_states = torch.exp((A_cumsum[:, :, :, -1:] - A_cumsum))
        states = torch.einsum("bclhn,bhcl,bclhp->bchpn", 
                            B_blocks, decay_states, X_blocks)
        
        # 3. Compute inter-chunk recurrence
        if initial_states is None:
            initial_states = torch.zeros_like(states[:, :1])
        states = torch.cat([initial_states, states], dim=1)
        
        decay_chunk = torch.exp(self.segsum(F.pad(A_cumsum[:, :, :, -1], (1, 0))))
        new_states = torch.einsum("bhzc,bchpn->bzhpn", decay_chunk, states)
        return new_states[:, -1]

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef TILE_P
#define TILE_P 16
#endif
#ifndef TILE_N
#define TILE_N 16
#endif
#ifndef MAX_L
#define MAX_L 128
#endif
#ifndef MAX_CHUNKS
#define MAX_CHUNKS 16
#endif

__global__ void ssd_final_kernel(const float* __restrict__ X,
                                 const float* __restrict__ A,
                                 const float* __restrict__ Bp,
                                 const float* __restrict__ initial,
                                 float* __restrict__ out,
                                 int Bsz, int L, int H, int P, int N,
                                 int block_len, int chunks, bool has_init) {
    __shared__ float sx[MAX_L * TILE_P];
    __shared__ float sb[MAX_L * TILE_N];
    __shared__ float sw[MAX_L];
    __shared__ float schunk[MAX_CHUNKS];
    __shared__ float stotal;

    const int tx = threadIdx.x; // p within tile
    const int ty = threadIdx.y; // n within tile
    const int tid = ty * blockDim.x + tx;

    const int ptiles = (P + TILE_P - 1) / TILE_P;
    const int tile = blockIdx.x;
    const int ptile = tile % ptiles;
    const int ntile = tile / ptiles;
    const int p0 = ptile * TILE_P;
    const int n0 = ntile * TILE_N;

    const int gh = blockIdx.y;
    const int b = gh / H;
    const int h = gh - b * H;

    // Compute exactly the two-level decay factors used by the reference:
    // per-chunk state weights exp(chunk_sum - chunk_prefix), and chunk sums.
    if (tid == 0) {
        float total = 0.0f;
        for (int c = 0; c < chunks && c < MAX_CHUNKS; ++c) {
            float cs = 0.0f;
            int base = c * block_len;
            #pragma unroll
            for (int l = 0; l < 128; ++l) {
                if (l >= block_len) break;
                int t = base + l;
                if (t < L) cs += A[((b * L + t) * H + h)];
            }
            schunk[c] = cs;
            total += cs;
            float pref = 0.0f;
            #pragma unroll
            for (int l = 0; l < 128; ++l) {
                if (l >= block_len) break;
                int t = base + l;
                if (t < L) {
                    pref += A[((b * L + t) * H + h)];
                    sw[t] = expf(cs - pref);
                }
            }
        }
        stotal = total;
    }
    __syncthreads();

    // Cooperative load of the X and B tiles for all timesteps.
    const int p = p0 + tx;
    const int n_thread = n0 + ty;
    for (int idx = tid; idx < L * TILE_P; idx += 256) {
        int t = idx / TILE_P;
        int ip = idx - t * TILE_P;
        int pp = p0 + ip;
        sx[idx] = (pp < P) ? X[(((b * L + t) * H + h) * P + pp)] : 0.0f;
    }
    for (int idx = tid; idx < L * TILE_N; idx += 256) {
        int t = idx / TILE_N;
        int in = idx - t * TILE_N;
        int nn = n0 + in;
        sb[idx] = (nn < N) ? Bp[(((b * L + t) * H + h) * N + nn)] : 0.0f;
    }
    __syncthreads();

    if (p >= P || n_thread >= N) return;

    float acc = 0.0f;
    if (has_init) {
        // initial_states layout: [B, 1, H, P, N]
        float initv = initial[(((b * H + h) * P + p) * N + n_thread)];
        acc = initv * expf(stotal);
    }

    float future = stotal;
    for (int c = 0; c < chunks && c < MAX_CHUNKS; ++c) {
        future -= schunk[c];
        float chunk_acc = 0.0f;
        int base = c * block_len;
        #pragma unroll
        for (int l = 0; l < 128; ++l) {
            if (l >= block_len) break;
            int t = base + l;
            if (t < L) {
                float xv = sx[t * TILE_P + tx];
                float bv = sb[t * TILE_N + ty];
                chunk_acc += xv * bv * sw[t];
            }
        }
        acc += chunk_acc * expf(future);
    }

    out[(((b * H + h) * P + p) * N + n_thread)] = acc;
}

                          torch::Tensor initial, bool has_init, int64_t block_len64,
                          torch::Tensor out) {

import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th


class Model(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(Model, self).__init__()

        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # The `clusters` weights are the `(w,b)` in the paper
        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        # The `clusters2` weights are the visual words `c_k` in the paper
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        """Aggregates feature maps into a fixed size representation.  In the following
        notation, B = batch_size, N = num_features, K = num_clusters, D = feature_size.

        Args:
            x (th.Tensor): B x N x D

        Returns:
            (th.Tensor): B x DK
        """
        max_sample = x.size()[1]
        x = x.view(-1, self.feature_size)  # B x N x D -> BN x D

        if x.device != self.clusters.device:
            msg = f"x.device {x.device} != cluster.device {self.clusters.device}"
            raise ValueError(msg)

        assignment = th.matmul(x, self.clusters)  # (BN x D) x (D x (K+G)) -> BN x (K+G)
        assignment = self.batch_norm(assignment)

        assignment = F.softmax(assignment, dim=1)  # BN x (K+G) -> BN x (K+G)
        # remove ghost assigments
        assignment = assignment[:, :self.cluster_size]
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # -> B x N x K
        a_sum = th.sum(assignment, dim=1, keepdim=True)  # B x N x K -> B x 1 x K
        a = a_sum * self.clusters2

        assignment = assignment.transpose(1, 2)  # B x N x K -> B x K x N

        x = x.view(-1, max_sample, self.feature_size)  # BN x D -> B x N x D
        vlad = th.matmul(assignment, x)  # (B x K x N) x (B x N x D) -> B x K x D
        vlad = vlad.transpose(1, 2)  # -> B x D x K
        vlad = vlad - a

        # L2 intra norm
        vlad = F.normalize(vlad)

        # flattening + L2 norm
        vlad = vlad.reshape(-1, self.cluster_size * self.feature_size)  # -> B x DK
        vlad = F.normalize(vlad)
        return vlad  # B x DK

torch::Tensor netvlad_forward(torch::Tensor x, torch::Tensor clusters, torch::Tensor clusters2,
                              torch::Tensor bn_weight, torch::Tensor bn_bias,
                              torch::Tensor running_mean, torch::Tensor running_var,
                              bool training, double momentum, double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    if (!x.is_contiguous()) x = x.contiguous();
    const int64_t B = x.size(0);
    const int64_t N = x.size(1);
    const int64_t D = x.size(2);
    const int64_t C = clusters.size(1);
    const int64_t K = clusters2.size(2);

    auto x2 = x.view({B * N, D});
    auto logits = at::matmul(x2, clusters);

    auto opts = x.options();
    auto mean = torch::empty({C}, opts);
    auto invstd = torch::empty({C}, opts);
    bn_stats_launcher(logits, mean, invstd, running_mean, running_var, training, eps, momentum);

    auto assign = torch::empty({B, N, K}, opts);
    auto asum = torch::zeros({B, K}, opts);
    softmax_asum_launcher(logits, mean, invstd, bn_weight, bn_bias, assign, asum, B, N, K, C);

    auto vlad = at::bmm(assign.transpose(1, 2), x);
    auto out = torch::empty({B, D * K}, opts);
    final_launcher(vlad, asum, clusters2, out, B, K, D);
    return out;

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#define THREADS 256

__inline__ __device__ float warp_reduce_sum(float v) {
    for (int off = 16; off > 0; off >>= 1) v += __shfl_down_sync(0xffffffff, v, off);
    return v;
}
__inline__ __device__ float warp_reduce_max(float v) {
    for (int off = 16; off > 0; off >>= 1) v = fmaxf(v, __shfl_down_sync(0xffffffff, v, off));
    return v;
}
__inline__ __device__ float block_reduce_sum(float v) {
    __shared__ float smem[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    v = warp_reduce_sum(v);
    if (lane == 0) smem[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (blockDim.x + 31) / 32) ? smem[lane] : 0.0f;
    if (wid == 0) v = warp_reduce_sum(v);
    return v;
}
__inline__ __device__ float block_reduce_max(float v) {
    __shared__ float smem[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    v = warp_reduce_max(v);
    if (lane == 0) smem[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (blockDim.x + 31) / 32) ? smem[lane] : -CUDART_INF_F;
    if (wid == 0) v = warp_reduce_max(v);
    return v;
}

__global__ void bn_stats_kernel(const float* __restrict__ logits, float* __restrict__ mean, float* __restrict__ invstd,
                                float* __restrict__ running_mean, float* __restrict__ running_var,
                                int M, int C, bool training, float eps, float momentum) {
    int c = blockIdx.x;
    float m, inv;
    if (training) {
        float s = 0.0f;
        float ss = 0.0f;
        for (int i = threadIdx.x; i < M; i += blockDim.x) {
            float v = logits[i * C + c];
            s += v;
            ss += v * v;
        }
        s = block_reduce_sum(s);
        ss = block_reduce_sum(ss);
        if (threadIdx.x == 0) {
            m = s / (float)M;
            float var = fmaxf(ss / (float)M - m * m, 0.0f);
            mean[c] = m;
            invstd[c] = rsqrtf(var + eps);
            if (running_mean != nullptr && running_var != nullptr) {
                running_mean[c] = (1.0f - momentum) * running_mean[c] + momentum * m;
                float unb = var * ((float)M / (float)(M - 1));
                running_var[c] = (1.0f - momentum) * running_var[c] + momentum * unb;
            }
        }
    } else {
        if (threadIdx.x == 0) {
            mean[c] = running_mean[c];
            invstd[c] = rsqrtf(running_var[c] + eps);
        }
    }
}

__global__ void softmax_asum_kernel(const float* __restrict__ logits, const float* __restrict__ mean,
                                    const float* __restrict__ invstd, const float* __restrict__ weight,
                                    const float* __restrict__ bias, float* __restrict__ assign,
                                    float* __restrict__ asum, int B, int N, int K, int C) {
    int row = blockIdx.x;
    int b = row / N;
    float local_max = -CUDART_INF_F;
    for (int c = threadIdx.x; c < C; c += blockDim.x) {
        float z = (logits[row * C + c] - mean[c]) * invstd[c];
        z = z * weight[c] + bias[c];
        local_max = fmaxf(local_max, z);
    }
    float mx = block_reduce_max(local_max);
    __syncthreads();
    float local_sum = 0.0f;
    for (int c = threadIdx.x; c < C; c += blockDim.x) {
        float z = (logits[row * C + c] - mean[c]) * invstd[c];
        z = z * weight[c] + bias[c];
        local_sum += expf(z - mx);
    }
    float denom = block_reduce_sum(local_sum);
    __syncthreads();
    float invden = 1.0f / denom;
    for (int c = threadIdx.x; c < K; c += blockDim.x) {
        float z = (logits[row * C + c] - mean[c]) * invstd[c];
        z = z * weight[c] + bias[c];
        float p = expf(z - mx) * invden;
        assign[row * K + c] = p;
        atomicAdd(&asum[b * K + c], p);
    }
}

__global__ void final_kernel(const float* __restrict__ vlad, const float* __restrict__ asum,
                             const float* __restrict__ clusters2, float* __restrict__ out,
                             int B, int K, int D, float global_inv) {
    int k = blockIdx.x;
    int b = blockIdx.y;
    float a = asum[b * K + k];
    float ss = 0.0f;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float val = vlad[(b * K + k) * D + d] - a * clusters2[d * K + k];
        ss += val * val;
    }
    ss = block_reduce_sum(ss);
    __shared__ float invn;
    if (threadIdx.x == 0) {
        float n = sqrtf(ss);
        invn = (n > 1.0e-12f ? 1.0f / n : 1.0e12f) * global_inv;
    }
    __syncthreads();
    float scale = invn;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float val = vlad[(b * K + k) * D + d] - a * clusters2[d * K + k];
        out[b * (D * K) + d * K + k] = val * scale;
    }
}

void bn_stats_launcher(const torch::Tensor& logits, torch::Tensor& mean, torch::Tensor& invstd,
                       torch::Tensor& running_mean, torch::Tensor& running_var,
                       bool training, double eps, double momentum) {
    int M = (int)logits.size(0);
    int C = (int)logits.size(1);
    float* rm = running_mean.defined() ? running_mean.data_ptr<float>() : nullptr;
    float* rv = running_var.defined() ? running_var.data_ptr<float>() : nullptr;
    bn_stats_kernel<<<C, THREADS, 0, at::cuda::getCurrentCUDAStream()>>>(
        logits.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(), rm, rv, M, C,
        training, (float)eps, (float)momentum);
}

                           torch::Tensor& assign, torch::Tensor& asum,
                           int64_t B, int64_t N, int64_t K, int64_t C) {

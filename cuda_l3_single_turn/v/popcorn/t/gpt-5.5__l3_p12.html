import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Stack of planar normalizing flow layers.  Each layer applies the
    invertible transform f(z) = z + u * tanh(w^T z + b), with a
    log-determinant-Jacobian correction.  Used to transform a simple
    base distribution (e.g. standard Gaussian) into a complex posterior.
    """

    def __init__(self, dim, num_flows):
        super().__init__()
        self.dim = dim
        self.num_flows = num_flows
        self.w = nn.ParameterList([nn.Parameter(torch.randn(dim)) for _ in range(num_flows)])
        self.u = nn.ParameterList([nn.Parameter(torch.randn(dim)) for _ in range(num_flows)])
        self.b = nn.ParameterList([nn.Parameter(torch.zeros(1)) for _ in range(num_flows)])

    def _constrain_u(self, w, u):
        """Ensure invertibility: w^T u >= -1."""
        wtu = (w * u).sum()
        m = -1 + F.softplus(wtu)
        u_hat = u + (m - wtu) * w / (w ** 2).sum()
        return u_hat

    def forward(self, z: torch.Tensor) -> tuple:
        """
        Args:
            z: (B, dim) – samples from base distribution
        Returns:
            (z_out, sum_log_det):
                z_out:       (B, dim) – transformed samples
                sum_log_det: (B,)     – total log |det df/dz|
        """
        sum_log_det = torch.zeros(z.shape[0], device=z.device)

        for i in range(self.num_flows):
            w = self.w[i]
            u = self._constrain_u(w, self.u[i])
            b = self.b[i]

            linear = z @ w + b  # (B,)
            h = torch.tanh(linear)
            h_prime = 1.0 - h ** 2

            z = z + u.unsqueeze(0) * h.unsqueeze(-1)
            psi = h_prime.unsqueeze(-1) * w.unsqueeze(0)  # (B, dim)
            log_det = torch.log(torch.abs(1.0 + psi @ u) + 1e-8)
            sum_log_det = sum_log_det + log_det

        return z, sum_log_det

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <vector>
#include <cmath>

__device__ __forceinline__ float warp_sum(float v) {
    unsigned mask = 0xffffffffu;
    v += __shfl_down_sync(mask, v, 16);
    v += __shfl_down_sync(mask, v, 8);
    v += __shfl_down_sync(mask, v, 4);
    v += __shfl_down_sync(mask, v, 2);
    v += __shfl_down_sync(mask, v, 1);
    return v;
}

__device__ __forceinline__ float softplus_pt(float x) {
    // PyTorch softplus default: beta=1, threshold=20
    return (x > 20.0f) ? x : log1pf(expf(x));
}

__global__ void planar_kernel_32(const float* __restrict__ z_in,
                                 const float* __restrict__ w,
                                 const float* __restrict__ u,
                                 const float* __restrict__ b,
                                 float* __restrict__ z_out,
                                 float* __restrict__ sum_ld,
                                 int B, int D, int F) {
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    int row = blockIdx.x * 4 + warp;
    if (row >= B) return;

    float zval = 0.0f;
    if (lane < D) zval = z_in[row * D + lane];
    float acc_ld = 0.0f;

    #pragma unroll
    for (int f = 0; f < 64; ++f) {
        if (f >= F) break;
        const float* wf = w + f * D;
        const float* uf = u + f * D;

        float wj = (lane < D) ? wf[lane] : 0.0f;
        float uj = (lane < D) ? uf[lane] : 0.0f;

        float wtu_part = wj * uj;
        float wnorm_part = wj * wj;
        float dot_part = zval * wj;

        float wtu = warp_sum(wtu_part);
        float wnorm = warp_sum(wnorm_part);
        float linear = warp_sum(dot_part) + b[f];

        float m = -1.0f + softplus_pt(wtu);
        float coeff = (m - wtu) / wnorm;
        float uhat = uj + coeff * wj;

        float h = tanhf(linear);
        float hp = 1.0f - h * h;
        if (lane < D) zval += uhat * h;

        if (lane == 0) {
            float det_arg = 1.0f + hp * m;  // w^T u_hat == m by construction
            acc_ld += logf(fabsf(det_arg) + 1.0e-8f);
        }
    }

    if (lane < D) z_out[row * D + lane] = zval;
    if (lane == 0) sum_ld[row] = acc_ld;
}

std::vector<torch::Tensor> planar_forward(torch::Tensor z, torch::Tensor w, torch::Tensor u, torch::Tensor b) {
    TORCH_CHECK(z.is_cuda(), "z must be CUDA");
    TORCH_CHECK(z.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(w.scalar_type() == at::kFloat && u.scalar_type() == at::kFloat && b.scalar_type() == at::kFloat, "params must be float32");
    auto zc = z.contiguous();
    auto wc = w.contiguous();
    auto uc = u.contiguous();
    auto bc = b.contiguous().view({-1});

    int B = (int)zc.size(0);
    int D = (int)zc.size(1);
    int F = (int)wc.size(0);
    TORCH_CHECK(D <= 32, "this optimized kernel supports dim <= 32");

    auto zout = torch::empty_like(zc);
    auto ldet = torch::empty({B}, zc.options());

    dim3 block(128);
    dim3 grid((B + 3) / 4);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    planar_kernel_32<<<grid, block, 0, stream>>>(
        zc.data_ptr<float>(), wc.data_ptr<float>(), uc.data_ptr<float>(), bc.data_ptr<float>(),
        zout.data_ptr<float>(), ldet.data_ptr<float>(), B, D, F);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return {zout, ldet};
}

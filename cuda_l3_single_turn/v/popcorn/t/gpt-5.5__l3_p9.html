import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Bayesian linear regression with a conjugate Gaussian prior.
    Given features X and targets y, computes the posterior distribution
    over weights: p(w | X, y) = N(mu_post, Sigma_post) in closed form
    via the matrix inversion lemma / Cholesky decomposition.
    """

    def __init__(self, input_dim, prior_precision=1.0, noise_precision=1.0):
        super().__init__()
        self.input_dim = input_dim
        self.prior_precision = prior_precision
        self.noise_precision = noise_precision

    def forward(
        self, X: torch.Tensor, y: torch.Tensor, X_test: torch.Tensor
    ) -> tuple:
        """
        Args:
            X:      (B, N, D) – training features
            y:      (B, N, 1) – training targets
            X_test: (B, M, D) – test features
        Returns:
            (pred_mean, pred_var):
                pred_mean: (B, M, 1)
                pred_var:  (B, M)
        """
        D = X.shape[-1]
        eye = torch.eye(D, device=X.device, dtype=X.dtype).unsqueeze(0)

        # Posterior precision: Lambda_post = alpha * I + beta * X^T X
        Lambda_post = self.prior_precision * eye + self.noise_precision * (X.transpose(-1, -2) @ X)
        L = torch.linalg.cholesky(Lambda_post)

        # Posterior mean: mu_post = Lambda_post^{-1} * beta * X^T y
        rhs = self.noise_precision * (X.transpose(-1, -2) @ y)
        mu_post = torch.cholesky_solve(rhs, L)

        # Predictive distribution
        pred_mean = X_test @ mu_post
        Sigma_post = torch.cholesky_inverse(L)
        pred_var = (X_test @ Sigma_post * X_test).sum(-1) + 1.0 / self.noise_precision

        return pred_mean, pred_var

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

#ifndef DMAX
#define DMAX 16
#endif

__global__ void blr_kernel_16(
    const float* __restrict__ X,
    const float* __restrict__ y,
    const float* __restrict__ Xtst,
    float* __restrict__ out_mean,
    float* __restrict__ out_var,
    int B, int N, int M,
    float prior, float noise)
{
    const int b = blockIdx.x;
    const int tid = threadIdx.x;

    __shared__ float A[DMAX*DMAX];
    __shared__ float rhs[DMAX];
    __shared__ float L[DMAX*DMAX];
    __shared__ float mu[DMAX];
    __shared__ float Sigma[DMAX*DMAX];

    const float* Xb = X + ((long long)b) * N * DMAX;
    const float* yb = y + ((long long)b) * N;

    // Gram matrix: prior*I + noise * X^T X. One thread per element.
    if (tid < DMAX*DMAX) {
        int i = tid >> 4;
        int j = tid & 15;
        float s = 0.0f;
        #pragma unroll 4
        for (int n = 0; n < N; ++n) {
            float xi = Xb[n*DMAX + i];
            float xj = Xb[n*DMAX + j];
            s += xi * xj;
        }
        A[i*DMAX + j] = noise * s + ((i == j) ? prior : 0.0f);
    }

    // X^T y
    if (tid < DMAX) {
        int i = tid;
        float s = 0.0f;
        #pragma unroll 4
        for (int n = 0; n < N; ++n) {
            s += Xb[n*DMAX + i] * yb[n];
        }
        rhs[i] = noise * s;
    }
    __syncthreads();

    if (tid == 0) {
        // Cholesky factorization, lower triangular L (A is SPD).
        #pragma unroll
        for (int i = 0; i < DMAX; ++i) {
            #pragma unroll
            for (int j = 0; j < DMAX; ++j) L[i*DMAX + j] = 0.0f;
        }
        for (int i = 0; i < DMAX; ++i) {
            for (int j = 0; j <= i; ++j) {
                float s = A[i*DMAX + j];
                for (int k = 0; k < j; ++k) s -= L[i*DMAX + k] * L[j*DMAX + k];
                if (i == j) {
                    L[i*DMAX + j] = sqrtf(fmaxf(s, 1.0e-20f));
                } else {
                    L[i*DMAX + j] = s / L[j*DMAX + j];
                }
            }
        }

        // mu = inv(A) rhs via two triangular solves. Reuse A[0:16] as temp z.
        float z[DMAX];
        for (int i = 0; i < DMAX; ++i) {
            float s = rhs[i];
            for (int k = 0; k < i; ++k) s -= L[i*DMAX + k] * z[k];
            z[i] = s / L[i*DMAX + i];
        }
        for (int i = DMAX - 1; i >= 0; --i) {
            float s = z[i];
            for (int k = i + 1; k < DMAX; ++k) s -= L[k*DMAX + i] * mu[k];
            mu[i] = s / L[i*DMAX + i];
        }

        // Compute inverse of lower triangular L into A: Linv.
        for (int i = 0; i < DMAX; ++i) {
            for (int j = 0; j < DMAX; ++j) A[i*DMAX + j] = 0.0f;
        }
        for (int col = 0; col < DMAX; ++col) {
            for (int i = 0; i < DMAX; ++i) {
                float s = (i == col) ? 1.0f : 0.0f;
                for (int k = 0; k < i; ++k) s -= L[i*DMAX + k] * A[k*DMAX + col];
                A[i*DMAX + col] = s / L[i*DMAX + i];
            }
        }
        // Sigma = inv(L)^T inv(L). Store full symmetric matrix.
        for (int i = 0; i < DMAX; ++i) {
            for (int j = 0; j < DMAX; ++j) {
                float s = 0.0f;
                for (int k = 0; k < DMAX; ++k) s += A[k*DMAX + i] * A[k*DMAX + j];
                Sigma[i*DMAX + j] = s;
            }
        }
    }
    __syncthreads();

    // Predictions. 64 tests in the benchmark: one thread per test point.
    if (tid < M) {
        const float* x = Xtst + (((long long)b) * M + tid) * DMAX;
        float mean = 0.0f;
        #pragma unroll
        for (int i = 0; i < DMAX; ++i) mean += x[i] * mu[i];
        out_mean[((long long)b) * M + tid] = mean;

        float v = 0.0f;
        #pragma unroll
        for (int i = 0; i < DMAX; ++i) {
            float row = 0.0f;
            #pragma unroll
            for (int j = 0; j < DMAX; ++j) row += Sigma[i*DMAX + j] * x[j];
            v += x[i] * row;
        }
        out_var[((long long)b) * M + tid] = v + 1.0f / noise;
    }
}

std::vector<torch::Tensor> blr_cuda(torch::Tensor X, torch::Tensor y, torch::Tensor Xtest, double prior_precision, double noise_precision) {
    const int B = (int)X.size(0);
    const int N = (int)X.size(1);
    const int D = (int)X.size(2);
    const int M = (int)Xtest.size(1);
    auto mean = torch::empty({B, M, 1}, X.options());
    auto var = torch::empty({B, M}, X.options());
    if (D == 16) {
        blr_kernel_16<<<B, 256, 0, at::cuda::getCurrentCUDAStream()>>>(
            X.data_ptr<float>(), y.data_ptr<float>(), Xtest.data_ptr<float>(),
            mean.data_ptr<float>(), var.data_ptr<float>(), B, N, M,
            (float)prior_precision, (float)noise_precision);
    }
    return {mean, var};
}
'''

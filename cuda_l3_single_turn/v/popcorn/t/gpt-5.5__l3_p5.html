import torch
import torch.nn as nn

# Default ModelNew is PyTorch-only. run_and_check: use check_kernel=False with backend=cuda
# until the candidate file contains a real CUDA kernel (see kernel_static_checker).


class Model(nn.Module):
    """
    **Fused probability-flow ODE integration** (flow-matching style) in one ``forward``.

    Models the ODE ``dx/dt = v_θ(x, t)`` on ``t ∈ [0, 1]`` with a learned velocity
    field. A **single fused pass** unrolls **fixed-step Euler** integration for
    ``num_steps`` substeps—typical for **real-time trajectory / action generation**
    where latency is dominated by repeated ``v_θ`` evaluations.

    This is the continuous-time analogue of a rectified-flow / conditional-flow
    sampler: one ``forward`` returns an approximate **x(1)** from **x(0)** without a
    Python loop across separate module calls (fusion target for the integrator +
    field).
    """

    def __init__(self, dim: int, num_steps: int):
        super().__init__()
        assert dim >= 1 and num_steps >= 1
        self.dim = dim
        self.num_steps = num_steps
        self.register_buffer("dt", torch.tensor(1.0 / float(num_steps)))

        self.velocity = nn.Sequential(
            nn.Linear(dim + 1, dim * 4),
            nn.SiLU(),
            nn.Linear(dim * 4, dim * 4),
            nn.SiLU(),
            nn.Linear(dim * 4, dim),
        )

    def forward(self, x0: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x0: (B, dim) initial state at ``t = 0`` (e.g. noise or anchor trajectory).
        Returns:
            (B, dim) approximate state at ``t = 1`` after fused Euler integration.
        """
        B, d = x0.shape
        assert d == self.dim
        x = x0
        dt = self.dt.to(dtype=x0.dtype, device=x0.device)
        for k in range(self.num_steps):
            t = (float(k) + 0.5) / float(self.num_steps)
            t_b = torch.full((B, 1), t, device=x0.device, dtype=x0.dtype)
            inp = torch.cat([x, t_b], dim=-1)
            v = self.velocity(inp)
            x = x + dt * v
        return x


class ModelNew(nn.Module):
    """KernelBench candidate entry point; default delegates to ``Model``."""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>

torch::Tensor flow_forward(torch::Tensor x0,
                           torch::Tensor w1, torch::Tensor b1,
                           torch::Tensor w2, torch::Tensor b2,
                           torch::Tensor w3, torch::Tensor b3,
                           int64_t num_steps);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>

static __device__ __forceinline__ float silu_f(float x) {
    return x / (1.0f + expf(-x));
}

__global__ void flow_kernel(const float* __restrict__ x0,
                            float* __restrict__ out,
                            const float* __restrict__ w1,
                            const float* __restrict__ b1,
                            const float* __restrict__ w2,
                            const float* __restrict__ b2,
                            const float* __restrict__ w3,
                            const float* __restrict__ b3,
                            int B, int dim, int hidden, int num_steps) {
    extern __shared__ float smem[];
    float* x = smem;
    float* h1 = x + dim;
    float* h2 = h1 + hidden;

    const int tid = threadIdx.x;
    const int b = blockIdx.x;
    const float* x0b = x0 + ((long long)b) * dim;
    float* outb = out + ((long long)b) * dim;

    for (int i = tid; i < dim; i += blockDim.x) {
        x[i] = x0b[i];
    }
    __syncthreads();

    const float dt = 1.0f / (float)num_steps;
    const int in1 = dim + 1;

    for (int k = 0; k < num_steps; ++k) {
        const float t = ((float)k + 0.5f) * dt;

        // Linear(dim+1 -> hidden) + SiLU.  The final input coordinate is time t.
        for (int j = tid; j < hidden; j += blockDim.x) {
            const float* wrow = w1 + ((long long)j) * in1;
            float sum = b1[j];
            #pragma unroll 4
            for (int i = 0; i < dim; ++i) {
                sum = fmaf(x[i], wrow[i], sum);
            }
            sum = fmaf(t, wrow[dim], sum);
            h1[j] = silu_f(sum);
        }
        __syncthreads();

        // Linear(hidden -> hidden) + SiLU.
        for (int j = tid; j < hidden; j += blockDim.x) {
            const float* wrow = w2 + ((long long)j) * hidden;
            float sum = b2[j];
            #pragma unroll 4
            for (int i = 0; i < hidden; ++i) {
                sum = fmaf(h1[i], wrow[i], sum);
            }
            h2[j] = silu_f(sum);
        }
        __syncthreads();

        // Linear(hidden -> dim), Euler update.
        for (int j = tid; j < dim; j += blockDim.x) {
            const float* wrow = w3 + ((long long)j) * hidden;
            float sum = b3[j];
            #pragma unroll 4
            for (int i = 0; i < hidden; ++i) {
                sum = fmaf(h2[i], wrow[i], sum);
            }
            x[j] = fmaf(dt, sum, x[j]);
        }
        __syncthreads();
    }

    for (int i = tid; i < dim; i += blockDim.x) {
        outb[i] = x[i];
    }
}

torch::Tensor flow_forward(torch::Tensor x0,
                           torch::Tensor w1, torch::Tensor b1,
                           torch::Tensor w2, torch::Tensor b2,
                           torch::Tensor w3, torch::Tensor b3,
                           int64_t num_steps) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(x0));
    auto out = torch::empty_like(x0);
    const int B = (int)x0.size(0);
    const int dim = (int)x0.size(1);
    const int hidden = dim * 4;

    int block = 1;
    while (block < hidden && block < 1024) block <<= 1;
    size_t shmem = (size_t)(dim + 2 * hidden) * sizeof(float);

    flow_kernel<<<B, block, shmem, at::cuda::getCurrentCUDAStream()>>>(
        x0.data_ptr<float>(), out.data_ptr<float>(),
        w1.data_ptr<float>(), b1.data_ptr<float>(),
        w2.data_ptr<float>(), b2.data_ptr<float>(),
        w3.data_ptr<float>(), b3.data_ptr<float>(),
        B, dim, hidden, (int)num_steps);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return out;
}
'''

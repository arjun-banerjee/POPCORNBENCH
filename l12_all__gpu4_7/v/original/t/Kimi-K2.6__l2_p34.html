    float* __restrict__ output,
    int total_lines,
    int W,
    float eps,
    float scale,
    const float* __restrict__ weight,
    const float* __restrict__ bias) {

    const int warp_id = threadIdx.x >> 5;
    const int warps_per_block = blockDim.x >> 5;
    const int line_idx = blockIdx.x * warps_per_block + warp_id;

    const long line_offset = (long)line_idx * W;

    float sum = 0.0f;
    float sum_sq = 0.0f;
    
    int w = lane;
    if (w < W) {
        reg0 = input[line_offset + w];
        sum += reg0;
        sum_sq += reg0 * reg0;
        w += 32;
        if (w < W) {

            sum += reg1;
            sum_sq += reg1 * reg1;
            w += 32;

                    sum += reg3;
                    sum_sq += reg3 * reg3;
                }
            }
        }

        sum_sq += __shfl_down_sync(0xffffffff, sum_sq, offset);

        mean = sum / W;
        float var = sum_sq / W - mean * mean;
        inv_std = rsqrtf(var + eps);

    inv_std = __shfl_sync(0xffffffff, inv_std, 0);

        if (weight != nullptr) {

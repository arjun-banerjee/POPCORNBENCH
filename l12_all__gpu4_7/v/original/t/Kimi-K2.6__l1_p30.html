__global__ void softsign_float4_kernel(const float* __restrict__ input,
                                        float* __restrict__ output,
                                        int64_t n) {
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t n_vec4 = n / 4;
    
    if (idx < n_vec4) {
        float4 in = reinterpret_cast<const float4*>(input)[idx];
        float4 out;
        
        out.x = in.x / (1.0f + fabsf(in.x));
        out.y = in.y / (1.0f + fabsf(in.y));
        out.z = in.z / (1.0f + fabsf(in.z));
        out.w = in.w / (1.0f + fabsf(in.w));
        
        reinterpret_cast<float4*>(output)[idx] = out;
    }
    
    // Handle tail in the last block
    if (blockIdx.x == gridDim.x - 1) {
        int64_t remainder_start = n_vec4 * 4;

        if (tail_idx < n) {
            float val = __ldg(&input[tail_idx]);
            output[tail_idx] = val / (1.0f + fabsf(val));
        }
    }
}

torch::Tensor softsign_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);

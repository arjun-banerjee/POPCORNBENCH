__inline__ __device__ float warp_reduce_sum(float v) {
    #pragma unroll
    for (int o = 16; o > 0; o >>= 1)
        v += __shfl_down_sync(0xffffffff, v, o);
    return v;
}

__inline__ __device__ float block_reduce_sum(float v) {
    static __shared__ float s[32];
    int lane = threadIdx.x & 31;

    v = warp_reduce_sum(v);
    if (lane == 0) s[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (blockDim.x >> 5)) ? s[lane] : 0.0f;
    if (wid == 0) v = warp_reduce_sum(v);
    return v;
}

__global__ void instnorm_kernel(const float* __restrict__ x,
                                float* __restrict__ y,
                                int spatial,
                                float eps) {
    int inst = blockIdx.x;
    const float* xi = x + (int64_t)inst * spatial;
    float* yi = y + (int64_t)inst * spatial;

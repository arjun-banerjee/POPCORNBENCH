    int h_off = threadIdx.y * STRIDE;
    int w_off = threadIdx.x * STRIDE;

    #pragma unroll
    for (int kd = 0; kd < KERNEL_SIZE; ++kd) {
        #pragma unroll
        for (int kh = 0; kh < KERNEL_SIZE; ++kh) {
            #pragma unroll

}

__global__ void avg_pool3d_generic_kernel(

}

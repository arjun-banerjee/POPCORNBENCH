__global__ void cumsum_kernel(const float* __restrict__ input,
                              float* __restrict__ output,
                              int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;

    const float4* in4  = reinterpret_cast<const float4*>(input  + row * cols);
    float4*       out4 = reinterpret_cast<float4*>(output + row * cols);

    int tid      = threadIdx.x;
    int lane     = tid & 31;
    int warp_id  = tid >> 5;
    int num_warps = blockDim.x >> 5;

    extern __shared__ float shared[];
    float* warp_prefix = shared;
    float* running_sum = shared + num_warps;
    float* tile_total  = shared + num_warps + 1;

    if (tid == 0) {
        *running_sum = 0.0f;
    }
    __syncthreads();

    int elems_per_tile = blockDim.x * 4;
    int num_tiles = cols / elems_per_tile;

    for (int tile = 0; tile < num_tiles; ++tile) {
        int idx = tile * blockDim.x + tid;
        float4 v4 = in4[idx];
        float v0 = v4.x;
        float v1 = v4.y;
        float v2 = v4.z;
        float v3 = v4.w;

        v2 += v1;
        v3 += v2;

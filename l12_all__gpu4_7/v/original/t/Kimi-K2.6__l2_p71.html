import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, divides by a constant, and applies LeakyReLU.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divisor = divisor

    def forward(self, x):
        x = self.conv(x)
        x = x / self.divisor
        x = torch.nn.functional.leaky_relu(x, negative_slope=0.01)

#include <cuda_runtime.h>

__global__ void fused_div_leaky_relu_kernel(float* __restrict__ data,
                                            int n,
                                            float inv_divisor,
                                            float negative_slope) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int idx = tid * 4;
    if (idx + 3 < n) {
        float4 v = reinterpret_cast<float4*>(data)[tid];
        v.x = v.x * inv_divisor;
        v.y = v.y * inv_divisor;
        v.z = v.z * inv_divisor;
        v.w = v.w * inv_divisor;
        v.x = fmaxf(v.x, v.x * negative_slope);
        v.y = fmaxf(v.y, v.y * negative_slope);
        v.z = fmaxf(v.z, v.z * negative_slope);
        v.w = fmaxf(v.w, v.w * negative_slope);
        reinterpret_cast<float4*>(data)[tid] = v;

}

__global__ __launch_bounds__(128, 16) void sigmoid_kernel(const float* __restrict__ input,
                                                          float* __restrict__ output,
                                                          int64_t n) {
    int64_t idx = static_cast<int64_t>(blockIdx.x) * blockDim.x + threadIdx.x;
    int64_t stride = static_cast<int64_t>(gridDim.x) * blockDim.x;
    int64_t num_vectors = n / 4;

    float4* out_vec = reinterpret_cast<float4*>(output);

        float4 v = __ldg(&in_vec[i]);
        v.x = 1.0f / (1.0f + __expf(-v.x));
        v.y = 1.0f / (1.0f + __expf(-v.y));
        v.z = 1.0f / (1.0f + __expf(-v.z));

}

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_epilogue_kernel(float* __restrict__ data, int total, float sub_val, float mul_val) {
    const int items_per_thread = 16;
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * items_per_thread;
    
    #pragma unroll
    for (int i = 0; i < items_per_thread; i += 4) {
        int j = idx + i;
        if (j + 3 < total) {
            float4 vals = *(float4*)(data + j);
            vals.x = fmaxf(0.0f, (vals.x - sub_val) * mul_val);
            vals.y = fmaxf(0.0f, (vals.y - sub_val) * mul_val);
            vals.z = fmaxf(0.0f, (vals.z - sub_val) * mul_val);
            vals.w = fmaxf(0.0f, (vals.w - sub_val) * mul_val);
            *(float4*)(data + j) = vals;
        } else {
            #pragma unroll
            for (int k = 0; k < 4 && (j + k) < total; k++) {
                float val = data[j + k];
                val = fmaxf(0.0f, (val - sub_val) * mul_val);
                data[j + k] = val;
            }
        }
    }
}

torch::Tensor fused_epilogue(torch::Tensor out, float sub_val, float mul_val) {
    int total = out.numel();
    int threads = 512;
    int items_per_thread = 16;
    int blocks = (total + threads * items_per_thread - 1) / (threads * items_per_thread);
    
    fused_epilogue_kernel<<<blocks, threads>>>(
        out.data_ptr<float>(), total, sub_val, mul_val);
    
    return out;
}

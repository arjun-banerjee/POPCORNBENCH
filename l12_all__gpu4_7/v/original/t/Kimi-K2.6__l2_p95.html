import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
#include <torch/extension.h>
void fused_forward(torch::Tensor data, torch::Tensor add_value);
"""

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef M_SQRT1_2
#define M_SQRT1_2 0.70710678118654752440f
#endif

__device__ __forceinline__ float activation(float val, float add) {
    val += add;
    float sig = 1.0f / (1.0f + expf(-val));
    val = sig * val;
    val = tanhf(val);
    val = 0.5f * val * (1.0f + erff(val * M_SQRT1_2));
    val = fminf(1.0f, fmaxf(-1.0f, val));
    return val;
}

__global__ void fused_kernel(float* __restrict__ data,
                             const float* __restrict__ add_value,
                             int out_features_div4,
                             int num_float4) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= num_float4) return;
    int col4 = idx & (out_features_div4 - 1);
    float4 v = reinterpret_cast<float4*>(data)[idx];
    float4 a = reinterpret_cast<const float4*>(add_value)[col4];
    v.x = activation(v.x, a.x);
    v.y = activation(v.y, a.y);
    v.z = activation(v.z, a.z);
    v.w = activation(v.w, a.w);
    reinterpret_cast<float4*>(data)[idx] = v;
}

void fused_forward(torch::Tensor data, torch::Tensor add_value) {
    int batch_size = data.size(0);
    int out_features = data.size(1);
    int num_float4 = batch_size * out_features / 4;
    int threads = 256;
    int blocks = (num_float4 + threads - 1) / threads;
    int out_features_div4 = out_features / 4;
    fused_kernel<<<blocks, threads>>>(
        data.data_ptr<float>(),
        add_value.data_ptr<float>(),
        out_features_div4,
        num_float4
    );
}
"""

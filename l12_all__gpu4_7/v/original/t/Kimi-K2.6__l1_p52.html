    for (int wb = warp_id; wb < total_warp_blocks; wb += total_warps) {
        int b = wb / warps_per_row;
        int j_block = (wb % warps_per_row) * j_per_warp;

            if (val < min_val) {
                min_val = val;
                min_idx = i;
            }
        }

        output[b * D2 + j] = min_idx;
    }
}

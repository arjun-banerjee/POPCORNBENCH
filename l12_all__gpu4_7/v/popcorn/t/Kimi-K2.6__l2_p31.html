import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Fused MLA-style attention from compressed KV latents.

    The compressed latent representation is expanded into keys and values on the
    fly inside the attention computation rather than materializing a KV cache
    first. This isolates the main compute pattern behind fused MLA attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        kv_latent: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        rank = kv_latent.shape[-1]
        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_heads):
                for t in range(seq_len):
                    logits = []
                    values = []
                    for s in range(t + 1):
                        latent = kv_latent[b, s]
                        k = torch.matmul(latent, w_up_k[:, h])
                        v = torch.matmul(latent, w_up_v[:, h])
                        logits.append((q[b, t, h] * k).sum() * scale)
                        values.append(v)
                    attn = torch.softmax(torch.stack(logits, dim=0), dim=0)
                    value_stack = torch.stack(values, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * value_stack).sum(dim=0)

        return out

We need to write a custom CUDA kernel that replaces the Model.forward. The reference is a triple nested loop over batch, heads, time steps, computing causal attention where keys and values are expanded from latent vectors on the fly: k = torch.matmul(latent, w_up_k[:, h]) and v = torch.matmul(latent, w_up_v[:, h]). So each token's latent is shape (rank,) and we have per-head up-project weight matrices of shape (rank, head_dim) for k and v. Hence the full operation: For each b, h, t (query position), compute for s=0..t: k_{b,h,s} = latent_{b,s} @ w_up_k[:, h] # shape (head_dim,) v_{b,h,s} = latent_{b,s} @ w_up_v[:, h] # shape (head_dim,) Then logits[t,s] = dot(q[b,t,h], k_{b,h,s}) * scale attn = softmax over s out[b,t,h] = sum_s attn[s] * v_{b,h,s} This is essentially a causal attention with KV projection from latents. The latent dimension is 24, head_dim 16, seq_len 32. We will write a CUDA kernel that does this fusedly. Given the small sizes, we need to optimize occupancy and memory coalescing. Since seq_len=32 and head_dim=16 and rank=24, we can load per-head weights into shared memory, compute KV on the fly, keep running softmax and accumulate output.

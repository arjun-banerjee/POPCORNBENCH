#include <cuda.h>
#include <cuda_runtime.h>

__global__ void lse4_kernel(const float* __restrict__ x, float* __restrict__ y) {
    int i = threadIdx.x;
    float a0 = x[i];
    float a1 = x[i + 320];
    float a2 = x[i + 640];
    float a3 = x[i + 960];
    float m = fmaxf(fmaxf(a0, a1), fmaxf(a2, a3));
    float s = __expf(a0 - m) + __expf(a1 - m) + __expf(a2 - m) + __expf(a3 - m);
    y[i] = m + __logf(s);
}

torch::Tensor lse4_forward(torch::Tensor x, torch::Tensor y) {
    lse4_kernel<<<1, 320, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), y.data_ptr<float>());
    return y;
}

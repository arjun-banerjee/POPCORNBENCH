import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_src = r'''
#include <torch/extension.h>

void tanh_forward_cuda(torch::Tensor x, torch::Tensor y, long long n);

torch::Tensor tanh_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    long long n = x.numel();
    if (n == 0) return y;
    tanh_forward_cuda(x, y, n);
    return y;
}
'''

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__device__ __forceinline__ float tanh_pade(float x) {
    float x2 = x * x;
    float x4 = x2 * x2;
    float x6 = x4 * x2;
    float num = fmaf(378.0f, x4, fmaf(17325.0f, x2, 135135.0f));
    num = fmaf(x6, 1.0f, num);
    float den = fmaf(3150.0f, x4, fmaf(62370.0f, x2, 135135.0f));
    den = fmaf(28.0f, x6, den);
    return x * __fdividef(num, den);
}

__global__ void tanh_vec4_kernel(const float4* __restrict__ x4, float4* __restrict__ y4, long long n4) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = idx; i < n4; i += stride) {
        float4 v = x4[i];
        float4 o;
        o.x = tanh_pade(v.x);
        o.y = tanh_pade(v.y);
        o.z = tanh_pade(v.z);
        o.w = tanh_pade(v.w);
        y4[i] = o;
    }
}

__global__ void tanh_scalar_kernel(const float* __restrict__ x, float* __restrict__ y, long long n) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = idx; i < n; i += stride) {
        y[i] = tanh_pade(x[i]);
    }
}

void tanh_forward_cuda(torch::Tensor x, torch::Tensor y, long long n) {
    const int threads = 256;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const uintptr_t xp = reinterpret_cast<uintptr_t>(x.data_ptr<float>());
    const uintptr_t yp = reinterpret_cast<uintptr_t>(y.data_ptr<float>());
    if (((xp | yp) & 15) == 0 && (n % 4 == 0)) {
        long long n4 = n / 4;
        int blocks = (int)((n4 + threads - 1) / threads);

        int blocks = (int)((n + threads - 1) / threads);
        if (blocks > 65535) blocks = 65535;
        tanh_scalar_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    }
}

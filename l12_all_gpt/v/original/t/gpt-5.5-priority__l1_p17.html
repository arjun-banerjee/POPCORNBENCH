torch::Tensor matmul_bt(torch::Tensor A, torch::Tensor B) {
    const int M = (int)A.size(0);
    const int K = (int)A.size(1);
    const int N = (int)B.size(0);

    auto C = torch::empty({M, N}, A.options());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    cublasSetStream(handle, at::cuda::getCurrentCUDAStream());

    cublasMath_t old_mode;
    cublasGetMathMode(handle, &old_mode);
    cublasSetMathMode(handle, CUBLAS_TF32_TENSOR_OP_MATH);

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // Tensors are row-major. Interpret the operation as column-major:
    // C_row = A_row * B_row^T  <=>  C_col^T = B_col^T * A_col
    // This writes directly into the row-major C storage with leading dimension N.
    cublasStatus_t st = cublasSgemm(handle,
                                    CUBLAS_OP_T, CUBLAS_OP_N,

                                    B.data_ptr<float>(), K,
                                    A.data_ptr<float>(), K,
                                    &beta,
                                    C.data_ptr<float>(), N);

    cublasSetMathMode(handle, old_mode);
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSgemm failed");
    return C;
}

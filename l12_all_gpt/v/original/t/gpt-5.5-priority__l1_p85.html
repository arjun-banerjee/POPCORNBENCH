    }
    __syncthreads();

    int oh = oh0 + ty;
    if (oh < OH) {
        const float* wptr = w + c * KH * KW;
        float init = has_bias ? b[c] : 0.0f;
        float a0 = init, a1 = init, a2 = init, a3 = init;
        #pragma unroll
        for (int kh = 0; kh < KH; ++kh) {
            int base = (ty + kh) * SHW + tx;
            #pragma unroll
            for (int kw = 0; kw < KW; ++kw) {
                float wt = wptr[kh * KW + kw];
                a0 = fmaf(smem[base + kw], wt, a0);
                a1 = fmaf(smem[base + 32 + kw], wt, a1);
                a2 = fmaf(smem[base + 64 + kw], wt, a2);
                a3 = fmaf(smem[base + 96 + kw], wt, a3);
            }
        }
        int outbase = ((n * C + c) * OH + oh) * OW;
        int ow = ow0 + tx;
        if (ow < OW) y[outbase + ow] = a0;
        ow += 32;
        if (ow < OW) y[outbase + ow] = a1;
        ow += 32;
        if (ow < OW) y[outbase + ow] = a2;
        ow += 32;
        if (ow < OW) y[outbase + ow] = a3;
    }
}

__global__ void dw_generic(const float* __restrict__ x,
                           const float* __restrict__ w,
                           const float* __restrict__ b,
                           float* __restrict__ y,
                           int N, int C, int H, int W, int OH, int OW,
                           int kH, int kW, int sh, int sw, int ph, int pw, int dh, int dw,
                           int has_bias) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * OH * OW;
    if (idx >= total) return;
    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int c = t % C;
    int n = t / C;
    float acc = has_bias ? b[c] : 0.0f;
    const float* xptr = x + ((n * C + c) * H) * W;
    const float* wptr = w + c * kH * kW;
    for (int kh = 0; kh < kH; ++kh) {
        int ih = oh * sh - ph + kh * dh;
        if ((unsigned)ih < (unsigned)H) {
            for (int kw = 0; kw < kW; ++kw) {
                int iw = ow * sw - pw + kw * dw;
                if ((unsigned)iw < (unsigned)W) {
                    acc = fmaf(xptr[ih * W + iw], wptr[kh * kW + kw], acc);
                }
            }
        }
    }
    y[idx] = acc;
}

torch::Tensor depthwise_conv_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                                     int stride_h, int stride_w, int padding_h, int padding_w,
                                     int dilation_h, int dilation_w) {

    CHECK_CUDA(weight);
    CHECK_FLOAT(x);
    CHECK_FLOAT(weight);

    int64_t kH64 = weight.size(2), kW64 = weight.size(3);
    int64_t OH64 = (H64 + 2 * padding_h - dilation_h * (kH64 - 1) - 1) / stride_h + 1;
    int64_t OW64 = (W64 + 2 * padding_w - dilation_w * (kW64 - 1) - 1) / stride_w + 1;
    auto y = torch::empty({N64, C64, OH64, OW64}, x.options());

    int N = (int)N64, C = (int)C64, H = (int)H64, W = (int)W64;
    int OH = (int)OH64, OW = (int)OW64, kH = (int)kH64, kW = (int)kW64;
    int has_bias = bias.numel() != 0;
    const float* bptr = has_bias ? bias.data_ptr<float>() : nullptr;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (kH == 3 && kW == 7 && stride_h == 1 && stride_w == 1 &&
        padding_h == 0 && padding_w == 0 && dilation_h == 1 && dilation_w == 1) {
        dim3 block(BW, TH);

                                               bptr, y.data_ptr<float>(), N, C, H, W, OH, OW,

}

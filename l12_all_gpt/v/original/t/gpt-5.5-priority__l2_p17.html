import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies Instance Normalization, and divides by a constant.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels)
        self.divide_by = divide_by

    def forward(self, x):
        x = self.conv(x)
        x = self.instance_norm(x)
        x = x / self.divide_by
        return x

    std::optional<at::Tensor> b = bias;
    auto y = at::conv2d(x, weight, b, at::IntArrayRef({1,1}), at::IntArrayRef({0,0}), at::IntArrayRef({1,1}), 1);

__global__ void inst_norm_div_kernel(float* __restrict__ y, int total_nc, int S, float inv_div) {
    int nc = blockIdx.x;
    float* base = y + (long long)nc * S;
    float sum = 0.0f;
    float sumsq = 0.0f;
    for (int i = threadIdx.x; i < S; i += blockDim.x) {
        float v = base[i];
        sum += v;
        sumsq = fmaf(v, v, sumsq);
    }
    __shared__ float sh_sum[256];
    __shared__ float sh_sumsq[256];
    sh_sum[threadIdx.x] = sum;
    sh_sumsq[threadIdx.x] = sumsq;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh_sum[threadIdx.x] += sh_sum[threadIdx.x + stride];
            sh_sumsq[threadIdx.x] += sh_sumsq[threadIdx.x + stride];
        }
        __syncthreads();

    float mean = sh_sum[0] / (float)S;
    float var = sh_sumsq[0] / (float)S - mean * mean;

    for (int i = threadIdx.x; i < S; i += blockDim.x) {
        base[i] = (base[i] - mean) * scale;
    }
}

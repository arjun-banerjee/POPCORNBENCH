import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a matrix multiplication, applies minimum, and subtracts a constant.
    """
    def __init__(self, in_features, out_features, constant):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.constant = nn.Parameter(torch.tensor(constant))

    def forward(self, x):
        x = self.linear(x)
        x = torch.min(x, self.constant)
        x = x - self.constant
        return x

__global__ void epilogue_kernel(float* __restrict__ y, const float* __restrict__ bias, const float* __restrict__ cptr, int M, int N) {
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (j < N && row < M) {
        float c = cptr[0];
        float v = y[row * N + j] + bias[j];
        y[row * N + j] = fminf(v, c) - c;
    }
}

torch::Tensor fused_linear_min_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, torch::Tensor constant) {
    int M = (int)x.size(0), K = (int)x.size(1), N = (int)weight.size(0);
    auto y = torch::empty({M, N}, x.options());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    cublasSetStream(handle, at::cuda::getCurrentCUDAStream());
    cublasSetMathMode(handle, CUBLAS_DEFAULT_MATH);
    float alpha = 1.0f, beta = 0.0f;

    // Compute row-major y = x * weight^T using cuBLAS' column-major interface:
    // the same memory is viewed as y^T = weight * x^T.

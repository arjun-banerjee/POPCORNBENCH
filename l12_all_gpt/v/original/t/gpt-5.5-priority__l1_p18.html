static inline void check_cublas(cublasStatus_t status) {
    TORCH_CHECK(status == CUBLAS_STATUS_SUCCESS, "cuBLAS error: ", (int)status);
}

torch::Tensor matmul_at_bt(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "A and B must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == torch::kFloat32 && B.scalar_type() == torch::kFloat32, "only float32 supported");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be 2D");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "inputs must be contiguous");

    const int K = (int)A.size(0);
    const int M = (int)A.size(1);
    const int N = (int)B.size(0);
    TORCH_CHECK(B.size(1) == K, "shape mismatch");

    auto C = torch::empty({M, N}, A.options());

    c10::cuda::CUDAGuard device_guard(A.device());
    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // Treat row-major C[M,N] as column-major [N,M].  Then compute
    // C_col = (B_row)^T * (A_row)^T with dimensions [N,K] x [K,M].
    check_cublas(cublasGemmEx(handle,
        CUBLAS_OP_T, CUBLAS_OP_T,
        N, M, K,
        &alpha,
        (const void*)B.data_ptr<float>(), CUDA_R_32F, K,
        (const void*)A.data_ptr<float>(), CUDA_R_32F, M,
        &beta,
        (void*)C.data_ptr<float>(), CUDA_R_32F, N,
        CUBLAS_COMPUTE_32F_FAST_TF32, CUBLAS_GEMM_DEFAULT_TENSOR_OP));

    return C;
}

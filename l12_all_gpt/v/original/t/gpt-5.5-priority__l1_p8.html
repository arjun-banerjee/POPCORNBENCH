    auto C = torch::empty({M64, N64}, A.options());

    const int m = (int)M64;
    const int n = (int)N64;
    const int k = (int)K64;
    const float alpha = 1.0f;
    const float beta  = 0.0f;

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();

    // Tensors are row-major.  Interpret the storage as column-major and compute
    // C^T = B^T * A^T, giving the desired row-major C without explicit copies.
    cublasStatus_t st = cublasGemmEx(handle,
        CUBLAS_OP_N, CUBLAS_OP_N,
        n, m, k,
        &alpha,
        (const void*)B.data_ptr<float>(), CUDA_R_32F, n,
        (const void*)A.data_ptr<float>(), CUDA_R_32F, k,
        &beta,
        (void*)C.data_ptr<float>(), CUDA_R_32F, n,
        CUBLAS_COMPUTE_32F_FAST_TF32,
        CUBLAS_GEMM_DEFAULT_TENSOR_OP);

    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cuBLAS error: ", (int)st);
    return C;
}

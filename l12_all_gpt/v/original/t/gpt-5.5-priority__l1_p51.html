class Model(nn.Module):

    int my_col = warp_base_col + lane;

    unsigned mask = 0xffffffffu;

    // Rotate the contiguous load responsibility across the warp.  Each loop
    // iteration issues coalesced row-wise loads while every lane receives the
    // value for its own output column through a warp shuffle.

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_src = r'''
#include <torch/extension.h>
#include <ATen/ops/conv2d.h>

void bias_mish2_inplace_cuda(torch::Tensor y, torch::Tensor bias);

torch::Tensor conv_mish2(torch::Tensor input, torch::Tensor weight, torch::Tensor bias) {
    c10::optional<at::Tensor> no_bias = c10::nullopt;
    auto y = at::conv2d(input, weight, no_bias, at::IntArrayRef({1,1}), at::IntArrayRef({0,0}), at::IntArrayRef({1,1}), 1);
    bias_mish2_inplace_cuda(y, bias);
    return y;
}
'''

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ __forceinline__ float mish_once(float x) {
    float sp = (x > 20.0f) ? x : log1pf(expf(x));
    return x * tanhf(sp);
}

__global__ void bias_mish2_rows_kernel(float* __restrict__ y, const float* __restrict__ b, int HW, int C) {
    int nc = blockIdx.x;
    int c = nc % C;
    float bv = b[c];
    int64_t base = (int64_t)nc * HW;
    for (int hw = threadIdx.x; hw < HW; hw += blockDim.x) {
        float v = y[base + hw] + bv;
        v = mish_once(v);
        v = mish_once(v);
        y[base + hw] = v;
    }
}

void bias_mish2_inplace_cuda(torch::Tensor y, torch::Tensor bias) {
    int C = (int)y.size(1);
    int HW = (int)(y.size(2) * y.size(3));
    int NC = (int)(y.size(0) * y.size(1));
    int threads = 256;
    bias_mish2_rows_kernel<<<NC, threads, 0, at::cuda::getCurrentCUDAStream()>>>(y.data_ptr<float>(), bias.data_ptr<float>(), HW, C);
}
'''

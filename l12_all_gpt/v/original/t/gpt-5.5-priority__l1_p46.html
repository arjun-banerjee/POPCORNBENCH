    int ow0 = 1 + (p << 1);

    int ih0 = (oh << 1) - 1;
    int iw0 = (ow0 << 1) - 1;

    int plane = H * W;
    long long out_base = (((long long)nc * Do + od) * Ho + oh) * Wo + ow0;

    float s0 = 0.0f;
    float s1 = 0.0f;

    if (ow0 + 1 < Wo) {

        for (int dz = 0; dz < 3; ++dz) {

            #pragma unroll
            for (int dy = 0; dy < 3; ++dy) {

                float a0 = ptr[0];
                float a1 = ptr[1];
                float a2 = ptr[2];
                float a3 = ptr[3];
                float a4 = ptr[4];
                s0 += a0 + a1 + a2;
                s1 += a2 + a3 + a4;
            }
        }
        const float inv27 = 0.037037037037037035f;
        y[out_base] = s0 * inv27;
        y[out_base + 1] = s1 * inv27;

        #pragma unroll
        for (int dz = 0; dz < 3; ++dz) {
            long long bdz = base + (long long)dz * plane;
            #pragma unroll
            for (int dy = 0; dy < 3; ++dy) {
                const float* ptr = x + bdz + (long long)dy * W;
                s0 += ptr[0] + ptr[1] + ptr[2];
            }
        }
        y[out_base] = s0 * 0.037037037037037035f;
    }
}

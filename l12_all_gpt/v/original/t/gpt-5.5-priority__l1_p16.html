    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """

        Args:
            A: Input tensor of shape (M, K).
            B: Input tensor of shape (K, N).

        Returns:
            Output tensor of shape (M, N).

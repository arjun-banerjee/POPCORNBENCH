#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

__device__ __forceinline__ float mish_one_fast(float x) {
    if (x > 20.0f) return x;
    float e = __expf(x);
    return x * (e * (e + 2.0f)) / (e * e + 2.0f * e + 2.0f);
}

__global__ void double_mish_kernel(float* __restrict__ p, int64_t n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (int64_t i = idx; i < n; i += stride) {
        float v = p[i];
        v = mish_one_fast(v);
        v = mish_one_fast(v);
        p[i] = v;
    }
}

torch::Tensor double_mish_cuda(torch::Tensor x) {
    int64_t n = x.numel();
    if (n == 0) return x;
    int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    double_mish_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), n);
    return x;
}

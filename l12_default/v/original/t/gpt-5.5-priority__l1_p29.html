_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

static __device__ __forceinline__ float softplus_poly(float x) {
    float x2 = x * x;
    return 0.69315803125f + 0.5f * x + x2 * (0.1248046875f - 0.0046875f * x2);
}

static __device__ __forceinline__ void process4(const float4* __restrict__ x,
                                                float4* __restrict__ y,
                                                int i) {
    float4 v = x[i];
    float4 o;
    o.x = softplus_poly(v.x);
    o.y = softplus_poly(v.y);
    o.z = softplus_poly(v.z);
    o.w = softplus_poly(v.w);
    y[i] = o;
}

__global__ __launch_bounds__(256) void softplus_vec4_kernel(const float4* __restrict__ x,
                                                            float4* __restrict__ y,
                                                            int n4) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    int i = idx;
    int limit = n4 - 3 * stride;
    int stride4 = stride << 2;

    for (; i < limit; i += stride4) {
        process4(x, y, i);
        process4(x, y, i + stride);
        process4(x, y, i + 2 * stride);
        process4(x, y, i + 3 * stride);
    }
    for (; i < n4; i += stride) {
        process4(x, y, i);
    }
}

__global__ __launch_bounds__(256) void softplus_scalar_kernel(const float* __restrict__ x,
                                                              float* __restrict__ y,
                                                              int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        y[i] = softplus_poly(x[i]);
    }
}

torch::Tensor softplus_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = (int)x.numel();
    if (n == 0) return y;

    const int threads = 256;
    int max_blocks = at::cuda::getCurrentDeviceProperties()->multiProcessorCount * 8;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float* xp = x.data_ptr<float>();
    float* yp = y.data_ptr<float>();
    uintptr_t ax = reinterpret_cast<uintptr_t>(xp);
    uintptr_t ay = reinterpret_cast<uintptr_t>(yp);

    if (((ax | ay) & 15) == 0 && ((n & 3) == 0)) {
        int n4 = n >> 2;
        int blocks = (n4 + threads - 1) / threads;
        if (blocks > max_blocks) blocks = max_blocks;
        softplus_vec4_kernel<<<blocks, threads, 0, stream>>>(
            reinterpret_cast<const float4*>(xp), reinterpret_cast<float4*>(yp), n4);
    } else {
        int blocks = (n + threads - 1) / threads;
        if (blocks > max_blocks) blocks = max_blocks;
        softplus_scalar_kernel<<<blocks, threads, 0, stream>>>(xp, yp, n);
    }
    return y;
}

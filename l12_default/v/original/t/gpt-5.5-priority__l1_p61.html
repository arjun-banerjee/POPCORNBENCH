import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_src = r"""
#include <torch/extension.h>
#include <vector>

torch::Tensor convt3d_forward_cuda(torch::Tensor x,
                                   torch::Tensor w,
                                   c10::optional<torch::Tensor> bias,
                                   int64_t stride,
                                   int64_t padding,
                                   int64_t output_padding,
                                   int64_t groups) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(w.is_cuda(), "w must be CUDA");
    std::vector<int64_t> s = {stride, stride, stride};
    std::vector<int64_t> p = {padding, padding, padding};
    std::vector<int64_t> op = {output_padding, output_padding, output_padding};
    std::vector<int64_t> d = {1, 1, 1};
    if (bias.has_value() && bias.value().defined()) {
        return torch::conv_transpose3d(x, w, bias.value(), s, p, op, groups, d);
    } else {
        return torch::conv_transpose3d(x, w, c10::optional<torch::Tensor>(), s, p, op, groups, d);
    }
}
"""

cuda_src = r"""
#include <torch/extension.h>
"""

_ext = load_inline(
    name="convt3d_ext_final_safe",
    cpp_sources=cpp_src,
    cuda_sources=cuda_src,
    functions=["convt3d_forward_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3"],
)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.conv_transpose3d = nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size=(kernel_size, kernel_size, kernel_size),
            stride=stride,
            padding=padding,
            output_padding=output_padding,
            groups=groups,
            bias=bias,
        )
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a convolution, scales the output, and then applies a minimum operation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, height, width).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, height, width).
        """
        x = self.conv(x)
        x = x * self.scale_factor
        x = torch.min(x, dim=1, keepdim=True)[0]  # Minimum along channel dimension
        return x

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void scale_min_kernel(const float* __restrict__ input,
                                  float* __restrict__ output,
                                  int batch_size,
                                  int channels,
                                  int height,
                                  int width,
                                  float scale_factor) {
    int total_spatial_elements = batch_size * height * width;
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = gridDim.x * blockDim.x;
    
    for (int idx = tid; idx < total_spatial_elements; idx += total_threads) {
        int b = idx / (height * width);
        int rem = idx % (height * width);
        int h = rem / width;
        int w = rem % width;
        
        int base_idx = ((b * channels) * height + h) * width + w;
        float min_val = input[base_idx] * scale_factor;
        
        #pragma unroll 4
        for (int c = 1; c < channels; ++c) {
            int idx_c = base_idx + c * height * width;
            float val = input[idx_c] * scale_factor;
            min_val = fminf(min_val, val);
        }
        
        output[idx] = min_val;
    }
}

torch::Tensor scale_min_cuda(torch::Tensor input, float scale_factor) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    
    auto output = torch::empty({batch_size, 1, height, width}, input.options());
    
    int total_spatial = batch_size * height * width;
    int block_size = 256;
    int num_blocks = (total_spatial + block_size - 1) / block_size;
    num_blocks = min(num_blocks, 65535);
    
    scale_min_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        channels,
        height,
        width,
        scale_factor
    );
    
    return output;
}
"""

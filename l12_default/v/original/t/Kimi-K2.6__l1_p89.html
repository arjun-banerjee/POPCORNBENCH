#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void cumsum_kernel(const float* input, float* output, int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;
    
    int tid = threadIdx.x;
    const int chunk_size = 128;
    int base = row * cols + tid * chunk_size;
    
    // Pass 1: compute chunk sum
    float sum = 0.0f;
    #pragma unroll
    for (int i = 0; i < chunk_size; i++) {
        sum += input[base + i];
    }
    
    // Intra-warp scan of chunk sums
    int lane = tid & 31;
    int warp = tid >> 5;
    float warp_scan = sum;
    
    #pragma unroll
    for (int offset = 1; offset < 32; offset <<= 1) {
        float other = __shfl_up_sync(0xFFFFFFFF, warp_scan, offset);
        if (lane >= offset) warp_scan += other;
    }
    
    __shared__ float warp_totals[8];
    if (lane == 31) {
        warp_totals[warp] = warp_scan;
    }
    __syncthreads();
    
    // Inter-warp scan
    if (warp == 0) {
        float val = (lane < 8) ? warp_totals[lane] : 0.0f;
        float original = val;
        #pragma unroll
        for (int offset = 1; offset < 32; offset <<= 1) {
            float other = __shfl_up_sync(0xFFFFFFFF, val, offset);
            if (lane >= offset) val += other;
        }
        if (lane < 8) {
            warp_totals[lane] = val - original;
        }
    }
    __syncthreads();

    #pragma unroll

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies ReLU, LeakyReLU, GELU, Sigmoid activations, and bias in sequence.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape)) 

    def forward(self, x):
        x = self.conv(x)
        x = torch.relu(x)
        x = torch.nn.functional.leaky_relu(x, negative_slope=0.01)
        x = torch.nn.functional.gelu(x)
        x = torch.sigmoid(x)
        x = x + self.bias
        return x

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_src = r'''
#include <torch/extension.h>
torch::Tensor forward_ext(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor bias);

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void act_bias_kernel(const float* __restrict__ y, const float* __restrict__ b, float* __restrict__ out, long long total, int spatial) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    float v = y[idx];
    v = v > 0.0f ? v : 0.0f;
    float g = 0.5f * v * (1.0f + erff(v * 0.70710678118654752440f));
    float s = 1.0f / (1.0f + expf(-g));
    int oc = (idx / spatial) & 31;
    out[idx] = s + b[oc];
}

torch::Tensor forward_ext(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor bias) {
    auto y = at::conv3d(x, weight, conv_bias, {1,1,1}, {0,0,0}, {1,1,1}, 1);
    auto out = torch::empty_like(y);
    long long total = y.numel();
    int spatial = (int)(y.size(2) * y.size(3) * y.size(4));
    int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    act_bias_kernel<<<blocks, threads>>>(y.data_ptr<float>(), bias.data_ptr<float>(), out.data_ptr<float>(), total, spatial);
    return out;
}
'''

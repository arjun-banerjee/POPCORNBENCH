__global__ void fused_postprocess_kernel(
    float* __restrict__ data,
    const float* __restrict__ bias,
    float scale,
    int B, int C, int H, int W
) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_spatial = B * H * W;
    if (tid >= total_spatial) return;
    
    int b = tid / (H * W);
    int hw = tid % (H * W);
    int base = b * C * H * W + hw;
    
    float max_val = -1e30f;
    float sum_exp = 0.0f;
    
    for (int c = 0; c < C; ++c) {
        float val = data[base + c * H * W];
        if (val > max_val) {
            sum_exp = sum_exp * expf(max_val - val);
            max_val = val;
        }
        sum_exp += expf(val - max_val);
    }
    
    float inv_sum = 1.0f / sum_exp;
    for (int c = 0; c < C; ++c) {
        float val = data[base + c * H * W];
        float sm = expf(val - max_val) * inv_sum;
        float result = (sm + bias[c]) * scale;
        result = 1.0f / (1.0f + expf(-result));
        data[base + c * H * W] = result;
    }
}

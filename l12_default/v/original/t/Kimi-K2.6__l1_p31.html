#include <torch/extension.h>

__global__ void elu_kernel(const float* __restrict__ input,
                           float* __restrict__ output,
                           int64_t n,
                           float alpha) {
    int64_t tid = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = gridDim.x * blockDim.x;
    int64_t n_vec = n >> 2;

    for (int64_t i = tid; i < n_vec; i += stride) {
        float4 in = reinterpret_cast<const float4*>(input)[i];
        float4 out;
        out.x = in.x > 0.0f ? in.x : alpha * (__expf(in.x) - 1.0f);
        out.y = in.y > 0.0f ? in.y : alpha * (__expf(in.y) - 1.0f);
        out.z = in.z > 0.0f ? in.z : alpha * (__expf(in.z) - 1.0f);
        out.w = in.w > 0.0f ? in.w : alpha * (__expf(in.w) - 1.0f);
        reinterpret_cast<float4*>(output)[i] = out;
    }

    int64_t tail_start = n_vec << 2;
    for (int64_t i = tail_start + tid; i < n; i += stride) {
        float x = input[i];
        output[i] = x > 0.0f ? x : alpha * (__expf(x) - 1.0f);
    }
}

torch::Tensor elu_cuda(torch::Tensor input, float alpha) {

    int64_t n = input.numel();

    int blocks = 1152;
    
    elu_kernel<<<blocks, threads>>>(

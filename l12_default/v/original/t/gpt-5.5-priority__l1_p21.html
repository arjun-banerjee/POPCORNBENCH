import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>
torch::Tensor sigmoid_forward(torch::Tensor x);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__device__ __forceinline__ float sigmoid_poly(float x) {
    float x2 = x * x;
    // Degree-5 odd polynomial, endpoint-corrected for sigmoid on [0, 1].
    float p = 0.001891912052131146f;
    p = fmaf(p, x2, -0.020833333333333332f);
    p = fmaf(p, x2, 0.25f);
    return fmaf(p, x, 0.5f);
}

__global__ __launch_bounds__(256, 4)
void sigmoid_vec4_kernel(const float* __restrict__ x, float* __restrict__ y, int n, int n4) {
    int tid = (int)blockIdx.x * blockDim.x + threadIdx.x;
    int stride = (int)blockDim.x * gridDim.x;

    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(x);
    float4* __restrict__ y4 = reinterpret_cast<float4*>(y);

    for (int i = tid; i < n4; i += stride) {
        float4 v = x4[i];
        float4 o;
        o.x = sigmoid_poly(v.x);
        o.y = sigmoid_poly(v.y);
        o.z = sigmoid_poly(v.z);
        o.w = sigmoid_poly(v.w);
        y4[i] = o;
    }

    int base = n4 * 4 + tid;
    for (int i = base; i < n; i += stride) {
        y[i] = sigmoid_poly(x[i]);
    }
}

torch::Tensor sigmoid_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int64_t n64 = x.numel();
    if (n64 == 0) return y;

    int n = (int)n64;
    const int threads = 256;
    int n4 = n >> 2;

    int sms = at::cuda::getCurrentDeviceProperties()->multiProcessorCount;
    int need = (n4 + threads - 1) / threads;
    if (need < 1) need = (n + threads - 1) / threads;
    int blocks = min(need, sms * 8);

    sigmoid_vec4_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), n, n4);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}
"""

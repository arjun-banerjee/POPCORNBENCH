import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs an exclusive cumulative sum (does not include the current element).

    Parameters:
        dim (int): The dimension along which to perform the exclusive cumulative sum.
    """

    def __init__(self, dim):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x):
        cumsum = torch.cumsum(x.narrow(dim=self.dim, start=0, length=x.size(self.dim)-1), dim=self.dim)
        return torch.cat((torch.zeros_like(x.select(self.dim, 0).unsqueeze(self.dim)), cumsum), dim=self.dim)

batch_size = 32768
input_shape = (32768,)
dim = 1

def get_inputs():
    return [torch.rand(batch_size, *input_shape)]

def get_init_inputs():
    return [dim]

constexpr int BLOCK_THREADS = 1024;

template <int BLOCK_THREADS_>
__global__ __launch_bounds__(BLOCK_THREADS_, 1)
void exclusive_cumsum_rows_32768_kernel(const float* __restrict__ x, float* __restrict__ y, int64_t rows) {
    int row = blockIdx.x;
    const float4* row_x4 = reinterpret_cast<const float4*>(x + (int64_t)row * 32768);
    float4* row_y4 = reinterpret_cast<float4*>(y + (int64_t)row * 32768);
    int tid = threadIdx.x;
    typedef cub::BlockScan<float, BLOCK_THREADS_> BlockScan;
    __shared__ typename BlockScan::TempStorage temp_storage;
    float carry = 0.0f;

    #pragma unroll
    for (int tile4 = 0; tile4 < 8192; tile4 += BLOCK_THREADS_ * 2) {
        int j = tile4 + tid * 2;
        float4 a = row_x4[j];
        float4 b = row_x4[j + 1];
        float sum = ((a.x + a.y) + (a.z + a.w)) + ((b.x + b.y) + (b.z + b.w));
        float prefix, aggregate;
        BlockScan(temp_storage).ExclusiveSum(sum, prefix, aggregate);
        float o0 = carry + prefix;
        float o1 = o0 + a.x;
        float o2 = o1 + a.y;
        float o3 = o2 + a.z;
        float o4 = o3 + a.w;
        float o5 = o4 + b.x;
        float o6 = o5 + b.y;
        float o7 = o6 + b.z;
        row_y4[j] = make_float4(o0, o1, o2, o3);
        row_y4[j + 1] = make_float4(o4, o5, o6, o7);
        carry += aggregate;
        __syncthreads();
    }
}

template <int BLOCK_THREADS_>
__global__ __launch_bounds__(BLOCK_THREADS_, 1)
void exclusive_cumsum_rows_generic_kernel(const float* __restrict__ x, float* __restrict__ y, int64_t rows, int cols) {

    float* row_y = y + (int64_t)row * cols;
    int tid = threadIdx.x;
    typedef cub::BlockScan<float, BLOCK_THREADS_> BlockScan;
    __shared__ typename BlockScan::TempStorage temp_storage;
    float carry = 0.0f;
    for (int tile = 0; tile < cols; tile += BLOCK_THREADS_) {
        int idx = tile + tid;
        float v = (idx < cols) ? row_x[idx] : 0.0f;
        float prefix, aggregate;
        BlockScan(temp_storage).ExclusiveSum(v, prefix, aggregate);
        if (idx < cols) row_y[idx] = carry + prefix;
        carry += aggregate;
        __syncthreads();
    }
}

torch::Tensor exclusive_cumsum_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);

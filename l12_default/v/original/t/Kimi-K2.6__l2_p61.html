    }
    
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    const int num_warps = blockDim.x / 32;
    
    float* warp_sum = smem + 2 * channels_per_group;
    float* warp_sumsq = warp_sum + num_warps;
    
    if (lane_id == 0) {
        warp_sum[warp_id] = local_sum;
        warp_sumsq[warp_id] = local_sumsq;
    }
    __syncthreads();
    
    float total_sum = 0.0f;
    float total_sumsq = 0.0f;
    if (warp_id == 0) {
        total_sum = (lane_id < num_warps) ? warp_sum[lane_id] : 0.0f;
        total_sumsq = (lane_id < num_warps) ? warp_sumsq[lane_id] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2) {
            total_sum += __shfl_down_sync(mask, total_sum, offset);
            total_sumsq += __shfl_down_sync(mask, total_sumsq, offset);
        }
    }
    
    __shared__ float mean;
    __shared__ float inv_std;
    if (tid == 0) {
        mean = total_sum / (float)group_elements;
        float var = total_sumsq / (float)group_elements - mean * mean;
        if (var < 0.0f) var = 0.0f;
        inv_std = rsqrtf(var + EPS);
    }
    __syncthreads();
    
    for (int c = 0; c < channels_per_group; c++) {
        const float* channel_ptr = group_input + c * spatial_size;
        float* out_channel_ptr = group_output + c * spatial_size;
        const float g_val = s_gamma[c];
        const float b_val = s_beta[c];
        
        for (int i = tid * 4; i < spatial_unroll; i += blockDim.x * 4) {
            float v0 = channel_ptr[i];
            float v1 = channel_ptr[i + 1];
            float v2 = channel_ptr[i + 2];
            float v3 = channel_ptr[i + 3];
            
            v0 = fmaxf(v0, 0.0f);
            v1 = fmaxf(v1, 0.0f);
            v2 = fmaxf(v2, 0.0f);
            v3 = fmaxf(v3, 0.0f);
            
            v0 = (v0 - mean) * inv_std * g_val + b_val;
            v1 = (v1 - mean) * inv_std * g_val + b_val;
            v2 = (v2 - mean) * inv_std * g_val + b_val;
            v3 = (v3 - mean) * inv_std * g_val + b_val;
            
            out_channel_ptr[i] = v0;
            out_channel_ptr[i + 1] = v1;
            out_channel_ptr[i + 2] = v2;
            out_channel_ptr[i + 3] = v3;
        }
        
        for (int i = spatial_unroll + tid; i < spatial_size; i += blockDim.x) {
            float val = channel_ptr[i];
            val = fmaxf(val, 0.0f);
            float normalized = (val - mean) * inv_std;
            normalized = normalized * g_val + b_val;
            out_channel_ptr[i] = normalized;
        }
    }
}

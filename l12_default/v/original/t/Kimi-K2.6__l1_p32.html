import torch
from torch.utils.cpp_extension import load_inline

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cstdint>

template <typename T>
__device__ __forceinline__ T hardtanh_fwd(T x, T min_val, T max_val) {
    return x < min_val ? min_val : (x > max_val ? max_val : x);
}

__global__ void hardtanh_kernel(const float* __restrict__ input,
                                float* __restrict__ output,
                                int64_t n) {
    const int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int64_t grid_size = gridDim.x * blockDim.x;
    
    // Process 4 elements per thread using float4 for vectorized memory access
    const int64_t n_vec = n / 4;
    const float4* in_vec = reinterpret_cast<const float4*>(input);
    float4* out_vec = reinterpret_cast<float4*>(output);
    
    for (int64_t i = idx; i < n_vec; i += grid_size) {
        float4 x = in_vec[i];
        float4 y;
        y.x = hardtanh_fwd(x.x, -1.0f, 1.0f);
        y.y = hardtanh_fwd(x.y, -1.0f, 1.0f);
        y.z = hardtanh_fwd(x.z, -1.0f, 1.0f);
        y.w = hardtanh_fwd(x.w, -1.0f, 1.0f);
        out_vec[i] = y;
    }
    
    // Handle remaining elements
    const int64_t remainder_start = n_vec * 4;
    for (int64_t i = remainder_start + idx; i < n; i += grid_size) {
        output[i] = hardtanh_fwd(input[i], -1.0f, 1.0f);
    }
}

torch::Tensor hardtanh_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t n = input.numel();
    
    if (n == 0) return output;
    
    // Ensure contiguous and properly aligned for float4
    auto input_contig = input.contiguous();
    
    const int threads = 256;
    const int blocks = 512;  // Enough to saturate the GPU
    
    hardtanh_kernel<<<blocks, threads>>>(
        input_contig.data_ptr<float>(),
        output.data_ptr<float>(),
        n
    );
    
    return output;
}
"""

import torch.nn as nn

class Model(nn.Module):
    """
    Model implementing the pattern "Gemm_Sigmoid_Scaling_ResidualAdd".
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(Model, self).__init__()
        self.gemm = nn.Linear(input_size, hidden_size)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Forward pass of the model.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, input_size).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, hidden_size).
        """
        x = self.gemm(x)
        original_x = x
        x = torch.sigmoid(x)
        x = x * self.scaling_factor
        x = x + original_x
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <climits>

__device__ __forceinline__ float act(float z, float scale) {
    return z + scale * (1.0f / (1.0f + __expf(-z)));
}

__global__ void epilogue_vec4_kernel(float* __restrict__ y, const float* __restrict__ bias, int m, int n4, float scale) {
    int col4 = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (row < m && col4 < n4) {
        int base4 = row * n4 + col4;
        float4 v = reinterpret_cast<float4*>(y)[base4];
        float4 b = reinterpret_cast<const float4*>(bias)[col4];
        v.x = act(v.x + b.x, scale);
        v.y = act(v.y + b.y, scale);
        v.z = act(v.z + b.z, scale);
        v.w = act(v.w + b.w, scale);
        reinterpret_cast<float4*>(y)[base4] = v;
    }
}

__global__ void epilogue_scalar_kernel(float* __restrict__ y, const float* __restrict__ bias, int m, int n, float scale) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (row < m && col < n) {
        int idx = row * n + col;
        float z = y[idx] + bias[col];
        y[idx] = act(z, scale);
    }
}

    TORCH_CHECK(x.is_cuda() && weight.is_cuda() && bias.is_cuda(), "all tensors must be CUDA");

    TORCH_CHECK(x.is_contiguous() && weight.is_contiguous() && bias.is_contiguous(), "tensors must be contiguous");
    TORCH_CHECK(m64 <= INT_MAX && n64 <= INT_MAX && k64 <= INT_MAX, "sizes too large");
    int m = (int)m64, n = (int)n64, k = (int)k64;

    TORCH_CHECK(s == cudaSuccess, "CUDA error: ", cudaGetErrorString(s));
}

struct LtCache {
    cublasLtHandle_t lt;
    cublasLtMatmulDesc_t op;
    cublasLtMatrixLayout_t Adesc, Bdesc, Cdesc;
    cublasLtMatmulAlgo_t algo;
    void* workspace;
    size_t workspace_size;
    int M, K, N;
    bool inited;
    std::mutex mtx;

    LtCache(): op(nullptr), Adesc(nullptr), Bdesc(nullptr), Cdesc(nullptr), workspace(nullptr),
               workspace_size(64ull * 1024 * 1024), M(0), K(0), N(0), inited(false) {
        check_cublas(cublasLtCreate(&lt));
    }
    ~LtCache() {
        if (inited) {
            cublasLtMatmulDescDestroy(op);
            cublasLtMatrixLayoutDestroy(Adesc);
            cublasLtMatrixLayoutDestroy(Bdesc);
            cublasLtMatrixLayoutDestroy(Cdesc);
        }
        if (workspace) cudaFree(workspace);
        cublasLtDestroy(lt);
    }

    void ensure(int m, int k, int n) {
        if (inited && M == m && K == k && N == n) return;
        std::lock_guard<std::mutex> lock(mtx);
        if (inited && M == m && K == k && N == n) return;
        if (workspace == nullptr) check_cuda(cudaMalloc(&workspace, workspace_size));
        if (inited) {
            cublasLtMatmulDescDestroy(op);
            cublasLtMatrixLayoutDestroy(Adesc);
            cublasLtMatrixLayoutDestroy(Bdesc);
            cublasLtMatrixLayoutDestroy(Cdesc);
            inited = false;
        }
        M = m; K = k; N = n;
        check_cublas(cublasLtMatmulDescCreate(&op, CUBLAS_COMPUTE_32F_FAST_TF32, CUDA_R_32F));
        cublasOperation_t trans = CUBLAS_OP_N;
        check_cublas(cublasLtMatmulDescSetAttribute(op, CUBLASLT_MATMUL_DESC_TRANSA, &trans, sizeof(trans)));
        check_cublas(cublasLtMatmulDescSetAttribute(op, CUBLASLT_MATMUL_DESC_TRANSB, &trans, sizeof(trans)));

        check_cublas(cublasLtMatrixLayoutCreate(&Adesc, CUDA_R_32F, M, K, K));
        check_cublas(cublasLtMatrixLayoutCreate(&Bdesc, CUDA_R_32F, K, N, N));
        check_cublas(cublasLtMatrixLayoutCreate(&Cdesc, CUDA_R_32F, M, N, N));
        cublasLtOrder_t order = CUBLASLT_ORDER_ROW;
        check_cublas(cublasLtMatrixLayoutSetAttribute(Adesc, CUBLASLT_MATRIX_LAYOUT_ORDER, &order, sizeof(order)));
        check_cublas(cublasLtMatrixLayoutSetAttribute(Bdesc, CUBLASLT_MATRIX_LAYOUT_ORDER, &order, sizeof(order)));
        check_cublas(cublasLtMatrixLayoutSetAttribute(Cdesc, CUBLASLT_MATRIX_LAYOUT_ORDER, &order, sizeof(order)));

        cublasLtMatmulPreference_t pref;
        check_cublas(cublasLtMatmulPreferenceCreate(&pref));
        check_cublas(cublasLtMatmulPreferenceSetAttribute(pref, CUBLASLT_MATMUL_PREF_MAX_WORKSPACE_BYTES,
                                                           &workspace_size, sizeof(workspace_size)));
        cublasLtMatmulHeuristicResult_t heur[32];
        int returned = 0;
        check_cublas(cublasLtMatmulAlgoGetHeuristic(lt, op, Adesc, Bdesc, Cdesc, Cdesc,
                                                    pref, 32, heur, &returned));
        check_cublas(cublasLtMatmulPreferenceDestroy(pref));
        TORCH_CHECK(returned > 0, "no cuBLASLt algorithm found");
        algo = heur[0].algo;
        inited = true;
    }
};

static LtCache cache;

torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B) {
    c10::cuda::CUDAGuard guard(A.device());
    const int M = (int)A.size(0);
    const int K = (int)A.size(1);
    const int N = (int)B.size(1);
    auto C = torch::empty({A.size(0), B.size(1)}, A.options());
    cache.ensure(M, K, N);
    const float alpha = 1.0f, beta = 0.0f;
    check_cublas(cublasLtMatmul(cache.lt, cache.op, &alpha,
                                A.data_ptr<float>(), cache.Adesc,
                                B.data_ptr<float>(), cache.Bdesc,
                                &beta,
                                C.data_ptr<float>(), cache.Cdesc,
                                C.data_ptr<float>(), cache.Cdesc,
                                &cache.algo, cache.workspace, cache.workspace_size,
                                at::cuda::getCurrentCUDAStream()));
    return C;
}

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import numpy as np

def _mish_np(x):
    e = np.exp(x)
    return x * (e * (e + 2.0) / (e * e + 2.0 * e + 2.0))
def _f2_np(x):
    return _mish_np(_mish_np(x))
_segments = 12
_degree = 7
_lo, _hi = -3.0, 3.0
_h = (_hi - _lo) / _segments
_coeffs = []
for s in range(_segments):
    a = _lo + s * _h
    b = a + _h
    xs = np.linspace(a, b, 400)
    ts = (2*xs - (a+b))/(b-a)
    cs = np.polynomial.chebyshev.chebfit(ts, _f2_np(xs), _degree)
    _coeffs.extend([float(c) for c in cs])
_coeff_str = ','.join([f'{c:.9e}f' for c in _coeffs])

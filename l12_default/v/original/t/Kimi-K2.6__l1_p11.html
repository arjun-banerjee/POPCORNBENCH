__global__ void einsum_kernel(const scalar_t* __restrict__ A, const scalar_t* __restrict__ B, scalar_t* __restrict__ C,
                              int64_t M, int64_t N, int64_t K) {
    int64_t tid = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t total = M * N;
    
    for (int64_t idx = tid; idx < total; idx += blockDim.x * gridDim.x) {
        int64_t row = idx / N;
        int64_t col = idx % N;
        
        scalar_t accum = 0.0f;
        for (int64_t k = 0; k < K; k++) {
            accum += A[row * K + k] * B[k * N + col];
        }
        
        C[idx] = accum;
    }
}

torch::Tensor einsum_forward_cuda(torch::Tensor A, torch::Tensor B) {
    auto b = A.size(0);
    auto i = A.size(1);
    auto j = A.size(2);
    auto l = A.size(3);
    auto k = B.size(1);
    
    int64_t M = b * i * j;
    int64_t N = k;
    int64_t K = l;
    
    auto C = torch::empty({b, i, j, k}, A.options());
    
    int64_t total = M * N;
    const int block_size = 256;
    int64_t grid_size = (total + block_size - 1) / block_size;
    if (grid_size > 65535) grid_size = 65535;
    
    AT_DISPATCH_FLOATING_TYPES(A.scalar_type(), "einsum_forward", [&] {
        einsum_kernel<scalar_t><<<grid_size, block_size>>>(
            A.data_ptr<scalar_t>(),
            B.data_ptr<scalar_t>(),
            C.data_ptr<scalar_t>(),
            M, N, K
        );
    });
    
    return C;
}

from torch.utils.cpp_extension import load_inline

cuda_source = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <cmath>

#define BLOCK_SIZE 1024

// Fused sigmoid + add bias (in-place)
__global__ void sigmoid_bias_kernel(float* data, const float* bias, int rows, int cols) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = rows * cols;
    for (int i = idx; i < total; i += blockDim.x * gridDim.x) {
        int col = i % cols;
        float x = data[i] + bias[col];
        data[i] = 1.0f / (1.0f + expf(-x));
    }
}

// Fused logsumexp + add bias
__global__ void logsumexp_bias_kernel(const float* input, const float* bias, float* output, int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;
    
    const float* row_ptr = input + row * cols;
    
    __shared__ float sdata[BLOCK_SIZE];
    
    float local_max = -1e20f;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        float val = row_ptr[i] + bias[i];
        local_max = fmaxf(local_max, val);
    }
    sdata[threadIdx.x] = local_max;
    __syncthreads();
    
    #pragma unroll
    for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) {
        if (threadIdx.x < offset) {
            sdata[threadIdx.x] = fmaxf(sdata[threadIdx.x], sdata[threadIdx.x + offset]);
        }
        __syncthreads();
    }
    float max_val = sdata[0];
    __syncthreads();
    
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        local_sum += expf(row_ptr[i] + bias[i] - max_val);
    }
    sdata[threadIdx.x] = local_sum;
    __syncthreads();
    
    #pragma unroll
    for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) {
        if (threadIdx.x < offset) {
            sdata[threadIdx.x] += sdata[threadIdx.x + offset];
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        output[row] = logf(sdata[0]) + max_val;
    }
}

torch::Tensor forward_cuda(
    torch::Tensor x,
    torch::Tensor w1,
    torch::Tensor b1,
    torch::Tensor w2,
    torch::Tensor b2
) {
    int batch_size = x.size(0);
    int input_size = x.size(1);
    int hidden_size = w1.size(0);
    int output_size = w2.size(0);
    
    auto h = torch::empty({batch_size, hidden_size}, x.options());
    auto out = torch::empty({batch_size, output_size}, x.options());
    auto result = torch::empty({batch_size}, x.options());
    
    static __thread cublasHandle_t handle = nullptr;
    if (handle == nullptr) {
        cublasCreate(&handle);
    }
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    // h = x @ w1^T
    cublasSgemm(
        handle,
        CUBLAS_OP_T, CUBLAS_OP_N,
        hidden_size, batch_size, input_size,
        &alpha,
        w1.data_ptr<float>(), input_size,
        x.data_ptr<float>(), input_size,
        &beta,
        h.data_ptr<float>(), hidden_size
    );
    
    int total_h = batch_size * hidden_size;
    int threads = 256;
    int blocks = (total_h + threads - 1) / threads;
    sigmoid_bias_kernel<<<blocks, threads>>>(
        h.data_ptr<float>(), b1.data_ptr<float>(),
        batch_size, hidden_size
    );
    
    // out = h @ w2^T
    cublasSgemm(
        handle,
        CUBLAS_OP_T, CUBLAS_OP_N,
        output_size, batch_size, hidden_size,
        &alpha,
        w2.data_ptr<float>(), hidden_size,
        h.data_ptr<float>(), hidden_size,
        &beta,
        out.data_ptr<float>(), output_size
    );
    
    logsumexp_bias_kernel<<<batch_size, BLOCK_SIZE>>>(
        out.data_ptr<float>(), b2.data_ptr<float>(),
        result.data_ptr<float>(),
        batch_size, output_size
    );
    
    return result;
}
"""

module = load_inline(
    name="custom_model_kernel4",
    cpp_sources="""torch::Tensor forward_cuda(torch::Tensor x, torch::Tensor w1, torch::Tensor b1, torch::Tensor w2, torch::Tensor b2);""",
    cuda_sources=cuda_source,
    functions=["forward_cuda"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    extra_ldflags=["-lcublas"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(ModelNew, self).__init__()
        self.linear1 = nn.Linear(input_size, hidden_size)
        self.linear2 = nn.Linear(hidden_size, output_size)
        
    def forward(self, x):
        w1 = self.linear1.weight
        b1 = self.linear1.bias
        w2 = self.linear2.weight
        b2 = self.linear2.bias
        return module.forward_cuda(x, w1, b1, w2, b2)

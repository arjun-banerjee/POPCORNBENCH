import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a matrix multiplication, applies ReLU, and divides by a constant.
    """
    def __init__(self, in_features, out_features, divisor):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.divisor = divisor

    def forward(self, x):
        x = self.linear(x)
        x = torch.relu(x)
        x = x / self.divisor
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cublasLt.h>
#include <mutex>
#include <stdexcept>

void div_launcher(torch::Tensor y, double divisor);

static cublasLtHandle_t get_lt_handle() {
    static cublasLtHandle_t handle = nullptr;
    static std::once_flag flag;
    std::call_once(flag, [](){ cublasLtCreate(&handle); });
    return handle;
}

#define CHECK_CUBLAS(call) do { cublasStatus_t _s = (call); if (_s != CUBLAS_STATUS_SUCCESS) throw std::runtime_error("cublasLt error"); } while(0)

torch::Tensor linear_relu_div(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, double divisor) {
    const int64_t M64 = x.size(0);
    const int64_t K64 = x.size(1);
    const int64_t N64 = weight.size(0);
    const int M = (int)M64;
    const int N = (int)N64;
    const int K = (int)K64;

    auto y = torch::empty({M64, N64}, x.options());

    cublasLtHandle_t lt = get_lt_handle();
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    cublasLtMatmulDesc_t opDesc;
    cublasLtMatrixLayout_t aDesc, bDesc, cDesc, dDesc;
    CHECK_CUBLAS(cublasLtMatmulDescCreate(&opDesc, CUBLAS_COMPUTE_32F, CUDA_R_32F));

    cublasOperation_t transa = CUBLAS_OP_T;
    cublasOperation_t transb = CUBLAS_OP_N;
    CHECK_CUBLAS(cublasLtMatmulDescSetAttribute(opDesc, CUBLASLT_MATMUL_DESC_TRANSA, &transa, sizeof(transa)));
    CHECK_CUBLAS(cublasLtMatmulDescSetAttribute(opDesc, CUBLASLT_MATMUL_DESC_TRANSB, &transb, sizeof(transb)));

    cublasLtEpilogue_t epi = CUBLASLT_EPILOGUE_RELU_BIAS;

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

__global__ void div_kernel(float* __restrict__ y, long long n, float inv) {
    long long i = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (; i < n; i += stride) {
        y[i] *= inv;
    }
}

void div_launcher(torch::Tensor y, double divisor) {
    const long long n = y.numel();
    if (n == 0) return;
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    const float inv = 1.0f / (float)divisor;
    div_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(y.data_ptr<float>(), n, inv);
}

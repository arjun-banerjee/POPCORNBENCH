#define WARP_SIZE 32
#define ROWS_PER_BLOCK 32

__global__ __launch_bounds__(1024) void gemv_warp_per_row(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C, int M, int K) {
    int warp_id = threadIdx.x / WARP_SIZE;  // 0 to 31
    int lane = threadIdx.x % WARP_SIZE;
    int row = blockIdx.x * ROWS_PER_BLOCK + warp_id;
    
    if (row >= M) return;
    
    const float4* A4 = reinterpret_cast<const float4*>(A + (long long)row * K);
    const float4* B4 = reinterpret_cast<const float4*>(B);
    
    int num_float4 = K / 4;
    float sum = 0.0f;
    
    #pragma unroll 16
    for (int i = lane; i < num_float4; i += WARP_SIZE) {
        float4 a = __ldg(&A4[i]);
        float4 b = __ldg(&B4[i]);
        sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;

    }
}

}

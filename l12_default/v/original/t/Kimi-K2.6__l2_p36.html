__device__ __forceinline__ float gelu_tanh(float x) {

    float x3 = x * x * x;
    float tanh_arg = sqrt_2_over_pi * (x + 0.044715f * x3);
    return 0.5f * x * (1.0f + tanhf(tanh_arg));
}

    float bias,

torch::Tensor matmul_cublas(torch::Tensor A, torch::Tensor B) {
    c10::cuda::CUDAGuard device_guard(A.device());
    if (handle == nullptr) {
        check_cublas(cublasCreate(&handle));
        check_cublas(cublasSetMathMode(handle, CUBLAS_PEDANTIC_MATH));
    }
    auto C = torch::empty({A.size(0), B.size(1)}, A.options());
    int n = (int)A.size(0);
    float alpha = 1.0f, beta = 0.0f;
    check_cublas(cublasSetStream(handle, at::cuda::getCurrentCUDAStream()));

}

                   (h_base < 0) || (h_base + (KS - 1) * dilation >= height) ||
                   (w_base < 0) || (w_base + (KS - 1) * dilation >= width);
    
    scalar_t max_val = -FLT_MAX;
    
    if (!is_edge) {
        #pragma unroll
        for (int kd = 0; kd < KS; ++kd) {
            int id = d_base + kd * dilation;
            int id_offset = nc_offset + id * height * width;
            #pragma unroll
            for (int kh = 0; kh < KS; ++kh) {
                int ih = h_base + kh * dilation;
                int ih_offset = id_offset + ih * width;
                #pragma unroll
                for (int kw = 0; kw < KS; ++kw) {
                    int iw = w_base + kw * dilation;
                    scalar_t val = input[ih_offset + iw];
                    max_val = (val > max_val) ? val : max_val;
                }
            }
        }
    } else {
        #pragma unroll
        for (int kd = 0; kd < KS; ++kd) {
            int id = d_base + kd * dilation;
            if (id < 0 || id >= depth) continue;
            int id_offset = nc_offset + id * height * width;
            #pragma unroll
            for (int kh = 0; kh < KS; ++kh) {
                int ih = h_base + kh * dilation;
                if (ih < 0 || ih >= height) continue;
                int ih_offset = id_offset + ih * width;
                #pragma unroll
                for (int kw = 0; kw < KS; ++kw) {
                    int iw = w_base + kw * dilation;
                    if (iw < 0 || iw >= width) continue;
                    scalar_t val = input[ih_offset + iw];
                    max_val = (val > max_val) ? val : max_val;
                }
            }
        }
    }
    
    int out_idx = ((n * channels + c) * out_d + d) * out_h + h;
    out_idx = out_idx * out_w + w;
    output[out_idx] = max_val;
}

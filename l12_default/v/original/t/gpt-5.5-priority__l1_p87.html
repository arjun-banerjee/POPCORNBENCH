__global__ void add_bias_kernel(float* __restrict__ y, const float* __restrict__ b, int N, int O, int HW) {
    long long total = (long long)N * O * HW;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        int o = (idx / HW) % O;
        y[idx] += b[o];
    }
}

torch::Tensor conv1x1_forward(torch::Tensor x, torch::Tensor w, torch::Tensor bias, bool has_bias) {
    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int Wd = (int)x.size(3);
    const int O = (int)w.size(0);
    const int HW = H * Wd;

    auto y = torch::empty({N, O, H, Wd}, x.options());

    // Each batch is a row-major GEMM:  Y[O,HW] = W[O,C] * X[C,HW].
    // cuBLAS is column-major, so compute Y^T[HW,O] = X^T[HW,C] * W^T[C,O]
    // without any data movement.  This gives the desired contiguous NCHW output.

    const float beta = 0.0f;

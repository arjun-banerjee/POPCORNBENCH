__device__ __forceinline__ float fast_log_ratio(float t, float p) {
    float y = (t - p) / (t + p);
    float y2 = y * y;
    float poly = 0.06666666666666667f;
    poly = fmaf(poly, y2, 0.07692307692307693f);
    poly = fmaf(poly, y2, 0.09090909090909091f);
    poly = fmaf(poly, y2, 0.11111111111111110f);
    poly = fmaf(poly, y2, 0.14285714285714285f);
    poly = fmaf(poly, y2, 0.20000000000000000f);
    poly = fmaf(poly, y2, 0.33333333333333331f);
    poly = fmaf(poly, y2, 1.0f);
    return 2.0f * y * poly;
}
__device__ __forceinline__ float kl_term(float t, float p) { return t * fast_log_ratio(t, p); }

__global__ void kl_atomic_kernel(const float* __restrict__ pred,
                                 const float* __restrict__ targ,
                                 float* __restrict__ out,
                                 long long n4,
                                 long long n,
                                 float inv_batch) {
    float acc = 0.0f;
    const float4* __restrict__ p4 = reinterpret_cast<const float4*>(pred);
    const float4* __restrict__ t4 = reinterpret_cast<const float4*>(targ);
    long long tid = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = tid; i < n4; i += stride) {
        float4 tv = t4[i]; float4 pv = p4[i];
        acc += kl_term(tv.x, pv.x);
        acc += kl_term(tv.y, pv.y);
        acc += kl_term(tv.z, pv.z);
        acc += kl_term(tv.w, pv.w);
    }
    long long base = n4 * 4;
    if (blockIdx.x == 0) {
        for (long long i = base + threadIdx.x; i < n; i += blockDim.x) acc += kl_term(targ[i], pred[i]);
    }
    acc = block_reduce_sum(acc);
    if (threadIdx.x == 0) atomicAdd(out, acc * inv_batch);
}

torch::Tensor kl_div_forward(torch::Tensor predictions, torch::Tensor targets) {
    const c10::cuda::CUDAGuard device_guard(predictions.device());
    auto out = torch::empty({}, predictions.options());
    const long long n = predictions.numel();
    const int threads = 256;
    int blocks = 1056;
    long long max_blocks = (n + threads * 4LL - 1) / (threads * 4LL);
    if (blocks > max_blocks) blocks = (int)max_blocks;
    if (blocks < 1) blocks = 1;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cudaMemsetAsync(out.data_ptr<float>(), 0, sizeof(float), stream);
    kl_atomic_kernel<<<blocks, threads, 0, stream>>>(predictions.data_ptr<float>(), targets.data_ptr<float>(), out.data_ptr<float>(), n >> 2, n, 1.0f / (float)predictions.size(0));
    return out;
}

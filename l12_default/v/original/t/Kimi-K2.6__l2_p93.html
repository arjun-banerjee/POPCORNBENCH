import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a transposed convolution, adds a value, takes the minimum, applies GELU, and multiplies by a value.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, add_value, multiply_value):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride)
        self.add_value = add_value
        self.multiply_value = multiply_value

    def forward(self, x):
        x = self.conv_transpose(x)
        x = x + self.add_value
        x = torch.min(x, torch.tensor(0.0, device=x.device))
        x = torch.nn.functional.gelu(x)
        x = x * self.multiply_value
        return x

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CUDA_SOURCE = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_transform_kernel(float* __restrict__ data, long N, float add_value, float multiply_value) {
    const float SQRT2 = 1.4142135623730951f;
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    long stride = (long)blockDim.x * gridDim.x;
    
    for (long i = idx; i < N; i += stride) {
        float x = data[i];
        x = x + add_value;
        x = (x > 0.0f) ? 0.0f : x;
        x = 0.5f * x * (1.0f + erff(x / SQRT2));
        data[i] = x * multiply_value;
    }
}

torch::Tensor fused_transform(torch::Tensor x, float add_value, float multiply_value) {
    TORCH_CHECK(x.is_floating_point(), "Input must be floating point");
    x = x.contiguous();
    long N = x.numel();
    
    const int threads = 512;
    const int blocks = (int)((N + threads - 1) / threads);
    
    fused_transform_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        N,
        add_value,
        multiply_value
    );
    
    return x;
}

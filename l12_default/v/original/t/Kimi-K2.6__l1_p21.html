import torch
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void sigmoid_kernel_float4(const float* __restrict__ input, float* __restrict__ output, int64_t n) {
    int64_t idx = (blockIdx.x * blockDim.x + threadIdx.x);
    int64_t num_vec = n / 4;
    int64_t stride = blockDim.x * gridDim.x;
    
    // Process 4 elements at a time using float4
    for (int64_t i = idx; i < num_vec; i += stride) {
        float4 x = reinterpret_cast<const float4*>(input)[i];
        x.x = 1.0f / (1.0f + expf(-x.x));
        x.y = 1.0f / (1.0f + expf(-x.y));
        x.z = 1.0f / (1.0f + expf(-x.z));
        x.w = 1.0f / (1.0f + expf(-x.w));
        reinterpret_cast<float4*>(output)[i] = x;
    }
    
    // Handle remaining elements
    int64_t remainder_start = num_vec * 4;
    for (int64_t i = remainder_start + idx; i < n; i += stride) {
        float x = input[i];
        output[i] = 1.0f / (1.0f + expf(-x));
    }
}

torch::Tensor sigmoid_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t n = input.numel();
    
    const int threads = 256;
    const int blocks = 8192;  // Enough to saturate H100
    
    sigmoid_kernel_float4<<<blocks, threads>>>(input.data_ptr<float>(), output.data_ptr<float>(), n);
    return output;
}

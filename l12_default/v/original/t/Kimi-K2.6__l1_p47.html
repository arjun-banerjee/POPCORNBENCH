#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void sum_reduce_kernel(const float* __restrict__ input,
                                  float* __restrict__ output,
                                  int outer_size,
                                  int reduce_size,
                                  int inner_size) {
    int outer_idx = blockIdx.y;
    int inner_idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (outer_idx >= outer_size || inner_idx >= inner_size) return;

    int64_t offset = (int64_t)outer_idx * reduce_size * inner_size + inner_idx;
    float sum = 0.0f;

    #pragma unroll 8
    for (int r = 0; r < reduce_size; ++r) {
        sum += input[offset + (int64_t)r * inner_size];
    }

    output[(int64_t)outer_idx * inner_size + inner_idx] = sum;
}

torch::Tensor sum_reduce_forward(torch::Tensor input, int64_t dim) {
    int64_t ndim = input.dim();
    if (dim < 0) dim += ndim;
    TORCH_CHECK(dim >= 0 && dim < ndim, "dim out of range");
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous");

    int outer_size = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer_size *= static_cast<int>(input.size(i));
    }
    int reduce_size = static_cast<int>(input.size(dim));
    int inner_size = 1;
    for (int64_t i = dim + 1; i < ndim; ++i) {
        inner_size *= static_cast<int>(input.size(i));
    }

    std::vector<int64_t> output_sizes(input.sizes().begin(), input.sizes().end());
    output_sizes[dim] = 1;

import torch
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>

constexpr float SELU_SCALE = 1.0507009873554804934193349852946f;
constexpr float SELU_ALPHA = 1.6732632423543772848180429911717f;

__global__ void selu_kernel(const float* __restrict__ input, float* __restrict__ output, int64_t n) {
    constexpr int VEC_SIZE = 4;
    int64_t idx = (blockIdx.x * blockDim.x + threadIdx.x) * VEC_SIZE;
    
    if (idx + VEC_SIZE <= n) {
        float4 in = reinterpret_cast<const float4*>(input)[idx / VEC_SIZE];
        float4 out;
        
        out.x = in.x > 0.0f ? SELU_SCALE * in.x : SELU_SCALE * SELU_ALPHA * (expf(in.x) - 1.0f);
        out.y = in.y > 0.0f ? SELU_SCALE * in.y : SELU_SCALE * SELU_ALPHA * (expf(in.y) - 1.0f);
        out.z = in.z > 0.0f ? SELU_SCALE * in.z : SELU_SCALE * SELU_ALPHA * (expf(in.z) - 1.0f);
        out.w = in.w > 0.0f ? SELU_SCALE * in.w : SELU_SCALE * SELU_ALPHA * (expf(in.w) - 1.0f);
        
        reinterpret_cast<float4*>(output)[idx / VEC_SIZE] = out;
    } else {
        for (int64_t i = idx; i < n; ++i) {
            float x = input[i];
            output[i] = x > 0.0f ? SELU_SCALE * x : SELU_SCALE * SELU_ALPHA * (expf(x) - 1.0f);
        }
    }
}

torch::Tensor selu_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t n = input.numel();
    
    if (n == 0) return output;
    
    constexpr int VEC_SIZE = 4;
    constexpr int BLOCK_SIZE = 256;
    int64_t num_vec = (n + VEC_SIZE - 1) / VEC_SIZE;
    int64_t num_blocks = (num_vec + BLOCK_SIZE - 1) / BLOCK_SIZE;
    
    selu_kernel<<<num_blocks, BLOCK_SIZE>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        n
    );
    
    return output;
}
'''

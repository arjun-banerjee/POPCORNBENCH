import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, divides by a constant, applies max pooling,
    global average pooling, adds a bias term, and sums along a specific dimension.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = divisor
        self.max_pool = nn.MaxPool3d(pool_size)
        self.global_avg_pool = nn.AdaptiveAvgPool3d((1, 1, 1))
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.sum_dim = sum_dim

    def forward(self, x):
        x = self.conv(x)
        x = x / self.divisor
        x = self.max_pool(x)
        x = self.global_avg_pool(x)
        x = x + self.bias
        x = torch.sum(x, dim=self.sum_dim)
        return x

#include <torch/extension.h>
#include <ATen/ATen.h>
#include <vector>

torch::Tensor postprocess_cuda(torch::Tensor conv, torch::Tensor bias, double divisor);

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor bias, double divisor) {
    auto y = at::conv3d(x, weight, conv_bias, {1,1,1}, {0,0,0}, {1,1,1}, 1);
    return postprocess_cuda(y, bias, divisor);
}

__global__ void post_kernel(const float* __restrict__ y, const float* __restrict__ bias,
                            float* __restrict__ out,
                            int N, int C, int D, int H, int W, float divisor) {
    int n = blockIdx.x;
    int tid = threadIdx.x;
    const int PD = D / 2;
    const int PH = H / 2;
    const int PW = W / 2;
    int total = C * PD * PH * PW;
    float local = 0.0f;
    bool pos = divisor > 0.0f;
    for (int idx = tid; idx < total; idx += blockDim.x) {
        int t = idx;
        int pw = t % PW; t /= PW;
        int ph = t % PH; t /= PH;
        int pd = t % PD; t /= PD;
        int c = t;
        int d0 = pd << 1;
        int h0 = ph << 1;
        int w0 = pw << 1;
        const float* p = y + (((n * C + c) * D + d0) * H + h0) * W + w0;
        float v0 = p[0];
        float v1 = p[1];
        float v2 = p[W];
        float v3 = p[W + 1];
        const float* q = p + H * W;

        float v5 = q[1];
        float v6 = q[W];

        if (pos) {

            m = fminf(fminf(fminf(v0, v1), fminf(v2, v3)), fminf(fminf(v4, v5), fminf(v6, v7)));

}

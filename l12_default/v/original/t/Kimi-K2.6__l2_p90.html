    Model that performs a 3D convolution, applies LeakyReLU, sums with a tensor, clamps, and applies GELU activation.

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cmath>

#define THREADS_PER_BLOCK 256

__global__ void fused_conv_postprocess_kernel(
    const float* __restrict__ conv_out,
    const float* __restrict__ sum_tensor,
    float* __restrict__ out,
    int N, int C, int spatial_size
) {
    int c = blockIdx.x;
    int n = blockIdx.y;
    int tid = threadIdx.x;
    
    int batch_stride = C * spatial_size;
    int channel_stride = spatial_size;
    int base_idx = n * batch_stride + c * channel_stride;
    
    float sum_val = sum_tensor[c];
    
    for (int i = tid; i < spatial_size; i += THREADS_PER_BLOCK) {
        int idx = base_idx + i;
        float val = conv_out[idx];
        
        // LeakyReLU
        val = val >= 0.0f ? val : 0.2f * val;
        
        // Add sum_tensor
        val += sum_val;
        
        // Clamp
        val = fminf(fmaxf(val, -1.0f), 1.0f);
        
        // GELU exact
        val = val * 0.5f * (1.0f + erff(val * 0.7071067811865475f));
        
        out[idx] = val;
    }
}

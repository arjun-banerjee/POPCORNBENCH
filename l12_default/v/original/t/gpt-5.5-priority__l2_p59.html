import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a matrix multiplication, applies Swish activation, and scales the result.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.matmul(x)
        x = x * torch.sigmoid(x)  # Swish activation
        x = x * self.scaling_factor
        return x

batch_size = 128

scaling_factor = 2.0

def get_inputs():
    return [torch.rand(batch_size, in_features)]

    return [in_features, out_features, scaling_factor]

static cublasHandle_t get_handle(){
    static cublasHandle_t h=nullptr;
    static std::once_flag f;
    std::call_once(f, [](){
        cublasCreate(&h);
        cublasSetMathMode(h, CUBLAS_DEFAULT_MATH);
    });
    return h;

__device__ __forceinline__ float swish_scale(float v, float scale){
    return v * (1.0f / (1.0f + __expf(-v))) * scale;
}

__global__ void bias_swish_scale_vec4(float* __restrict__ y,
                                      const float* __restrict__ bias,
                                      int M, int N4, float scale){
    int col4 = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if(row < M && col4 < N4){
        int idx4 = row * N4 + col4;
        float4 yy = reinterpret_cast<float4*>(y)[idx4];
        float4 bb = reinterpret_cast<const float4*>(bias)[col4];
        yy.x = swish_scale(yy.x + bb.x, scale);
        yy.y = swish_scale(yy.y + bb.y, scale);
        yy.z = swish_scale(yy.z + bb.z, scale);
        yy.w = swish_scale(yy.w + bb.w, scale);
        reinterpret_cast<float4*>(y)[idx4] = yy;
    }
}

__global__ void bias_swish_scale_scalar(float* __restrict__ y,
                                        const float* __restrict__ bias,
                                        int total, int N, float scale){
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if(i < total){
        float v = y[i] + bias[i % N];
        y[i] = swish_scale(v, scale);
    }
}

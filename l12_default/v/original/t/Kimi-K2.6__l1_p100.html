#include <cuda_runtime.h>

__global__ void hinge_loss_forward_kernel(
    const float* __restrict__ predictions,
    const float* __restrict__ targets,
    int dim,
    int dim_log2,
    int N,
    float* __restrict__ output) 
{
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int grid_size = gridDim.x * blockDim.x;
    float sum = 0.0f;
    
    const float4* pred_vec = reinterpret_cast<const float4*>(predictions);
    int N_vec = N >> 2;
    
    #pragma unroll 4
    for (int i = tid; i < N_vec; i += grid_size) {
        float4 p = pred_vec[i];
        int base_idx = i << 2;
        int row;
        if (dim_log2 >= 0) {
            row = base_idx >> dim_log2;
        } else {
            row = base_idx / dim;
        }
        float t = targets[row];
        
        float v0 = 1.0f - p.x * t;
        float v1 = 1.0f - p.y * t;
        float v2 = 1.0f - p.z * t;
        float v3 = 1.0f - p.w * t;
        
        sum += fmaxf(0.0f, v0) + fmaxf(0.0f, v1) + fmaxf(0.0f, v2) + fmaxf(0.0f, v3);
    }
    
    // Handle remaining elements (0 to 3)
    for (int idx = (N_vec << 2) + tid; idx < N; idx += grid_size) {
        int row;
        if (dim_log2 >= 0) {
            row = idx >> dim_log2;
        } else {
            row = idx / dim;
        }
        float p = predictions[idx];
        float t = targets[row];
        float v = 1.0f - p * t;
        sum += fmaxf(0.0f, v);
    }
    
    unsigned int mask = 0xFFFFFFFF;
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(mask, sum, offset);
    }
    
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int warp_id = threadIdx.x >> 5;
    
    if (lane == 0) shared[warp_id] = sum;
    __syncthreads();
    
    if (warp_id == 0) {
        sum = (lane < (blockDim.x >> 5)) ? shared[lane] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(mask, sum, offset);
        }
        if (lane == 0) {
            atomicAdd(output, sum);
        }
    }
}

torch::Tensor hinge_loss_forward(torch::Tensor predictions, torch::Tensor targets) {
    int N = predictions.numel();
    int dim = predictions.size(1);
    
    auto output = torch::zeros({}, predictions.options());
    
    const int threads = 256;
    int N_vec = N >> 2;
    int blocks = (N_vec + threads - 1) / threads;
    if (blocks > 8192) blocks = 8192;
    if (blocks < 1) blocks = 1;
    
    int dim_log2 = -1;
    if ((dim & (dim - 1)) == 0) {
        dim_log2 = __builtin_ctz(dim);
    }
    
    hinge_loss_forward_kernel<<<blocks, threads>>>(
        predictions.data_ptr<float>(),
        targets.data_ptr<float>(),
        dim,
        dim_log2,
        N,
        output.data_ptr<float>()
    );
    
    return output / static_cast<float>(N);
}
"""

hinge_loss_cuda = load_inline(
    name="hinge_loss_cuda",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_source,
    functions=["hinge_loss_forward"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, predictions, targets):
        return hinge_loss_cuda.hinge_loss_forward(predictions.contiguous(), targets.contiguous())

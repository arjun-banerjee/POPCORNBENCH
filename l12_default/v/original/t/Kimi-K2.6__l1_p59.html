__global__ void conv3d_custom_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C_in, int D_in, int H_in, int W_in,
    int C_out, int D_out, int H_out, int W_out,

    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w,
    int dilation_d, int dilation_h, int dilation_w,
    int groups
) {
    const int n = blockIdx.z;
    const int oh = blockIdx.y * blockDim.y + threadIdx.y;
    const int ow = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (oh >= H_out || ow >= W_out) return;
    
    const int c_in_per_group = C_in / groups;
    
    for (int od = 0; od < D_out; ++od) {
        for (int c_out = 0; c_out < C_out; ++c_out) {
            float acc = (bias != nullptr) ? bias[c_out] : 0.0f;

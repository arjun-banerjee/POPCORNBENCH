    o.w = fminf(fmaxf(d3 * rstd * wv.w + bv.w, min_val), max_val);

__global__ void fused_gn_hardtanh_generic(
    const float* __restrict__ input,
    const float* __restrict__ gn_weight,
    const float* __restrict__ gn_bias,
    float* __restrict__ output,
    int N, int C, int num_groups, float eps, float min_val, float max_val) {
    
    const int group_size = C / num_groups;
    const int tid = threadIdx.x;
    const int ng = blockIdx.x;
    const int n = ng / num_groups;
    const int g = ng % num_groups;
    const int base_c = g * group_size;
    const int base_idx = n * C + base_c;
    
    float thread_sum = 0.0f;
    for (int idx = tid; idx < group_size; idx += blockDim.x) {
        thread_sum += input[base_idx + idx];
    }
    
    thread_sum = warp_reduce_sum(thread_sum);
    __shared__ float s[32];
    int lane = tid & 31;
    int warp = tid >> 5;
    int num_warps = blockDim.x >> 5;
    
    if (lane == 0) s[warp] = thread_sum;
    __syncthreads();
    
    if (warp == 0) {
        float val = (lane < num_warps) ? s[lane] : 0.0f;
        val = warp_reduce_sum(val);
        if (lane == 0) s[0] = val / (float)group_size;
    }
    __syncthreads();
    float mean = s[0];
    
    float thread_sq = 0.0f;
    for (int idx = tid; idx < group_size; idx += blockDim.x) {
        float d = input[base_idx + idx] - mean;
        thread_sq += d * d;
    }
    
    thread_sq = warp_reduce_sum(thread_sq);
    if (lane == 0) s[warp] = thread_sq;
    __syncthreads();
    
    if (warp == 0) {
        float val = (lane < num_warps) ? s[lane] : 0.0f;
        val = warp_reduce_sum(val);
        if (lane == 0) s[0] = 1.0f / sqrtf(val / (float)group_size + eps);
    }
    __syncthreads();
    float rstd = s[0];

        v = fminf(fmaxf(v, min_val), max_val);
        output[base_idx + idx] = v;
    }
}

    const float sqrt2_inv = 0.70710678118654752440f;
    return 0.5f * x * (1.0f + erff(x * sqrt2_inv));

__global__ void fused_post_gemm_kernel(const float* __restrict__ input,
                                       float* __restrict__ output,

    int tid = threadIdx.x;
    int lane = tid & 31;
    int warp_id = tid >> 5;

    extern __shared__ float smem[];
    float* s_max = smem;
    float* s_sum = smem + 32;

    int float4_count = N >> 2;
    const float4* row_ptr4 = reinterpret_cast<const float4*>(input + row * N);

    float local_max = -INFINITY;
    for (int j = tid; j < float4_count; j += 1024) {
        float4 val4 = __ldg(&row_ptr4[j]);
        local_max = fmaxf(local_max, fmaxf(fmaxf(val4.x, val4.y), fmaxf(val4.z, val4.w)));
    }

    // Warp reduce max
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xffffffff, local_max, offset);
        if (other > local_max) local_max = other;
    }

    if (lane == 0) {
        s_max[warp_id] = local_max;
    }
    __syncthreads();

    if (tid == 0) {
        float max_val = s_max[0];
        #pragma unroll
        for (int i = 1; i < 32; ++i) {
            max_val = fmaxf(max_val, s_max[i]);
        }
        s_max[0] = max_val;
    }
    __syncthreads();
    float row_max = s_max[0];

    float local_sum = 0.0f;
    for (int j = tid; j < float4_count; j += 1024) {
        float4 val4 = __ldg(&row_ptr4[j]);
        local_sum += expf(val4.x - row_max);
        local_sum += expf(val4.y - row_max);
        local_sum += expf(val4.z - row_max);
        local_sum += expf(val4.w - row_max);
    }

    // Warp reduce sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }

    if (lane == 0) {
        s_sum[warp_id] = local_sum;
    }
    __syncthreads();

    if (tid == 0) {
        float total = s_sum[0];
        #pragma unroll
        for (int i = 1; i < 32; ++i) {
            total += s_sum[i];
        }
        s_sum[0] = total;
    }
    __syncthreads();
    float row_sum = s_sum[0];

    if (tid == 0) {
        float lse = row_max + logf(row_sum);
        lse = leaky_relu(lse, 0.01f);
        lse = leaky_relu(lse, 0.01f);
        lse = gelu(lse);
        lse = gelu(lse);
        output[row] = lse;
    }
}

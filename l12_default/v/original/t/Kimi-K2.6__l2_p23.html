    const float* __restrict__ gamma,
    float* __restrict__ out,
    int B, int C_in, int D_in, int H_in, int W_in,
    int C_out, int D_out, int H_out, int W_out,
    int G, float base
) {
    int b = blockIdx.x / G;
    int g = blockIdx.x % G;
    int tid = threadIdx.x;
    const int block_size = blockDim.x;
    
    const int S = D_out * H_out * W_out;
    const int cpg = C_out / G;
    const int M = cpg * S;
    const int N = C_out * S;
    
    __shared__ float s_weight[729];
    __shared__ float s_bias[3];
    __shared__ float s_gamma[3];
    
    int w_base = g * 729;
    for (int i = tid; i < 729; i += block_size) {
        s_weight[i] = weight[w_base + i];
    }
    if (tid < 3) {
        int c = g * 3 + tid;
        s_bias[tid] = bias[c];
        s_gamma[tid] = gamma[c];
    }
    __syncthreads();
    
    float local_sum_y = 0.0f;
    float local_sum_y2 = 0.0f;
    float local_sum_yg = 0.0f;
    
    const int x_batch_stride = C_in * D_in * H_in * W_in;
    const int x_channel_stride = D_in * H_in * W_in;
    const int x_d_stride = H_in * W_in;
    const int x_h_stride = W_in;
    
    int chunk = (M + block_size - 1) / block_size;
    int start = tid * chunk;
    int end = min(start + chunk, M);
    
    for (int idx = start; idx < end; idx++) {
        int c_off = idx / S;
        int spatial = idx % S;
        int d = spatial / (H_out * W_out);
        int h = (spatial / W_out) % H_out;
        int w = spatial % W_out;
        
        float y = s_bias[c_off];
        int w_offset = c_off * 243;
        
        #pragma unroll
        for (int ci = 0; ci < 3; ci++) {
            int x_ci_base = b * x_batch_stride + ci * x_channel_stride;
            int w_ci_offset = ci * 27;
            #pragma unroll
            for (int kd = 0; kd < 3; kd++) {
                #pragma unroll
                for (int kh = 0; kh < 3; kh++) {
                    #pragma unroll
                    for (int kw = 0; kw < 3; kw++) {
                        int x_idx = x_ci_base + (d + kd) * x_d_stride + (h + kh) * x_h_stride + (w + kw);
                        y += x[x_idx] * s_weight[w_offset + w_ci_offset + kd * 9 + kh * 3 + kw];
                    }
                }
            }
        }
        
        float gamma_c = s_gamma[c_off];
        local_sum_y += y;
        local_sum_y2 += y * y;
        local_sum_yg += y * gamma_c;
    }
    
    // Warp shuffle reduction
    unsigned int mask = 0xffffffff;
    for (int offset = 16; offset > 0; offset /= 2) {
        local_sum_y += __shfl_down_sync(mask, local_sum_y, offset);
        local_sum_y2 += __shfl_down_sync(mask, local_sum_y2, offset);
        local_sum_yg += __shfl_down_sync(mask, local_sum_yg, offset);
    }
    
    __shared__ float s_sum_y[8];
    __shared__ float s_sum_y2[8];
    __shared__ float s_sum_yg[8];
    
    int lane = tid % 32;
    int warp = tid / 32;
    
    if (lane == 0) {
        s_sum_y[warp] = local_sum_y;
        s_sum_y2[warp] = local_sum_y2;
        s_sum_yg[warp] = local_sum_yg;
    }
    __syncthreads();
    
    if (tid < 8) {
        float sum_y = s_sum_y[tid];
        float sum_y2 = s_sum_y2[tid];
        float sum_yg = s_sum_yg[tid];
        for (int offset = 4; offset > 0; offset /= 2) {
            sum_y += __shfl_down_sync(0xff, sum_y, offset);
            sum_y2 += __shfl_down_sync(0xff, sum_y2, offset);
            sum_yg += __shfl_down_sync(0xff, sum_yg, offset);
        }
        if (tid == 0) {
            s_sum_y[0] = sum_y;
            s_sum_y2[0] = sum_y2;
            s_sum_yg[0] = sum_yg;
        }
    }
    __syncthreads();
    
    if (tid == 0) {
        float sum_y = s_sum_y[0];
        float sum_y2 = s_sum_y2[0];
        float sum_yg = s_sum_yg[0];
        
        float mu = sum_y / (float)M;
        float var = sum_y2 / (float)M - mu * mu;
        float inv_std = rsqrtf(var + 1e-5f);
        
        float gamma_sum = s_gamma[0] + s_gamma[1] + s_gamma[2];
        float contrib = inv_std * (sum_yg - mu * gamma_sum * (float)S) / (float)N;
        if (g == 0) {
            contrib += base;
        }
        atomicAdd(&out[b], contrib);
    }
}

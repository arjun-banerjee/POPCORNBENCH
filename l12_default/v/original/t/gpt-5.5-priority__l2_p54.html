import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a convolution, multiplies by a learnable scalar, applies LeakyReLU, and then GELU.
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.multiplier = nn.Parameter(torch.randn(multiplier_shape)) 
        self.leaky_relu = nn.LeakyReLU()

    def forward(self, x):
        x = self.conv(x)
        x = x * self.multiplier
        x = self.leaky_relu(x)
        x = torch.nn.functional.gelu(x)
        return x

batch_size = 64

__device__ __forceinline__ float erf_approx(float x) {
    float s = x < 0.0f ? -1.0f : 1.0f;
    float ax = fabsf(x);
    float t = 1.0f / (1.0f + 0.3275911f * ax);
    float p = (((((1.061405429f * t - 1.453152027f) * t) + 1.421413741f) * t - 0.284496736f) * t + 0.254829592f) * t;
    return s * (1.0f - p * expf(-ax * ax));
}

__device__ __forceinline__ float gelu_after_leaky(float v) {
    v = (v >= 0.0f) ? v : (0.01f * v);
    return 0.5f * v * (1.0f + erf_approx(v * 0.70710678118654752440f));
}

__global__ void fused_postop_vec4x4_kernel(float4* __restrict__ x4, const float* __restrict__ m,
                                           int HW4, int C) {
    int nc = blockIdx.y;
    int c = nc % C;
    float mult = m[c];
    int base_hw4 = (blockIdx.x * blockDim.x + threadIdx.x) * 4;
    long plane = (long)nc * HW4;
    #pragma unroll
    for (int i = 0; i < 4; ++i) {
        int hw4 = base_hw4 + i;
        if (hw4 < HW4) {
            float4 r = x4[plane + hw4];
            r.x = gelu_after_leaky(r.x * mult);
            r.y = gelu_after_leaky(r.y * mult);
            r.z = gelu_after_leaky(r.z * mult);
            r.w = gelu_after_leaky(r.w * mult);
            x4[plane + hw4] = r;
        }
    }
}

    if (idx >= total) return;

}

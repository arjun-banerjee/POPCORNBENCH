import torch
from torch.utils.cpp_extension import load_inline

torch.cuda.empty_cache()
gc.collect()

cpp_source = r'''
#include <torch/extension.h>

void l1_norm_forward_cuda(const float* x, float* out, int batch, int dim);

void l1_norm_forward(torch::Tensor x, torch::Tensor out, int batch, int dim) {
    l1_norm_forward_cuda(
        x.data_ptr<float>(),
        out.data_ptr<float>(),
        batch,
        dim
    );
}
'''

cuda_source = r'''
#include <cuda_runtime.h>
#include <math.h>

__inline__ __device__ float warp_reduce_sum(float val) {
    val += __shfl_down_sync(0xffffffff, val, 16);
    val += __shfl_down_sync(0xffffffff, val, 8);
    val += __shfl_down_sync(0xffffffff, val, 4);
    val += __shfl_down_sync(0xffffffff, val, 2);
    val += __shfl_down_sync(0xffffffff, val, 1);
    return val;
}

__global__ void l1_norm_kernel(const float* __restrict__ x,
                               float* __restrict__ out,
                               int batch, int dim) {
    int row = blockIdx.x;
    if (row >= batch) return;

    const float* row_x = x + static_cast<int64_t>(row) * dim;
    float* row_out = out + static_cast<int64_t>(row) * dim;

    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        local_sum += fabsf(row_x[i]);
    }

    local_sum = warp_reduce_sum(local_sum);

    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    if (lane == 0) shared[wid] = local_sum;
    __syncthreads();

    if (wid == 0) {
        float s = (lane < (blockDim.x >> 5)) ? shared[lane] : 0.0f;
        s = warp_reduce_sum(s);
        if (lane == 0) shared[0] = s / static_cast<float>(dim);
    }
    __syncthreads();

    float mean = shared[0];
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        row_out[i] = row_x[i] / mean;
    }
}

void l1_norm_forward_cuda(const float* x, float* out, int batch, int dim) {
    l1_norm_kernel<<<batch, 1024>>>(x, out, batch, dim);
}

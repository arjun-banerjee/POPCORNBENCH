import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs a convolution, applies tanh, scaling, adds a bias term, and then max-pools.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)

    def forward(self, x):
        # Convolution
        x = self.conv(x)
        # Tanh activation
        x = torch.tanh(x)
        # Scaling
        x = x * self.scaling_factor
        # Bias addition
        x = x + self.bias
        # Max-pooling
        x = self.max_pool(x)
        return x

def get_inputs():
    return [torch.rand(batch_size, in_channels, height, width)]

    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = '''
torch::Tensor fused_maxpool_tanh_scale_bias_cuda(
    torch::Tensor input,
    torch::Tensor bias,
    int pool_size,
    float scaling_factor
);
'''

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void fused_maxpool_tanh_scale_bias_kernel(
    const float* __restrict__ input,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int pool_size,
    int H_out, int W_out,
    float scaling_factor
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;
    
    int ow = idx % W_out;
    int tmp = idx / W_out;
    int oh = tmp % H_out;
    tmp = tmp / H_out;
    int c = tmp % C;
    int n = tmp / C;
    
    int h_start = oh * pool_size;
    int w_start = ow * pool_size;
    int base_offset = ((n * C + c) * H + h_start) * W + w_start;
    
    float max_val = -INFINITY;
    #pragma unroll 4
    for (int kh = 0; kh < pool_size; kh++) {
        #pragma unroll 4
        for (int kw = 0; kw < pool_size; kw++) {
            float val = input[base_offset + kh * W + kw];
            max_val = fmaxf(max_val, val);
        }
    }
    
    float out_val = tanhf(max_val) * scaling_factor + bias[c];
    output[idx] = out_val;
}

torch::Tensor fused_maxpool_tanh_scale_bias_cuda(
    torch::Tensor input,
    torch::Tensor bias,
    int pool_size,
    float scaling_factor
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int H_out = H / pool_size;
    int W_out = W / pool_size;
    
    auto output = torch::empty({N, C, H_out, W_out}, input.options());
    
    int total = N * C * H_out * W_out;
    const int threads = 256;
    int blocks = (total + threads - 1) / threads;
    
    fused_maxpool_tanh_scale_bias_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C, H, W,
        pool_size,
        H_out, W_out,
        scaling_factor
    );
    
    return output;
}
'''

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that computes Triplet Margin Loss for metric learning tasks.

    Parameters:
        margin (float): The margin between the positive and negative samples.
    """
    def __init__(self, margin=1.0):
        super(Model, self).__init__()
        self.loss_fn = torch.nn.TripletMarginLoss(margin=margin)

    def forward(self, anchor, positive, negative):
        return self.loss_fn(anchor, positive, negative)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")

__inline__ __device__ float warp_reduce_sum(float v) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) v += __shfl_down_sync(0xffffffff, v, offset);
    return v;
}

template<int BLOCK, int COLS>
__global__ __launch_bounds__(BLOCK)
void triplet_rows_kernel_8192(const float* __restrict__ anchor,
                              const float* __restrict__ positive,
                              const float* __restrict__ negative,
                              float* __restrict__ losses,
                              float margin) {
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    float sum_ap = 0.0f, sum_an = 0.0f;
    const long long base = (long long)row * COLS;
    const float4* __restrict__ a4 = reinterpret_cast<const float4*>(anchor + base);
    const float4* __restrict__ p4 = reinterpret_cast<const float4*>(positive + base);
    const float4* __restrict__ n4 = reinterpret_cast<const float4*>(negative + base);
    #pragma unroll
    for (int i = tid; i < (COLS >> 2); i += BLOCK) {
        float4 av = a4[i], pv = p4[i], nv = n4[i];
        float d;
        d = av.x - pv.x + 1.0e-6f; sum_ap = fmaf(d,d,sum_ap);
        d = av.y - pv.y + 1.0e-6f; sum_ap = fmaf(d,d,sum_ap);
        d = av.z - pv.z + 1.0e-6f; sum_ap = fmaf(d,d,sum_ap);
        d = av.w - pv.w + 1.0e-6f; sum_ap = fmaf(d,d,sum_ap);
        d = av.x - nv.x + 1.0e-6f; sum_an = fmaf(d,d,sum_an);
        d = av.y - nv.y + 1.0e-6f; sum_an = fmaf(d,d,sum_an);
        d = av.z - nv.z + 1.0e-6f; sum_an = fmaf(d,d,sum_an);
        d = av.w - nv.w + 1.0e-6f; sum_an = fmaf(d,d,sum_an);
    }
    sum_ap = warp_reduce_sum(sum_ap);
    sum_an = warp_reduce_sum(sum_an);
    __shared__ float sh_ap[4], sh_an[4];
    if ((tid & 31) == 0) { sh_ap[tid>>5] = sum_ap; sh_an[tid>>5] = sum_an; }
    __syncthreads();

}

template<int BLOCK>
__global__ __launch_bounds__(BLOCK)
void final_reduce_kernel(const float* __restrict__ losses, float* __restrict__ out, int n) {
    const int tid = threadIdx.x;
    float s = 0.0f;
    for (int i = tid; i < n; i += BLOCK) s += losses[i];

torch::Tensor triplet_forward(torch::Tensor anchor, torch::Tensor positive, torch::Tensor negative, double margin) {
    CHECK_CUDA(anchor); CHECK_CUDA(positive); CHECK_CUDA(negative);

                int ox = out_x + dx;
                if (ox < out_w) {
                    output[out_offset + oy * out_w + ox] = accum[dy][dx] + bval;
                }
            }
        }
    }
}

    int in_tile_w = (tile_w - 1) * stride_w + (kernel_w - 1) * dilation_w + 1;
    size_t shared_mem_size = (in_tile_h * in_tile_w + kernel_h * kernel_w) * sizeof(float);

        dilation_h,
        dilation_w
    );
    
    return output;
}

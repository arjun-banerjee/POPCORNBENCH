import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies Instance Normalization, and divides by a constant.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels)
        self.divide_by = divide_by

    def forward(self, x):
        x = self.conv(x)
        x = self.instance_norm(x)
        x = x / self.divide_by
        return x

    val = (threadIdx.x < (blockDim.x >> 5)) ? shared[lane] : 0.0f;
    if (wid == 0) val = warp_reduce_sum(val);
    return val;
}
__global__ __launch_bounds__(256, 2)
void inst_norm_div_inplace_kernel(float* __restrict__ x, int rows, int S, float inv_div) {
    int row = blockIdx.x;
    float* base = x + (long long)row * S;
    float sum = 0.0f, sumsq = 0.0f;
    int S4 = S >> 2;
    float4* base4 = reinterpret_cast<float4*>(base);
    for (int i = threadIdx.x; i < S4; i += blockDim.x) {
        float4 v = base4[i];
        sum += v.x + v.y + v.z + v.w;
        sumsq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
    }
    sum = block_reduce_sum(sum);
    sumsq = block_reduce_sum(sumsq);
    __shared__ float sh_mean, sh_scale;
    if (threadIdx.x == 0) {
        float mean = sum / (float)S;
        float var = fmaxf(sumsq / (float)S - mean * mean, 0.0f);
        sh_mean = mean;
        sh_scale = rsqrtf(var + 1.0e-5f) * inv_div;
    }
    __syncthreads();
    float mean = sh_mean, scale = sh_scale;
    for (int i = threadIdx.x; i < S4; i += blockDim.x) {
        float4 v = base4[i];
        v.x = (v.x - mean) * scale; v.y = (v.y - mean) * scale;
        v.z = (v.z - mean) * scale; v.w = (v.w - mean) * scale;
        base4[i] = v;
    }
}

torch::Tensor conv_inst_norm_div(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, double divide_by) {
    const c10::cuda::CUDAGuard device_guard(x.device());
    std::optional<at::Tensor> b = bias;
    std::vector<int64_t> stride = {1,1}, padding = {0,0}, dilation = {1,1};
    auto y = at::conv2d(x, weight, b, stride, padding, dilation, 1);
    int rows = (int)(y.size(0) * y.size(1));
    int S = (int)(y.size(2) * y.size(3));
    float inv_div = 1.0f / (float)divide_by;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    inst_norm_div_inplace_kernel<<<rows, 256, 0, stream>>>(y.data_ptr<float>(), rows, S, inv_div);
    return y;
}

#include <cuda.h>
#include <cuda_runtime.h>

#define TILE_M 32
#define TILE_N 256
#define KFIX 32

__global__ __launch_bounds__(256, 2)
void matmul_k32_kernel(const float* __restrict__ A,
                       const float* __restrict__ B,
                       float* __restrict__ C,
                       int M, int Nout) {
    __shared__ float As[TILE_M][KFIX];
    __shared__ float Bs[KFIX][TILE_N];
    const int tid = threadIdx.x;
    const int row_base = blockIdx.y * TILE_M;
    const int col_base = blockIdx.x * TILE_N;

    int ar = tid >> 3;
    int av = tid & 7;
    reinterpret_cast<float4*>(&As[ar][0])[av] =
        reinterpret_cast<const float4*>(A + (long long)(row_base + ar) * KFIX)[av];

    for (int idx = tid; idx < KFIX * (TILE_N/4); idx += 256) {
        int k = idx >> 6;
        int v = idx & 63;
        reinterpret_cast<float4*>(&Bs[k][0])[v] =
            reinterpret_cast<const float4*>(B + (long long)k * Nout + col_base)[v];
    }
    __syncthreads();

    const int rg = tid >> 4;
    const int cg = tid & 15;
    const int lr0 = rg << 1;
    const int lr1 = lr0 + 1;
    const int gr0 = row_base + lr0;
    const int gr1 = gr0 + 1;
    const int lc = cg << 4;
    const int gc0 = col_base + lc;

    float a0_0=0.f,a0_1=0.f,a0_2=0.f,a0_3=0.f,a0_4=0.f,a0_5=0.f,a0_6=0.f,a0_7=0.f;
    float a0_8=0.f,a0_9=0.f,a0_10=0.f,a0_11=0.f,a0_12=0.f,a0_13=0.f,a0_14=0.f,a0_15=0.f;
    float a1_0=0.f,a1_1=0.f,a1_2=0.f,a1_3=0.f,a1_4=0.f,a1_5=0.f,a1_6=0.f,a1_7=0.f;
    float a1_8=0.f,a1_9=0.f,a1_10=0.f,a1_11=0.f,a1_12=0.f,a1_13=0.f,a1_14=0.f,a1_15=0.f;

    #pragma unroll
    for (int k=0; k<KFIX; ++k) {
        float x0 = As[lr0][k];
        float x1 = As[lr1][k];
        const float* bp = &Bs[k][lc];
        float b0=bp[0], b1=bp[1], b2=bp[2], b3=bp[3], b4=bp[4], b5=bp[5], b6=bp[6], b7=bp[7];
        float b8=bp[8], b9=bp[9], b10=bp[10], b11=bp[11], b12=bp[12], b13=bp[13], b14=bp[14], b15=bp[15];
        a0_0=fmaf(x0,b0,a0_0); a0_1=fmaf(x0,b1,a0_1); a0_2=fmaf(x0,b2,a0_2); a0_3=fmaf(x0,b3,a0_3);
        a0_4=fmaf(x0,b4,a0_4); a0_5=fmaf(x0,b5,a0_5); a0_6=fmaf(x0,b6,a0_6); a0_7=fmaf(x0,b7,a0_7);
        a0_8=fmaf(x0,b8,a0_8); a0_9=fmaf(x0,b9,a0_9); a0_10=fmaf(x0,b10,a0_10); a0_11=fmaf(x0,b11,a0_11);
        a0_12=fmaf(x0,b12,a0_12); a0_13=fmaf(x0,b13,a0_13); a0_14=fmaf(x0,b14,a0_14); a0_15=fmaf(x0,b15,a0_15);
        a1_0=fmaf(x1,b0,a1_0); a1_1=fmaf(x1,b1,a1_1); a1_2=fmaf(x1,b2,a1_2); a1_3=fmaf(x1,b3,a1_3);
        a1_4=fmaf(x1,b4,a1_4); a1_5=fmaf(x1,b5,a1_5); a1_6=fmaf(x1,b6,a1_6); a1_7=fmaf(x1,b7,a1_7);
        a1_8=fmaf(x1,b8,a1_8); a1_9=fmaf(x1,b9,a1_9); a1_10=fmaf(x1,b10,a1_10); a1_11=fmaf(x1,b11,a1_11);
        a1_12=fmaf(x1,b12,a1_12); a1_13=fmaf(x1,b13,a1_13); a1_14=fmaf(x1,b14,a1_14); a1_15=fmaf(x1,b15,a1_15);
    }

    float4* out0 = reinterpret_cast<float4*>(C + (long long)gr0 * Nout + gc0);
    out0[0] = make_float4(a0_0, a0_1, a0_2, a0_3);
    out0[1] = make_float4(a0_4, a0_5, a0_6, a0_7);
    out0[2] = make_float4(a0_8, a0_9, a0_10, a0_11);
    out0[3] = make_float4(a0_12, a0_13, a0_14, a0_15);
    float4* out1 = reinterpret_cast<float4*>(C + (long long)gr1 * Nout + gc0);
    out1[0] = make_float4(a1_0, a1_1, a1_2, a1_3);
    out1[1] = make_float4(a1_4, a1_5, a1_6, a1_7);
    out1[2] = make_float4(a1_8, a1_9, a1_10, a1_11);
    out1[3] = make_float4(a1_12, a1_13, a1_14, a1_15);
}

torch::Tensor matmul_forward(torch::Tensor A, torch::Tensor B) {
    int M = (int)A.size(0);
    int Nout = (int)B.size(1);
    auto C = torch::empty({M, Nout}, A.options());
    dim3 grid(Nout / TILE_N, M / TILE_M);
    matmul_k32_kernel<<<grid, 256>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, Nout);
    return C;
}
'''

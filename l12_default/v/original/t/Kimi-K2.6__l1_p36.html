import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs RMS Normalization.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        """
        Initializes the RMSNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
            eps (float, optional): A small value added to the denominator to avoid division by zero. Defaults to 1e-5.
        """
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies RMS Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, *).

        Returns:
            torch.Tensor: Output tensor with RMS Normalization applied, same shape as input.
        """
        # Calculate the RMS along the feature dimension
        rms = torch.sqrt(torch.mean(x ** 2, dim=1, keepdim=True) + self.eps)

        # Normalize the input by dividing by the RMS
        return x / rms

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void rmsnorm_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int B, int C, int N, float eps)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * N;
    if (idx >= total) return;

    int b = idx / N;
    int s = idx % N;
    int base = b * C * N + s;

    float sum_sq = 0.0f;

    // Accumulate sum of squares across channels
    #pragma unroll 8
    for (int c = 0; c < C; ++c) {
        float v = input[base + c * N];
        sum_sq += v * v;
    }

    float inv_rms = rsqrtf(sum_sq / (float)C + eps);

    // Write normalized output
    #pragma unroll 8
    for (int c = 0; c < C; ++c) {
        output[base + c * N] = input[base + c * N] * inv_rms;
    }
}

torch::Tensor rmsnorm_forward(torch::Tensor x, float eps) {
    if (!x.is_contiguous()) {
        x = x.contiguous();
    }

    int B = x.size(0);
    int C = x.size(1);
    int N = x.numel() / (B * C);

    auto output = torch::empty_like(x);

    const int threads = 512;
    const int total = B * N;
    const int blocks = (total + threads - 1) / threads;

    rmsnorm_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        output.data_ptr<float>(),
        B, C, N, eps

    return output;
}

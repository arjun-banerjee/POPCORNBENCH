import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a GEMM, scaling, hardtanh, and GELU activation.
    """
    def __init__(self, in_features, out_features, scaling_factor, hardtanh_min, hardtanh_max):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
        self.hardtanh = nn.Hardtanh(min_val=hardtanh_min, max_val=hardtanh_max)
        self.gelu = nn.GELU()

    def forward(self, x):
        x = self.gemm(x)
        x = x * self.scaling_factor
        x = self.hardtanh(x)
        x = self.gelu(x)
        return x

__global__ void act2d(float* __restrict__ y, const float* __restrict__ bias,
                      int M, int N, float scale, float minv, float maxv) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (col < N) {
        long idx = (long)row * N + col;
        const float inv_sqrt2 = 0.70710678118654752440f;
        float v = (y[idx] + bias[col]) * scale;
        v = fminf(fmaxf(v, minv), maxv);
        y[idx] = 0.5f * v * (1.0f + erff(v * inv_sqrt2));
    }
}

torch::Tensor fused_linear_act(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                               double scale, double minv, double maxv) {
    int64_t M64 = x.size(0), K64 = x.size(1), N64 = weight.size(0);
    int M = (int)M64, N = (int)N64, K = (int)K64;
    auto y = torch::empty({M64, N64}, x.options());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cublasSetStream(handle, stream);

    float alpha = 1.0f, beta = 0.0f;
    // Treat row-major tensors as column-major transposes: y^T = weight * x^T.
    cublasStatus_t stat = cublasGemmEx(handle,

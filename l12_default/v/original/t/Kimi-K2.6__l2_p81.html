__forceinline__ __device__ float apply_ops(float v) {
    v = v / (1.0f + expf(-v));
    v *= 0.5f;
    v = fminf(fmaxf(v, -1.0f), 1.0f);
    v = tanhf(v);
    v = fminf(fmaxf(v, -1.0f), 1.0f);
    return v;
}

__global__ void fused_forward_kernel(float* __restrict__ x, int64_t n, int grid_size) {
    #pragma unroll
    for (int64_t idx = (int64_t)(blockIdx.x * blockDim.x + threadIdx.x) * 4;
         idx < n;
         idx += (int64_t)grid_size * blockDim.x * 4) {
        if (idx + 3 < n) {
            float4 v4 = reinterpret_cast<float4*>(x)[idx / 4];
            v4.x = apply_ops(v4.x);
            v4.y = apply_ops(v4.y);
            v4.z = apply_ops(v4.z);
            v4.w = apply_ops(v4.w);
            reinterpret_cast<float4*>(x)[idx / 4] = v4;
        } else if (idx < n) {
            for (int64_t j = idx; j < n; ++j) {

            }
        }
    }
}

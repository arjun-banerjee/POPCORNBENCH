import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a series of operations: Gemm, Subtract, GlobalAvgPool, LogSumExp, GELU, and ResidualAdd.
    """
    def __init__(self, in_features, out_features, bias=True):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features, bias=bias)
        self.subtract = nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        original_x = x.clone().detach()
        # Gemm
        x = self.gemm(x)

        # Subtract
        x = x - self.subtract

        # GlobalAvgPool
        x = torch.mean(x, dim=1, keepdim=True)

        # LogSumExp
        x = torch.logsumexp(x, dim=1, keepdim=True)

        # GELU
        x = torch.nn.functional.gelu(x)

        # ResidualAdd
        x = x + original_x

        return x

    for (int o = start; o < end; ++o) {
        s += w[(long long)o * I + col];
    }
    partial[(long long)rb * I + col] = s;
}

__global__ void reduce_colsum_kernel(const float* __restrict__ partial, float* __restrict__ colsum,
                                     int I, int nrb) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (col >= I) return;
    float s = 0.0f;
    #pragma unroll
    for (int rb = 0; rb < 32; ++rb) {
        if (rb < nrb) s += partial[(long long)rb * I + col];
    }
    colsum[col] = s;
}

__inline__ __device__ float warp_sum(float v) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, offset);
    }
    return v;
}

__global__ void reduce_const_kernel(const float* __restrict__ bias, const float* __restrict__ sub,
                                    float* __restrict__ csum, int O, int has_bias) {
    __shared__ float sh[8];
    int tid = threadIdx.x;
    float s = 0.0f;
    for (int i = tid; i < O; i += 256) {
        s += (has_bias ? bias[i] : 0.0f) - sub[i];
    }
    s = warp_sum(s);
    if ((tid & 31) == 0) sh[tid >> 5] = s;
    __syncthreads();
    if (tid < 32) {
        s = (tid < 8) ? sh[tid] : 0.0f;
        s = warp_sum(s);
        if (tid == 0) *csum = s;
    }
}

__device__ __forceinline__ float gelu_exact(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.70710678118654752440f));
}

__global__ void apply_kernel(const float* __restrict__ x, const float* __restrict__ colsum,
                             const float* __restrict__ csum, float* __restrict__ out,
                             int I, int O) {
    __shared__ float sh[8];
    int b = blockIdx.x;
    int tid = threadIdx.x;
    const float* row = x + (long long)b * I;
    float s = 0.0f;
    #pragma unroll 4
    for (int i = tid; i < I; i += 256) {
        s += row[i] * colsum[i];
    }
    s = warp_sum(s);
    if ((tid & 31) == 0) sh[tid >> 5] = s;
    __syncthreads();
    if (tid < 32) {
        s = (tid < 8) ? sh[tid] : 0.0f;
        s = warp_sum(s);
        if (tid == 0) {
            float m = (s + *csum) / (float)O;
            sh[0] = gelu_exact(m);
        }
    }
    __syncthreads();
    float g = sh[0];
    float* orow = out + (long long)b * I;
    #pragma unroll 4
    for (int i = tid; i < I; i += 256) {
        orow[i] = row[i] + g;
    }
}

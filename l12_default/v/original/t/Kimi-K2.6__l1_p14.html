from torch.utils.cpp_extension import load_inline

#include <torch/extension.h>

    // Since A and B are upper triangular, their product is also upper triangular.
    // Therefore torch.triu(torch.matmul(A, B)) is equivalent to at::mm(A, B).
    return at::mm(A, B);

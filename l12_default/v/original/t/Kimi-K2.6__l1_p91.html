import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor reverse_cumsum_forward(torch::Tensor input);
'''

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <int BLOCK_THREADS, int TILE_SIZE>
__global__ void reverse_cumsum_kernel(const float* __restrict__ input,
                                      float* __restrict__ output,
                                      int64_t batch,
                                      int64_t N) {
    int64_t row = blockIdx.x;
    if (row >= batch) return;

    const float* in_row = input + row * N;
    float* out_row = output + row * N;

    __shared__ float smem[TILE_SIZE];
    __shared__ float s_carry;

    if (threadIdx.x == 0) {
        s_carry = 0.0f;
    }
    __syncthreads();

    int num_tiles = (N + TILE_SIZE - 1) / TILE_SIZE;
    for (int tile = num_tiles - 1; tile >= 0; --tile) {
        int tile_start = tile * TILE_SIZE;
        int this_tile_size = (int)min((int64_t)TILE_SIZE, N - tile_start);

        // Coalesced load into shared memory
        for (int i = threadIdx.x; i < this_tile_size; i += BLOCK_THREADS) {
            smem[i] = in_row[tile_start + i];
        }
        __syncthreads();

        // Serial reverse cumulative sum within tile
        if (threadIdx.x == 0) {
            for (int i = this_tile_size - 2; i >= 0; --i) {
                smem[i] += smem[i + 1];
            }
        }
        __syncthreads();

        float carry = s_carry;

        // Coalesced write to output
        for (int i = threadIdx.x; i < this_tile_size; i += BLOCK_THREADS) {
            out_row[tile_start + i] = smem[i] + carry;
        }
        __syncthreads();

        if (threadIdx.x == 0) {
            s_carry = carry + smem[0];
        }
        __syncthreads();
    }
}

torch::Tensor reverse_cumsum_forward(torch::Tensor input) {
    int64_t batch = 1;
    for (int i = 0; i < input.dim() - 1; ++i) {
        batch *= input.size(i);
    }
    int64_t N = input.size(input.dim() - 1);
    auto output = torch::empty_like(input);

    const float* in_ptr = input.data_ptr<float>();
    float* out_ptr = output.data_ptr<float>();

    const int BLOCK_THREADS = 256;
    const int TILE_SIZE = BLOCK_THREADS * 16; // 4096

    reverse_cumsum_kernel<BLOCK_THREADS, TILE_SIZE><<<batch, BLOCK_THREADS>>>(
        in_ptr, out_ptr, batch, N);

    return output;
}
'''

class ModelNew(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim

    def forward(self, x):
        if x.dtype != torch.float32 or self.dim != x.dim() - 1 or not x.is_contiguous():
            return torch.cumsum(x.flip(self.dim), dim=self.dim).flip(self.dim)
        shape = x.shape
        batch = 1
        for i in range(self.dim):
            batch *= shape[i]
        N = shape[self.dim]
        x_2d = x.view(batch, N)
        out_2d = reverse_cumsum.reverse_cumsum_forward(x_2d)
        return out_2d.view(shape)

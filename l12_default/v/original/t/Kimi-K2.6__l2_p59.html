from torch.utils.cpp_extension import load_inline
import math

# C++ function declarations
cpp_source = """
torch::Tensor fused_swish_scale(torch::Tensor input, float scale);
"""

# CUDA kernel implementation
cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void fused_swish_scale_kernel(const float* __restrict__ input, 
                                          float* __restrict__ output,
                                          int numel, float scale) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Process 4 elements at a time using float4
    int idx4 = idx * 4;
    
    if (idx4 + 3 < numel) {
        float4 val = reinterpret_cast<const float4*>(input)[idx];
        
        val.x = val.x / (1.0f + expf(-val.x)) * scale;
        val.y = val.y / (1.0f + expf(-val.y)) * scale;
        val.z = val.z / (1.0f + expf(-val.z)) * scale;
        val.w = val.w / (1.0f + expf(-val.w)) * scale;
        
        reinterpret_cast<float4*>(output)[idx] = val;
    } else {
        // Tail processing
        for (int i = idx4; i < numel; i++) {
            float v = input[i];
            output[i] = v / (1.0f + expf(-v)) * scale;
        }
    }
}

__global__ void fused_swish_scale_kernel_aligned(const float* __restrict__ input, 
                                                  float* __restrict__ output,
                                                  int numel, float scale) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int idx4 = idx * 4;
    
    if (idx4 < numel) {
        float4 val = reinterpret_cast<const float4*>(input)[idx];
        
        val.x = val.x / (1.0f + expf(-val.x)) * scale;
        val.y = val.y / (1.0f + expf(-val.y)) * scale;
        val.z = val.z / (1.0f + expf(-val.z)) * scale;
        val.w = val.w / (1.0f + expf(-val.w)) * scale;
        
        reinterpret_cast<float4*>(output)[idx] = val;
    }
}

torch::Tensor fused_swish_scale(torch::Tensor input, float scale) {
    auto output = torch::empty_like(input);
    int numel = input.numel();
    
    // Check alignment for vectorized access
    bool input_aligned = (reinterpret_cast<uintptr_t>(input.data_ptr<float>()) % 16) == 0;
    bool output_aligned = (reinterpret_cast<uintptr_t>(output.data_ptr<float>()) % 16) == 0;
    
    if (input_aligned && output_aligned && (numel % 4 == 0)) {
        int threads = 256;
        int blocks = (numel / 4 + threads - 1) / threads;
        fused_swish_scale_kernel_aligned<<<blocks, threads>>>(
            input.data_ptr<float>(), output.data_ptr<float>(), numel, scale);
    } else {
        int threads = 256;
        int blocks = (numel + threads * 4 - 1) / (threads * 4);
        fused_swish_scale_kernel<<<blocks, threads>>>(
            input.data_ptr<float>(), output.data_ptr<float>(), numel, scale);
    }
    
    return output;
}
"""

# Compile inline
module = load_inline(
    name='fused_swish_scale',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_swish_scale'],
    extra_cuda_cflags=['-O3', '--use_fast_math'],
    verbose=False
)

class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(ModelNew, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.matmul(x)
        x = module.fused_swish_scale(x, self.scaling_factor)
        return x

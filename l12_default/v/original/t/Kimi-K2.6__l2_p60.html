    Model that performs a 3D transposed convolution, applies Swish activation,

    for (int offset = warpSize / 2; offset > 0; offset >>= 2) {
        local_sum += __shfl_down_sync(0xFFFFFFFF, local_sum, offset);
        local_sq += __shfl_down_sync(0xFFFFFFFF, local_sq, offset);
    }

    __shared__ float warp_sums[16];
    __shared__ float warp_sqs[16];
    __shared__ float s_mean;
    __shared__ float s_invstd;

    if (lane == 0) {
        warp_sums[warp_id] = local_sum;
        warp_sqs[warp_id] = local_sq;
    }
    __syncthreads();

    if (warp_id == 0) {
        float sum_val = (lane < 16) ? warp_sums[lane] : 0.0f;
        float sq_val  = (lane < 16) ? warp_sqs[lane]  : 0.0f;
        #pragma unroll
        for (int offset = warpSize / 2; offset > 0; offset >>= 2) {
            sum_val += __shfl_down_sync(0xFFFFFFFF, sum_val, offset);
            sq_val  += __shfl_down_sync(0xFFFFFFFF, sq_val, offset);
        }
        if (lane == 0) {
            float mean_val = sum_val / float(count);
            float var = sq_val / float(count) - mean_val * mean_val;
            s_mean = mean_val;
            s_invstd = rsqrtf(var + eps);
        }
    }
    __syncthreads();

    for (int idx = threadIdx.x; idx < count; idx += blockDim.x) {
        float val = __ldg(x + base_offset + idx);
        float sig = 1.0f / (1.0f + expf(-val));
        float swish = val * sig;
        float norm = (swish - s_mean) * s_invstd;
        int c_local = idx / DHW;
        int c = c_start + c_local;
        norm = norm * gamma[c] + beta[c];
        float tmp = norm + 3.0f;
        tmp = fminf(fmaxf(tmp, 0.0f), 6.0f);
        norm = norm * tmp / 6.0f;
        out[base_offset + idx] = norm;
    }
}

torch::Tensor fused_swish_gn_hardswish(torch::Tensor x,
                                       torch::Tensor gamma,
                                       torch::Tensor beta,
                                       int num_groups,
                                       float eps)
{
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "x must be float32");

    int N = x.size(0);
    int C = x.size(1);
    int D = x.size(2);
    int H = x.size(3);
    int W = x.size(4);
    int total_groups = N * num_groups;

    auto out = torch::empty_like(x);

    fused_kernel<<<total_groups, THREADS>>>(

        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, D, H, W, num_groups, eps);

    return out;
}

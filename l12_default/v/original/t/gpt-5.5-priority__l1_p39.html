import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs L2 normalization.
    """
    def __init__(self):
        """
        Initializes the L2Norm layer.

        Args:
            dim (int): Dimension along which to normalize.
        """
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies L2 normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, dim, *).

        Returns:
            torch.Tensor: Output tensor with L2 normalization applied, same shape as input.
        """
        return x / torch.norm(x, p=2, dim=1, keepdim=True)

__inline__ __device__ float warp_sum(float v) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) v += __shfl_down_sync(0xffffffff, v, offset);
    return v;
}

template<int BLOCK>
__global__ void l2norm_inplace_kernel(float* __restrict__ x, long long rows, long long cols) {
    int row = (int)blockIdx.x;
    float* __restrict__ xp = x + ((long long)row) * cols;
    int tid = threadIdx.x;
    float sum = 0.0f;
    long long c = tid;
    const long long step = (long long)BLOCK * 4;
    for (; c + (long long)3 * BLOCK < cols; c += step) {
        float v0 = xp[c];
        float v1 = xp[c + BLOCK];
        float v2 = xp[c + (long long)2 * BLOCK];
        float v3 = xp[c + (long long)3 * BLOCK];

        sum = fmaf(v2, v2, sum);
        sum = fmaf(v3, v3, sum);

        sum = fmaf(v, v, sum);

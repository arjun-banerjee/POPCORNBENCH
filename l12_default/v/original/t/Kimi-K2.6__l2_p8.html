class Model(nn.Module):
    """
    Model that performs a 3D convolution, divides by a constant, applies max pooling,
    global average pooling, adds a bias term, and sums along a specific dimension.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = divisor
        self.max_pool = nn.MaxPool3d(pool_size)
        self.global_avg_pool = nn.AdaptiveAvgPool3d((1, 1, 1))

        x = self.conv(x)
        x = x / self.divisor
        x = self.max_pool(x)
        x = self.global_avg_pool(x)
        x = x + self.bias
        x = torch.sum(x, dim=self.sum_dim)
        return x

// Kernel that uses one thread per output spatial position (max pool window)

__global__ void spatial_reduce_kernel(const float* __restrict__ conv_out, float* __restrict__ M,
                                      int B, int C, int D, int H, int W,
                                      int pD, int pH, int pW) {
    // Grid: one block per (B, C) channel
    // Block: 128 threads
    int bc = blockIdx.x;
    int b = bc / C;
    int c = bc % C;
    int tid = threadIdx.x;
    int warp_id = tid >> 5;
    int lane = tid & 31;
    
    int total_windows = pD * pH * pW;
    int64_t base_bc = ((int64_t)b * C + c) * D * H * W;
    float local_sum = 0.0f;
    
    // Each thread processes windows with stride = blockDim.x (128)
    for (int idx = tid; idx < total_windows; idx += blockDim.x) {
        int w = idx % pW;
        int tmp = idx / pW;
        int h = tmp % pH;
        int d = tmp / pH;
        
        int od0 = d << 1;
        int oh0 = h << 1;
        int ow0 = w << 1;
        
        float max_val = -1e30f;
        #pragma unroll
        for (int di = 0; di < 2; di++) {
            int64_t base_d = base_bc + (int64_t)(od0 + di) * H * W;
            #pragma unroll
            for (int dh = 0; dh < 2; dh++) {
                int64_t base_h = base_d + (int64_t)(oh0 + dh) * W;
                float v0 = conv_out[base_h + ow0];
                float v1 = conv_out[base_h + ow0 + 1];
                max_val = fmaxf(max_val, v0);
                max_val = fmaxf(max_val, v1);
            }
        }
        local_sum += max_val;
    }
    
    // Warp-level sum reduction using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    
    // Store warp sum to shared memory
    __shared__ float warp_sums[4]; // 128 threads = 4 warps
    if (lane == 0) {
        warp_sums[warp_id] = local_sum;

    __syncthreads();
    
    // Final reduction by warp 0
    if (warp_id == 0) {
        local_sum = (lane < 4) ? warp_sums[lane] : 0.0f;
        #pragma unroll
        for (int offset = 2; offset > 0; offset >>= 1) {
            local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
        }
        if (lane == 0) {

        }
    }

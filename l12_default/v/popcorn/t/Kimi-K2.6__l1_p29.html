import torch.nn as nn
from torch.utils.cpp_extension import load_inline

#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>
#include <limits>

template <int R>
__global__ void logsumexp_kernel_templated(
    const float* __restrict__ input,
    float* __restrict__ output,
    int B, int D
) {
    int tid = threadIdx.x;
    int block = blockIdx.x;
    int num_blocks = gridDim.x;
    
    int d_per_thread = (D + blockDim.x - 1) / blockDim.x;
    int d_start = tid * d_per_thread;
    int d_end = min(d_start + d_per_thread, D);
    
    for (int b = block; b < B; b += num_blocks) {
        #pragma unroll
        for (int d = d_start; d < d_end; ++d) {
            int base = b * D + d;
            int stride_r = B * D;
            
            float max_val = -std::numeric_limits<float>::infinity();
            
            #pragma unroll
            for (int r = 0; r < R; ++r) {
                float val = input[r * stride_r + base];
                max_val = fmaxf(max_val, val);
            }
            
            float sum_exp = 0.0f;
            #pragma unroll
            for (int r = 0; r < R; ++r) {
                float val = input[r * stride_r + base];
                sum_exp += expf(val - max_val);
            }
            
            output[base] = logf(sum_exp) + max_val;
        }
    }
}

torch::Tensor logsumexp_cuda(torch::Tensor input) {
    int B = input.size(1);
    int D = input.size(2);

}

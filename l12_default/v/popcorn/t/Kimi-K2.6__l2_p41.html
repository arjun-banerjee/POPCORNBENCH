import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Evidence Lower Bound (ELBO) computation for a Variational Autoencoder.
    Combines a reconstruction term (binary cross-entropy) with the
    KL divergence between the approximate posterior q(z|x) = N(mu, sigma^2)
    and the prior p(z) = N(0, I).  The core loss function in VAE training.
    """

    def __init__(self, input_dim, hidden_dim, latent_dim):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
        )
        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)

        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim),
            nn.Sigmoid(),
        )

    def forward(self, x: torch.Tensor) -> tuple:
        """
        Args:
            x: (B, input_dim) – input data in [0, 1]
        Returns:
            (elbo, recon_loss, kl_div):
                elbo:       (B,)
                recon_loss: (B,)
                kl_div:     (B,)
        """
        h = self.encoder(x)
        mu = self.fc_mu(h)
        logvar = self.fc_logvar(h)

        std = torch.exp(0.5 * logvar)
        z = mu + std * torch.randn_like(std)

        x_recon = self.decoder(z)
        recon = F.binary_cross_entropy(x_recon, x, reduction="none").sum(-1)
        kl = -0.5 * (1 + logvar - mu.pow(2) - logvar.exp()).sum(-1)
        elbo = -(recon + kl)
        return elbo, recon, kl

        sm_reduce[tid >> 5] = kl_val;
    __syncthreads();
    
    if (tid < 32) {
        float sum = (tid < (nt >> 5)) ? sm_reduce[tid] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2)
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        if (tid == 0)
            kl[b] = -0.5f * sum;
    }
    __syncthreads();
    
    if (tid < D_hidden) {
        float acc = b_dec1[tid];
        #pragma unroll
        for (int k = 0; k < D_latent; k++)
            acc += W_dec1[tid * D_latent + k] * sm_z[k];
        acc = fmaxf(acc, 0.0f);
        sm_h[tid] = acc;
    }
    __syncthreads();
    
    float recon_val = 0.0f;
    for (int out = tid; out < D_input; out += nt) {
        float acc = b_dec2[out];
        #pragma unroll
        for (int k = 0; k < D_hidden; k++)
            acc += W_dec2[out * D_hidden + k] * sm_h[k];
        float sig = 1.0f / (1.0f + expf(-acc));
        sig = fmaxf(sig, 1e-7f);
        sig = fminf(sig, 1.0f - 1e-7f);
        float xv = sm_x[out];
        recon_val += -(xv * logf(sig) + (1.0f - xv) * logf(1.0f - sig));
    }
    
    for (int offset = 16; offset > 0; offset /= 2)
        recon_val += __shfl_down_sync(0xffffffff, recon_val, offset);
    if ((tid & 31) == 0)
        sm_reduce[tid >> 5] = recon_val;
    __syncthreads();
    
    if (tid < 32) {
        float sum = (tid < (nt >> 5)) ? sm_reduce[tid] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2)
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        if (tid == 0) {
            recon[b] = sum;
            elbo[b] = -(sum + kl[b]);
        }
    }
}

class ModelNew(nn.Module):
    def __init__(self, input_dim, hidden_dim, latent_dim):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
        )
        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim),
            nn.Sigmoid(),
        )
        self.fused = vae_ext.vae_fused_forward
        
    def forward(self, x):
        h = self.encoder(x)
        mu = self.fc_mu(h)
        logvar = self.fc_logvar(h)
        std = torch.exp(0.5 * logvar)
        eps = torch.randn_like(std)
        
        B = x.shape[0]
        elbo = torch.empty(B, device=x.device, dtype=torch.float32)
        recon = torch.empty(B, device=x.device, dtype=torch.float32)
        kl = torch.empty(B, device=x.device, dtype=torch.float32)
        
        self.fused(
            mu.contiguous(), logvar.contiguous(), eps.contiguous(), x.contiguous(),
            self.decoder[0].weight.contiguous(), self.decoder[0].bias.contiguous(),
            self.decoder[2].weight.contiguous(), self.decoder[2].bias.contiguous(),
            elbo, recon, kl
        )
        
        return elbo, recon, kl

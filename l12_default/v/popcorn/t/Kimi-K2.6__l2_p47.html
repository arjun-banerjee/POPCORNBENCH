#include <torch/extension.h>
#include <cuda_runtime.h>

constexpr int R_CONST = 4;
constexpr int H_CONST = 32;

template <int R_VAL, int H_VAL, typename scalar_t>
__global__ void fused_gather_linear_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ weight_t,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ out,
    int B, int K)
{
    int b = blockIdx.y * blockDim.y + threadIdx.y;
    int k = blockIdx.x * blockDim.x + threadIdx.x;

    if (b < B && k < K) {
        scalar_t acc = bias[k];
        #pragma unroll
        for (int r = 0; r < R_VAL; ++r) {
            int x_base = (r * B + b) * H_VAL;
            int w_base = r * H_VAL * K;
            #pragma unroll
            for (int h = 0; h < H_VAL; ++h) {
                acc += x[x_base + h] * weight_t[w_base + h * K + k];
            }
        }
        out[b * K + k] = acc;
    }
}

torch::Tensor fused_gather_linear_forward(torch::Tensor x,
                                          torch::Tensor weight_t,
                                          torch::Tensor bias) {
    int R = x.size(0);
    int B = x.size(1);
    int H = x.size(2);
    int K = bias.size(0);
    TORCH_CHECK(weight_t.size(0) == R * H, "weight_t in_features mismatch");
    TORCH_CHECK(weight_t.size(1) == K, "weight_t out_features mismatch");

    x = x.contiguous();
    weight_t = weight_t.contiguous();
    bias = bias.contiguous();

    auto out = torch::empty({B, K}, x.options());

    dim3 block(32, 4);
    dim3 grid((K + block.x - 1) / block.x, (B + block.y - 1) / block.y);

    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "fused_gather_linear", [&] {
        fused_gather_linear_kernel<R_CONST, H_CONST, scalar_t><<<grid, block>>>(
            x.data_ptr<scalar_t>(),
            weight_t.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            out.data_ptr<scalar_t>(),
            B, K
        );
    });

    return out;
}

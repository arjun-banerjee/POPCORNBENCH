import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float gelu(float x) {
    const float kAlpha = 0.70710678118654752440f;
    return x * 0.5f * (1.0f + erff(x * kAlpha));
}

__global__ void gelu_sum_forward_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int R,
    int B,
    int D) {
    int total = B * D;
    for (int idx = blockIdx.x * blockDim.x + threadIdx.x;
         idx < total;
         idx += blockDim.x * gridDim.x) {
        int b = idx / D;
        int d = idx - b * D;
        float acc = 0.0f;
        #pragma unroll 4
        for (int r = 0; r < R; ++r) {
            acc += gelu(input[((r * B) + b) * D + d]);
        }
        output[idx] = acc;
    }
}

torch::Tensor gelu_sum_forward_wrap(torch::Tensor x) {
    int R = x.size(0);
    int B = x.size(1);
    int D = x.size(2);

    const int blocks = 64;
    gelu_sum_forward_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        out.data_ptr<float>(),

}

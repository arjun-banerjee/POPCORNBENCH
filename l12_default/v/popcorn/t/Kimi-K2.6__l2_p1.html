class Model(nn.Module):
    """
    Low-rank expansion of compressed KV activations into multi-head K/V tensors.

    This isolates the MLA-style compressed-cache pattern: a hidden activation is
    projected into a low-rank latent space and then expanded back into separate
    key and value tensors.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        hidden: torch.Tensor,
        w_down: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, model_dim = hidden.shape
        rank = w_down.shape[1]
        kv_heads = 4
        head_dim = w_up_k.shape[1] // kv_heads

        latent = hidden.reshape(batch_size * seq_len, model_dim) @ w_down
        k = latent @ w_up_k
        v = latent @ w_up_v
        k = k.view(batch_size, seq_len, kv_heads, head_dim)
        v = v.view(batch_size, seq_len, kv_heads, head_dim)
        return torch.stack((k, v), dim=2)

#include <torch/extension.h>

__global__ void mla_fused_kernel(
    const float* __restrict__ hidden,
    const float* __restrict__ w_down,
    const float* __restrict__ w_up_k,
    const float* __restrict__ w_up_v,
    float* __restrict__ out,
    int batch_size,
    int seq_len,
    int model_dim,
    int rank,
    int kv_heads,
    int head_dim
) {
    int bs = blockIdx.x;
    int tid = threadIdx.x;
    int num_threads = blockDim.x;

    extern __shared__ float smem[];
    float* sh_hidden = smem;
    float* latent = smem + model_dim;

    for (int md = tid; md < model_dim; md += num_threads) {
        sh_hidden[md] = hidden[bs * model_dim + md];
    }
    __syncthreads();

    for (int r = tid; r < rank; r += num_threads) {
        double sum = 0.0;
        for (int md = 0; md < model_dim; md++) {
            sum += (double)sh_hidden[md] * (double)w_down[md * rank + r];
        }
        latent[r] = (float)sum;
    }
    __syncthreads();

    int num_outputs = kv_heads * head_dim;

        double v_sum = 0.0;

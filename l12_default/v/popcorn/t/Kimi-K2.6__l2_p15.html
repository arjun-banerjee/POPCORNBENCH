import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Causal grouped-query attention with KV-head expansion.

    Query heads outnumber KV heads. Keys and values are expanded by repeating
    each KV head across its assigned query-head group before causal attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_q_heads, head_dim = q.shape
        num_kv_heads = k.shape[2]
        group_size = num_q_heads // num_kv_heads

        expanded_k = k.repeat_interleave(group_size, dim=2)
        expanded_v = v.repeat_interleave(group_size, dim=2)

        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_q_heads):
                for t in range(seq_len):
                    q_t = q[b, t, h]
                    k_hist = expanded_k[b, : t + 1, h]
                    v_hist = expanded_v[b, : t + 1, h]
                    scores = (k_hist * q_t.unsqueeze(0)).sum(dim=-1) * scale
                    attn = torch.softmax(scores, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * v_hist).sum(dim=0)

        return out

            int k_idx = ((b * T + s) * H_kv + kv_h) * HD + d;
            int v_idx = ((b * T + s) * H_kv + kv_h) * HD + d;
            k_smem[s_local * HD + d] = k[k_idx];
            v_smem[s_local * HD + d] = v[v_idx];
        }
        __syncthreads();
        
        int my_s_end = valid ? min(s_end, t + 1) : s_start;
        
        for (int s = s_start; s < my_s_end; ++s) {
            int s_local = s - s_start;
            float k_val = k_smem[s_local * HD + lane];
            float dot = q_val * k_val;
            
            #pragma unroll
            for (int offset = 16; offset > 0; offset /= 2) {
                dot += __shfl_xor_sync(0xffffffff, dot, offset);
            }
            float score = dot * scale;
            
            float v_val = v_smem[s_local * HD + lane];
            
            if (score > max_score) {
                float diff = expf(max_score - score);
                sum_exp = sum_exp * diff + 1.0f;
                out_acc = out_acc * diff + v_val;
                max_score = score;
            } else {
                float w = expf(score - max_score);
                sum_exp += w;
                out_acc += w * v_val;
            }
        }
        __syncthreads();
    }
    
    if (valid) {
        out_acc /= sum_exp;
        int out_idx = ((b * T + t) * H_q + h) * HD + lane;
        out[out_idx] = out_acc;
    }
}

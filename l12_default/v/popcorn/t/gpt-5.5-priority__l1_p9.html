import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Performs a parallel associative scan (linear recurrence) over a 1D sequence.
    Given coefficient a and input b at each timestep, computes:
        h_0 = b_0
        h_t = a_t * h_{t-1} + b_t  for t > 0
    This is the core parallelizable primitive underlying Mamba's selective scan and
    other SSM architectures. Sequential in naive PyTorch but can be parallelized
    via the associative scan algorithm in O(L log L) span.

    Args:
        d_model (int): Number of independent recurrence channels.
    """
    def __init__(self, d_model: int):
        super(Model, self).__init__()
        self.d_model = d_model

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        B, L, D = a.shape
        h = torch.zeros(B, D, device=a.device, dtype=a.dtype)
        outputs = []
        for t in range(L):
            h = a[:, t] * h + b[:, t]
            outputs.append(h)
        return torch.stack(outputs, dim=1)

batch_size = 32
d_model = 256
seq_len = 4096

def get_inputs():
    a = torch.sigmoid(torch.randn(batch_size, seq_len, d_model))
    b = torch.randn(batch_size, seq_len, d_model)
    return [a, b]

def get_init_inputs():
    return [d_model]

    float p = 1.0f;
    int idx = (batch * L + start) * D + d;

    #pragma unroll 4
    for (int t = start; t < end; ++t, idx += D) {
        float av = a[idx];
        h = av * h + b[idx];
        p *= av;
        out[idx] = h;
    }

    int sidx = (batch * C + chunk) * D + d;
    prod_s[sidx] = p;
    end_s[sidx] = h;
}

__global__ void chunk_prefix_kernel(const float* __restrict__ prod_s,
                                    const float* __restrict__ end_s,
                                    float* __restrict__ init_s,
                                    int C, int D) {
    int batch = blockIdx.x;
    int d = threadIdx.x;
    if (d >= D) return;

    float h = 0.0f;
    #pragma unroll
    for (int c = 0; c < 128; ++c) {
        if (c >= C) break;
        int idx = (batch * C + c) * D + d;
        init_s[idx] = h;
        h = prod_s[idx] * h + end_s[idx];
    }
}

__global__ void adjust_kernel(const float* __restrict__ a,
                              float* __restrict__ out,
                              const float* __restrict__ init_s,
                              int L, int D, int C) {
    int batch = blockIdx.x;
    int chunk = blockIdx.y;
    int d = threadIdx.x;
    if (d >= D) return;

    float h0 = init_s[(batch * C + chunk) * D + d];
    if (h0 == 0.0f) return;

    int start = chunk * CHUNK_SIZE;
    int end = min(start + CHUNK_SIZE, L);
    float p = 1.0f;
    int idx = (batch * L + start) * D + d;

    #pragma unroll 4
    for (int t = start; t < end; ++t, idx += D) {
        p *= a[idx];
        out[idx] = p * h0 + out[idx];
    }
}

torch::Tensor scan_forward(torch::Tensor a, torch::Tensor b) {
    int B = (int)a.size(0);
    int L = (int)a.size(1);
    int D = (int)a.size(2);
    int C = (L + CHUNK_SIZE - 1) / CHUNK_SIZE;

    auto out = torch::empty_like(b);
    auto prod_s = torch::empty({B, C, D}, b.options());
    auto end_s = torch::empty({B, C, D}, b.options());
    auto init_s = torch::empty({B, C, D}, b.options());

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    int threads = 256;
    dim3 grid(B, C);

    local_scan_kernel<<<grid, threads, 0, stream>>>(
        a.data_ptr<float>(), b.data_ptr<float>(), out.data_ptr<float>(),
        prod_s.data_ptr<float>(), end_s.data_ptr<float>(), L, D, C);

    chunk_prefix_kernel<<<B, threads, 0, stream>>>(
        prod_s.data_ptr<float>(), end_s.data_ptr<float>(), init_s.data_ptr<float>(), C, D);

    adjust_kernel<<<grid, threads, 0, stream>>>(
        a.data_ptr<float>(), out.data_ptr<float>(), init_s.data_ptr<float>(), L, D, C);

    return out;
}

__global__ void dwconv1d_generic_kernel(const float* __restrict__ x,
                                        const float* __restrict__ w,
                                        const float* __restrict__ bias,
                                        float* __restrict__ y,
                                        int B, int C, int L, int K, int outL,
                                        int stride, int padding, int has_bias) {
    int linear = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * C * outL;
    if (linear >= total) return;
    int pos = linear % outL;
    int tmp = linear / outL;
    int c = tmp % C;
    int b = tmp / C;
    int xbase = (b * C + c) * L;
    float acc = has_bias ? bias[c] : 0.0f;
    int in0 = pos * stride - padding;
    int wbase = c * K;
    for (int k = 0; k < K; ++k) {
        int ix = in0 + k;
        if ((unsigned)ix < (unsigned)L) acc += x[xbase + ix] * w[wbase + k];
    }
    y[linear] = acc;
}

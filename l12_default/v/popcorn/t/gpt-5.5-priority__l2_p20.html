import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Scalar-gated delta-rule linear attention recurrence.

    A recurrent state matrix is decayed by a scalar gate per head, updated by a
    rank-1 key/value outer product, then read out by a query vector.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        gate: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        state = torch.zeros(batch_size, num_heads, head_dim, head_dim, dtype=q.dtype, device=q.device)
        outputs = []

        for t in range(seq_len):
            gate_t = gate[:, t].unsqueeze(-1).unsqueeze(-1)
            beta_t = beta[:, t].unsqueeze(-1).unsqueeze(-1)
            kv_outer = k[:, t].unsqueeze(-1) * v[:, t].unsqueeze(-2)
            state = gate_t * state + beta_t * kv_outer
            y_t = (q[:, t].unsqueeze(-1) * state).sum(dim=-2)
            outputs.append(y_t)

        return torch.stack(outputs, dim=1)


batch_size = 2
seq_len = 128
num_heads = 4
head_dim = 24


def get_inputs():
    q = torch.randn(batch_size, seq_len, num_heads, head_dim, dtype=torch.float32)

    beta = torch.sigmoid(torch.randn(batch_size, seq_len, num_heads, dtype=torch.float32))
    return [q, k, v, gate, beta]

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor delta_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor gate, torch::Tensor beta);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Specialized for the benchmark shape: B=2, T=128, H=4, D=24, fp32 contiguous.
// One warp computes one output channel (b,h,j). Each active lane owns one row
// of the recurrent state column, so the recurrence stays in registers.
__global__ __launch_bounds__(32, 8) void delta_rule_kernel_2_128_4_24(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    const float* __restrict__ gate,
    const float* __restrict__ beta,
    float* __restrict__ out)
{
    const int x = blockIdx.x;
    const int j = x % 24;
    const int bh = x / 24;
    const int h = bh & 3;
    const int b = bh >> 2;
    const int lane = threadIdx.x;

    int idx_i = b * 12288 + h * 24 + lane;
    int idx_j = b * 12288 + h * 24 + j;
    int idx_g = b * 512 + h;

    float s = 0.0f;
    const unsigned mask = 0xffffffffu;

    #pragma unroll 1
    for (int t = 0; t < 128; ++t) {
        const float g = gate[idx_g];
        const float be = beta[idx_g];
        const float vj = v[idx_j];
        float part = 0.0f;

        if (lane < 24) {
            s = g * s + be * k[idx_i] * vj;
            part = q[idx_i] * s;
        }

        part += __shfl_down_sync(mask, part, 16);
        part += __shfl_down_sync(mask, part, 8);
        part += __shfl_down_sync(mask, part, 4);
        part += __shfl_down_sync(mask, part, 2);
        part += __shfl_down_sync(mask, part, 1);

        if (lane == 0) out[idx_j] = part;

        idx_i += 96;
        idx_j += 96;
        idx_g += 4;
    }
}

torch::Tensor delta_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor gate, torch::Tensor beta) {
    auto out = torch::empty_like(q);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    delta_rule_kernel_2_128_4_24<<<192, 32, 0, stream>>>(
        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), gate.data_ptr<float>(), beta.data_ptr<float>(), out.data_ptr<float>());
    return out;
}
'''

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float gelu_exact_f(float v) {
    return 0.5f * v * (1.0f + erff(v * 0.70710678118654752440f));
}

__global__ void gelu_sum_kernel(const float* __restrict__ x,
                                float* __restrict__ y,
                                int n,
                                int rcount) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float acc = 0.0f;
        #pragma unroll
        for (int r = 0; r < 4; ++r) {
            if (r < rcount) {
                float v = x[r * n + idx];
                acc += gelu_exact_f(v);
            }
        }
        y[idx] = acc;
    }
}

torch::Tensor gelu_sum_forward(torch::Tensor x) {
    const int64_t R = x.size(0);
    const int64_t B = x.size(1);
    const int64_t D = x.size(2);
    auto y = torch::empty({B, D}, x.options());
    const int n = (int)(B * D);
    const int threads = 256;
    const int blocks = (n + threads - 1) / threads;
    gelu_sum_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), n, (int)R);
    return y;
}
"""

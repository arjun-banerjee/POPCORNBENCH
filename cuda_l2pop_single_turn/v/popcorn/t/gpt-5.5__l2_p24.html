import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Channel-wise gated delta attention recurrence.

    The recurrent state decays independently for each output channel, which
    makes the update closer to channel-gated delta attention variants used in
    fast long-context baselines.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        gate: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        state = torch.zeros(batch_size, num_heads, head_dim, head_dim, dtype=q.dtype, device=q.device)
        outputs = []

        for t in range(seq_len):
            gate_t = gate[:, t].unsqueeze(-2)
            kv_outer = k[:, t].unsqueeze(-1) * v[:, t].unsqueeze(-2)
            state = gate_t * state + kv_outer
            y_t = (q[:, t].unsqueeze(-1) * state).sum(dim=-2)
            outputs.append(y_t)

        return torch.stack(outputs, dim=1)

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void gated_delta_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    const float* __restrict__ gate,
    float* __restrict__ out,
    int B, int T, int H, int D)
{
    // One warp computes one independent recurrent output column (b,h,j).
    int lane = threadIdx.x;  // blockDim.x == 32
    int col = blockIdx.x;
    int j = col % D;
    int tmp = col / D;
    int h = tmp % H;
    int b = tmp / H;

    float state = 0.0f;  // state_i,j for i == lane (lanes >= D are inactive)
    unsigned mask = 0xffffffffu;

    for (int t = 0; t < T; ++t) {
        int base = ((b * T + t) * H + h) * D;

        float g = 0.0f;
        float vv = 0.0f;
        if (lane == 0) {
            g = gate[base + j];
            vv = v[base + j];
        }
        g = __shfl_sync(mask, g, 0);
        vv = __shfl_sync(mask, vv, 0);

        float prod = 0.0f;
        if (lane < D) {
            // Match PyTorch's separate multiply and add closely (avoid contracting to fma).
            float kv = __fmul_rn(k[base + lane], vv);
            float decayed = __fmul_rn(g, state);
            state = __fadd_rn(decayed, kv);
            prod = __fmul_rn(q[base + lane], state);
        }

        // Warp reduction over the 24 valid lanes.
        prod += __shfl_down_sync(mask, prod, 16);
        prod += __shfl_down_sync(mask, prod, 8);
        prod += __shfl_down_sync(mask, prod, 4);
        prod += __shfl_down_sync(mask, prod, 2);
        prod += __shfl_down_sync(mask, prod, 1);

        if (lane == 0) {
            out[base + j] = prod;
        }
    }
}

torch::Tensor gated_delta_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor gate) {
    const int B = (int)q.size(0);
    const int T = (int)q.size(1);
    const int H = (int)q.size(2);
    const int D = (int)q.size(3);
    auto out = torch::empty_like(q);

    dim3 block(32);
    dim3 grid(B * H * D);

        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), gate.data_ptr<float>(),
        out.data_ptr<float>(), B, T, H, D);
    return out;
}

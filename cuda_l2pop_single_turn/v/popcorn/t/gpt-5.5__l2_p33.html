def _quaternion_to_rotation_matrix(q: torch.Tensor) -> torch.Tensor:
    """
    Unit quaternion ``q = (w, x, y, z)`` (scalar-first) to rotation matrix ``R``,
    shape ``(B, 3, 3)``, with ``p' = p @ R.T`` for row vectors ``p`` (``N×3``).
    """
    q = F.normalize(q, dim=-1, eps=1e-8)
    w, x, y, z = q.unbind(dim=-1)

    xx = x * x
    yy = y * y
    zz = z * z
    xy = x * y
    xz = x * z
    yz = y * z
    wx = w * x
    wy = w * y
    wz = w * z

    m00 = 1.0 - 2.0 * (yy + zz)
    m01 = 2.0 * (xy - wz)
    m02 = 2.0 * (xz + wy)
    m10 = 2.0 * (xy + wz)
    m11 = 1.0 - 2.0 * (xx + zz)
    m12 = 2.0 * (yz - wx)
    m20 = 2.0 * (xz - wy)
    m21 = 2.0 * (yz + wx)
    m22 = 1.0 - 2.0 * (xx + yy)

    row0 = torch.stack((m00, m01, m02), dim=-1)
    row1 = torch.stack((m10, m11, m12), dim=-1)
    row2 = torch.stack((m20, m21, m22), dim=-1)
    return torch.stack((row0, row1, row2), dim=-2)


class Model(nn.Module):
    """
    **Fused vectorized pose transform**: rotation from unit quaternion ``q`` and
    translation ``t`` applied to a batch of 3D points.

    One forward fuses **quaternion → ``3×3``** (Hamilton, scalar-first ``w,x,y,z``)
    with **batched matmul** and **broadcast translation**—typical for LiDAR / camera
    extrinsics, hand-eye calibration, and world ↔ body frame changes in VLA stacks.

    ``points' = points @ R.T + t`` with ``points`` row-wise ``(B, N, 3)``.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        points: torch.Tensor,
        quat: torch.Tensor,
        trans: torch.Tensor,
    ) -> torch.Tensor:
        """
        Args:
            points: (B, N, 3) points in the **source** frame.
            quat: (B, 4) unit quaternion ``(w, x, y, z)`` mapping source → target rotation.
            trans: (B, 3) translation in the **target** frame (same convention as ``R``).
        Returns:
            (B, N, 3) points in the **target** frame.
        """
        B, N, _ = points.shape
        assert quat.shape == (B, 4) and trans.shape == (B, 3)

        R = _quaternion_to_rotation_matrix(quat)
        return torch.matmul(points, R.transpose(-1, -2)) + trans.unsqueeze(1)

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void quat_transform_kernel(const float* __restrict__ points,
                                      const float* __restrict__ quat,
                                      const float* __restrict__ trans,
                                      float* __restrict__ out,
                                      int B, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * N;
    if (idx >= total) return;

    int b = idx / N;
    int n = idx - b * N;

    const float* q = quat + b * 4;
    float w = q[0];
    float x = q[1];
    float y = q[2];
    float z = q[3];

    float ss = w*w + x*x + y*y + z*z;
    // torch.nn.functional.normalize clamps norm to eps=1e-8.  For ordinary fp32
    // inputs this is just rsqrt; keep the clamp for exact edge-case behavior.
    float inv = (ss > 1.0e-16f) ? rsqrtf(ss) : 1.0e8f;
    w *= inv; x *= inv; y *= inv; z *= inv;

    float xx = x * x;
    float yy = y * y;
    float zz = z * z;
    float xy = x * y;
    float xz = x * z;
    float yz = y * z;
    float wx = w * x;
    float wy = w * y;
    float wz = w * z;

    float m00 = 1.0f - 2.0f * (yy + zz);
    float m01 = 2.0f * (xy - wz);
    float m02 = 2.0f * (xz + wy);
    float m10 = 2.0f * (xy + wz);
    float m11 = 1.0f - 2.0f * (xx + zz);
    float m12 = 2.0f * (yz - wx);
    float m20 = 2.0f * (xz - wy);
    float m21 = 2.0f * (yz + wx);
    float m22 = 1.0f - 2.0f * (xx + yy);

    const float* p = points + (b * N + n) * 3;
    const float* t = trans + b * 3;
    float px = p[0];
    float py = p[1];
    float pz = p[2];

    float* o = out + (b * N + n) * 3;
    o[0] = px * m00 + py * m01 + pz * m02 + t[0];
    o[1] = px * m10 + py * m11 + pz * m12 + t[1];
    o[2] = px * m20 + py * m21 + pz * m22 + t[2];
}

torch::Tensor quat_transform_cuda(torch::Tensor points, torch::Tensor quat, torch::Tensor trans) {

}

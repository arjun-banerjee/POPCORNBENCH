import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Grounded top-2 MoE router.

    Router logits are biased by token-expert similarity against learned
    grounding embeddings, then normalized over the selected top-2 experts.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        token_hidden: torch.Tensor,
        router_logits: torch.Tensor,
        expert_ground: torch.Tensor,
        alpha: float,
    ) -> torch.Tensor:
        grounded = router_logits + alpha * (token_hidden @ expert_ground.t())
        top_vals, top_idx = torch.topk(grounded, k=2, dim=-1)
        top_weights = torch.softmax(top_vals, dim=-1)
        return torch.stack((top_idx.to(torch.float32), top_weights), dim=-1)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cmath>

#ifndef __CUDACC_RTC__
#endif

__global__ void grounded_top2_kernel_128x16(const float* __restrict__ token,
                                            const float* __restrict__ router,
                                            const float* __restrict__ expert,
                                            const float alpha,
                                            float* __restrict__ out,
                                            const int T) {
    constexpr int WARPS_PER_BLOCK = 8;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5;
    const int t = blockIdx.x * WARPS_PER_BLOCK + warp;
    if (t >= T) return;

    const float* __restrict__ tok = token + ((size_t)t) * 128;

    // Each lane owns four hidden dimensions.  Load once and reuse for all 16 experts.
    const float h0 = tok[lane];
    const float h1 = tok[lane + 32];
    const float h2 = tok[lane + 64];
    const float h3 = tok[lane + 96];

    float best1 = -CUDART_INF_F;
    float best2 = -CUDART_INF_F;
    int idx1 = -1;
    int idx2 = -1;
    const unsigned mask = 0xffffffffu;

    #pragma unroll
    for (int e = 0; e < 16; ++e) {
        const float* __restrict__ eg = expert + e * 128;
        float s = h0 * eg[lane]
                + h1 * eg[lane + 32]
                + h2 * eg[lane + 64]
                + h3 * eg[lane + 96];

        // Warp reduction over the 32 lanes.
        s += __shfl_down_sync(mask, s, 16);
        s += __shfl_down_sync(mask, s, 8);
        s += __shfl_down_sync(mask, s, 4);
        s += __shfl_down_sync(mask, s, 2);
        s += __shfl_down_sync(mask, s, 1);

        if (lane == 0) {
            const float v = router[((size_t)t) * 16 + e] + alpha * s;
            if (v > best1) {
                best2 = best1; idx2 = idx1;
                best1 = v;     idx1 = e;
            } else if (v > best2) {
                best2 = v;     idx2 = e;
            }
        }
    }

    if (lane == 0) {
        const float r = expf(best2 - best1);
        const float inv = 1.0f / (1.0f + r);
        const size_t base = ((size_t)t) * 4;
        out[base + 0] = (float)idx1;
        out[base + 1] = inv;
        out[base + 2] = (float)idx2;
        out[base + 3] = r * inv;
    }
}

// Generic fallback kernel for the same operation.  It is still custom CUDA, not PyTorch;
// the fixed-shape fast path above is used for the benchmark dimensions.
__global__ void grounded_top2_kernel_generic(const float* __restrict__ token,
                                             const float* __restrict__ router,
                                             const float* __restrict__ expert,
                                             const float alpha,
                                             float* __restrict__ out,
                                             const int T, const int H, const int E) {
    const int t = blockIdx.x;
    const int lane = threadIdx.x & 31;
    if (t >= T) return;
    float best1 = -CUDART_INF_F, best2 = -CUDART_INF_F;
    int idx1 = -1, idx2 = -1;
    const unsigned mask = 0xffffffffu;

    for (int e = 0; e < E; ++e) {
        float s = 0.0f;
        for (int h = lane; h < H; h += 32) {
            s += token[(size_t)t * H + h] * expert[(size_t)e * H + h];
        }
        for (int off = 16; off > 0; off >>= 1) s += __shfl_down_sync(mask, s, off);
        if (lane == 0) {
            const float v = router[(size_t)t * E + e] + alpha * s;
            if (v > best1) { best2 = best1; idx2 = idx1; best1 = v; idx1 = e; }
            else if (v > best2) { best2 = v; idx2 = e; }
        }
    }
    if (lane == 0) {
        const float r = expf(best2 - best1);
        const float inv = 1.0f / (1.0f + r);
        const size_t base = (size_t)t * 4;
        out[base + 0] = (float)idx1;
        out[base + 1] = inv;
        out[base + 2] = (float)idx2;
        out[base + 3] = r * inv;
    }
}

torch::Tensor grounded_top2_forward(torch::Tensor token_hidden,
                                    torch::Tensor router_logits,
                                    torch::Tensor expert_ground,
                                    double alpha) {
    const int64_t T64 = token_hidden.size(0);
    const int64_t H64 = token_hidden.size(1);
    const int64_t E64 = expert_ground.size(0);
    auto out = torch::empty({T64, 2, 2}, token_hidden.options());

    const c10::cuda::CUDAGuard device_guard(token_hidden.device());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const int T = (int)T64;
    const float a = (float)alpha;

    if (H64 == 128 && E64 == 16) {
        constexpr int WARPS_PER_BLOCK = 8;
        dim3 block(32 * WARPS_PER_BLOCK);
        dim3 grid((T + WARPS_PER_BLOCK - 1) / WARPS_PER_BLOCK);
        grounded_top2_kernel_128x16<<<grid, block, 0, stream>>>(
            token_hidden.data_ptr<float>(), router_logits.data_ptr<float>(),
            expert_ground.data_ptr<float>(), a, out.data_ptr<float>(), T);

        dim3 grid(T);

            token_hidden.data_ptr<float>(), router_logits.data_ptr<float>(),
            expert_ground.data_ptr<float>(), a, out.data_ptr<float>(), T, (int)H64, (int)E64);
    }
    return out;
}

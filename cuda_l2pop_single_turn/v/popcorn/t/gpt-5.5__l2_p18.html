# Build a tiny CUDA extension for y = x * scale.  The benchmark input is small
# (8 x 64), so the host wrapper does only one allocation and one kernel launch.
_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor scale_forward(torch::Tensor x, torch::Tensor scale);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void scale_kernel_f32(const float* __restrict__ x,
                                 const float* __restrict__ s,
                                 float* __restrict__ y,
                                 int64_t n) {
    float a = s[0];
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) y[idx] = x[idx] * a;
}

// Generic contiguous strided copy*scale kernel for the expected fp32 path.
torch::Tensor scale_forward(torch::Tensor x, torch::Tensor scale) {
    auto y = torch::empty_like(x);
    int64_t n = x.numel();
    if (n == 0) return y;
    const int threads = 256;
    const int blocks = (int)((n + threads - 1) / threads);
    scale_kernel_f32<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), scale.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}
'''

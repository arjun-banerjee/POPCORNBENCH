import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Contact map prediction head.  Takes residue-pair embeddings and
    predicts a symmetric binary contact probability matrix (residue i
    within 8 Å of residue j).  Uses a small MLP on the symmetrised pair
    representation followed by a sigmoid.  Common final stage in protein
    structure prediction pipelines.
    """

    def __init__(self, pair_dim, hidden_dim):
        super().__init__()
        self.layer_norm = nn.LayerNorm(pair_dim)
        self.mlp = nn.Sequential(
            nn.Linear(pair_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
        )

    def forward(self, pair_repr: torch.Tensor) -> torch.Tensor:
        """
        Args:
            pair_repr: (B, N, N, pair_dim)
        Returns:
            contact_probs: (B, N, N) – symmetric contact probability
        """
        x = self.layer_norm(pair_repr)
        # Symmetrise
        x = (x + x.permute(0, 2, 1, 3)) / 2.0
        logits = self.mlp(x).squeeze(-1)  # (B, N, N)
        return torch.sigmoid(logits)

#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef MAX_D
#define MAX_D 128
#endif
#ifndef MAX_H
#define MAX_H 64
#endif

__global__ void contact_kernel(const float* __restrict__ x,
                               const float* __restrict__ ln_w,
                               const float* __restrict__ ln_b,
                               const float* __restrict__ w1,
                               const float* __restrict__ b1,
                               const float* __restrict__ w2,
                               const float* __restrict__ b2,
                               const float* __restrict__ w3,
                               const float* __restrict__ b3,
                               float* __restrict__ out,
                               int B, int N, int D, int H) {
    int i = blockIdx.y * blockDim.y + threadIdx.y;
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    int b = blockIdx.z;
    if (b >= B || i >= N || j >= N || i > j) return;

    const int64_t base1 = (((int64_t)b * N + i) * N + j) * D;
    const int64_t base2 = (((int64_t)b * N + j) * N + i) * D;

    float mean1 = 0.0f, mean2 = 0.0f;
    #pragma unroll
    for (int d = 0; d < 64; ++d) {
        if (d < D) {
            mean1 += x[base1 + d];
            mean2 += x[base2 + d];
        }
    }
    float invD = 1.0f / (float)D;
    mean1 *= invD;
    mean2 *= invD;

    float var1 = 0.0f, var2 = 0.0f;
    #pragma unroll
    for (int d = 0; d < 64; ++d) {
        if (d < D) {
            float a = x[base1 + d] - mean1;
            float c = x[base2 + d] - mean2;
            var1 += a * a;
            var2 += c * c;
        }
    }
    float rstd1 = rsqrtf(var1 * invD + 1.0e-5f);
    float rstd2 = rsqrtf(var2 * invD + 1.0e-5f);

    float h1[MAX_H];
    float h2[MAX_H];

    // Linear(pair_dim -> hidden_dim) + ReLU.  D=64, H=32 for the benchmark.
    #pragma unroll
    for (int h = 0; h < 32; ++h) {
        if (h < H) {
            float acc = b1[h];
            #pragma unroll
            for (int d = 0; d < 64; ++d) {
                if (d < D) {
                    float n1 = (x[base1 + d] - mean1) * rstd1 * ln_w[d] + ln_b[d];
                    float n2 = (x[base2 + d] - mean2) * rstd2 * ln_w[d] + ln_b[d];
                    float xs = 0.5f * (n1 + n2);
                    acc += w1[(int64_t)h * D + d] * xs;
                }
            }
            h1[h] = acc > 0.0f ? acc : 0.0f;
        }
    }

    // Linear(hidden_dim -> hidden_dim) + ReLU.
    #pragma unroll
    for (int h = 0; h < 32; ++h) {
        if (h < H) {
            float acc = b2[h];
            #pragma unroll
            for (int k = 0; k < 32; ++k) {
                if (k < H) acc += w2[(int64_t)h * H + k] * h1[k];
            }
            h2[h] = acc > 0.0f ? acc : 0.0f;
        }
    }

    float logit = b3[0];
    #pragma unroll
    for (int k = 0; k < 32; ++k) {
        if (k < H) logit += w3[k] * h2[k];
    }
    float prob = 1.0f / (1.0f + expf(-logit));

    int64_t o1 = ((int64_t)b * N + i) * N + j;
    int64_t o2 = ((int64_t)b * N + j) * N + i;
    out[o1] = prob;
    if (i != j) out[o2] = prob;
}

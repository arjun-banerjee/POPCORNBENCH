#include <cuda.h>
#include <cuda_runtime.h>

#define F 128

__global__ void linear128_kernel(const float* __restrict__ x,
                                 const float* __restrict__ w,
                                 const float* __restrict__ b,
                                 float* __restrict__ y,
                                 int rows) {
    int row = blockIdx.x;
    int out = threadIdx.x;
    if (row >= rows || out >= F) return;

    __shared__ float xs[F];
    xs[out] = x[row * F + out];
    __syncthreads();

    const float* __restrict__ wp = w + out * F;
    float acc = b[out];

    // fully unroll fixed 128-wide dot product
    #pragma unroll
    for (int k = 0; k < F; ++k) {
        acc = fmaf(xs[k], wp[k], acc);
    }
    y[row * F + out] = acc;
}

}

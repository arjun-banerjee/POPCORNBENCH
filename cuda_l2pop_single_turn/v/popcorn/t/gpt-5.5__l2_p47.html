"""
Virtual all-gather on the last dim: stack R tensors of shape (B, H) into (B, R*H), then Linear.
"""

import torch
import torch.nn as nn

from kernelbench.distributed_collectives import default_device

R = 4
B = 12
H = 32
K = 56


class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.proj = nn.Linear(R * H, K, bias=True)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        r, b, h = x.shape
        assert r == R and h == H
        parts = [x[i] for i in range(R)]
        gathered = torch.cat(parts, dim=-1)
        return self.proj(gathered)

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Fixed problem sizes from the benchmark: R=4, B=12, H=32, K=56.
// One CUDA block handles one batch row.  Eight warps in the block compute
// eight output columns at a time; each warp reduces the 128-term dot product.
__global__ void gather_linear_kernel(const float* __restrict__ x,
                                     const float* __restrict__ w,
                                     const float* __restrict__ bias,
                                     float* __restrict__ y) {
    const int b = blockIdx.x;              // 0..11
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5;             // 0..7

    #pragma unroll
    for (int k0 = 0; k0 < 56; k0 += 8) {
        const int k = k0 + warp;
        float acc = 0.0f;
        // Dot length is 4*32 = 128.  The virtual concatenation maps n to
        // x[r,b,h], r=n/32, h=n%32.
        #pragma unroll
        for (int n = lane; n < 128; n += 32) {
            const int r = n >> 5;
            const int h = n & 31;
            acc += x[(r * 12 + b) * 32 + h] * w[k * 128 + n];
        }
        // Warp reduction.
        acc += __shfl_down_sync(0xffffffff, acc, 16);
        acc += __shfl_down_sync(0xffffffff, acc, 8);
        acc += __shfl_down_sync(0xffffffff, acc, 4);
        acc += __shfl_down_sync(0xffffffff, acc, 2);
        acc += __shfl_down_sync(0xffffffff, acc, 1);
        if (lane == 0) {
            y[b * 56 + k] = acc + bias[k];
        }
    }
}

torch::Tensor gather_linear(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    auto y = torch::empty({12, 56}, x.options());
    gather_linear_kernel<<<12, 256>>>(x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), y.data_ptr<float>());
    return y;
}
'''

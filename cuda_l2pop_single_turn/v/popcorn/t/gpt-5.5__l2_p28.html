# Default ModelNew is PyTorch-only. run_and_check with check_kernel=True and backend=cuda

class Model(nn.Module):
    """
    Fused GroupNorm + SiLU (Swish), a common post-conv / residual-branch pattern in
    ResNet-family backbones (SiLU instead of ReLU) and in spatial blocks of diffusion
    transformers (DiT) where features are (N, C, H, W).

    Reference pipeline for a single fused kernel: normalize channels with
    `GroupNorm`, then apply `SiLU` elementwise (`x * sigmoid(x)`).
    """

    def __init__(
        self,
        num_channels: int,
        num_groups: int,
        eps: float = 1e-5,
        affine: bool = True,
    ):
        super().__init__()
        assert num_channels % num_groups == 0
        self.norm = nn.GroupNorm(
            num_groups=num_groups,
            num_channels=num_channels,
            eps=eps,
            affine=affine,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (N, C, H, W) feature map
        Returns:
            Same shape as x
        """
        return F.silu(self.norm(x))

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


_cpp_src = r"""
torch::Tensor gn_silu_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, int64_t num_groups, double eps);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>


template<int BLOCK>
__global__ void gn_silu_kernel(const float* __restrict__ x,
                               const float* __restrict__ weight,
                               const float* __restrict__ bias,
                               float* __restrict__ y,
                               int N, int C, int H, int W,
                               int num_groups, float eps, bool affine) {
    __shared__ float sdata[BLOCK];
    __shared__ float s_mean;
    __shared__ float s_invstd;

    const int tid = threadIdx.x;
    const int bid = blockIdx.x;          // one block per (n, group)
    const int n = bid / num_groups;
    const int g = bid - n * num_groups;
    const int channels_per_group = C / num_groups;
    const int S = H * W;
    const int M = channels_per_group * S;
    const long long base = ((long long)n * C + (long long)g * channels_per_group) * (long long)S;

    float sum = 0.0f;
    for (int i = tid; i < M; i += BLOCK) {
        sum += x[base + i];
    }
    sdata[tid] = sum;
    __syncthreads();

    #pragma unroll
    for (int stride = BLOCK / 2; stride > 0; stride >>= 1) {
        if (tid < stride) sdata[tid] += sdata[tid + stride];
        __syncthreads();
    }
    if (tid == 0) {
        s_mean = sdata[0] / (float)M;
    }
    __syncthreads();

    const float mean = s_mean;
    float var_sum = 0.0f;
    for (int i = tid; i < M; i += BLOCK) {
        const float d = x[base + i] - mean;
        var_sum += d * d;
    }
    sdata[tid] = var_sum;
    __syncthreads();

    #pragma unroll
    for (int stride = BLOCK / 2; stride > 0; stride >>= 1) {
        if (tid < stride) sdata[tid] += sdata[tid + stride];
        __syncthreads();
    }
    if (tid == 0) {
        s_invstd = rsqrtf(sdata[0] / (float)M + eps);
    }
    __syncthreads();

    const float invstd = s_invstd;
    for (int i = tid; i < M; i += BLOCK) {
        const int c = g * channels_per_group + i / S;
        float v = (x[base + i] - mean) * invstd;
        if (affine) {
            v = v * weight[c] + bias[c];
        }
        // SiLU / swish: x * sigmoid(x).  __expf is accurate enough for fp32 tolerance
        // and much faster than the full-precision libdevice expf.
        const float sig = 1.0f / (1.0f + __expf(-v));
        y[base + i] = v * sig;
    }
}


torch::Tensor gn_silu_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, int64_t num_groups, double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(x.dim() == 4, "expected NCHW tensor");

    const c10::cuda::CUDAGuard device_guard(x.device());
    auto y = torch::empty_like(x);

    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int G = (int)num_groups;
    const bool affine = (weight.defined() && bias.defined() && weight.numel() == C && bias.numel() == C);

    const float* wptr = affine ? weight.data_ptr<float>() : nullptr;
    const float* bptr = affine ? bias.data_ptr<float>() : nullptr;

    constexpr int BLOCK = 256;
    const int grid = N * G;
    gn_silu_kernel<BLOCK><<<grid, BLOCK, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), wptr, bptr, y.data_ptr<float>(), N, C, H, W, G, (float)eps, affine);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}
"""

class ModelNew(nn.Module):
    def __init__(self, num_channels: int, num_groups: int, eps: float = 1e-5, affine: bool = True):
        super().__init__()
        assert num_channels % num_groups == 0
        self.num_groups = int(num_groups)
        self.eps = float(eps)
        self.norm = nn.GroupNorm(num_groups=num_groups, num_channels=num_channels, eps=eps, affine=affine)
        self.register_buffer("_empty", torch.empty(0), persistent=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.norm.weight is None:
            return _ext.gn_silu_forward(x, self._empty, self._empty, self.num_groups, self.eps)
        return _ext.gn_silu_forward(x, self.norm.weight, self.norm.bias, self.num_groups, self.eps)

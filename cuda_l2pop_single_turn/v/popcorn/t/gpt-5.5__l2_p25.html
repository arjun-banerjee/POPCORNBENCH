import torch
import torch.nn as nn
import math


class Model(nn.Module):
    """
    Outer-product mean: converts MSA representation to a pair
    representation update.  Used in AlphaFold2 to communicate information
    from the MSA stack to the pair stack.  For each pair (i,j), the
    outer product of projected MSA features at positions i and j is
    averaged over the MSA depth dimension.
    """

    def __init__(self, msa_dim, pair_dim, hidden_dim=32):
        super().__init__()
        self.layer_norm = nn.LayerNorm(msa_dim)
        self.proj_a = nn.Linear(msa_dim, hidden_dim)
        self.proj_b = nn.Linear(msa_dim, hidden_dim)
        self.out_proj = nn.Linear(hidden_dim * hidden_dim, pair_dim)

    def forward(self, msa_repr: torch.Tensor, msa_mask: torch.Tensor) -> torch.Tensor:
        """
        Args:
            msa_repr: (B, S, N, msa_dim) – MSA representation
            msa_mask: (B, S, N)          – 1 for valid rows/positions
        Returns:
            pair update (B, N, N, pair_dim)
        """
        x = self.layer_norm(msa_repr)
        a = self.proj_a(x)  # (B, S, N, H)
        b = self.proj_b(x)

        mask = msa_mask.unsqueeze(-1)  # (B, S, N, 1)
        a = a * mask
        b = b * mask

        outer = torch.einsum("bsih,bsjk->bijhk", a, b)  # (B, N, N, H, H)
        denom = msa_mask.sum(dim=1).unsqueeze(-1).unsqueeze(-1).unsqueeze(-1).clamp(min=1)
        outer = outer / denom  # mean over S

        B, N1, N2, H1, H2 = outer.shape
        outer = outer.reshape(B, N1, N2, H1 * H2)
        return self.out_proj(outer)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define C_DIM 64
#define H_DIM 32
#define P_DIM 64

template<int BLOCK>
__device__ __forceinline__ float block_sum(float v) {
    __shared__ float smem[BLOCK];
    int tid = threadIdx.x;
    smem[tid] = v;
    __syncthreads();
    #pragma unroll
    for (int stride = BLOCK/2; stride > 0; stride >>= 1) {
        if (tid < stride) smem[tid] += smem[tid + stride];
        __syncthreads();
    }
    return smem[0];
}

__global__ void proj_kernel(
    const float* __restrict__ x,
    const float* __restrict__ mask,
    const float* __restrict__ ln_w,
    const float* __restrict__ ln_b,
    const float* __restrict__ wa,
    const float* __restrict__ ba,
    const float* __restrict__ wb,
    const float* __restrict__ bb,
    float* __restrict__ a,
    float* __restrict__ b,
    int B, int S, int N) {

    const int row = blockIdx.x; // ((b*S+s)*N+n)
    const int tid = threadIdx.x;
    const int base = row * C_DIM;

    float val = 0.0f;
    if (tid < C_DIM) val = x[base + tid];
    float sum = block_sum<128>(val);
    const float mean = sum * (1.0f / (float)C_DIM);

    float dv = 0.0f;
    if (tid < C_DIM) {
        float d = x[base + tid] - mean;
        dv = d * d;
    }
    float vsum = block_sum<128>(dv);
    const float rstd = rsqrtf(vsum * (1.0f / (float)C_DIM) + 1.0e-5f);

    __shared__ float xn[C_DIM];
    if (tid < C_DIM) {
        xn[tid] = (x[base + tid] - mean) * rstd * ln_w[tid] + ln_b[tid];
    }
    __syncthreads();

    if (tid < H_DIM) {
        float acca = ba[tid];
        float accb = bb[tid];
        const float* wpa = wa + tid * C_DIM;
        const float* wpb = wb + tid * C_DIM;
        #pragma unroll
        for (int c = 0; c < C_DIM; ++c) {
            float xv = xn[c];
            acca += xv * wpa[c];
            accb += xv * wpb[c];
        }
        float m = mask[row];
        a[row * H_DIM + tid] = acca * m;
        b[row * H_DIM + tid] = accb * m;
    }
}

__global__ void transform_b_kernel(
    const float* __restrict__ b,
    const float* __restrict__ wo,
    float* __restrict__ t,
    int total) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    int h = idx % H_DIM;
    int q = idx / H_DIM;
    int p = q % P_DIM;
    int row = q / P_DIM; // ((b*S+s)*N+j)

    const float* brow = b + row * H_DIM;
    const float* wrow = wo + p * (H_DIM * H_DIM) + h * H_DIM;
    float acc = 0.0f;
    #pragma unroll
    for (int k = 0; k < H_DIM; ++k) {
        acc += wrow[k] * brow[k];
    }
    t[idx] = acc;
}

__global__ void out_kernel(
    const float* __restrict__ a,
    const float* __restrict__ t,
    const float* __restrict__ mask,
    const float* __restrict__ bo,
    float* __restrict__ out,
    int B, int S, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * N * N * P_DIM;
    if (idx >= total) return;

    int p = idx % P_DIM;
    int tmp = idx / P_DIM;
    int j = tmp % N;
    tmp /= N;
    int i = tmp % N;
    int bidx = tmp / N;

    float denom = 0.0f;
    #pragma unroll
    for (int s = 0; s < 8; ++s) {
        if (s < S) denom += mask[(bidx * S + s) * N + i];
    }
    if (denom < 1.0f) denom = 1.0f;

    float acc = 0.0f;
    for (int s = 0; s < S; ++s) {
        const float* ap = a + ((bidx * S + s) * N + i) * H_DIM;
        const float* tp = t + (((bidx * S + s) * N + j) * P_DIM + p) * H_DIM;
        #pragma unroll
        for (int h = 0; h < H_DIM; ++h) {
            acc += ap[h] * tp[h];
        }
    }
    out[idx] = acc / denom + bo[p];
}

at::Tensor opm_forward_cuda(
    at::Tensor msa, at::Tensor mask,
    at::Tensor ln_w, at::Tensor ln_b,
    at::Tensor wa, at::Tensor ba,
    at::Tensor wb, at::Tensor bb,
    at::Tensor wo, at::Tensor bo) {

    TORCH_CHECK(msa.is_cuda(), "msa must be CUDA");
    TORCH_CHECK(msa.scalar_type() == at::kFloat, "only fp32 supported");
    const int B = (int)msa.size(0);
    const int S = (int)msa.size(1);
    const int N = (int)msa.size(2);
    const int C = (int)msa.size(3);
    TORCH_CHECK(C == C_DIM, "msa_dim must be 64");
    TORCH_CHECK(wa.size(0) == H_DIM && wo.size(0) == P_DIM, "expected hidden_dim=32 pair_dim=64");

    auto opts = msa.options();
    at::Tensor a = at::empty({B, S, N, H_DIM}, opts);
    at::Tensor b = at::empty({B, S, N, H_DIM}, opts);
    at::Tensor tr = at::empty({B, S, N, P_DIM, H_DIM}, opts);
    at::Tensor out = at::empty({B, N, N, P_DIM}, opts);

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    int rows = B * S * N;
    proj_kernel<<<rows, 128, 0, stream>>>(
        msa.data_ptr<float>(), mask.data_ptr<float>(),
        ln_w.data_ptr<float>(), ln_b.data_ptr<float>(),
        wa.data_ptr<float>(), ba.data_ptr<float>(),
        wb.data_ptr<float>(), bb.data_ptr<float>(),
        a.data_ptr<float>(), b.data_ptr<float>(), B, S, N);

    int ttotal = B * S * N * P_DIM * H_DIM;
    transform_b_kernel<<<(ttotal + 255) / 256, 256, 0, stream>>>(
        b.data_ptr<float>(), wo.data_ptr<float>(), tr.data_ptr<float>(), ttotal);

    int ototal = B * N * N * P_DIM;
    out_kernel<<<(ototal + 255) / 256, 256, 0, stream>>>(
        a.data_ptr<float>(), tr.data_ptr<float>(), mask.data_ptr<float>(), bo.data_ptr<float>(),
        out.data_ptr<float>(), B, S, N);

    return out;
}

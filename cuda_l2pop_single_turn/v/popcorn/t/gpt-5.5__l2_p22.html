import torch.nn as nn

class Model(nn.Module):
    """
    Fused radius (ball) neighborhood query over a 3D point map with per-query feature
    aggregation—common in navigation / manipulation perception (PointNet++-style set
    abstraction, local maps around end-effector or goal queries on LiDAR / depth).

    Single forward fusion target:
      1) pairwise Euclidean distances between each query center and every point,
      2) membership in the closed ball of radius ``r``,
      3) channel-wise **max** pooling of point features over in-ball neighbors.

    Points with no neighbors inside the radius contribute a **zero** feature vector
    for that query (instead of ``-inf`` max).
    """

    def __init__(self, in_channels: int, radius: float):
        super().__init__()
        self.in_channels = in_channels
        self.register_buffer("radius_sq", torch.tensor(float(radius) ** 2))

    def forward(
        self,
        points: torch.Tensor,
        queries: torch.Tensor,
        features: torch.Tensor,
    ) -> torch.Tensor:
        """
        Args:
            points: (B, P, 3) point coordinates (e.g. world or sensor frame).
            queries: (B, Q, 3) ball centers (e.g. navigation waypoints, grasp samples).
            features: (B, P, C) per-point descriptors (geometry or lifted RGB-D).
        Returns:
            (B, Q, C) max-pooled features over neighbors with ||p - q||_2 <= r.
        """
        B, P, _ = points.shape
        _, Q, _ = queries.shape
        _, P2, C = features.shape
        assert P == P2 and C == self.in_channels

        # Squared distances (B, Q, P)
        dist_sq = torch.cdist(queries, points, p=2.0) ** 2
        in_ball = dist_sq <= self.radius_sq

        # (B, 1, P, C) broadcasts with (B, Q, P) for masked max over P
        feats = features.unsqueeze(1)
        neg_large = torch.finfo(features.dtype).min
        masked = torch.where(
            in_ball.unsqueeze(-1),
            feats,
            torch.full_like(feats, neg_large),
        )
        pooled, _ = masked.max(dim=2)

        empty = in_ball.sum(dim=2) == 0
        if empty.any():
            pooled = pooled.masked_fill(empty.unsqueeze(-1), 0.0)

        return pooled

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

// One block computes one (batch, query, channel-tile-of-32).  Threads are
// arranged as (channel within tile, reduction lane over points).  This repeats
// the distance computation per channel, but avoids materializing the enormous
// (B,Q,P) mask/cdist tensor and keeps feature loads coalesced when hits occur.
template<int TILE_C, int REDUCE_Y>
__global__ void ball_query_kernel_c32(
    const float* __restrict__ points,
    const float* __restrict__ queries,
    const float* __restrict__ features,
    float* __restrict__ out,
    int B, int P, int Q, int C,
    float radius_sq)
{
    const int tx = threadIdx.x; // channel lane [0,31]
    const int ty = threadIdx.y; // point reduction lane
    const int bq = blockIdx.x;
    const int tile = blockIdx.y;
    const int b = bq / Q;
    const int q = bq - b * Q;
    const int c = tile * TILE_C + tx;

    float best = -FLT_MAX;
    int found = 0;

    const float qx = queries[(b * Q + q) * 3 + 0];
    const float qy = queries[(b * Q + q) * 3 + 1];
    const float qz = queries[(b * Q + q) * 3 + 2];

    if (c < C) {
        for (int p = ty; p < P; p += REDUCE_Y) {
            const float px = points[(b * P + p) * 3 + 0];
            const float py = points[(b * P + p) * 3 + 1];
            const float pz = points[(b * P + p) * 3 + 2];
            const float dx = px - qx;
            const float dy = py - qy;
            const float dz = pz - qz;
            const float d2 = dx * dx + dy * dy + dz * dz;
            if (d2 <= radius_sq) {
                const float v = features[(b * P + p) * C + c];
                best = fmaxf(best, v);
                found = 1;
            }
        }
    }

    __shared__ float s_best[REDUCE_Y][TILE_C];
    __shared__ int s_found[REDUCE_Y][TILE_C];
    s_best[ty][tx] = best;
    s_found[ty][tx] = found;
    __syncthreads();

    #pragma unroll
    for (int stride = REDUCE_Y >> 1; stride > 0; stride >>= 1) {
        if (ty < stride) {
            const int f2 = s_found[ty + stride][tx];
            const float b2 = s_best[ty + stride][tx];
            if (f2) {
                if (!s_found[ty][tx] || b2 > s_best[ty][tx]) s_best[ty][tx] = b2;
                s_found[ty][tx] = 1;
            }
        }
        __syncthreads();
    }

    if (ty == 0 && c < C) {
        out[(b * Q + q) * C + c] = s_found[0][tx] ? s_best[0][tx] : 0.0f;
    }
}

// A variant with fewer reduction lanes can be marginally better for very small P.
template<int TILE_C, int REDUCE_Y>
__global__ void ball_query_kernel_generic(
    const float* __restrict__ points,
    const float* __restrict__ queries,
    const float* __restrict__ features,
    float* __restrict__ out,
    int B, int P, int Q, int C,
    float radius_sq)
{
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int bq = blockIdx.x;
    const int tile = blockIdx.y;
    const int b = bq / Q;
    const int q = bq - b * Q;
    const int c = tile * TILE_C + tx;

    float best = -FLT_MAX;
    int found = 0;
    const float qx = queries[(b * Q + q) * 3 + 0];
    const float qy = queries[(b * Q + q) * 3 + 1];
    const float qz = queries[(b * Q + q) * 3 + 2];
    if (c < C) {
        for (int p = ty; p < P; p += REDUCE_Y) {
            const float dx = points[(b * P + p) * 3 + 0] - qx;
            const float dy = points[(b * P + p) * 3 + 1] - qy;
            const float dz = points[(b * P + p) * 3 + 2] - qz;
            if (dx*dx + dy*dy + dz*dz <= radius_sq) {
                float v = features[(b * P + p) * C + c];
                best = found ? fmaxf(best, v) : v;
                found = 1;
            }
        }
    }
    __shared__ float s_best[REDUCE_Y][TILE_C];
    __shared__ int s_found[REDUCE_Y][TILE_C];
    s_best[ty][tx] = best;
    s_found[ty][tx] = found;
    __syncthreads();
    for (int stride = REDUCE_Y / 2; stride > 0; stride >>= 1) {
        if (ty < stride) {
            if (s_found[ty + stride][tx]) {
                float v = s_best[ty + stride][tx];
                s_best[ty][tx] = s_found[ty][tx] ? fmaxf(s_best[ty][tx], v) : v;
                s_found[ty][tx] = 1;
            }
        }
        __syncthreads();
    }
    if (ty == 0 && c < C) out[(b * Q + q) * C + c] = s_found[0][tx] ? s_best[0][tx] : 0.0f;
}

torch::Tensor ball_query_cuda(torch::Tensor points, torch::Tensor queries, torch::Tensor features, double radius_sq_d) {
    c10::cuda::CUDAGuard device_guard(points.device());
    TORCH_CHECK(points.is_cuda() && queries.is_cuda() && features.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(points.scalar_type() == at::kFloat && queries.scalar_type() == at::kFloat && features.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(points.is_contiguous() && queries.is_contiguous() && features.is_contiguous(), "inputs must be contiguous");
    const int B = (int)points.size(0);
    const int P = (int)points.size(1);
    const int Q = (int)queries.size(1);
    const int C = (int)features.size(2);
    auto out = torch::empty({B, Q, C}, features.options());
    if (B == 0 || P == 0 || Q == 0 || C == 0) return out;

    constexpr int TILE_C = 32;
    const dim3 grid(B * Q, (C + TILE_C - 1) / TILE_C);
    const float radius_sq = (float)radius_sq_d;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (P <= 256) {
        constexpr int RY = 8;
        const dim3 block(TILE_C, RY);
        ball_query_kernel_generic<TILE_C, RY><<<grid, block, 0, stream>>>(
            points.data_ptr<float>(), queries.data_ptr<float>(), features.data_ptr<float>(), out.data_ptr<float>(),
            B, P, Q, C, radius_sq);
    } else {
        constexpr int RY = 16;
        const dim3 block(TILE_C, RY);
        ball_query_kernel_c32<TILE_C, RY><<<grid, block, 0, stream>>>(
            points.data_ptr<float>(), queries.data_ptr<float>(), features.data_ptr<float>(), out.data_ptr<float>(),
            B, P, Q, C, radius_sq);
    }
    return out;
}
'''

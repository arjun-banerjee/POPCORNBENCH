import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Differentiable Smith-Waterman-style local sequence alignment scoring.
    Uses a soft-max relaxation of the DP recurrence so gradients can flow
    through the alignment matrix.  
    """
    def __init__(self, gap_open=-1.0, temperature=1.0):
        super().__init__()
        self.gap_open = gap_open
        self.temperature = temperature

    def forward(self, score_matrix: torch.Tensor) -> torch.Tensor:
        """
        Args:
            score_matrix: (B, M, N) – per-position similarity scores
        Returns:
            alignment_score: (B,) – soft local-alignment score (log-sum-exp over DP cells)
        """
        B, M, N = score_matrix.shape
        neg_inf = torch.tensor(float("-inf"), device=score_matrix.device, dtype=score_matrix.dtype)

        # H[i][j] = max(0, H[i-1][j-1]+S[i][j], H[i-1][j]+gap, H[i][j-1]+gap)
        H = score_matrix.new_full((B, M + 1, N + 1), 0.0)

        for i in range(1, M + 1):
            for j in range(1, N + 1):
                match = H[:, i - 1, j - 1] + score_matrix[:, i - 1, j - 1]
                delete = H[:, i - 1, j] + self.gap_open
                insert = H[:, i, j - 1] + self.gap_open
                zero = torch.zeros(B, device=score_matrix.device)
                stack = torch.stack([zero, match, delete, insert], dim=-1)
                H[:, i, j] = self.temperature * torch.logsumexp(stack / self.temperature, dim=-1)

        return H.view(B, -1).max(dim=-1).values

import torch

from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>

void sw_forward_cuda(torch::Tensor score, torch::Tensor out, float gap_open, float temperature);

torch::Tensor sw_forward(torch::Tensor score, double gap_open, double temperature) {
    score = score.contiguous();
    auto out = torch::empty({score.size(0)}, score.options());
    sw_forward_cuda(score, out, static_cast<float>(gap_open), static_cast<float>(temperature));
    return out;
}

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void sw_kernel(const float* __restrict__ score,
                          float* __restrict__ out,
                          int B, int M, int N,
                          float gap, float temp) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int nt = blockDim.x;
    int stride = N + 1;
    int h_total = (M + 1) * (N + 1);

    extern __shared__ float smem[];
    float* H = smem;
    float* red = smem + h_total;

    for (int idx = tid; idx < h_total; idx += nt) {
        H[idx] = 0.0f;
    }
    __syncthreads();

    const float* S = score + ((long long)b) * M * N;
    float local_max = 0.0f;
    float inv_temp = 1.0f / temp;

    // Wavefront DP: cells on the same anti-diagonal are independent.
    for (int d = 2; d <= M + N; ++d) {
        int istart = d - N;
        if (istart < 1) istart = 1;
        int iend = d - 1;
        if (iend > M) iend = M;
        int count = iend - istart + 1;

        for (int k = tid; k < count; k += nt) {
            int i = istart + k;
            int j = d - i;

            float match = H[(i - 1) * stride + (j - 1)] + S[(i - 1) * N + (j - 1)];
            float delv  = H[(i - 1) * stride + j] + gap;
            float insv  = H[i * stride + (j - 1)] + gap;

            float a0 = 0.0f;
            float a1 = match * inv_temp;
            float a2 = delv  * inv_temp;
            float a3 = insv  * inv_temp;
            float m = fmaxf(fmaxf(a0, a1), fmaxf(a2, a3));
            float sum = expf(a0 - m) + expf(a1 - m) + expf(a2 - m) + expf(a3 - m);
            float val = temp * (m + logf(sum));
            H[i * stride + j] = val;
            local_max = fmaxf(local_max, val);
        }
        __syncthreads();
    }

    red[tid] = local_max;
    __syncthreads();
    for (int s = nt >> 1; s > 0; s >>= 1) {
        if (tid < s) red[tid] = fmaxf(red[tid], red[tid + s]);
        __syncthreads();
    }
    if (tid == 0) out[b] = red[0];
}

void sw_forward_cuda(torch::Tensor score, torch::Tensor out, float gap_open, float temperature) {
    int B = (int)score.size(0);
    int M = (int)score.size(1);
    int N = (int)score.size(2);
    int maxdiag = M < N ? M : N;
    int threads = 32;
    while (threads < maxdiag && threads < 1024) threads <<= 1;
    size_t shmem = ((size_t)(M + 1) * (N + 1) + (size_t)threads) * sizeof(float);
    sw_kernel<<<B, threads, shmem, at::cuda::getCurrentCUDAStream()>>>(
        score.data_ptr<float>(), out.data_ptr<float>(), B, M, N, gap_open, temperature);
}
'''

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Selective scan (Mamba's S6 layer): projects input to derive input-dependent SSM parameters
    (delta, B, C), discretizes a learned state matrix A, runs the linear recurrence, and
    applies a skip connection via D. The fusion opportunity is keeping all intermediates
    (discretized parameters, hidden states) in SRAM rather than materializing them to global memory.

    Args:
        d_model (int): Input feature dimension.
        d_state (int): Hidden state dimension per channel (N in Mamba, typically 16).
    """
    def __init__(self, d_model: int, d_state: int):
        super(Model, self).__init__()
        self.d_model = d_model
        self.d_state = d_state
        self.A_log = nn.Parameter(torch.randn(d_model, d_state))
        self.proj_B = nn.Linear(d_model, d_state, bias=False)
        self.proj_C = nn.Linear(d_model, d_state, bias=False)
        self.proj_delta = nn.Linear(d_model, d_model, bias=True)
        self.D = nn.Parameter(torch.ones(d_model))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, L, D = x.shape
        N = self.d_state

        delta = torch.nn.functional.softplus(self.proj_delta(x))
        B_t = self.proj_B(x)
        C_t = self.proj_C(x)

        A = -torch.exp(self.A_log)
        dA = torch.exp(delta.unsqueeze(-1) * A.unsqueeze(0).unsqueeze(0))
        dB = delta.unsqueeze(-1) * B_t.unsqueeze(2)
        x_db = x.unsqueeze(-1) * dB

        h = torch.zeros(B, D, N, device=x.device, dtype=x.dtype)
        outputs = []
        for t in range(L):
            h = dA[:, t] * h + x_db[:, t]
            y_t = (h * C_t[:, t].unsqueeze(1)).sum(dim=-1)
            outputs.append(y_t)

        y = torch.stack(outputs, dim=1)
        y = y + x * self.D.unsqueeze(0).unsqueeze(0)
        return y

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// One warp computes one (batch, channel) recurrent stream.  The state dimension
// is small (16 for the benchmark), so lanes hold individual state entries in
// registers and scan through time without materializing dA/dB/h.
__global__ void selective_scan_kernel(const float* __restrict__ x,
                                      const float* __restrict__ delta,
                                      const float* __restrict__ B_t,
                                      const float* __restrict__ C_t,
                                      const float* __restrict__ A_log,
                                      const float* __restrict__ D_param,
                                      float* __restrict__ y,
                                      int B, int L, int D, int N) {
    int bd = blockIdx.x;
    int b = bd / D;
    int d = bd - b * D;
    int lane = threadIdx.x; // blockDim.x == 32

    float h = 0.0f;
    float Aval = 0.0f;
    if (lane < N) {
        Aval = -expf(A_log[d * N + lane]);
    }
    const float Dv = D_param[d];
    const int base_x = (b * L) * D + d;
    const int base_n = (b * L) * N;

    #pragma unroll 1
    for (int t = 0; t < L; ++t) {
        const float dt = delta[base_x + t * D];
        const float xv = x[base_x + t * D];
        float prod = 0.0f;
        if (lane < N) {
            const int offn = base_n + t * N + lane;
            const float bt = B_t[offn];
            const float ct = C_t[offn];
            const float e = expf(dt * Aval);
            h = e * h + xv * dt * bt;
            prod = h * ct;
        }

        // Sum the N<=32 state contributions within the warp.
        unsigned mask = 0xffffffffu;
        prod += __shfl_down_sync(mask, prod, 16);
        prod += __shfl_down_sync(mask, prod, 8);
        prod += __shfl_down_sync(mask, prod, 4);
        prod += __shfl_down_sync(mask, prod, 2);
        prod += __shfl_down_sync(mask, prod, 1);

        if (lane == 0) {
            y[base_x + t * D] = prod + xv * Dv;
        }
    }
}

torch::Tensor selective_scan_forward(torch::Tensor x,
                                    torch::Tensor delta,
                                    torch::Tensor B_t,
                                    torch::Tensor C_t,
                                    torch::Tensor A_log,
                                    torch::Tensor D_param) {
    const int B = (int)x.size(0);
    const int L = (int)x.size(1);
    const int D = (int)x.size(2);
    const int N = (int)A_log.size(1);
    auto y = torch::empty_like(x);

    dim3 block(32);
    dim3 grid(B * D);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    selective_scan_kernel<<<grid, block, 0, stream>>>(
        x.data_ptr<float>(), delta.data_ptr<float>(), B_t.data_ptr<float>(), C_t.data_ptr<float>(),
        A_log.data_ptr<float>(), D_param.data_ptr<float>(), y.data_ptr<float>(), B, L, D, N);
    return y;
}
"""

    def __init__(self, d_model: int, d_state: int):
        super(ModelNew, self).__init__()
        self.d_model = d_model
        self.d_state = d_state
        self.A_log = nn.Parameter(torch.randn(d_model, d_state))
        self.proj_B = nn.Linear(d_model, d_state, bias=False)
        self.proj_C = nn.Linear(d_model, d_state, bias=False)
        self.proj_delta = nn.Linear(d_model, d_model, bias=True)
        self.D = nn.Parameter(torch.ones(d_model))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use cuBLAS for the dense projections, then a custom fused CUDA scan for
        # the recurrent SSM portion, avoiding the huge dA/dB/x_db materializations
        # and the Python-level loop in the reference implementation.
        delta = F.softplus(F.linear(x, self.proj_delta.weight, self.proj_delta.bias))
        B_t = F.linear(x, self.proj_B.weight, None)
        C_t = F.linear(x, self.proj_C.weight, None)
        return _ext.selective_scan_forward(x, delta, B_t, C_t, self.A_log, self.D)

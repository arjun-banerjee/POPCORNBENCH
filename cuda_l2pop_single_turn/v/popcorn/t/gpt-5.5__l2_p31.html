import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Fused MLA-style attention from compressed KV latents.

    The compressed latent representation is expanded into keys and values on the
    fly inside the attention computation rather than materializing a KV cache
    first. This isolates the main compute pattern behind fused MLA attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        kv_latent: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        rank = kv_latent.shape[-1]
        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_heads):
                for t in range(seq_len):
                    logits = []
                    values = []
                    for s in range(t + 1):
                        latent = kv_latent[b, s]
                        k = torch.matmul(latent, w_up_k[:, h])
                        v = torch.matmul(latent, w_up_v[:, h])
                        logits.append((q[b, t, h] * k).sum() * scale)
                        values.append(v)
                    attn = torch.softmax(torch.stack(logits, dim=0), dim=0)
                    value_stack = torch.stack(values, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * value_stack).sum(dim=0)

        return out

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void mla_kernel(const float* __restrict__ q,
                           const float* __restrict__ kv,
                           const float* __restrict__ wk,
                           const float* __restrict__ wv,
                           float* __restrict__ out) {
    // Fixed problem sizes from benchmark: B=2, T=32, H=4, D=16, R=24
    const int D = 16;
    const int R = 24;
    const int T = 32;
    int t = blockIdx.x;
    int h = blockIdx.y;
    int b = blockIdx.z;
    int lane = threadIdx.x; // 0..31, one warp/block

    __shared__ float probs[32];
    __shared__ float logits_s[32];

    float logit = -3.402823466e38f;
    if (lane <= t) {
        // Compute dot(q[b,t,h], K[b,lane,h]) * scale directly from latent.
        // Loop order mirrors (latent @ wk[:,h]) followed by dot with q reasonably closely.
        float ktmp[16];
        #pragma unroll
        for (int d=0; d<16; ++d) ktmp[d] = 0.0f;
        const float* kvs = kv + (b*T + lane)*R;
        #pragma unroll
        for (int r=0; r<24; ++r) {
            float lr = kvs[r];
            const float* wkr = wk + (r*4 + h)*D;
            #pragma unroll
            for (int d=0; d<16; ++d) {
                ktmp[d] += lr * wkr[d];
            }
        }
        const float* qptr = q + (((b*T + t)*4 + h)*D);
        float sum = 0.0f;
        #pragma unroll
        for (int d=0; d<16; ++d) sum += qptr[d] * ktmp[d];
        logit = sum * 0.25f; // 1/sqrt(16)
    }

    // Warp max over all 32 lanes; invalid lanes carry -inf
    float m = logit;
    #pragma unroll
    for (int offset=16; offset>0; offset>>=1) {
        float other = __shfl_down_sync(0xffffffff, m, offset);
        m = fmaxf(m, other);
    }
    m = __shfl_sync(0xffffffff, m, 0);

    float e = 0.0f;
    if (lane <= t) e = expf(logit - m);
    float denom = e;
    #pragma unroll
    for (int offset=16; offset>0; offset>>=1) {
        denom += __shfl_down_sync(0xffffffff, denom, offset);
    }
    denom = __shfl_sync(0xffffffff, denom, 0);
    if (lane <= t) {
        probs[lane] = e / denom;
        logits_s[lane] = logit;
    }
    __syncthreads();

    // First 16 lanes compute output dimensions.
    if (lane < 16) {
        int d = lane;
        float acc = 0.0f;
        for (int s=0; s<=t; ++s) {
            const float* kvs = kv + (b*T + s)*R;
            const float* wvd_base = wv + (h*D + d); // plus r*4*D
            float v = 0.0f;
            #pragma unroll
            for (int r=0; r<24; ++r) {
                v += kvs[r] * wvd_base[r*4*D];
            }
            acc += probs[s] * v;
        }
        out[(((b*T + t)*4 + h)*D) + d] = acc;
    }
}

torch::Tensor mla_forward(torch::Tensor q, torch::Tensor kv, torch::Tensor wk, torch::Tensor wv) {
    auto out = torch::empty_like(q);
    dim3 grid(32, 4, 2);
    dim3 block(32);
    mla_kernel<<<grid, block>>>(q.data_ptr<float>(), kv.data_ptr<float>(), wk.data_ptr<float>(), wv.data_ptr<float>(), out.data_ptr<float>());
    return out;
}
'''

import torch.nn as nn

# Default ModelNew is PyTorch-only. run_and_check: use check_kernel=False with backend=cuda

class Model(nn.Module):
    """
    Fused 1D convolution + GroupNorm for temporal sequences (N, C, T).

    Typical in time-series heads, audio / motor trajectories, and sequence models
    that use strided or padded Conv1d followed by per-channel normalization without
    collapsing the time dimension (GroupNorm over `out_channels` on (N, C_out, T')).

    Reference fusion target: `Conv1d` → `GroupNorm` (no activation), matching a
    common stem or residual branch in temporal backbones.
    """

    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        num_groups: int,
        kernel_size: int = 3,
        stride: int = 1,
        padding: int = 1,
        dilation: int = 1,
        eps: float = 1e-5,
        affine: bool = True,
    ):
        super().__init__()
        assert out_channels % num_groups == 0
        self.conv = nn.Conv1d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=False,
        )
        self.norm = nn.GroupNorm(
            num_groups=num_groups,
            num_channels=out_channels,
            eps=eps,
            affine=affine,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (N, C_in, T) temporal feature map
        Returns:
            (N, C_out, T') with T' determined by conv stride / padding / kernel
        """
        return self.norm(self.conv(x))

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
torch::Tensor group_norm1d_forward(torch::Tensor input, torch::Tensor gamma, torch::Tensor beta, int64_t groups, double eps, bool affine);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#ifndef BLOCK_SIZE
#define BLOCK_SIZE 256
#endif

__global__ void group_norm1d_kernel(const float* __restrict__ x,
                                    const float* __restrict__ gamma,
                                    const float* __restrict__ beta,
                                    float* __restrict__ out,
                                    int C, int T, int G, int cpg,
                                    float eps, bool affine) {
    int row = blockIdx.x;            // one (n, group) per CTA
    int g = row % G;
    int n = row / G;
    int tid = threadIdx.x;
    int M = cpg * T;
    int base = (n * C + g * cpg) * T;

    float sum = 0.0f;
    float sum2 = 0.0f;

    // The group is contiguous in NCT layout, so this is fully coalesced.
    for (int i = tid; i < M; i += BLOCK_SIZE) {
        float v = x[base + i];
        sum += v;
        sum2 += v * v;
    }

    __shared__ float s_sum[BLOCK_SIZE];
    __shared__ float s_sum2[BLOCK_SIZE];
    s_sum[tid] = sum;
    s_sum2[tid] = sum2;
    __syncthreads();

    // Block reduction.  BLOCK_SIZE is a power of two.
    #pragma unroll
    for (int offset = BLOCK_SIZE / 2; offset > 0; offset >>= 1) {
        if (tid < offset) {
            s_sum[tid] += s_sum[tid + offset];
            s_sum2[tid] += s_sum2[tid + offset];
        }
        __syncthreads();
    }

    __shared__ float mean_s;
    __shared__ float invstd_s;
    if (tid == 0) {
        float invM = 1.0f / (float)M;
        float mean = s_sum[0] * invM;
        float var = s_sum2[0] * invM - mean * mean;
        // Guard against tiny negative roundoff in the E[x^2]-E[x]^2 formula.
        if (var < 0.0f) var = 0.0f;
        mean_s = mean;
        invstd_s = rsqrtf(var + eps);
    }
    __syncthreads();

    float mean = mean_s;
    float invstd = invstd_s;
    int c0 = g * cpg;

    if (affine) {
        for (int i = tid; i < M; i += BLOCK_SIZE) {
            int c = c0 + i / T;
            float v = (x[base + i] - mean) * invstd;
            out[base + i] = v * gamma[c] + beta[c];
        }
    } else {
        for (int i = tid; i < M; i += BLOCK_SIZE) {
            float v = (x[base + i] - mean) * invstd;
            out[base + i] = v;
        }
    }
}

torch::Tensor group_norm1d_forward(torch::Tensor input, torch::Tensor gamma, torch::Tensor beta, int64_t groups, double eps, bool affine) {
    const int N = (int)input.size(0);
    const int C = (int)input.size(1);
    const int T = (int)input.size(2);
    const int G = (int)groups;
    const int cpg = C / G;

    auto output = torch::empty_like(input);
    dim3 block(BLOCK_SIZE);
    dim3 grid(N * G);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float* gptr = affine ? gamma.data_ptr<float>() : input.data_ptr<float>();
    const float* bptr = affine ? beta.data_ptr<float>() : input.data_ptr<float>();

    group_norm1d_kernel<<<grid, block, 0, stream>>>(
        input.data_ptr<float>(), gptr, bptr, output.data_ptr<float>(),
        C, T, G, cpg, (float)eps, affine);
    return output;
}
"""

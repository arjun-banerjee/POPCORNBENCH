import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Causal grouped-query attention with KV-head expansion.

    Query heads outnumber KV heads. Keys and values are expanded by repeating
    each KV head across its assigned query-head group before causal attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_q_heads, head_dim = q.shape
        num_kv_heads = k.shape[2]
        group_size = num_q_heads // num_kv_heads

        expanded_k = k.repeat_interleave(group_size, dim=2)
        expanded_v = v.repeat_interleave(group_size, dim=2)

        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_q_heads):
                for t in range(seq_len):
                    q_t = q[b, t, h]
                    k_hist = expanded_k[b, : t + 1, h]
                    v_hist = expanded_v[b, : t + 1, h]
                    scores = (k_hist * q_t.unsqueeze(0)).sum(dim=-1) * scale
                    attn = torch.softmax(scores, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * v_hist).sum(dim=0)

        return out

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef WARP_SIZE
#define WARP_SIZE 32
#endif

__inline__ __device__ float warp_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// One warp/block computes one (batch, time, query-head) output vector (D=32).
__global__ void gqa_kernel_d32(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    float* __restrict__ out,
    int B, int T, int Hq, int Hkv)
{
    int idx = blockIdx.x;
    int t = idx % T;
    int tmp = idx / T;
    int h = tmp % Hq;
    int b = tmp / Hq;
    int lane = threadIdx.x;  // 0..31

    int group = Hq / Hkv;
    int hk = h / group;

    constexpr int D = 32;
    const float scale = 0.17677669529663687f; // 1/sqrt(32)

    __shared__ float scores[80];
    __shared__ float smem[2];

    const float qd = q[((b * T + t) * Hq + h) * D + lane];

    // Compute all causal scores. Accumulation is within one warp.
    for (int s = 0; s <= t; ++s) {
        float kd = k[((b * T + s) * Hkv + hk) * D + lane];
        float prod = qd * kd;
        float dot = warp_sum(prod);
        if (lane == 0) scores[s] = dot * scale;
    }

    // Stable softmax statistics (serial over <=80 in lane 0; cheap and deterministic).
    if (lane == 0) {
        float m = scores[0];
        #pragma unroll
        for (int s = 1; s < 80; ++s) {
            if (s <= t) {
                float x = scores[s];
                m = x > m ? x : m;
            }
        }
        float denom = 0.0f;
        #pragma unroll
        for (int s = 0; s < 80; ++s) {
            if (s <= t) {
                float e = expf(scores[s] - m);
                scores[s] = e;
                denom += e;
            }
        }
        smem[0] = 1.0f / denom;
    }
    __syncthreads();

    // Each lane computes one output dimension.
    float acc = 0.0f;
    float inv_denom = smem[0];
    for (int s = 0; s <= t; ++s) {
        float w = scores[s] * inv_denom;
        float vd = v[((b * T + s) * Hkv + hk) * D + lane];
        acc += w * vd;
    }
    out[((b * T + t) * Hq + h) * D + lane] = acc;
}

// Generic fallback for D not equal 32, still optimized for the benchmark's small sizes.
__global__ void gqa_kernel_generic(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    float* __restrict__ out,
    int B, int T, int Hq, int Hkv, int D)
{
    int idx = blockIdx.x;
    int t = idx % T;
    int tmp = idx / T;
    int h = tmp % Hq;
    int b = tmp / Hq;
    int lane = threadIdx.x;
    int group = Hq / Hkv;
    int hk = h / group;
    float scale = rsqrtf((float)D);
    extern __shared__ float sh[];
    float* scores = sh;          // T floats
    float* scratch = sh + T;     // blockDim floats

    for (int s = 0; s <= t; ++s) {
        float sum = 0.0f;
        for (int d = lane; d < D; d += blockDim.x) {
            sum += q[((b*T + t)*Hq + h)*D + d] * k[((b*T + s)*Hkv + hk)*D + d];
        }
        scratch[lane] = sum;
        __syncthreads();
        for (int stride = blockDim.x/2; stride > 0; stride >>= 1) {
            if (lane < stride) scratch[lane] += scratch[lane + stride];
            __syncthreads();
        }
        if (lane == 0) scores[s] = scratch[0] * scale;
        __syncthreads();
    }
    if (lane == 0) {
        float m = scores[0];
        for (int s = 1; s <= t; ++s) m = fmaxf(m, scores[s]);
        float denom = 0.0f;
        for (int s = 0; s <= t; ++s) { float e = expf(scores[s] - m); scores[s] = e; denom += e; }
        scratch[0] = 1.0f / denom;
    }
    __syncthreads();
    float inv = scratch[0];
    for (int d = lane; d < D; d += blockDim.x) {
        float acc = 0.0f;
        for (int s = 0; s <= t; ++s) acc += (scores[s] * inv) * v[((b*T + s)*Hkv + hk)*D + d];
        out[((b*T + t)*Hq + h)*D + d] = acc;
    }
}

void gqa_forward_cuda(torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor out) {
    const int B = (int)q.size(0);
    const int T = (int)q.size(1);
    const int Hq = (int)q.size(2);
    const int D = (int)q.size(3);
    const int Hkv = (int)k.size(2);
    const int blocks = B * Hq * T;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    if (D == 32 && T <= 80) {
        gqa_kernel_d32<<<blocks, 32, 0, stream>>>(q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), out.data_ptr<float>(), B, T, Hq, Hkv);
    } else {
        int threads = 128;
        size_t shmem = (size_t)(T + threads) * sizeof(float);
        gqa_kernel_generic<<<blocks, threads, shmem, stream>>>(q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), out.data_ptr<float>(), B, T, Hq, Hkv, D);
    }
}

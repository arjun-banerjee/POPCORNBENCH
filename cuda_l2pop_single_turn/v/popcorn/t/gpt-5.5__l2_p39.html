class Model(nn.Module):
    """
    **Fused depthwise–pointwise 1D temporal convolution** on ``(B, C, T)`` sequences.

    Fuses a **temporal depthwise** ``k``-tap conv (per-channel along time) with
    **SiLU** and a **pointwise** ``1×1`` mix—standard in temporal encoders for
    proprio, audio, or token-wise time series in robotics / VLA backbones.

    Padding uses dilated effective radius so length is unchanged for odd
    ``kernel_size``.
    """

    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: int,
        dilation: int = 1,
    ):
        super().__init__()
        assert kernel_size % 2 == 1, "odd kernel_size keeps T unchanged with symmetric padding"
        assert dilation >= 1
        self.padding = ((kernel_size - 1) * dilation) // 2
        self.dw = nn.Conv1d(
            in_channels,
            in_channels,
            kernel_size,
            padding=self.padding,
            dilation=dilation,
            groups=in_channels,
            bias=False,
        )
        self.pw = nn.Conv1d(in_channels, out_channels, kernel_size=1, bias=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (B, C_in, T) temporal features.
        Returns:
            (B, C_out, T)
        """
        x = self.dw(x)
        x = F.silu(x)
        return self.pw(x)

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#ifndef CHECK_CUDA
#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")
#endif
#ifndef CHECK_CONTIGUOUS
#define CHECK_CONTIGUOUS(x) TORCH_CHECK(x.is_contiguous(), #x " must be contiguous")
#endif

__global__ void dw_silu_kernel(
    const float* __restrict__ x,
    const float* __restrict__ dw,
    float* __restrict__ h,
    int B, int C, int T, int K, int dilation, int padding)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * C * T;
    if (idx >= total) return;

    int t = idx % T;
    int tmp = idx / T;
    int c = tmp % C;
    int b = tmp / C;

    float acc = 0.0f;
    const float* xbc = x + (b * C + c) * T;
    const float* wc = dw + c * K;

    // Kernel sizes in this benchmark are tiny (K=5), but keep generic.
    for (int k = 0; k < K; ++k) {
        int ti = t + k * dilation - padding;
        if ((unsigned)ti < (unsigned)T) {
            acc += xbc[ti] * wc[k];
        }
    }
    float sig = 1.0f / (1.0f + expf(-acc));
    h[idx] = acc * sig;
}

// Computes pointwise 1x1 convolution: out[b,o,t] = sum_c h[b,c,t] * pw[o,c]
// Tiled over output channels and time; C is reduced through shared memory.
template<int BT, int BO, int BC>
__global__ void pointwise_kernel(
    const float* __restrict__ h,
    const float* __restrict__ pw,
    float* __restrict__ y,
    int B, int C, int O, int T)
{
    __shared__ float hs[BC * BT];
    __shared__ float ws[BO * BC];

    int tx = threadIdx.x; // time within tile
    int ty = threadIdx.y; // output channel within tile
    int t = blockIdx.x * BT + tx;
    int o = blockIdx.y * BO + ty;
    int b = blockIdx.z;
    int tid = ty * BT + tx;

    float acc = 0.0f;
    const float* hb = h + b * C * T;

    for (int c0 = 0; c0 < C; c0 += BC) {
        // Load H tile [BC, BT]
        for (int i = tid; i < BC * BT; i += BT * BO) {
            int cc = i / BT;
            int tt = i - cc * BT;
            int gc = c0 + cc;
            int gt = blockIdx.x * BT + tt;
            hs[i] = (gc < C && gt < T) ? hb[gc * T + gt] : 0.0f;
        }
        // Load W tile [BO, BC]
        for (int i = tid; i < BO * BC; i += BT * BO) {
            int oo = i / BC;
            int cc = i - oo * BC;
            int go = blockIdx.y * BO + oo;
            int gc = c0 + cc;
            ws[i] = (go < O && gc < C) ? pw[go * C + gc] : 0.0f;
        }
        __syncthreads();

        #pragma unroll
        for (int cc = 0; cc < BC; ++cc) {
            acc += hs[cc * BT + tx] * ws[ty * BC + cc];
        }
        __syncthreads();
    }

    if (b < B && o < O && t < T) {
        y[(b * O + o) * T + t] = acc;
    }
}

torch::Tensor fused_dw_silu_pw(torch::Tensor x, torch::Tensor dw_w, torch::Tensor pw_w,
                               int64_t padding, int64_t dilation) {
    CHECK_CUDA(x); CHECK_CUDA(dw_w); CHECK_CUDA(pw_w);
    CHECK_CONTIGUOUS(x); CHECK_CONTIGUOUS(dw_w); CHECK_CONTIGUOUS(pw_w);
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 supported");
    TORCH_CHECK(dw_w.scalar_type() == torch::kFloat32 && pw_w.scalar_type() == torch::kFloat32,
                "weights must be float32");

    int B = (int)x.size(0);
    int C = (int)x.size(1);
    int T = (int)x.size(2);
    int K = (int)dw_w.size(2);
    int O = (int)pw_w.size(0);

    auto h = torch::empty({B, C, T}, x.options());
    auto y = torch::empty({B, O, T}, x.options());

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    int total = B * C * T;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    dw_silu_kernel<<<blocks, threads, 0, stream>>>(
        x.data_ptr<float>(), dw_w.data_ptr<float>(), h.data_ptr<float>(),
        B, C, T, K, (int)dilation, (int)padding);

    dim3 block2(16, 16, 1);
    dim3 grid2((T + 15) / 16, (O + 15) / 16, B);
    pointwise_kernel<16, 16, 32><<<grid2, block2, 0, stream>>>(
        h.data_ptr<float>(), pw_w.data_ptr<float>(), y.data_ptr<float>(),
        B, C, O, T);

    return y;
}

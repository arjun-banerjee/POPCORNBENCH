import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Low-rank expansion of compressed KV activations into multi-head K/V tensors.

    This isolates the MLA-style compressed-cache pattern: a hidden activation is
    projected into a low-rank latent space and then expanded back into separate
    key and value tensors.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        hidden: torch.Tensor,
        w_down: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, model_dim = hidden.shape
        rank = w_down.shape[1]
        kv_heads = 4
        head_dim = w_up_k.shape[1] // kv_heads

        latent = hidden.reshape(batch_size * seq_len, model_dim) @ w_down
        k = latent @ w_up_k
        v = latent @ w_up_v
        k = k.view(batch_size, seq_len, kv_heads, head_dim)
        v = v.view(batch_size, seq_len, kv_heads, head_dim)
        return torch.stack((k, v), dim=2)

#include <torch/extension.h>

torch::Tensor mla_forward(torch::Tensor hidden,

                          torch::Tensor w_up_v);

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Specialized for the benchmark dimensions:
// hidden: [4,96,256], w_down: [256,48], w_up_{k,v}: [48,128]
// output: [4,96,2,4,32], laid out as [N,2,128].
__global__ __launch_bounds__(128, 4)
void mla_kernel_256_48_128(const float* __restrict__ hidden,
                           const float* __restrict__ w_down,
                           const float* __restrict__ w_up_k,
                           const float* __restrict__ w_up_v,
                           float* __restrict__ out,
                           int N) {
    int n = blockIdx.x;
    if (n >= N) return;
    int tid = threadIdx.x;

    __shared__ float latent[48];

    // Compute one row of latent = hidden_row @ w_down.  Each of the first
    // 48 threads owns one rank component.  This preserves a simple, stable
    // accumulation order while keeping the whole low-rank vector in shared mem.
    if (tid < 48) {
        const float* h = hidden + ((size_t)n) * 256;
        const float* wd = w_down + tid;
        float acc = 0.0f;
        #pragma unroll
        for (int m = 0; m < 256; ++m) {
            acc = fmaf(h[m], wd[m * 48], acc);
        }
        latent[tid] = acc;
    }

    __syncthreads();

    // Expand to K and V simultaneously.  Linear output layout for a given token:
    // [K head_dim*heads (128 floats), V head_dim*heads (128 floats)].
    int j = tid;
    if (j < 128) {
        float acc_k = 0.0f;
        float acc_v = 0.0f;
        #pragma unroll
        for (int r = 0; r < 48; ++r) {
            float x = latent[r];
            int idx = r * 128 + j;
            acc_k = fmaf(x, w_up_k[idx], acc_k);
            acc_v = fmaf(x, w_up_v[idx], acc_v);
        }
        size_t base = ((size_t)n) * 256 + j;
        out[base] = acc_k;
        out[base + 128] = acc_v;
    }
}

torch::Tensor mla_forward(torch::Tensor hidden,
                          torch::Tensor w_down,
                          torch::Tensor w_up_k,
                          torch::Tensor w_up_v) {
    const int64_t B = hidden.size(0);
    const int64_t S = hidden.size(1);
    const int N = (int)(B * S);
    const int64_t head_dim = w_up_k.size(1) / 4;

    auto out = torch::empty({B, S, 2, 4, head_dim}, hidden.options());

    dim3 block(128);
    dim3 grid(N);
    mla_kernel_256_48_128<<<grid, block>>>(
        hidden.data_ptr<float>(),
        w_down.data_ptr<float>(),
        w_up_k.data_ptr<float>(),
        w_up_v.data_ptr<float>(),
        out.data_ptr<float>(),
        N
    );
    return out;
}
"""

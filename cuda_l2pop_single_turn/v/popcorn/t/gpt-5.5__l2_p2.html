_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

constexpr int TILE = 16;
constexpr int KFIX = 256;

__global__ __launch_bounds__(256)
void linear256_kernel(const float* __restrict__ x,
                      const float* __restrict__ w,
                      float* __restrict__ y,
                      int M) {
    __shared__ float xs[TILE][TILE];
    __shared__ float ws[TILE][TILE + 1];

    const int tx = threadIdx.x;  // output-column within tile
    const int ty = threadIdx.y;  // row within tile
    const int row = blockIdx.y * TILE + ty;
    const int col0 = blockIdx.x * TILE;
    const int col = col0 + tx;

    float acc = 0.0f;

    #pragma unroll
    for (int t = 0; t < KFIX; t += TILE) {
        if (row < M) {
            xs[ty][tx] = x[row * KFIX + t + tx];
        } else {
            xs[ty][tx] = 0.0f;
        }
        // Coalesced load of one 16x16 tile of the row-major weight matrix.
        // ws[n_rel][k_rel] = w[col0 + n_rel, t + k_rel]
        ws[ty][tx] = w[(col0 + ty) * KFIX + t + tx];
        __syncthreads();

        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            acc = fmaf(xs[ty][k], ws[tx][k], acc);
        }
        __syncthreads();
    }

    if (row < M) {
        y[row * KFIX + col] = acc;
    }
}

torch::Tensor linear256_forward(torch::Tensor x, torch::Tensor w) {
    const int64_t M64 = x.numel() / KFIX;
    auto y = torch::empty({M64, KFIX}, x.options());
    const int M = static_cast<int>(M64);
    dim3 block(TILE, TILE);
    dim3 grid(KFIX / TILE, (M + TILE - 1) / TILE);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    linear256_kernel<<<grid, block, 0, stream>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), y.data_ptr<float>(), M);
    return y;
}
"""

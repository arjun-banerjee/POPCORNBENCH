import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Evidence Lower Bound (ELBO) computation for a Variational Autoencoder.
    Combines a reconstruction term (binary cross-entropy) with the
    KL divergence between the approximate posterior q(z|x) = N(mu, sigma^2)
    and the prior p(z) = N(0, I).  The core loss function in VAE training.
    """

    def __init__(self, input_dim, hidden_dim, latent_dim):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
        )
        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)

        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim),
            nn.Sigmoid(),
        )

    def forward(self, x: torch.Tensor) -> tuple:
        """
        Args:
            x: (B, input_dim) – input data in [0, 1]
        Returns:
            (elbo, recon_loss, kl_div):
                elbo:       (B,)
                recon_loss: (B,)
                kl_div:     (B,)
        """
        h = self.encoder(x)
        mu = self.fc_mu(h)
        logvar = self.fc_logvar(h)

        std = torch.exp(0.5 * logvar)
        z = mu + std * torch.randn_like(std)

        x_recon = self.decoder(z)
        recon = F.binary_cross_entropy(x_recon, x, reduction="none").sum(-1)
        kl = -0.5 * (1 + logvar - mu.pow(2) - logvar.exp()).sum(-1)
        elbo = -(recon + kl)
        return elbo, recon, kl


input_dim = 784
hidden_dim = 256
latent_dim = 32
batch_size = 128


def get_inputs():
    return [torch.rand(batch_size, input_dim)]


def get_init_inputs():
    return [input_dim, hidden_dim, latent_dim]

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>
#include <vector>

std::vector<torch::Tensor> reparam_kl(torch::Tensor mu, torch::Tensor logvar, torch::Tensor eps);
std::vector<torch::Tensor> bce_elbo(torch::Tensor logits, torch::Tensor x, torch::Tensor kl);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>
#include <math.h>

__inline__ __device__ float warp_reduce_sum(float v) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, offset);
    }
    return v;
}

__inline__ __device__ float block_reduce_sum(float v) {
    static __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    v = warp_reduce_sum(v);
    if (lane == 0) shared[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (blockDim.x >> 5)) ? shared[lane] : 0.0f;
    if (wid == 0) v = warp_reduce_sum(v);
    return v;
}

__global__ void reparam_kl_kernel(const float* __restrict__ mu,
                                  const float* __restrict__ logvar,
                                  const float* __restrict__ eps,
                                  float* __restrict__ z,
                                  float* __restrict__ kl,
                                  int L) {
    int b = blockIdx.x;
    int base = b * L;
    float sum = 0.0f;
    for (int j = threadIdx.x; j < L; j += blockDim.x) {
        float m = mu[base + j];
        float lv = logvar[base + j];
        float ev = expf(lv);
        float stdv = expf(0.5f * lv);
        z[base + j] = m + stdv * eps[base + j];
        sum += 1.0f + lv - m * m - ev;
    }
    sum = block_reduce_sum(sum);
    if (threadIdx.x == 0) kl[b] = -0.5f * sum;
}

__global__ void bce_elbo_kernel(const float* __restrict__ logits,
                                const float* __restrict__ x,
                                const float* __restrict__ kl,
                                float* __restrict__ recon,
                                float* __restrict__ elbo,
                                int D) {
    int b = blockIdx.x;
    int base = b * D;
    float sum = 0.0f;
    for (int j = threadIdx.x; j < D; j += blockDim.x) {
        float a = logits[base + j];
        float t = x[base + j];
        // Stable BCEWithLogits: max(a,0) - a*t + log(1 + exp(-abs(a))).
        // This is algebraically BCE(sigmoid(a), t), avoiding separate sigmoid/BCE kernels.
        float aa = fabsf(a);
        float mx = a > 0.0f ? a : 0.0f;
        sum += mx - a * t + log1pf(expf(-aa));
    }
    sum = block_reduce_sum(sum);
    if (threadIdx.x == 0) {
        recon[b] = sum;
        elbo[b] = -(sum + kl[b]);
    }
}

std::vector<torch::Tensor> reparam_kl(torch::Tensor mu, torch::Tensor logvar, torch::Tensor eps) {
    const int B = (int)mu.size(0);
    const int L = (int)mu.size(1);
    auto z = torch::empty_like(mu);
    auto kl = torch::empty({B}, mu.options());
    const int threads = 64;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    reparam_kl_kernel<<<B, threads, 0, stream>>>(
        (const float*)mu.data_ptr<float>(),
        (const float*)logvar.data_ptr<float>(),
        (const float*)eps.data_ptr<float>(),
        (float*)z.data_ptr<float>(),
        (float*)kl.data_ptr<float>(),
        L);
    return {z, kl};
}

std::vector<torch::Tensor> bce_elbo(torch::Tensor logits, torch::Tensor x, torch::Tensor kl) {
    const int B = (int)logits.size(0);
    const int D = (int)logits.size(1);
    auto recon = torch::empty({B}, logits.options());
    auto elbo = torch::empty({B}, logits.options());
    const int threads = 256;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    bce_elbo_kernel<<<B, threads, 0, stream>>>(
        (const float*)logits.data_ptr<float>(),
        (const float*)x.data_ptr<float>(),
        (const float*)kl.data_ptr<float>(),
        (float*)recon.data_ptr<float>(),
        (float*)elbo.data_ptr<float>(),

    return {recon, elbo};
}

    def __init__(self, input_dim, hidden_dim, latent_dim):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
        )
        self.fc_mu = nn.Linear(hidden_dim, latent_dim)
        self.fc_logvar = nn.Linear(hidden_dim, latent_dim)
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim),
            nn.Sigmoid(),
        )

    def forward(self, x: torch.Tensor) -> tuple:
        # Keep the same parameter structure as the reference for state_dict compatibility,
        # but fuse the stochastic reparameterization/KL and final BCE/ELBO reductions.
        h = F.linear(x, self.encoder[0].weight, self.encoder[0].bias)
        h = F.relu(h, inplace=False)
        mu = F.linear(h, self.fc_mu.weight, self.fc_mu.bias)
        logvar = F.linear(h, self.fc_logvar.weight, self.fc_logvar.bias)

        eps = torch.randn_like(logvar)
        z, kl = _vae_ext.reparam_kl(mu.contiguous(), logvar.contiguous(), eps)

        d = F.linear(z, self.decoder[0].weight, self.decoder[0].bias)
        d = F.relu(d, inplace=False)
        logits = F.linear(d, self.decoder[2].weight, self.decoder[2].bias)

        recon, elbo = _vae_ext.bce_elbo(logits.contiguous(), x.contiguous(), kl)
        return elbo, recon, kl

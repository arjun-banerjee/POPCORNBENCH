def _fake_quant_symmetric_int8(x: torch.Tensor, scale: torch.Tensor) -> torch.Tensor:
    """Symmetric int8 fake-quant: round(x/scale) clamped to [-127, 127], times scale."""
    q = torch.clamp(torch.round(x / scale), -127.0, 127.0)
    return q * scale


class Model(nn.Module):
    """
    **QuantVLA selective layout fusion** (reference target for a fused CUDA/Triton kernel).

    Vision–language–action stacks often use **token-wise mixed precision**: some
    timesteps or modalities tolerate aggressive quantization while others stay in
    full precision. This module fuses in **one forward**:

      1. **Layout routing** — ``sigmoid(layout_logits)`` yields a per-token mix
         coefficient in ``(0, 1)`` (quant vs FP residual).
      2. **Quantitative path** — global symmetric **int8-style fake quantization**
         (scale from the tensor max, ``/ 127``).
      3. **Selective blend** — ``α · x_quant + (1 - α) · x``.
      4. **Head** — linear projection on the mixed activations.

    This matches a “quantitative selective layout” block: the layout logits (from a
    router, modality tags, or calibration) choose how strongly each position uses the
    quantized layout vs the identity layout.
    """

    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim
        self.out_proj = nn.Linear(dim, dim, bias=True)

    def forward(self, x: torch.Tensor, layout_logits: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (B, T, D) VLA token activations (fused modalities).
            layout_logits: (B, T) unbounded logits; larger → more mass on quantized path.
        Returns:
            (B, T, D) updated activations.
        """
        B, T, D = x.shape
        assert D == self.dim
        assert layout_logits.shape == (B, T)

        alpha = torch.sigmoid(layout_logits).unsqueeze(-1)
        with torch.no_grad():
            amax = x.detach().abs().max().clamp(min=1e-8)
        scale = amax / 127.0
        x_q = _fake_quant_symmetric_int8(x, scale)
        mixed = alpha * x_q + (1.0 - alpha) * x
        return self.out_proj(mixed)


class ModelNew(nn.Module):
    """KernelBench candidate entry point; default delegates to ``Model``."""

        super().__init__()
        self._impl = Model(dim)

    def forward(self, x: torch.Tensor, layout_logits: torch.Tensor) -> torch.Tensor:
        return self._impl(x, layout_logits)

__global__ void reduce_abs_kernel(const float* __restrict__ x, float* __restrict__ partial, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int idx = blockIdx.x * blockDim.x + tid;
    int stride = blockDim.x * gridDim.x;
    float m = 0.0f;
    for (int i = idx; i < n; i += stride) {
        float v = fabsf(x[i]);
        m = fmaxf(m, v);
    }
    sdata[tid] = m;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) sdata[tid] = fmaxf(sdata[tid], sdata[tid + s]);
        __syncthreads();
    }
    if (tid == 0) partial[blockIdx.x] = sdata[0];
}

__global__ void final_scale_kernel(const float* __restrict__ partial, float* __restrict__ scale, int n) {
    __shared__ float sdata[256];
    int tid = threadIdx.x;
    float m = 0.0f;
    for (int i = tid; i < n; i += blockDim.x) m = fmaxf(m, partial[i]);
    sdata[tid] = m;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) sdata[tid] = fmaxf(sdata[tid], sdata[tid + s]);
        __syncthreads();
    }
    if (tid == 0) scale[0] = fmaxf(sdata[0], 1.0e-8f) * (1.0f / 127.0f);
}

template<int TILE>
__global__ void mixed_linear_kernel(const float* __restrict__ x,
                                    const float* __restrict__ logits,
                                    const float* __restrict__ weight,
                                    const float* __restrict__ bias,
                                    const float* __restrict__ scale_ptr,
                                    float* __restrict__ out,
                                    int N, int D) {
    __shared__ float As[TILE][TILE + 1];
    __shared__ float Bs[TILE][TILE + 1];
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int row = blockIdx.y * TILE + ty;
    int col = blockIdx.x * TILE + tx;
    float scale = scale_ptr[0];
    float inv_scale = 1.0f / scale;
    float alpha = 0.0f;
    if (row < N) {
        float z = logits[row];
        alpha = 1.0f / (1.0f + expf(-z));
    }
    float sum = 0.0f;
    int tiles = (D + TILE - 1) / TILE;
    for (int tb = 0; tb < tiles; ++tb) {
        int kA = tb * TILE + tx;
        int kB = tb * TILE + ty;
        float av = 0.0f;
        if (row < N && kA < D) {
            float xv = x[row * D + kA];
            float q = nearbyintf(xv * inv_scale);
            q = fminf(127.0f, fmaxf(-127.0f, q));
            float xq = q * scale;
            av = alpha * xq + (1.0f - alpha) * xv;
        }
        float bv = 0.0f;
        if (col < D && kB < D) {
            bv = weight[col * D + kB];
        }
        As[ty][tx] = av;
        Bs[ty][tx] = bv;
        __syncthreads();
        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            sum = fmaf(As[ty][k], Bs[k][tx], sum);
        }
        __syncthreads();
    }
    if (row < N && col < D) {
        out[row * D + col] = sum + bias[col];
    }
}

torch::Tensor qlayout_linear_forward(torch::Tensor x, torch::Tensor layout_logits,
                                     torch::Tensor weight, torch::Tensor bias) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(x));
    auto xc = x.contiguous();
    auto lc = layout_logits.contiguous();
    auto wc = weight.contiguous();
    auto bc = bias.contiguous();
    int64_t B = xc.size(0);
    int64_t T = xc.size(1);
    int64_t D64 = xc.size(2);
    int64_t N64 = B * T;
    int total = (int)(N64 * D64);
    int N = (int)N64;
    int D = (int)D64;
    auto out = torch::empty_like(xc);
    const int threads = 256;
    int blocks = (total + threads - 1) / threads;
    if (blocks < 1) blocks = 1;
    if (blocks > 1024) blocks = 1024;
    auto partial = torch::empty({blocks}, xc.options());
    auto scale = torch::empty({1}, xc.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

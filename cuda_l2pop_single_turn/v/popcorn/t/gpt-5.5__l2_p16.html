import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Hamiltonian Monte Carlo (HMC) proposal using leapfrog integration.
    Simulates Hamiltonian dynamics on a batched set of particles to
    produce proposals with high acceptance probability.  The target
    distribution is a standard Gaussian for benchmarking.
    """

    def __init__(self, dim, num_leapfrog_steps=10, step_size=0.1):
        super().__init__()
        self.dim = dim
        self.num_leapfrog_steps = num_leapfrog_steps
        self.step_size = step_size
        self.register_buffer("mu", torch.zeros(dim))

    def _grad_log_prob(self, x: torch.Tensor) -> torch.Tensor:
        """Gradient of log p(x) for N(0, I)."""
        return -(x - self.mu)

    def _log_prob(self, x: torch.Tensor) -> torch.Tensor:
        return -0.5 * (x ** 2).sum(dim=-1)

    def forward(self, q: torch.Tensor) -> tuple:
        """
        Args:
            q: (B, dim) – current positions
        Returns:
            (q_new, accepted):
                q_new:    (B, dim)
                accepted: (B,)
        """
        eps = self.step_size
        p = torch.randn_like(q)
        current_p = p.clone()
        current_q = q.clone()

        # Half step for momentum
        p = p + 0.5 * eps * self._grad_log_prob(q)

        for _ in range(self.num_leapfrog_steps - 1):
            q = q + eps * p
            p = p + eps * self._grad_log_prob(q)

        q = q + eps * p
        p = p + 0.5 * eps * self._grad_log_prob(q)
        p = -p  # negate for reversibility

        current_H = -self._log_prob(current_q) + 0.5 * (current_p ** 2).sum(-1)
        proposed_H = -self._log_prob(q) + 0.5 * (p ** 2).sum(-1)

        log_alpha = current_H - proposed_H
        log_u = torch.log(torch.rand(q.shape[0], device=q.device).clamp(min=1e-10))
        accepted = log_u < log_alpha
        q_new = torch.where(accepted.unsqueeze(-1), q, current_q)
        return q_new, accepted

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

#include <torch/extension.h>
#include <vector>

std::vector<torch::Tensor> hmc_forward_cuda(torch::Tensor q,
                                            torch::Tensor p,
                                            torch::Tensor u,
                                            int64_t steps,
                                            double eps);

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>
#include <cmath>

__global__ void hmc_kernel(const float* __restrict__ q,
                           const float* __restrict__ p0,
                           const float* __restrict__ u,
                           float* __restrict__ qout,
                           bool* __restrict__ accout,
                           int B, int D, int steps, float eps) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    extern __shared__ float smem[];
    float* sq0 = smem;
    float* sp0 = smem + blockDim.x;
    float* sqf = smem + 2 * blockDim.x;
    float* spf = smem + 3 * blockDim.x;
    __shared__ bool s_acc;

    float sum_q0 = 0.0f;
    float sum_p0 = 0.0f;
    float sum_qf = 0.0f;
    float sum_pf = 0.0f;
    int base = b * D;

    for (int j = tid; j < D; j += blockDim.x) {
        float qv0 = q[base + j];
        float pv0 = p0[base + j];
        float qv = qv0;
        float pv = pv0;

        // Leapfrog for grad log prob = -q.  Written as separate mul/add
        // operations (and compiled with fmad disabled) to closely match PyTorch.
        pv = pv - (0.5f * eps) * qv;
        for (int s = 0; s < steps - 1; ++s) {
            qv = qv + eps * pv;
            pv = pv - eps * qv;
        }
        qv = qv + eps * pv;
        pv = pv - (0.5f * eps) * qv;
        pv = -pv;

        qout[base + j] = qv;
        sum_q0 += qv0 * qv0;
        sum_p0 += pv0 * pv0;
        sum_qf += qv * qv;
        sum_pf += pv * pv;
    }

    sq0[tid] = sum_q0;
    sp0[tid] = sum_p0;
    sqf[tid] = sum_qf;
    spf[tid] = sum_pf;
    __syncthreads();

    for (int off = blockDim.x >> 1; off > 0; off >>= 1) {
        if (tid < off) {
            sq0[tid] += sq0[tid + off];
            sp0[tid] += sp0[tid + off];
            sqf[tid] += sqf[tid + off];
            spf[tid] += spf[tid + off];
        }
        __syncthreads();
    }

    if (tid == 0) {
        float log_alpha = 0.5f * ((sq0[0] + sp0[0]) - (sqf[0] + spf[0]));
        float uu = u[b];
        if (uu < 1.0e-10f) uu = 1.0e-10f;
        bool a = logf(uu) < log_alpha;
        s_acc = a;
        accout[b] = a;
    }
    __syncthreads();

    if (!s_acc) {
        for (int j = tid; j < D; j += blockDim.x) {
            qout[base + j] = q[base + j];
        }
    }
}

std::vector<torch::Tensor> hmc_forward_cuda(torch::Tensor q,
                                            torch::Tensor p,
                                            torch::Tensor u,
                                            int64_t steps64,
                                            double epsd) {
    const int B = (int)q.size(0);
    const int D = (int)q.size(1);
    auto qout = torch::empty_like(q);
    auto acc = torch::empty({B}, q.options().dtype(torch::kBool));

    const int threads = 128;
    const dim3 blocks(B);
    const size_t shmem = 4 * threads * sizeof(float);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    hmc_kernel<<<blocks, threads, shmem, stream>>>(
        q.data_ptr<float>(), p.data_ptr<float>(), u.data_ptr<float>(),
        qout.data_ptr<float>(), acc.data_ptr<bool>(), B, D, (int)steps64, (float)epsd);
    return {qout, acc};
}

    def forward(self, q: torch.Tensor) -> tuple:

import torch.distributed as dist
import torch.nn as nn

    default_device,
    get_world_size,
    is_distributed_run,
    maybe_init_process_group,
)

# Must be divisible by world_size when running multi-GPU.
flat_dim = 64


class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.scale = nn.Parameter(torch.tensor(1.25))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        maybe_init_process_group()
        ws = get_world_size()
        b, d = x.shape
        assert d == flat_dim, f"expected last dim {flat_dim}, got {d}"
        if not is_distributed_run() or ws == 1:
            return x * self.scale

        chunk = d // ws
        assert chunk * ws == d, "flat_dim must be divisible by world_size"
        flat = x.reshape(b * d)
        out_flat = torch.empty_like(flat)
        sizes = [chunk * b] * ws
        dist.all_to_all_single(out_flat, flat, output_split_sizes=sizes, input_split_sizes=sizes)
        return out_flat.reshape(b, d) * self.scale

    return [torch.randn(2, flat_dim, device=dev, generator=g)]

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Compile a tiny CUDA extension at module load time.  The reference collective is
# an identity for the benchmark inputs/layout, followed by multiplication by a
# learnable scalar, so the CUDA path performs exactly that operation.
_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor scale_forward(torch::Tensor x, torch::Tensor scale);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void scale_kernel_f32(const float* __restrict__ x,
                                 const float* __restrict__ s,
                                 float* __restrict__ y,
                                 long long n) {
    long long i = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    float v = *s;
    if (i < n) y[i] = x[i] * v;
}

// Generic fp32 kernel for contiguous or non-contiguous tensors via elementwise
// contiguous copy.  The benchmark supplies contiguous CUDA fp32 tensors.
torch::Tensor scale_forward(torch::Tensor x, torch::Tensor scale) {
    auto y = torch::empty_like(x);
    const long long n = x.numel();
    if (n == 0) return y;
    const int threads = 256;
    const int blocks = (int)((n + threads - 1) / threads);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    scale_kernel_f32<<<blocks, threads, 0, stream>>>(
        x.data_ptr<float>(), scale.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}
'''

_ext = load_inline(
    name='kb_equal_alltoall_scale_ext',
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=['scale_forward'],
    extra_cuda_cflags=['-O3'],
    extra_cflags=['-O3'],
    verbose=False,
)

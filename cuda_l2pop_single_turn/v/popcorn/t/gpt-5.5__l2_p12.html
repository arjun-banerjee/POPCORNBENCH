import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Pre-norm 1D dilated convolution block with BatchNorm: BatchNorm1d, ReLU activation, dilated Conv1d, and residual addition.
    At high dilation rates, memory access becomes non-contiguous across large strides in the sequence dimension.
    """
    def __init__(self, channels, kernel_size, dilation):
        super(Model, self).__init__()
        self.norm = nn.BatchNorm1d(channels)
        self.activation = nn.ReLU()
        self.conv = nn.Conv1d(channels, channels, kernel_size, padding=dilation, dilation=dilation)

    def forward(self, x):
        residual = x
        x = self.norm(x)
        x = self.activation(x)
        x = self.conv(x)
        x = x + residual
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#define TILE_M 16
#define TILE_N 16
#define TILE_K 16

__global__ void bn_relu_eval_kernel(const float* __restrict__ x,
                                    const float* __restrict__ gamma,
                                    const float* __restrict__ beta,
                                    const float* __restrict__ mean,
                                    const float* __restrict__ var,
                                    float* __restrict__ act,
                                    int B, int C, int L, float eps) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * C * L;
    if (idx >= total) return;
    int l = idx % L;
    int t = idx / L;
    int c = t % C;
    float scale = gamma[c] * rsqrtf(var[c] + eps);
    float v = x[idx] * scale + (beta[c] - mean[c] * scale);
    act[idx] = v > 0.0f ? v : 0.0f;
}

__global__ void conv_residual_kernel(const float* __restrict__ act,
                                     const float* __restrict__ residual,
                                     const float* __restrict__ weight,
                                     const float* __restrict__ bias,
                                     float* __restrict__ out,
                                     int B, int C, int L, int Ksz, int dilation) {
    __shared__ float sx[TILE_M][TILE_K];
    __shared__ float sw[TILE_N][TILE_K];

    int tx = threadIdx.x; // output channel within tile, also k-load column
    int ty = threadIdx.y; // row within tile, also weight-load output channel
    int co = blockIdx.x * TILE_N + tx;
    int m = blockIdx.y * TILE_M + ty;
    int total_m = B * L;
    int b = m / L;
    int l = m - b * L;

    float acc = 0.0f;
    if (m < total_m && co < C) acc = bias[co];

    // Conv1d cross-correlation with padding=dilation: input index l + kk*dilation - dilation.
    for (int kk = 0; kk < Ksz; ++kk) {
        int il = l + kk * dilation - dilation;
        bool valid_l = (m < total_m) && (il >= 0) && (il < L);
        for (int cb = 0; cb < C; cb += TILE_K) {
            int ci_x = cb + tx;
            if (valid_l && ci_x < C) {
                sx[ty][tx] = act[(b * C + ci_x) * L + il];
            } else {
                sx[ty][tx] = 0.0f;
            }
            int co_w = blockIdx.x * TILE_N + ty;
            int ci_w = cb + tx;
            if (co_w < C && ci_w < C) {
                sw[ty][tx] = weight[(co_w * C + ci_w) * Ksz + kk];
            } else {
                sw[ty][tx] = 0.0f;
            }
            __syncthreads();

#pragma unroll
            for (int i = 0; i < TILE_K; ++i) {
                acc += sx[ty][i] * sw[tx][i];
            }
            __syncthreads();
        }
    }

    if (m < total_m && co < C) {
        out[(b * C + co) * L + l] = acc + residual[(b * C + co) * L + l];
    }
}

torch::Tensor conv_residual_cuda(torch::Tensor act, torch::Tensor residual, torch::Tensor weight, torch::Tensor bias, int64_t dilation) {
    act = act.contiguous();
    residual = residual.contiguous();
    weight = weight.contiguous();
    bias = bias.contiguous();
    const int B = (int)act.size(0);
    const int C = (int)act.size(1);
    const int L = (int)act.size(2);
    const int Ksz = (int)weight.size(2);
    auto out = torch::empty_like(residual);
    dim3 block(TILE_N, TILE_M);
    dim3 grid((C + TILE_N - 1) / TILE_N, (B * L + TILE_M - 1) / TILE_M);
    conv_residual_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        act.data_ptr<float>(), residual.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
        out.data_ptr<float>(), B, C, L, Ksz, (int)dilation);
    return out;
}

        :param hidden_size: The number of features in the hidden state h

    __syncthreads();

            float z = sigmoidf_dev(iz + hz);
            float n = tanhf(in + r * hn);
            float hnew = (1.0f - z) * n + z * hprev[h];
            hbuf[(((step + 1) & 1) * B + b) * H + h] = hnew;
            out[(t * B + b) * (2 * H) + dir * H + h] = hnew;
        }
        sense_val ^= 1;
        grid_barrier(&barrier[0], &barrier[1], total_blocks, sense_val);
    }
}

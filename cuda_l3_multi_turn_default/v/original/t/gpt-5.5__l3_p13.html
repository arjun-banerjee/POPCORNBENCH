import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        """
        :param num_input_features: The number of input feature maps
        :param num_output_features: The number of output feature maps
        """
        super(Model, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        """
        :param x: Input tensor of shape (batch_size, num_input_features, height, width)
        :return: Downsampled tensor with reduced number of feature maps
        """
        return self.transition(x)

batch_size = 128
num_input_features = 32
num_output_features = 64
height, width = 256, 256

def get_inputs():
    return [torch.rand(batch_size, num_input_features, height, width)]

def get_init_inputs():
    return [num_input_features, num_output_features]

__device__ __forceinline__ float to_tf32(float x){
    unsigned int u = __float_as_uint(x);
    u += 0x00001000u + ((u >> 13) & 1u);
    u &= 0xFFFFE000u;
    return __uint_as_float(u);
}
__device__ __forceinline__ float tf32muladd(float a, float b, float c){
    return fmaf(to_tf32(a), to_tf32(b), c);
}

// Specialized fast path for the benchmark shape C=32, O=64.  It uses the
// identity AvgPool(Conv1x1(y)) == Conv1x1(AvgPool(y)), but preserves cuDNN's
// TF32 multiply semantics by applying TF32 rounding before every per-pixel
// multiply, then averaging the four independently accumulated conv results.
__global__ void relu_conv_pool_tiled64_kernel(const float* __restrict__ bn,
                                              float* __restrict__ out,
                                              const float* __restrict__ w,
                                              int N, int H, int W, int OH, int OW) {
    __shared__ float sh[4 * 32 * 4]; // four pooled sites, 32 channels, four pixels
    int tid = threadIdx.x;
    int64_t P = (int64_t)N * OH * OW;
    int64_t pbase = (int64_t)blockIdx.x * 4;

    for (int k = tid; k < 4*32*4; k += blockDim.x) {
        int item = k / (32*4);
        int rem = k - item * (32*4);
        int c = rem >> 2;
        int q = rem & 3;
        int64_t p = pbase + item;
        float v = 0.0f;
        if (p < P) {
            int ow = p % OW;
            int64_t t = p / OW;
            int oh = t % OH;
            int n = t / OH;
            int h0 = oh * 2 + (q >> 1);
            int w0 = ow * 2 + (q & 1);
            v = bn[((int64_t)n * 32 + c) * H * W + h0 * W + w0];
            v = v > 0.0f ? v : 0.0f;
        }
        sh[k] = v;
    }
    __syncthreads();

    int item = tid >> 6;   // 0..3
    int o = tid & 63;      // output channel
    int64_t p = pbase + item;
    if (p < P) {
        float s0 = 0.0f, s1 = 0.0f, s2 = 0.0f, s3 = 0.0f;
        #pragma unroll
        for (int c = 0; c < 32; ++c) {
            float ww = w[o * 32 + c];
            int off = item * (32*4) + c * 4;
            s0 = tf32muladd(sh[off + 0], ww, s0);
            s1 = tf32muladd(sh[off + 1], ww, s1);
            s2 = tf32muladd(sh[off + 2], ww, s2);
            s3 = tf32muladd(sh[off + 3], ww, s3);
        }
        int ow = p % OW;
        int64_t t = p / OW;
        int oh = t % OH;
        int n = t / OH;
        out[((int64_t)n * 64 + o) * OH * OW + oh * OW + ow] = (s0 + s1 + s2 + s3) * 0.25f;
    }
}

__global__ void relu_conv_pool_generic_kernel(const float* __restrict__ bn,float* __restrict__ out,const float* __restrict__ w,int N,int C,int H,int W,int O,int OH,int OW){
    int64_t total=(int64_t)N*O*OH*OW;
    for(int64_t idx=(int64_t)blockIdx.x*blockDim.x+threadIdx.x; idx<total; idx+=(int64_t)gridDim.x*blockDim.x){
        int ow=idx%OW; int64_t t=idx/OW; int oh=t%OH; t/=OH; int o=t%O; int n=t/O;
        int h0=oh*2, w0=ow*2; float s0=0.f,s1=0.f,s2=0.f,s3=0.f;
        for(int c=0;c<C;c++){
            int64_t base=((int64_t)n*C+c)*H*W+h0*W+w0; float ww=w[o*C+c];
            float v0=bn[base]; v0=v0>0.f?v0:0.f; float v1=bn[base+1]; v1=v1>0.f?v1:0.f;
            float v2=bn[base+W]; v2=v2>0.f?v2:0.f; float v3=bn[base+W+1]; v3=v3>0.f?v3:0.f;
            s0=tf32muladd(v0,ww,s0); s1=tf32muladd(v1,ww,s1); s2=tf32muladd(v2,ww,s2); s3=tf32muladd(v3,ww,s3);
        }
        out[idx]=(s0+s1+s2+s3)*0.25f;
    }
}

torch::Tensor transition_forward(torch::Tensor x, torch::Tensor bn_weight, torch::Tensor bn_bias,
                                 torch::Tensor running_mean, torch::Tensor running_var,
                                 torch::Tensor conv_weight, bool training, double eps, double momentum) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(x));
    x = x.contiguous();
    conv_weight = conv_weight.contiguous();
    auto bn = at::batch_norm(x, bn_weight, bn_bias, running_mean, running_var, training, momentum, eps, true);
    int N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3), O=conv_weight.size(0);
    int OH=H/2, OW=W/2;
    auto out = torch::empty({N,O,OH,OW}, x.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    if (C==32 && O==64) {
        int64_t P = (int64_t)N * OH * OW;
        int blocks = (int)((P + 3) / 4);
        relu_conv_pool_tiled64_kernel<<<blocks, 256, 0, stream>>>(bn.data_ptr<float>(), out.data_ptr<float>(), conv_weight.data_ptr<float>(), N, H, W, OH, OW);
    } else {
        int threads=256;
        int blocks=(int)std::min<int64_t>(65535, ((int64_t)N*O*OH*OW+threads-1)/threads);
        relu_conv_pool_generic_kernel<<<blocks,threads,0,stream>>>(bn.data_ptr<float>(), out.data_ptr<float>(), conv_weight.data_ptr<float>(), N,C,H,W,O,OH,OW);
    }
    return out;
}

class ModelNew(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super(ModelNew, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2)
        )
    def forward(self, x):
        bn = self.transition[0]
        conv = self.transition[2]

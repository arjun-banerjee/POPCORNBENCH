// Specialized fast path for the benchmark shape C=32, O=64.  It uses the
// identity AvgPool(Conv1x1(y)) == Conv1x1(AvgPool(y)), but preserves cuDNN's
// TF32 multiply semantics by applying TF32 rounding before every per-pixel
// multiply, then averaging the four independently accumulated conv results.
__global__ void relu_conv_pool_tiled64_kernel(const float* __restrict__ bn,
                                              float* __restrict__ out,
                                              const float* __restrict__ w,
                                              int N, int H, int W, int OH, int OW) {
    __shared__ float sh[4 * 32 * 4]; // four pooled sites, 32 channels, four pixels
    int tid = threadIdx.x;
    int64_t P = (int64_t)N * OH * OW;
    int64_t pbase = (int64_t)blockIdx.x * 4;

    for (int k = tid; k < 4*32*4; k += blockDim.x) {
        int item = k / (32*4);
        int rem = k - item * (32*4);
        int c = rem >> 2;
        int q = rem & 3;
        int64_t p = pbase + item;
        float v = 0.0f;
        if (p < P) {
            int ow = p % OW;
            int64_t t = p / OW;
            int oh = t % OH;
            int n = t / OH;
            int h0 = oh * 2 + (q >> 1);
            int w0 = ow * 2 + (q & 1);
            v = bn[((int64_t)n * 32 + c) * H * W + h0 * W + w0];
            v = v > 0.0f ? v : 0.0f;
        }
        sh[k] = v;
    }
    __syncthreads();

    int item = tid >> 6;   // 0..3
    int o = tid & 63;      // output channel
    int64_t p = pbase + item;

        int n = t / OH;
        out[((int64_t)n * 64 + o) * OH * OW + oh * OW + ow] = (s0 + s1 + s2 + s3) * 0.25f;
    }
}

__global__ void relu_conv_pool_generic_kernel(const float* __restrict__ bn,float* __restrict__ out,const float* __restrict__ w,int N,int C,int H,int W,int O,int OH,int OW){
    int64_t total=(int64_t)N*O*OH*OW;
    for(int64_t idx=(int64_t)blockIdx.x*blockDim.x+threadIdx.x; idx<total; idx+=(int64_t)gridDim.x*blockDim.x){
        int ow=idx%OW; int64_t t=idx/OW; int oh=t%OH; t/=OH; int o=t%O; int n=t/O;
        int h0=oh*2, w0=ow*2; float s0=0.f,s1=0.f,s2=0.f,s3=0.f;
        for(int c=0;c<C;c++){
            int64_t base=((int64_t)n*C+c)*H*W+h0*W+w0; float ww=w[o*C+c];
            float v0=bn[base]; v0=v0>0.f?v0:0.f; float v1=bn[base+1]; v1=v1>0.f?v1:0.f;
            float v2=bn[base+W]; v2=v2>0.f?v2:0.f; float v3=bn[base+W+1]; v3=v3>0.f?v3:0.f;

        }
        out[idx]=(s0+s1+s2+s3)*0.25f;
    }
}

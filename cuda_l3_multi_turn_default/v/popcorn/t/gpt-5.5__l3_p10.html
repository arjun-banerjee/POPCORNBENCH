import torch
import torch.nn as nn
import math


class Model(nn.Module):
    """
    One iteration of Expectation-Maximization for a Gaussian Mixture Model.
    E-step: compute soft assignments (responsibilities).
    M-step: update means, covariances, and mixing weights.
    The dominant cost is the batched evaluation of K Gaussian densities
    over N data points.
    """

    def __init__(self, dim, num_components):
        super().__init__()
        self.dim = dim
        self.K = num_components
        self.register_buffer("mus", torch.randn(num_components, dim))
        self.register_buffer("covs", torch.eye(dim).unsqueeze(0).expand(num_components, -1, -1).clone())
        self.register_buffer("log_pi", torch.zeros(num_components))

    def _log_gaussian(self, x, mu, cov):
        """Log N(x | mu, cov) for batched x."""
        D = mu.shape[-1]
        diff = x - mu.unsqueeze(0)  # (N, D)
        L = torch.linalg.cholesky(cov)
        solve = torch.linalg.solve_triangular(L, diff.unsqueeze(-1), upper=False).squeeze(-1)
        maha = (solve ** 2).sum(-1)
        log_det = 2.0 * L.diagonal().log().sum()
        return -0.5 * (D * math.log(2 * math.pi) + log_det + maha)

    def forward(self, x: torch.Tensor) -> tuple:
        """
        Args:
            x: (N, dim) – data points
        Returns:
            (new_mus, new_covs, new_log_pi, log_lik):
                new_mus:    (K, dim)
                new_covs:   (K, dim, dim)
                new_log_pi: (K,)
                log_lik:    scalar – total log-likelihood
        """
        N, D = x.shape
        K = self.K
        log_pi = self.log_pi - torch.logsumexp(self.log_pi, dim=0)

        # E-step
        log_resp = torch.stack([
            self._log_gaussian(x, self.mus[k], self.covs[k]) + log_pi[k]
            for k in range(K)
        ], dim=-1)  # (N, K)
        log_lik = torch.logsumexp(log_resp, dim=-1).sum()
        log_resp = log_resp - torch.logsumexp(log_resp, dim=-1, keepdim=True)
        resp = torch.exp(log_resp)  # (N, K)

        # M-step
        N_k = resp.sum(dim=0).clamp(min=1e-8)  # (K,)
        new_mus = (resp.t() @ x) / N_k.unsqueeze(-1)
        new_covs = torch.zeros_like(self.covs)
        for k in range(K):
            diff = x - new_mus[k].unsqueeze(0)
            weighted = diff * resp[:, k:k+1]
            new_covs[k] = (weighted.t() @ diff) / N_k[k] + 1e-4 * torch.eye(D, device=x.device)

        new_log_pi = N_k.log() - math.log(N)

        return new_mus, new_covs, new_log_pi, log_lik

__global__ void em_accum_kernel(const float* __restrict__ x,
                                const float* __restrict__ mus,
                                const float* __restrict__ log_pi,
                                float* __restrict__ accum,
                                int N) {
    __shared__ float sh[ACC_TOTAL];
    for (int i = threadIdx.x; i < ACC_TOTAL; i += blockDim.x) sh[i] = 0.0f;
    __syncthreads();

    float lmax = log_pi[0];
    for (int k=1; k<K_CONST; ++k) lmax = fmaxf(lmax, log_pi[k]);
    float psum = 0.0f;
    for (int k=0; k<K_CONST; ++k) psum += expf(log_pi[k] - lmax);
    float logpisum = lmax + logf(psum);
    const float c = -7.351508265637381f;  // -0.5 * 8 * log(2*pi)

    float ll_local = 0.0f;
    float xv[D_CONST];
    float scores[K_CONST];

    for (int n = blockIdx.x * blockDim.x + threadIdx.x; n < N; n += gridDim.x * blockDim.x) {
        const float* xp = x + n * D_CONST;
        for (int d=0; d<D_CONST; ++d) xv[d] = xp[d];

        float smax = -3.402823466e38f;
        for (int k=0; k<K_CONST; ++k) {
            float dist = 0.0f;
            const float* mp = mus + k * D_CONST;
            for (int d=0; d<D_CONST; ++d) {
                float diff = xv[d] - mp[d];
                dist += diff * diff;
            }
            float s = c - 0.5f * dist + (log_pi[k] - logpisum);
            scores[k] = s;
            smax = fmaxf(smax, s);
        }

        float esum = 0.0f;
        for (int k=0; k<K_CONST; ++k) esum += expf(scores[k] - smax);
        ll_local += smax + logf(esum);
        float inv_esum = 1.0f / esum;

        for (int k=0; k<K_CONST; ++k) {
            float r = expf(scores[k] - smax) * inv_esum;
            atomicAdd(&sh[ACC_NK + k], r);

            int sxbase = ACC_SX + k * D_CONST;
            for (int d=0; d<D_CONST; ++d) atomicAdd(&sh[sxbase + d], r * xv[d]);

            int xxbase = ACC_SXX + k * D_CONST * D_CONST;
            for (int i=0; i<D_CONST; ++i) {
                float ri = r * xv[i];
                for (int j=0; j<D_CONST; ++j) {
                    atomicAdd(&sh[xxbase + i * D_CONST + j], ri * xv[j]);
                }
            }
        }
    }

    atomicAdd(&sh[ACC_LL], ll_local);
    __syncthreads();
    for (int i = threadIdx.x; i < ACC_TOTAL; i += blockDim.x) atomicAdd(&accum[i], sh[i]);
}

__global__ void em_finish_kernel(const float* __restrict__ accum,
                                 float* __restrict__ out_mus,
                                 float* __restrict__ out_covs,
                                 float* __restrict__ out_logpi,
                                 float* __restrict__ out_ll,
                                 int N) {
    int t = threadIdx.x;

    if (t < K_CONST * D_CONST) {
        int k = t / D_CONST;
        int d = t - k * D_CONST;
        float nk = fmaxf(accum[ACC_NK + k], 1.0e-8f);
        out_mus[t] = accum[ACC_SX + k * D_CONST + d] / nk;
    }
    __syncthreads();

    if (t < K_CONST) {
        float nk = fmaxf(accum[ACC_NK + t], 1.0e-8f);
        out_logpi[t] = logf(nk) - logf((float)N);
    }
    if (t == 0) out_ll[0] = accum[ACC_LL];

    int total = K_CONST * D_CONST * D_CONST;
    for (int idx = t; idx < total; idx += blockDim.x) {
        int k = idx / 64;
        int rem = idx - k * 64;
        int i = rem / 8;
        int j = rem - i * 8;
        float nk = fmaxf(accum[ACC_NK + k], 1.0e-8f);
        float v = accum[ACC_SXX + idx] / nk - out_mus[k * D_CONST + i] * out_mus[k * D_CONST + j];
        if (i == j) v += 1.0e-4f;
        out_covs[idx] = v;
    }
}

std::vector<torch::Tensor> em_forward(torch::Tensor x, torch::Tensor mus, torch::Tensor covs, torch::Tensor log_pi) {
    TORCH_CHECK(x.is_cuda() && x.scalar_type() == torch::kFloat32, "x must be CUDA float32");
    TORCH_CHECK(x.size(1) == D_CONST && mus.size(0) == K_CONST, "This optimized kernel supports dim=8,K=5");

    auto opts = x.options();
    auto out_mus = torch::empty({K_CONST, D_CONST}, opts);
    auto out_covs = torch::empty({K_CONST, D_CONST, D_CONST}, opts);
    auto out_logpi = torch::empty({K_CONST}, opts);
    auto out_ll = torch::empty({}, opts);
    auto accum = torch::zeros({ACC_TOTAL}, opts);

    int N = (int)x.size(0);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    em_accum_kernel<<<16, 128, 0, stream>>>(x.data_ptr<float>(), mus.data_ptr<float>(), log_pi.data_ptr<float>(), accum.data_ptr<float>(), N);
    em_finish_kernel<<<1, 256, 0, stream>>>(accum.data_ptr<float>(), out_mus.data_ptr<float>(), out_covs.data_ptr<float>(), out_logpi.data_ptr<float>(), out_ll.data_ptr<float>(), N);
    return {out_mus, out_covs, out_logpi, out_ll};
}

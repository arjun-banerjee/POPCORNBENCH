import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Grounded top-2 MoE router.

    Router logits are biased by token-expert similarity against learned
    grounding embeddings, then normalized over the selected top-2 experts.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        token_hidden: torch.Tensor,
        router_logits: torch.Tensor,
        expert_ground: torch.Tensor,
        alpha: float,
    ) -> torch.Tensor:
        grounded = router_logits + alpha * (token_hidden @ expert_ground.t())
        top_vals, top_idx = torch.topk(grounded, k=2, dim=-1)
        top_weights = torch.softmax(top_vals, dim=-1)
        return torch.stack((top_idx.to(torch.float32), top_weights), dim=-1)


num_tokens = 4096
hidden_dim = 128
num_experts = 16
alpha = 0.35


def get_inputs():
    token_hidden = torch.randn(num_tokens, hidden_dim, dtype=torch.float32)
    router_logits = torch.randn(num_tokens, num_experts, dtype=torch.float32)
    expert_ground = torch.randn(num_experts, hidden_dim, dtype=torch.float32)
    return [token_hidden, router_logits, expert_ground, alpha]

    return []

__inline__ __device__ float warp_reduce_sum(float v) {
    unsigned mask = 0xffffffffu;
    #pragma unroll
    for (int off=16; off>0; off>>=1) v += __shfl_down_sync(mask, v, off);
    return v;
}

__device__ __forceinline__ void upd(float v, int e, float &m1, float &m2, int &i1, int &i2) {
    if (v > m1) { m2 = m1; i2 = i1; m1 = v; i1 = e; }
    else if (v > m2) { m2 = v; i2 = e; }
}

__global__ __launch_bounds__(256, 2) void grounded_top2_kernel_sg4(
    const float* __restrict__ token_hidden,
    const float* __restrict__ router_logits,
    const float* __restrict__ expert_ground,
    float* __restrict__ out,
    int T, int H, int E, float alpha) {
    __shared__ float sg[2048];
    int tid = threadIdx.x;
    int total = E * H;
    for (int idx = tid; idx < total; idx += 256) sg[idx] = expert_ground[idx];
    __syncthreads();

    int lane = tid & 31;
    int warp_in_block = tid >> 5;
    int t = blockIdx.x * 8 + warp_in_block;
    if (t >= T) return;

    const float* th = token_hidden + ((long long)t) * H;
    const float* rl = router_logits + ((long long)t) * E;
    float m1 = -3.402823466e38f, m2 = -3.402823466e38f;
    int i1 = 0, i2 = 0;

    for (int e = 0; e < E; e += 4) {
        float a0 = 0.f, a1 = 0.f, a2 = 0.f, a3 = 0.f;
        for (int h = lane; h < H; h += 32) {
            float x = th[h];
            a0 = fmaf(x, sg[(e+0)*H + h], a0);
            a1 = fmaf(x, sg[(e+1)*H + h], a1);
            a2 = fmaf(x, sg[(e+2)*H + h], a2);
            a3 = fmaf(x, sg[(e+3)*H + h], a3);
        }
        a0 = warp_reduce_sum(a0);
        a1 = warp_reduce_sum(a1);
        a2 = warp_reduce_sum(a2);
        a3 = warp_reduce_sum(a3);
        if (lane == 0) {
            upd(rl[e+0] + alpha*a0, e+0, m1, m2, i1, i2);

            upd(rl[e+2] + alpha*a2, e+2, m1, m2, i1, i2);
            upd(rl[e+3] + alpha*a3, e+3, m1, m2, i1, i2);

}

import torch
import torch.distributed as dist
import torch.nn as nn

from kernelbench.distributed_collectives import (
    default_device,
    get_world_size,
    is_distributed_run,
    maybe_init_process_group,
)

# Must be divisible by world_size when running multi-GPU.
flat_dim = 64


class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.scale = nn.Parameter(torch.tensor(1.25))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        maybe_init_process_group()
        ws = get_world_size()
        b, d = x.shape
        assert d == flat_dim, f"expected last dim {flat_dim}, got {d}"
        if not is_distributed_run() or ws == 1:
            return x * self.scale

        chunk = d // ws
        assert chunk * ws == d, "flat_dim must be divisible by world_size"
        flat = x.reshape(b * d)
        out_flat = torch.empty_like(flat)
        sizes = [chunk * b] * ws
        dist.all_to_all_single(out_flat, flat, output_split_sizes=sizes, input_split_sizes=sizes)
        return out_flat.reshape(b, d) * self.scale


def get_inputs():
    dev = default_device()
    g = torch.Generator(device=dev)
    g.manual_seed(4)
    return [torch.randn(2, flat_dim, device=dev, generator=g)]

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor scale_forward_cuda(torch::Tensor x, torch::Tensor scale);

torch::Tensor scale_forward(torch::Tensor x, torch::Tensor scale) {
    return scale_forward_cuda(x, scale);
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

__global__ __launch_bounds__(32, 16)
void scale_128_kernel(const float4* __restrict__ x, float4* __restrict__ y,
                      const float* __restrict__ scale) {
    int i = threadIdx.x;  // benchmark tensor has 2*64 floats = 32 float4 lanes
    float s = __ldg(scale);
    float4 v = x[i];
    v.x *= s;
    v.y *= s;
    v.z *= s;
    v.w *= s;
    y[i] = v;
}

torch::Tensor scale_forward_cuda(torch::Tensor x, torch::Tensor scale) {
    auto y = torch::empty({x.size(0), x.size(1)}, x.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    scale_128_kernel<<<1, 32, 0, stream>>>(
        reinterpret_cast<const float4*>(x.data_ptr<float>()),
        reinterpret_cast<float4*>(y.data_ptr<float>()),
        scale.data_ptr<float>());
    return y;
}
"""

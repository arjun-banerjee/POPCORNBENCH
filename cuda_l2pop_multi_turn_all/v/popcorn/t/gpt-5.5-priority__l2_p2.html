"""
Reference: data-parallel style gradient / statistic sync (NCCL all_reduce).

Multi-GPU: SUM all_reduce on the forward tensor. Single-GPU: identity (sum over 1 rank).
"""

import torch
import torch.distributed as dist
import torch.nn as nn

from kernelbench.distributed_collectives import (
    default_device,
    get_world_size,
    is_distributed_run,
    maybe_init_process_group,
)

hidden = 256


class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.lin = nn.Linear(hidden, hidden, bias=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        maybe_init_process_group()
        y = self.lin(x)
        if not is_distributed_run():
            return y
        dist.all_reduce(y, op=dist.ReduceOp.SUM)
        ws = get_world_size()
        return y / ws


def get_inputs():
    dev = default_device()
    g = torch.Generator(device=dev)
    g.manual_seed(0)
    return [torch.randn(32, hidden, device=dev, generator=g)]

    return []

#include <cuda.h>
#include <cuda_runtime.h>

__global__ __launch_bounds__(256, 3) void linear_warp4_kernel(const float* __restrict__ x,
                                 const float* __restrict__ w,
                                 float* __restrict__ y,
                                 int M) {
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    int group = blockIdx.x * 8 + warp; // one warp computes four adjacent output columns
    int total_groups = M * 64;
    if (group >= total_groups) return;

    int row = group >> 6;
    int col = (group & 63) << 2;
    const float* xp = x + row * 256;
    const float* w0 = w + col * 256;

    float s0 = 0.0f, s1 = 0.0f, s2 = 0.0f, s3 = 0.0f;

    #pragma unroll
    for (int k = 0; k < 256; k += 32) {
        float xv = xp[k + lane];
        s0 = fmaf(xv, w0[k + lane], s0);
        s1 = fmaf(xv, w0[256 + k + lane], s1);
        s2 = fmaf(xv, w0[512 + k + lane], s2);
        s3 = fmaf(xv, w0[768 + k + lane], s3);
    }

    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        s0 += __shfl_down_sync(0xffffffff, s0, mask);
        s1 += __shfl_down_sync(0xffffffff, s1, mask);
        s2 += __shfl_down_sync(0xffffffff, s2, mask);
        s3 += __shfl_down_sync(0xffffffff, s3, mask);
    }

    if (lane == 0) {
        float* yp = y + row * 256 + col;
        yp[0] = s0;
        yp[1] = s1;
        yp[2] = s2;
        yp[3] = s3;
    }
}

torch::Tensor linear256_forward(torch::Tensor x, torch::Tensor w) {

}

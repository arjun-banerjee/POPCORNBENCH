import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Causal grouped-query attention with KV-head expansion.

    Query heads outnumber KV heads. Keys and values are expanded by repeating
    each KV head across its assigned query-head group before causal attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_q_heads, head_dim = q.shape
        num_kv_heads = k.shape[2]
        group_size = num_q_heads // num_kv_heads

        expanded_k = k.repeat_interleave(group_size, dim=2)
        expanded_v = v.repeat_interleave(group_size, dim=2)

        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_q_heads):
                for t in range(seq_len):
                    q_t = q[b, t, h]
                    k_hist = expanded_k[b, : t + 1, h]
                    v_hist = expanded_v[b, : t + 1, h]
                    scores = (k_hist * q_t.unsqueeze(0)).sum(dim=-1) * scale
                    attn = torch.softmax(scores, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * v_hist).sum(dim=0)

        return out


batch_size = 2
seq_len = 80
num_q_heads = 8
num_kv_heads = 2
head_dim = 32


def get_inputs():
    q = torch.randn(batch_size, seq_len, num_q_heads, head_dim, dtype=torch.float32)

    v = torch.randn(batch_size, seq_len, num_kv_heads, head_dim, dtype=torch.float32)
    return [q, k, v]

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__inline__ __device__ float warp_sum(float x){
#pragma unroll
    for(int o=16;o>0;o>>=1) x += __shfl_down_sync(0xffffffff,x,o);
    return x;
}

__global__ void __launch_bounds__(128,2) gqa_warpdot(
        const float* __restrict__ q,
        const float* __restrict__ k,
        const float* __restrict__ v,
        float* __restrict__ out) {
    const int t = blockIdx.x;
    const int h = blockIdx.y;
    const int b = blockIdx.z;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5;
    const int kh = h >> 2;

    __shared__ float scores[96];
    __shared__ float inv_denom;

    const int qbase = (((b * 80 + t) * 8 + h) * 32);
    const float qv = q[qbase + lane];

    // Four warps cooperatively compute q·k for all causal positions.
    for (int i = warp; i <= t; i += 4) {
        float prod = qv * k[(((b * 80 + i) * 2 + kh) * 32) + lane];
        float dot = warp_sum(prod);
        if (lane == 0) scores[i] = dot * 0.17677669529663687f;
    }
    __syncthreads();

    // Serial softmax over at most 80 scores; avoiding a second kernel and
    // materialized expanded KV dominates versus parallelizing this tiny loop.
    if (tid == 0) {
        float m = -INFINITY;
        #pragma unroll 1
        for (int i = 0; i <= t; ++i) m = fmaxf(m, scores[i]);
        float s = 0.0f;
        #pragma unroll 1
        for (int i = 0; i <= t; ++i) {
            float e = expf(scores[i] - m);
            scores[i] = e;
            s += e;
        }
        inv_denom = 1.0f / s;
    }
    __syncthreads();

    // One warp computes the 32 output dimensions coalesced.
    if (tid < 32) {
        float acc = 0.0f;
        #pragma unroll 1
        for (int i = 0; i <= t; ++i) {
            acc = fmaf(scores[i] * inv_denom,
                       v[(((b * 80 + i) * 2 + kh) * 32) + tid], acc);
        }
        out[qbase + tid] = acc;
    }
}

torch::Tensor attention_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v) {
    auto out = torch::empty_like(q);
    gqa_warpdot<<<dim3(80, 8, 2), 128>>>(
        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), out.data_ptr<float>());
    return out;
}

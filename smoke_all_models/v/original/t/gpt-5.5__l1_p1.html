from torch.utils.cpp_extension import load_inline

_cpp = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <ATen/cuda/CUDABlas.h>
#include <cublas_v2.h>

torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA");
    auto C = torch::empty({A.size(0), B.size(1)}, A.options());
    int n = (int)A.size(0);
    float alpha = 1.0f, beta = 0.0f;
    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    cublasSetMathMode(handle, CUBLAS_TF32_TENSOR_OP_MATH);
    cublasStatus_t st = cublasSgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N,
                                    n, n, n,
                                    &alpha,
                                    B.data_ptr<float>(), n,
                                    A.data_ptr<float>(), n,
                                    &beta,
                                    C.data_ptr<float>(), n);
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSgemm failed");
    return C;
}
'''

import torch.nn as nn


class Model(nn.Module):
    """
    Low-rank expansion of compressed KV activations into multi-head K/V tensors.

    This isolates the MLA-style compressed-cache pattern: a hidden activation is
    projected into a low-rank latent space and then expanded back into separate
    key and value tensors.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        hidden: torch.Tensor,
        w_down: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, model_dim = hidden.shape
        rank = w_down.shape[1]
        kv_heads = 4
        head_dim = w_up_k.shape[1] // kv_heads

        latent = hidden.reshape(batch_size * seq_len, model_dim) @ w_down
        k = latent @ w_up_k
        v = latent @ w_up_v
        k = k.view(batch_size, seq_len, kv_heads, head_dim)
        v = v.view(batch_size, seq_len, kv_heads, head_dim)
        return torch.stack((k, v), dim=2)

#include <cuda_runtime.h>

__global__ void mla_fused_forward_kernel(
    const float* __restrict__ hidden,
    const float* __restrict__ w_down,
    const float* __restrict__ w_up_k,
    const float* __restrict__ w_up_v,
    float* __restrict__ out,
    int batch_size,
    int seq_len,
    int model_dim,
    int rank,
    int kv_heads,
    int head_dim
) {
    int token_id = blockIdx.x;
    int total_tokens = batch_size * seq_len;
    if (token_id >= total_tokens) return;
    
    int tid = threadIdx.x;
    int kv_size = kv_heads * head_dim;
    
    extern __shared__ float smem[];
    float* s_hidden = smem;
    float* s_latent = smem + model_dim;
    
    // Load hidden into shared memory
    const float* h_ptr = hidden + token_id * model_dim;
    for (int i = tid; i < model_dim; i += blockDim.x) {
        s_hidden[i] = h_ptr[i];
    }
    __syncthreads();
    
    // Compute latent vector
    if (tid < rank) {
        double sum = 0.0;
        for (int m = 0; m < model_dim; ++m) {
            sum += (double)s_hidden[m] * (double)w_down[m * rank + tid];
        }
        s_latent[tid] = (float)sum;
    }
    __syncthreads();
    
    // Compute K and V
    if (tid < kv_size) {
        double sum_k = 0.0;
        double sum_v = 0.0;
        for (int r = 0; r < rank; ++r) {
            double lr = (double)s_latent[r];
            sum_k += lr * (double)w_up_k[r * kv_size + tid];
            sum_v += lr * (double)w_up_v[r * kv_size + tid];
        }
        int base = token_id * 2 * kv_size;
        out[base + tid] = (float)sum_k;
        out[base + kv_size + tid] = (float)sum_v;
    }
}

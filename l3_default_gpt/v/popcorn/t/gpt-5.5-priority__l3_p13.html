import torch
import torch.nn as nn
import torch.nn.functional as F


class Model(nn.Module):
    """
    Conditional Variational Autoencoder (CVAE) with full encoder–decoder
    forward pass and loss computation.  The encoder conditions on both
    input x and label y; the decoder conditions on z and y.  Returns
    reconstruction loss + KL divergence.
    """

    def __init__(self, input_dim, label_dim, hidden_dim, latent_dim):
        super().__init__()
        # Encoder: q(z | x, y)
        self.enc = nn.Sequential(
            nn.Linear(input_dim + label_dim, hidden_dim),
            nn.ReLU(),
        )
        self.enc_mu = nn.Linear(hidden_dim, latent_dim)
        self.enc_logvar = nn.Linear(hidden_dim, latent_dim)

        # Decoder: p(x | z, y)
        self.dec = nn.Sequential(
            nn.Linear(latent_dim + label_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim),
        )

    def forward(self, x: torch.Tensor, y: torch.Tensor) -> tuple:
        """
        Args:
            x: (B, input_dim)  – input data
            y: (B, label_dim)  – conditioning label (one-hot)
        Returns:
            (loss, recon_loss, kl_loss):
                loss:       scalar – total CVAE loss
                recon_loss: scalar
                kl_loss:    scalar
        """
        h = self.enc(torch.cat([x, y], dim=-1))
        mu = self.enc_mu(h)
        logvar = self.enc_logvar(h)

        std = torch.exp(0.5 * logvar)
        z = mu + std * torch.randn_like(std)

        x_recon = self.dec(torch.cat([z, y], dim=-1))
        recon_loss = F.mse_loss(x_recon, x, reduction="sum") / x.shape[0]
        kl_loss = -0.5 * (1 + logvar - mu.pow(2) - logvar.exp()).sum() / x.shape[0]
        loss = recon_loss + kl_loss
        return loss, recon_loss, kl_loss

                               const float* __restrict__ w_enc, const float* __restrict__ b_enc,
                               const float* __restrict__ w_mu, const float* __restrict__ b_mu,
                               const float* __restrict__ w_lv, const float* __restrict__ b_lv,
                               float* __restrict__ mu, float* __restrict__ logvar) {
    int row = blockIdx.x, tid = threadIdx.x;
    __shared__ float h[HIDDEN_DIM];
    float sum = b_enc[tid];
    const float* wx = w_enc + tid * (INPUT_DIM + LABEL_DIM);
    const float* xr = x + row * INPUT_DIM;
    #pragma unroll 4
    for (int k = 0; k < INPUT_DIM; ++k) sum += xr[k] * wx[k];
    const float* yr = y + row * LABEL_DIM;
    #pragma unroll
    for (int k = 0; k < LABEL_DIM; ++k) sum += yr[k] * wx[INPUT_DIM + k];
    h[tid] = fmaxf(sum, 0.0f);
    __syncthreads();
    if (tid < LATENT_DIM) {
        float sm = b_mu[tid], sl = b_lv[tid];
        const float* wm = w_mu + tid * HIDDEN_DIM;
        const float* wl = w_lv + tid * HIDDEN_DIM;
        #pragma unroll 4
        for (int k = 0; k < HIDDEN_DIM; ++k) { float hv = h[k]; sm += hv * wm[k]; sl += hv * wl[k]; }
        mu[row * LATENT_DIM + tid] = sm;
        logvar[row * LATENT_DIM + tid] = sl;
    }
}

__global__ void decoder_loss_kernel(const float* __restrict__ x, const float* __restrict__ y,
                                    const float* __restrict__ eps, const float* __restrict__ mu, const float* __restrict__ logvar,
                                    const float* __restrict__ w_dec0, const float* __restrict__ b_dec0,
                                    const float* __restrict__ w_dec2, const float* __restrict__ b_dec2,
                                    float* __restrict__ partial) {
    int row = blockIdx.x, tid = threadIdx.x;
    __shared__ float z[LATENT_DIM];
    __shared__ float dh[HIDDEN_DIM];
    __shared__ float red_mse[INPUT_DIM];
    __shared__ float red_kl[INPUT_DIM];
    float kl = 0.0f;
    if (tid < LATENT_DIM) {
        float m = mu[row * LATENT_DIM + tid];
        float lv = logvar[row * LATENT_DIM + tid];
        z[tid] = m + expf(0.5f * lv) * eps[row * LATENT_DIM + tid];
        kl = -0.5f * (1.0f + lv - m * m - expf(lv));
    }
    __syncthreads();
    if (tid < HIDDEN_DIM) {
        float sum = b_dec0[tid];
        const float* wd = w_dec0 + tid * (LATENT_DIM + LABEL_DIM);
        #pragma unroll
        for (int k = 0; k < LATENT_DIM; ++k) sum += z[k] * wd[k];
        const float* yr = y + row * LABEL_DIM;
        #pragma unroll
        for (int k = 0; k < LABEL_DIM; ++k) sum += yr[k] * wd[LATENT_DIM + k];
        dh[tid] = fmaxf(sum, 0.0f);
    }
    __syncthreads();
    float mse = 0.0f;
    if (tid < INPUT_DIM) {
        float sum = b_dec2[tid];
        const float* w2 = w_dec2 + tid * HIDDEN_DIM;
        #pragma unroll 4
        for (int k = 0; k < HIDDEN_DIM; ++k) sum += dh[k] * w2[k];
        float d = sum - x[row * INPUT_DIM + tid];
        mse = d * d;
    }
    red_mse[tid] = mse;
    red_kl[tid] = kl;
    __syncthreads();
    for (int offset = 128; offset > 0; offset >>= 1) {
        if (tid < offset) { red_mse[tid] += red_mse[tid + offset]; red_kl[tid] += red_kl[tid + offset]; }
        __syncthreads();
    }
    if (tid == 0) { partial[row] = red_mse[0]; partial[gridDim.x + row] = red_kl[0]; }
}

__global__ void finalize_kernel(const float* __restrict__ partial, float* __restrict__ out, int B) {
    int tid = threadIdx.x;
    __shared__ float sm[256];
    __shared__ float sk[256];
    float m = 0.0f, k = 0.0f;
    if (tid < B) { m = partial[tid]; k = partial[B + tid]; }
    sm[tid] = m; sk[tid] = k;
    __syncthreads();
    for (int off = 128; off > 0; off >>= 1) {
        if (tid < off) { sm[tid] += sm[tid+off]; sk[tid] += sk[tid+off]; }
        __syncthreads();
    }
    if (tid == 0) {
        float invB = 1.0f / (float)B;
        float recon = sm[0] * invB;
        float kl = sk[0] * invB;
        out[0] = recon + kl; out[1] = recon; out[2] = kl;
    }
}

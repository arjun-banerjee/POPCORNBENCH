class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        super(ModelNew, self).__init__()
        layers = []
        current_input_size = input_size
        linears = []
        for hidden_size in hidden_layer_sizes:
            lin = nn.Linear(current_input_size, hidden_size)
            layers.append(lin)
            linears.append(lin)
            layers.append(nn.ReLU())
            current_input_size = hidden_size
        lin = nn.Linear(current_input_size, output_size)
        layers.append(lin)
        linears.append(lin)
        self.network = nn.Sequential(*layers)
        self._weights = [m.weight for m in linears]
        self._biases = [m.bias for m in linears]
        self.register_buffer('_workspace', torch.empty(0, dtype=torch.uint8), persistent=False)

    def forward(self, x):
        if self._workspace.device != x.device or self._workspace.numel() < 32 * 1024 * 1024:
            self._workspace = torch.empty(32 * 1024 * 1024, device=x.device, dtype=torch.uint8)
        return _ext.mlp_forward(x, self._weights, self._biases, self._workspace)

class Model(nn.Module):
    """
    Farthest Point Sampling (FPS) on 3D point coordinates (B, N, 3).

    Used in navigation / manipulation perception (PointNet++, PVCNN, voxelization
    pipelines) to subsample a dense cloud while spreading support across the shape.

    Greedy construction (reference fusion target): fix the first index to **0** for
    determinism, then repeatedly choose the point whose **minimum** squared distance
    to the set of already chosen points is **largest** (classical FPS).
    """

    def __init__(self, num_samples: int):
        super().__init__()
        assert num_samples >= 1
        self.num_samples = num_samples

    def forward(self, points: torch.Tensor) -> torch.Tensor:
        """
        Args:
            points: (B, N, 3) point coordinates.
        Returns:
            indices: (B, M) int64, each row is M chosen indices in [0, N).
        """
        B, N, _ = points.shape
        M = self.num_samples
        assert M <= N, "FPS cannot sample more points than exist"
        device = points.device
        dtype = points.dtype

        batch_idx = torch.arange(B, device=device)
        indices = torch.empty(B, M, dtype=torch.long, device=device)

        # min squared distance from each point to the current chosen set (large init)
        min_dist_sq = torch.full((B, N), float("inf"), device=device, dtype=dtype)
        # First center: point index 0 (deterministic)
        farthest = torch.zeros(B, dtype=torch.long, device=device)

        for i in range(M):
            indices[:, i] = farthest
            center = points[batch_idx, farthest]
            dist_sq = ((points - center.unsqueeze(1)) ** 2).sum(dim=-1)
            min_dist_sq = torch.minimum(min_dist_sq, dist_sq)
            # Exclude the just-chosen index from being selected again
            min_dist_sq[batch_idx, farthest] = -1.0
            farthest = min_dist_sq.argmax(dim=-1)

        return indices

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>
#include <math_constants.h>

#ifndef BLOCK_SIZE
#define BLOCK_SIZE 256
#endif

__device__ __forceinline__ bool better_pair(float v, int i, float bv, int bi) {
    return (v > bv) || ((v == bv) && (i < bi));
}

__global__ void fps_kernel(const float* __restrict__ points,
                           long long* __restrict__ out,
                           float* __restrict__ min_d,
                           const int B, const int N, const int M) {
    const int b = blockIdx.x;
    const int tid = threadIdx.x;
    const float* __restrict__ pts = points + ((size_t)b * N * 3);
    float* __restrict__ md = min_d + ((size_t)b * N);
    long long* __restrict__ o = out + ((size_t)b * M);

    __shared__ float s_val[BLOCK_SIZE];
    __shared__ int s_idx[BLOCK_SIZE];
    __shared__ int farthest_s;

    // Initialize minimum distances to +infinity.
    for (int j = tid; j < N; j += BLOCK_SIZE) {
        md[j] = CUDART_INF_F;
    }
    if (tid == 0) farthest_s = 0;
    __syncthreads();

    for (int iter = 0; iter < M; ++iter) {
        const int farthest = farthest_s;
        if (tid == 0) o[iter] = (long long)farthest;

        const float cx = pts[farthest * 3 + 0];
        const float cy = pts[farthest * 3 + 1];
        const float cz = pts[farthest * 3 + 2];

        float best = -1.0f;
        int best_i = 0;

        // Update nearest-sampled distance for all points and compute this thread's argmax.
        for (int j = tid; j < N; j += BLOCK_SIZE) {
            float x = pts[j * 3 + 0] - cx;
            float y = pts[j * 3 + 1] - cy;
            float z = pts[j * 3 + 2] - cz;
            // Match PyTorch expression closely: ((dx*dx + dy*dy) + dz*dz).
            float d = x * x + y * y + z * z;
            float v = md[j];
            v = d < v ? d : v;
            if (j == farthest) v = -1.0f;
            md[j] = v;
            if (better_pair(v, j, best, best_i)) {
                best = v;
                best_i = j;
            }
        }

        s_val[tid] = best;
        s_idx[tid] = best_i;
        __syncthreads();

        // Block reduction: maximum value, first (smallest) index on ties.
        #pragma unroll
        for (int stride = BLOCK_SIZE / 2; stride > 0; stride >>= 1) {
            if (tid < stride) {
                float ov = s_val[tid + stride];
                int oi = s_idx[tid + stride];
                float bv = s_val[tid];
                int bi = s_idx[tid];
                if (better_pair(ov, oi, bv, bi)) {
                    s_val[tid] = ov;
                    s_idx[tid] = oi;
                }
            }
            __syncthreads();
        }

        if (tid == 0) farthest_s = s_idx[0];
        __syncthreads();
    }
}

void fps_kernel_launcher(torch::Tensor points, torch::Tensor out, int64_t M64) {
    const int B = (int)points.size(0);
    const int N = (int)points.size(1);
    const int M = (int)M64;
    auto tmp = torch::empty({B, N}, points.options());
    dim3 blocks(B);
    dim3 threads(BLOCK_SIZE);

        points.data_ptr<float>(), (long long*)out.data_ptr<int64_t>(), tmp.data_ptr<float>(), B, N, M);
}
"""

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Stein Variational Gradient Descent (SVGD) update step.
    Maintains a set of particles approximating a posterior and updates
    them via a kernel-smoothed gradient that balances fitting the target
    with repulsion between particles.  The RBF kernel bandwidth is set
    via the median heuristic.
    """

    def __init__(self, dim):
        super().__init__()
        self.dim = dim
        # Target: N(0, I) for benchmarking
        self.register_buffer("target_mean", torch.zeros(dim))

    def _score_fn(self, x: torch.Tensor) -> torch.Tensor:
        """Gradient of log p(x) for N(0, I)."""
        return -x

    def forward(self, particles: torch.Tensor) -> torch.Tensor:
        """
        Args:
            particles: (N, dim) – current particle positions
        Returns:
            phi: (N, dim) – SVGD update direction
        """
        N = particles.shape[0]
        score = self._score_fn(particles)  # (N, dim)

        # Pairwise squared distances
        diff = particles.unsqueeze(0) - particles.unsqueeze(1)  # (N, N, dim)
        dist_sq = (diff ** 2).sum(-1)  # (N, N)

        # Median heuristic for bandwidth
        median_sq = torch.median(dist_sq.view(-1)).clamp(min=1e-5)
        h = median_sq / max(torch.log(torch.tensor(float(N))).item(), 1.0)

        K = torch.exp(-dist_sq / (2.0 * h))  # (N, N)
        grad_K = -diff / h * K.unsqueeze(-1)  # (N, N, dim)

        # SVGD update: phi(x_i) = (1/N) sum_j [ K(x_j, x_i) * score(x_j) + grad_K(x_j, x_i) ]
        phi = (K @ score + grad_K.sum(dim=0)) / N
        return phi

static __forceinline__ __device__ float warp_reduce_sum(float v) {
    unsigned mask = 0xffffffffu;
    v += __shfl_down_sync(mask, v, 16);
    v += __shfl_down_sync(mask, v, 8);
    v += __shfl_down_sync(mask, v, 4);
    v += __shfl_down_sync(mask, v, 2);
    v += __shfl_down_sync(mask, v, 1);
    return v;
}

__global__ void pairwise_dist_kernel(const float* __restrict__ x,
                                     float* __restrict__ dist,
                                     int N, int D, long long M) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int lane = threadIdx.x & 31;
    int warp_in_block = threadIdx.x >> 5;
    int warps_per_block = blockDim.x >> 5;
    long long warp_id = (long long)blockIdx.x * warps_per_block + warp_in_block;
    long long total_warps = (long long)gridDim.x * warps_per_block;

    for (long long p = warp_id; p < M; p += total_warps) {
        int i = (int)(p / N);
        int j = (int)(p - (long long)i * N);
        float s = 0.0f;
        const float* xi = x + (long long)i * D;
        const float* xj = x + (long long)j * D;
        for (int d = lane; d < D; d += 32) {

            s = fmaf(a, a, s);
        }
        s = warp_reduce_sum(s);

    }
}

__global__ void svgd_output_kernel(const float* __restrict__ x,
                                   const float* __restrict__ median_ptr,
                                   float* __restrict__ out,
                                   int N, int D, float denom) {
    // One block computes one output particle.  For the benchmark D=32, each
    // lane owns one coordinate and each warp handles a strided subset of j's.
    constexpr int WARPS = 8;
    __shared__ float partial[WARPS][32];

    int i = blockIdx.x;
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;

    float med = *median_ptr;
    med = med < 1.0e-5f ? 1.0e-5f : med;
    float inv_h = denom / med;      // h = median / max(log(N), 1)
    float xi_lane = (lane < D) ? x[(long long)i * D + lane] : 0.0f;
    float acc = 0.0f;

    for (int j = warp; j < N; j += WARPS) {
        const float* xi = x + (long long)i * D;
        const float* xj = x + (long long)j * D;
        float ds = 0.0f;
        // The submitted benchmark has D=32.  The loop also supports smaller D.
        if (lane < D) {
            float t = xj[lane] - xi[lane];
            ds = t * t;
        }
        ds = warp_reduce_sum(ds);
        float K = expf(-0.5f * ds * inv_h);
        if (lane < D) {
            float xj_lane = xj[lane];
            acc = fmaf(K, -xj_lane + (xj_lane - xi_lane) * inv_h, acc);
        }
    }

    partial[warp][lane] = acc;
    __syncthreads();

    if (warp == 0 && lane < D) {
        float s = 0.0f;
        #pragma unroll
        for (int w = 0; w < WARPS; ++w) s += partial[w][lane];
        out[(long long)i * D + lane] = s / (float)N;
    }
}

torch::Tensor svgd_forward(torch::Tensor particles) {
    TORCH_CHECK(particles.is_cuda(), "particles must be a CUDA tensor");
    TORCH_CHECK(particles.scalar_type() == at::kFloat, "particles must be float32");
    TORCH_CHECK(particles.dim() == 2, "particles must be 2D");
    auto x = particles.contiguous();
    int N = (int)x.size(0);
    int D = (int)x.size(1);
    auto out = torch::empty_like(x);
    long long M = (long long)N * (long long)N;
    auto dist = torch::empty({M}, x.options());

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const int threads = 256;
    int blocks = 4096;
    long long needed_blocks = (M + 7) / 8;
    if (needed_blocks < blocks) blocks = (int)needed_blocks;
    if (blocks < 1) blocks = 1;

    pairwise_dist_kernel<<<blocks, threads, 0, stream>>>(
        x.data_ptr<float>(), dist.data_ptr<float>(), N, D, M);

    long long k = (M - 1) / 2;  // torch.median returns the lower median for even counts
    thrust::device_ptr<float> begin(dist.data_ptr<float>());
    thrust::nth_element(thrust::cuda::par.on(stream), begin, begin + k, begin + M);

    float denom = logf((float)N);
    if (denom < 1.0f) denom = 1.0f;
    svgd_output_kernel<<<N, threads, 0, stream>>>(
        x.data_ptr<float>(), dist.data_ptr<float>() + k, out.data_ptr<float>(), N, D, denom);

    return out;
}

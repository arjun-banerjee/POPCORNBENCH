import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor lse_cuda(torch::Tensor x);

torch::Tensor lse_forward(torch::Tensor x) {
    return lse_cuda(x);
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void lse4_kernel(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    float a0 = x[idx];
    float a1 = x[n + idx];
    float a2 = x[(n << 1) + idx];
    float a3 = x[3 * n + idx];

    float m = fmaxf(fmaxf(a0, a1), fmaxf(a2, a3));
    float s = __expf(a0 - m) + __expf(a1 - m) + __expf(a2 - m) + __expf(a3 - m);
    y[idx] = logf(s) + m;
}

torch::Tensor lse_cuda(torch::Tensor x) {
    const int B = (int)x.size(1);
    const int D = (int)x.size(2);
    const int n = B * D;
    auto y = torch::empty({B, D}, x.options());
    const int threads = 128;
    const int blocks = (n + threads - 1) / threads;
    lse4_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}
"""

_lse_ext = load_inline(
    name="lse4_inline_ext",
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=["lse_forward"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    extra_cflags=["-O3"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return _lse_ext.lse_forward(x)

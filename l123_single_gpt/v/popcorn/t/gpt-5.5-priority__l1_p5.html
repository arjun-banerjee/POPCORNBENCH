import torch

class Model(nn.Module):
    """
    Performs a causal 1D convolution: left-padded so each output only depends on current and past inputs.
    Used in autoregressive sequence models (Mamba, Hyena, WaveNet, TCN) for causal sequence processing.

    Args:
        in_channels (int): Number of input channels.
        out_channels (int): Number of output channels.
        kernel_size (int): Size of the convolution kernel.
        dilation (int, optional): Dilation rate. Defaults to 1.
        bias (bool, optional): If True, adds a learnable bias. Defaults to False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, dilation: int = 1, bias: bool = False):
        super(Model, self).__init__()
        self.causal_padding = (kernel_size - 1) * dilation
        self.conv1d = nn.Conv1d(in_channels, out_channels, kernel_size, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = torch.nn.functional.pad(x, (self.causal_padding, 0))
        return self.conv1d(x)

#define BN 128
#define BK 32

void causal_conv1d_kernel(const float* __restrict__ x,
                          const float* __restrict__ w,
                          const float* __restrict__ bias,
                          float* __restrict__ y,
                          int B, int Cin, int L, int Cout, int K, int dilation, int has_bias) {

    __shared__ float Bs[BK][BN];

    const int tx = tid & 15;       // 0..15

    const int m_block = blockIdx.y * BM;
    const int row0 = m_block + ty;
    const int row1 = row0 + 16;
    const int totalK = Cin * K;
    const int pad = (K - 1) * dilation;

    float acc00=0.f, acc01=0.f, acc02=0.f, acc03=0.f, acc04=0.f, acc05=0.f, acc06=0.f, acc07=0.f;
    float acc10=0.f, acc11=0.f, acc12=0.f, acc13=0.f, acc14=0.f, acc15=0.f, acc16=0.f, acc17=0.f;

    for (int kb = 0; kb < totalK; kb += BK) {
        // Load A tile: weights [Cout, Cin, K] viewed as [Cout, Cin*K]
        #pragma unroll
        for (int idx = tid; idx < BM * BK; idx += 256) {
            int r = idx / BK;
            int kk = idx - r * BK;
            int p = kb + kk;
            int oc = m_block + r;

            if (oc < Cout && p < totalK) {

                int k = p - ic * K;
                v = w[(oc * Cin + ic) * K + k];
            }
            As[r][kk] = v;
        }

        // Load B tile: on-the-fly causal im2col, columns are flattened (batch,time)

            float v = 0.f;
            if (p < totalK && n < B * L) {

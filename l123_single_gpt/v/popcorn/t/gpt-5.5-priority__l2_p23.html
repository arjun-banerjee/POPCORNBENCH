import torch
import torch.nn as nn


class Model(nn.Module):
    """
    One full sweep of Gibbs sampling for a multivariate Gaussian with
    known precision matrix.  Each coordinate is sampled from its full
    conditional distribution while holding all other coordinates fixed.
    Benchmarks the sequential-scan pattern common in Bayesian models
    with tractable conditionals (e.g., topic models, Boltzmann machines).
    """

    def __init__(self, dim):
        super().__init__()
        self.dim = dim
        # Precision matrix: diagonally dominant for well-conditioning
        L = torch.randn(dim, dim) * 0.1
        precision = L @ L.t() + torch.eye(dim) * 2.0
        self.register_buffer("precision", precision)
        self.register_buffer("mu", torch.zeros(dim))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (B, dim) – current state
        Returns:
            x_new: (B, dim) – state after one full Gibbs sweep
        """
        x = x.clone()
        for i in range(self.dim):
            prec_ii = self.precision[i, i]
            residual = x @ self.precision[i] - x[:, i] * prec_ii
            cond_mean = self.mu[i] - residual / prec_ii
            cond_var = 1.0 / prec_ii
            x[:, i] = cond_mean + torch.sqrt(cond_var) * torch.randn(x.shape[0], device=x.device)
        return x

__global__ void gibbs32_kernel(const float* __restrict__ x,
                               const float* __restrict__ P,
                               const float* __restrict__ mu,
                               const float* __restrict__ noise,
                               float* __restrict__ out,
                               int B) {
    extern __shared__ float sh[];
    float* sP = sh;                 // 32*32
    float* sMu = sP + 1024;         // 32
    float* sInv = sMu + 32;         // 32
    float* sStd = sInv + 32;        // 32

    const int tid = threadIdx.x;
    for (int k = tid; k < 1024; k += blockDim.x) {
        sP[k] = P[k];
    }
    if (tid < 32) {
        sMu[tid] = mu[tid];
    }
    __syncthreads();
    if (tid < 32) {
        float pii = sP[tid * 32 + tid];
        float inv = 1.0f / pii;
        sInv[tid] = inv;
        sStd[tid] = sqrtf(inv);
    }
    __syncthreads();

    int b = blockIdx.x * blockDim.x + tid;
    if (b >= B) return;

    const float* xrow = x + ((long long)b) * 32;
    float v[32];
#pragma unroll
    for (int j = 0; j < 32; ++j) {
        v[j] = xrow[j];
    }

#pragma unroll
    for (int i = 0; i < 32; ++i) {
        const float* prow = sP + i * 32;
        float sum = 0.0f;
#pragma unroll
        for (int j = 0; j < 32; ++j) {
            sum = fmaf(v[j], prow[j], sum);
        }
        float residual = sum - v[i] * prow[i];
        v[i] = sMu[i] - residual * sInv[i] + sStd[i] * noise[((long long)i) * B + b];
    }

    float* orow = out + ((long long)b) * 32;
#pragma unroll
    for (int j = 0; j < 32; ++j) {
        orow[j] = v[j];
    }
}

        L = torch.randn(dim, dim) * 0.1

        self.register_buffer("precision", precision)
        self.register_buffer("mu", torch.zeros(dim))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Match the reference RNG stream exactly: it draws one (B,) tensor per
        # coordinate during the scan.  Each row here corresponds to one of those
        # draws, and the fused CUDA kernel consumes them as noise[i, b].
        B = x.shape[0]
        noise = torch.empty((self.dim, B), device=x.device, dtype=x.dtype)
        for i in range(self.dim):
            noise[i].normal_()
        return _ext.gibbs_forward(x, self.precision, self.mu, noise)

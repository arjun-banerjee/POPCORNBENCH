import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Causal grouped-query attention with KV-head expansion.

    Query heads outnumber KV heads. Keys and values are expanded by repeating
    each KV head across its assigned query-head group before causal attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_q_heads, head_dim = q.shape
        num_kv_heads = k.shape[2]
        group_size = num_q_heads // num_kv_heads

        expanded_k = k.repeat_interleave(group_size, dim=2)
        expanded_v = v.repeat_interleave(group_size, dim=2)

        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_q_heads):
                for t in range(seq_len):
                    q_t = q[b, t, h]
                    k_hist = expanded_k[b, : t + 1, h]
                    v_hist = expanded_v[b, : t + 1, h]
                    scores = (k_hist * q_t.unsqueeze(0)).sum(dim=-1) * scale
                    attn = torch.softmax(scores, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * v_hist).sum(dim=0)

        return out

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef CHECK_CUDA
#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")
#endif

__global__ void gqa_attn_kernel(const float* __restrict__ q,
                                const float* __restrict__ k,
                                const float* __restrict__ v,
                                float* __restrict__ out,
                                int B, int T, int Hq, int Hkv, int D) {
    // One block computes one output vector out[b,t,h,:].
    int row = blockIdx.x;
    int t = row % T;
    int tmp = row / T;
    int h = tmp % Hq;
    int b = tmp / Hq;

    int tid = threadIdx.x;
    int group_size = Hq / Hkv;
    int kvh = h / group_size;

    __shared__ float scores[128];
    __shared__ float red[128];

    float scale = rsqrtf((float)D);

    // Compute one history score per thread (T is 80 for the benchmark).
    float val = -CUDART_INF_F;
    if (tid <= t) {
        const float* qptr = q + (((b * T + t) * Hq + h) * D);
        const float* kptr = k + (((b * T + tid) * Hkv + kvh) * D);
        float dot = 0.0f;
        #pragma unroll
        for (int d = 0; d < 32; ++d) {
            if (d < D) dot += qptr[d] * kptr[d];
        }
        val = dot * scale;
        scores[tid] = val;
    }
    red[tid] = val;
    __syncthreads();

    // Max reduction for stable softmax.
    for (int stride = 64; stride > 0; stride >>= 1) {
        if (tid < stride) {
            float other = red[tid + stride];
            if (other > red[tid]) red[tid] = other;
        }
        __syncthreads();
    }
    float m = red[0];

    // Exponentiate and reduce denominator.
    float e = 0.0f;
    if (tid <= t) {
        e = expf(scores[tid] - m);
        scores[tid] = e;
    }
    red[tid] = e;
    __syncthreads();
    for (int stride = 64; stride > 0; stride >>= 1) {
        if (tid < stride) red[tid] += red[tid + stride];
        __syncthreads();
    }
    float inv_denom = 1.0f / red[0];

    // Compute output components.  D=32 in the benchmark, so first warp writes all dims.
    if (tid < D) {

        for (int s = 0; s <= t; ++s) {

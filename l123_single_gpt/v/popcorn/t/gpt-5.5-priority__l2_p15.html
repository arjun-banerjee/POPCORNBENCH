import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Causal grouped-query attention with KV-head expansion.

    Query heads outnumber KV heads. Keys and values are expanded by repeating
    each KV head across its assigned query-head group before causal attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_q_heads, head_dim = q.shape
        num_kv_heads = k.shape[2]
        group_size = num_q_heads // num_kv_heads

        expanded_k = k.repeat_interleave(group_size, dim=2)
        expanded_v = v.repeat_interleave(group_size, dim=2)

        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_q_heads):
                for t in range(seq_len):
                    q_t = q[b, t, h]
                    k_hist = expanded_k[b, : t + 1, h]
                    v_hist = expanded_v[b, : t + 1, h]
                    scores = (k_hist * q_t.unsqueeze(0)).sum(dim=-1) * scale
                    attn = torch.softmax(scores, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * v_hist).sum(dim=0)

        return out

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor gqa_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef MAX_SEQ
#define MAX_SEQ 128
#endif

__global__ void gqa_attn_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    float* __restrict__ out,
    int B, int T, int Hq, int Hkv, int D)
{
    const int tid = threadIdx.x;
    const int row = blockIdx.x;
    const int t = row % T;
    const int h = (row / T) % Hq;
    const int b = row / (T * Hq);
    const int group = Hq / Hkv;
    const int kh = h / group;

    __shared__ float scores[MAX_SEQ];
    __shared__ float red[128];

    const float scale = 1.0f / sqrtf((float)D);
    const int q_base = ((b * T + t) * Hq + h) * D;
    const int kv_b_base = b * T * Hkv * D;

    // One thread computes one historical score.  For this benchmark D=32,T=80,
    // so the simple per-thread dot product is faster than launching many tiny ops.
    if (tid <= t) {
        const int k_base = kv_b_base + (tid * Hkv + kh) * D;
        float dot = 0.0f;
        #pragma unroll
        for (int d = 0; d < 32; ++d) {
            if (d < D) dot += q[q_base + d] * k[k_base + d];
        }
        // In case a different head_dim is used, finish the tail.
        for (int d = 32; d < D; ++d) dot += q[q_base + d] * k[k_base + d];
        scores[tid] = dot * scale;
        red[tid] = scores[tid];
    } else {
        red[tid] = -INFINITY;
    }
    __syncthreads();

    // max reduction over 128 threads
    for (int offset = 64; offset > 0; offset >>= 1) {
        if (tid < offset) {
            float other = red[tid + offset];
            red[tid] = red[tid] > other ? red[tid] : other;
        }
        __syncthreads();
    }
    const float m = red[0];

    float e = 0.0f;
    if (tid <= t) {
        e = expf(scores[tid] - m);
        scores[tid] = e;
    }
    red[tid] = e;
    __syncthreads();

    // sum reduction
    for (int offset = 64; offset > 0; offset >>= 1) {
        if (tid < offset) red[tid] += red[tid + offset];
        __syncthreads();
    }
    const float inv_sum = 1.0f / red[0];

    // Compute output vector.  For D=32, the first warp covers all channels.
    for (int d = tid; d < D; d += blockDim.x) {
        float acc = 0.0f;
        for (int s = 0; s <= t; ++s) {
            const int v_idx = kv_b_base + (s * Hkv + kh) * D + d;
            acc += (scores[s] * inv_sum) * v[v_idx];
        }
        out[q_base + d] = acc;
    }
}

torch::Tensor gqa_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v) {
    const at::cuda::CUDAGuard device_guard(q.device());
    auto out = torch::empty_like(q);

    const int B = (int)q.size(0);
    const int T = (int)q.size(1);
    const int Hq = (int)q.size(2);
    const int D = (int)q.size(3);
    const int Hkv = (int)k.size(2);

    const int threads = 128;
    auto stream = at::cuda::getCurrentCUDAStream();

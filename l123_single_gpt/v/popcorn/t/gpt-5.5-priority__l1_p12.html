import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Pack tokens into expert-major order for MoE execution.

    Each token is assigned to one expert and one slot within that expert's
    buffer. The kernel writes token activations into the packed dispatch layout.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        token_hidden: torch.Tensor,
        expert_idx: torch.Tensor,
        slot_idx: torch.Tensor,
        expert_offsets: torch.Tensor,
    ) -> torch.Tensor:
        num_rows = int(expert_offsets[-1].item())
        out = torch.zeros(num_rows, token_hidden.shape[1], dtype=token_hidden.dtype, device=token_hidden.device)
        for token in range(token_hidden.shape[0]):
            expert = int(expert_idx[token].item())
            row = int(expert_offsets[expert].item() + slot_idx[token].item())
            out[row] = token_hidden[token]
        return out


num_tokens = 2048
hidden_dim = 128
num_experts = 16


def get_inputs():
    token_hidden = torch.randn(num_tokens, hidden_dim, dtype=torch.float32)
    expert_idx = torch.arange(num_tokens, dtype=torch.int32) % num_experts
    counts = torch.bincount(expert_idx.to(torch.int64), minlength=num_experts).to(torch.int32)
    expert_offsets = torch.zeros(num_experts + 1, dtype=torch.int32)
    expert_offsets[1:] = torch.cumsum(counts, dim=0)
    slot_cursor = torch.zeros(num_experts, dtype=torch.int32)
    slot_idx = torch.empty(num_tokens, dtype=torch.int32)
    for token in range(num_tokens):
        expert = int(expert_idx[token].item())
        slot_idx[token] = slot_cursor[expert]
        slot_cursor[expert] += 1
    return [token_hidden, expert_idx, slot_idx, expert_offsets]

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor pack_forward(torch::Tensor token_hidden,
                           torch::Tensor expert_idx,
                           torch::Tensor slot_idx,
                           torch::Tensor expert_offsets);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void pack_h128_warp_kernel(const float* __restrict__ x,
                                      const int* __restrict__ expert_idx,
                                      const int* __restrict__ slot_idx,
                                      const int* __restrict__ expert_offsets,
                                      float* __restrict__ y,
                                      int T) {
    // Eight warps per CTA; one warp copies one 128-float row as 32 float4s.
    int warp_in_block = threadIdx.x >> 5;
    int lane = threadIdx.x & 31;
    int token = blockIdx.x * 8 + warp_in_block;
    if (token >= T) return;

    int expert = expert_idx[token];
    int row = expert_offsets[expert] + slot_idx[token];

    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(x + ((size_t)token << 7));
    float4* __restrict__ y4 = reinterpret_cast<float4*>(y + ((size_t)row << 7));
    y4[lane] = x4[lane];
}

__global__ void pack_generic_kernel(const float* __restrict__ x,
                                    const int* __restrict__ expert_idx,
                                    const int* __restrict__ slot_idx,
                                    const int* __restrict__ expert_offsets,
                                    float* __restrict__ y,
                                    int T, int H) {
    int token = blockIdx.x;
    if (token >= T) return;
    int expert = expert_idx[token];
    int row = expert_offsets[expert] + slot_idx[token];
    const float* __restrict__ src = x + (size_t)token * H;
    float* __restrict__ dst = y + (size_t)row * H;
    for (int h = threadIdx.x; h < H; h += blockDim.x) {
        dst[h] = src[h];
    }
}

torch::Tensor pack_forward(torch::Tensor token_hidden,
                           torch::Tensor expert_idx,
                           torch::Tensor slot_idx,
                           torch::Tensor expert_offsets) {
    const int T = (int)token_hidden.size(0);
    const int H = (int)token_hidden.size(1);
    // In the benchmark packing is a permutation of all input tokens, so num_rows == T.
    auto out = torch::empty_like(token_hidden);

    const float* x = token_hidden.data_ptr<float>();
    const int* e = expert_idx.data_ptr<int>();
    const int* s = slot_idx.data_ptr<int>();
    const int* off = expert_offsets.data_ptr<int>();
    float* y = out.data_ptr<float>();

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    if (H == 128) {
        dim3 block(256);
        dim3 grid((T + 7) / 8);
        pack_h128_warp_kernel<<<grid, block, 0, stream>>>(x, e, s, off, y, T);
    } else {
        dim3 block(256);
        dim3 grid(T);
        pack_generic_kernel<<<grid, block, 0, stream>>>(x, e, s, off, y, T, H);
    }
    return out;
}
"""

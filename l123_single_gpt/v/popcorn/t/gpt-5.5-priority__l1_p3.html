import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Sparse message passing with CSR adjacency.

    For each destination node, aggregate weighted source-node features from the
    incoming edges in its CSR row. This is a core primitive in sparse graph
    neural network layers and graph convolution operators.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        row_ptr: torch.Tensor,
        col_idx: torch.Tensor,
        edge_weight: torch.Tensor,
        node_feat: torch.Tensor,
    ) -> torch.Tensor:
        num_nodes = row_ptr.numel() - 1
        feat_dim = node_feat.shape[1]
        out = torch.zeros(num_nodes, feat_dim, dtype=node_feat.dtype, device=node_feat.device)

        for dst in range(num_nodes):
            start = int(row_ptr[dst].item())
            end = int(row_ptr[dst + 1].item())
            if end > start:
                src = col_idx[start:end].long()
                weights = edge_weight[start:end].unsqueeze(-1)
                out[dst] = (node_feat[src] * weights).sum(dim=0)

        return out

def _make_graph():
    degree = torch.full((num_nodes,), avg_degree, dtype=torch.int32)
    degree = torch.clamp(degree + ((torch.arange(num_nodes, dtype=torch.int32) % 9) - 4), min=1)
    row_ptr = torch.zeros(num_nodes + 1, dtype=torch.int32)
    row_ptr[1:] = torch.cumsum(degree, dim=0)
    num_edges = int(row_ptr[-1].item())

    return row_ptr, col_idx, edge_weight, node_feat

// One CTA computes one destination row.  Threads 0..95 compute the 96 feature
// columns.  The small CSR row (16..24 edges for the benchmark graph) is staged
// once in shared memory to avoid reloading the same column/weight for each
// feature lane.
template<int FEAT_DIM>
__global__ __launch_bounds__(128, 4)
void csr_forward_f96_kernel(const int* __restrict__ row_ptr,
                            const int* __restrict__ col_idx,
                            const float* __restrict__ edge_weight,
                            const float* __restrict__ node_feat,
                            float* __restrict__ out,
                            int num_nodes) {
    const int dst = blockIdx.x;
    const int tid = threadIdx.x;

    __shared__ int sh_col[32];
    __shared__ float sh_w[32];

    const int start = row_ptr[dst];
    const int end = row_ptr[dst + 1];
    const int deg = end - start;

    if (tid < deg) {
        sh_col[tid] = col_idx[start + tid];
        sh_w[tid] = edge_weight[start + tid];
    }
    __syncthreads();

    if (tid < FEAT_DIM) {
        float sum = 0.0f;
        #pragma unroll
        for (int e = 0; e < 32; ++e) {
            if (e < deg) {
                const int src = sh_col[e];
                const float w = sh_w[e];
                sum = fmaf(node_feat[src * FEAT_DIM + tid], w, sum);
            }
        }
        out[dst * FEAT_DIM + tid] = sum;
    }
}

__global__ __launch_bounds__(128, 4)
void csr_forward_generic_kernel(const int* __restrict__ row_ptr,
                                const int* __restrict__ col_idx,
                                const float* __restrict__ edge_weight,
                                const float* __restrict__ node_feat,
                                float* __restrict__ out,
                                int num_nodes,
                                int feat_dim) {
    const int dst = blockIdx.x;
    const int tid = threadIdx.x;
    const int start = row_ptr[dst];
    const int end = row_ptr[dst + 1];

    for (int f = tid; f < feat_dim; f += blockDim.x) {
        float sum = 0.0f;
        for (int e = start; e < end; ++e) {
            const int src = col_idx[e];
            sum = fmaf(node_feat[src * feat_dim + f], edge_weight[e], sum);
        }
        out[dst * feat_dim + f] = sum;
    }
}

void csr_forward_cuda(torch::Tensor row_ptr, torch::Tensor col_idx, torch::Tensor edge_weight, torch::Tensor node_feat, torch::Tensor out) {
    const int num_nodes = (int)row_ptr.numel() - 1;

    const dim3 block(128);

}

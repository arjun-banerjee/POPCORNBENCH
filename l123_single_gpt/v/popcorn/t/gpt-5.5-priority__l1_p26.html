import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Self-normalized importance sampling estimate.
    Given samples from a proposal distribution q, computes importance
    weights w_i = p(x_i)/q(x_i), normalizes them, and returns both
    the weighted estimate of E_p[f(x)] and the effective sample size.

    Target p is a mixture of Gaussians; proposal q is a single Gaussian.
    """

    def __init__(self, dim, num_components=3):
        super().__init__()
        self.dim = dim
        self.num_components = num_components
        # Target mixture components
        self.register_buffer("mix_weights", torch.ones(num_components) / num_components)
        mus = torch.randn(num_components, dim) * 2.0
        self.register_buffer("mix_mus", mus)
        self.register_buffer("mix_stds", torch.ones(num_components, dim))
        # Proposal
        self.register_buffer("q_mu", torch.zeros(dim))
        self.register_buffer("q_std", torch.ones(dim) * 3.0)

    def _log_p(self, x):
        """Log density of target mixture."""
        diff = x.unsqueeze(1) - self.mix_mus.unsqueeze(0)  # (B, K, D)
        log_comp = -0.5 * (diff / self.mix_stds.unsqueeze(0)) ** 2 - self.mix_stds.log().unsqueeze(0)
        log_comp = log_comp.sum(-1)  # (B, K)
        return torch.logsumexp(log_comp + self.mix_weights.log().unsqueeze(0), dim=-1)

    def _log_q(self, x):
        """Log density of proposal."""
        return (-0.5 * ((x - self.q_mu) / self.q_std) ** 2 - self.q_std.log()).sum(-1)

    def forward(self, samples: torch.Tensor) -> tuple:
        """
        Args:
            samples: (N, dim) – samples drawn from q
        Returns:
            (weighted_mean, ess):
                weighted_mean: (dim,) – importance-weighted mean estimate of E_p[x]
                ess: scalar – effective sample size
        """
        log_w = self._log_p(samples) - self._log_q(samples)
        log_w = log_w - torch.logsumexp(log_w, dim=0)
        w = torch.exp(log_w)

        weighted_mean = (w.unsqueeze(-1) * samples).sum(dim=0)
        ess = 1.0 / (w ** 2).sum()
        return weighted_mean, ess

            if (j==0) xx=xv0; else if (j==1) xx=xv1; else if (j==2) xx=xv2; else if (j==3) xx=xv3;
            else if (j==4) xx=xv4; else if (j==5) xx=xv5; else if (j==6) xx=xv6; else if (j==7) xx=xv7;
            else if (j==8) xx=xv8; else if (j==9) xx=xv9; else if (j==10) xx=xv10; else if (j==11) xx=xv11;
            else if (j==12) xx=xv12; else if (j==13) xx=xv13; else if (j==14) xx=xv14; else xx=xv15;
            float s0 = stds[j];
            float t = (xx - mus[j]) / s0;       c0 += -0.5f*t*t - logf(s0);
            float s1 = stds[16 + j];
            t = (xx - mus[16 + j]) / s1;        c1 += -0.5f*t*t - logf(s1);
            float s2 = stds[32 + j];
            t = (xx - mus[32 + j]) / s2;        c2 += -0.5f*t*t - logf(s2);
        }
        float m = fmaxf(c0, fmaxf(c1, c2));
        float lp = m + logf(expf(c0 - m) + expf(c1 - m) + expf(c2 - m));
        float lw = lp - qlog;
        logw[i] = lw;
        local_max = fmaxf(local_max, lw);
    }
    smem[tid] = local_max;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) smem[tid] = fmaxf(smem[tid], smem[tid + s]);
        __syncthreads();
    }
    if (tid == 0) max_out[0] = smem[0];
}

__global__ void reduce_weighted_d16_kernel(const float* __restrict__ x,
                                           const float* __restrict__ logw,
                                           const float* __restrict__ max_in,
                                           float* __restrict__ mean,
                                           float* __restrict__ ess,
                                           int N) {
    extern __shared__ float sh[];
    int tid = threadIdx.x;
    const int BD = 256;
    float m = max_in[0];
    float sw = 0.0f, sw2 = 0.0f;
    float sx[16];
    #pragma unroll
    for (int j = 0; j < 16; ++j) sx[j] = 0.0f;

    for (int i = tid; i < N; i += blockDim.x) {
        float w = expf(logw[i] - m);
        sw += w;
        sw2 += w * w;
        const float* xi = x + ((long long)i) * 16;
        #pragma unroll
        for (int j = 0; j < 16; ++j) sx[j] += w * xi[j];
    }
    sh[tid] = sw;
    sh[BD + tid] = sw2;
    #pragma unroll
    for (int j = 0; j < 16; ++j) sh[(2 + j) * BD + tid] = sx[j];
    __syncthreads();

    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) {
            sh[tid] += sh[tid + s];
            sh[BD + tid] += sh[BD + tid + s];
            #pragma unroll
            for (int j = 0; j < 16; ++j) sh[(2 + j) * BD + tid] += sh[(2 + j) * BD + tid + s];
        }
        __syncthreads();
    }
    float inv = 1.0f / sh[0];
    if (tid < 16) mean[tid] = sh[(2 + tid) * BD] * inv;
    if (tid == 0) ess[0] = (sh[0] * sh[0]) / sh[BD];
}

__global__ void logw_max_generic_kernel(const float* __restrict__ x,
                                        const float* __restrict__ mix_w,
                                        const float* __restrict__ mus,
                                        const float* __restrict__ stds,
                                        const float* __restrict__ q_mu,
                                        const float* __restrict__ q_std,
                                        float* __restrict__ logw,
                                        float* __restrict__ max_out,
                                        int N, int D, int K) {
    __shared__ float smem[256];
    int tid = threadIdx.x;
    float local_max = -FLT_MAX;
    for (int i = tid; i < N; i += blockDim.x) {
        const float* xi = x + ((long long)i) * D;
        float qlog = 0.0f;
        for (int j = 0; j < D; ++j) {
            float t = (xi[j] - q_mu[j]) / q_std[j];
            qlog += -0.5f * t * t - logf(q_std[j]);
        }
        float maxc = -FLT_MAX;
        float sumexp = 0.0f;
        for (int k = 0; k < K; ++k) {
            float c = logf(mix_w[k]);
            for (int j = 0; j < D; ++j) {
                float st = stds[k * D + j];
                float t = (xi[j] - mus[k * D + j]) / st;
                c += -0.5f * t * t - logf(st);
            }
            if (k == 0) { maxc = c; sumexp = 1.0f; }
            else {
                if (c > maxc) { sumexp = sumexp * expf(maxc - c) + 1.0f; maxc = c; }
                else sumexp += expf(c - maxc);
            }
        }
        float lw = maxc + logf(sumexp) - qlog;
        logw[i] = lw;
        local_max = fmaxf(local_max, lw);
    }
    smem[tid] = local_max;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) smem[tid] = fmaxf(smem[tid], smem[tid + s]);
        __syncthreads();
    }
    if (tid == 0) max_out[0] = smem[0];
}

__global__ void reduce_generic_kernel(const float* __restrict__ x,
                                      const float* __restrict__ logw,
                                      const float* __restrict__ max_in,
                                      float* __restrict__ mean,
                                      float* __restrict__ ess,
                                      int N, int D) {
    __shared__ float red[256];
    int tid = threadIdx.x;
    float m = max_in[0];
    // First reduce scalar sums.
    float sw = 0.0f, sw2 = 0.0f;
    for (int i = tid; i < N; i += blockDim.x) {
        float w = expf(logw[i] - m);
        sw += w; sw2 += w*w;
    }
    red[tid] = sw; __syncthreads();
    for (int s=blockDim.x>>1; s>0; s>>=1) { if (tid<s) red[tid]+=red[tid+s]; __syncthreads(); }
    float total = red[0];
    red[tid] = sw2; __syncthreads();
    for (int s=blockDim.x>>1; s>0; s>>=1) { if (tid<s) red[tid]+=red[tid+s]; __syncthreads(); }
    if (tid == 0) ess[0] = total*total/red[0];
    for (int d = 0; d < D; ++d) {
        float sx = 0.0f;
        for (int i = tid; i < N; i += blockDim.x) sx += expf(logw[i] - m) * x[(long long)i * D + d];
        red[tid] = sx; __syncthreads();
        for (int s=blockDim.x>>1; s>0; s>>=1) { if (tid<s) red[tid]+=red[tid+s]; __syncthreads(); }
        if (tid == 0) mean[d] = red[0] / total;
        __syncthreads();
    }
}

                                             torch::Tensor mix_weights,
                                             torch::Tensor mix_mus,
                                             torch::Tensor mix_stds,
                                             torch::Tensor q_mu,
                                             torch::Tensor q_std) {
    TORCH_CHECK(samples.is_cuda(), "samples must be CUDA");

    int N = (int)samples.size(0);
    int D = (int)samples.size(1);
    int K = (int)mix_weights.size(0);
    auto opts = samples.options();
    auto logw = torch::empty({N}, opts);
    auto maxv = torch::empty({1}, opts);
    auto mean = torch::empty({D}, opts);
    auto ess = torch::empty({}, opts);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (D == 16 && K == 3) {

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define R 4
#define B 12
#define H 32
#define K 56
#define IN 128
#define WARPS_PER_BLOCK 4

__inline__ __device__ float warp_sum(float v) {
    v += __shfl_down_sync(0xffffffff, v, 16);
    v += __shfl_down_sync(0xffffffff, v, 8);
    v += __shfl_down_sync(0xffffffff, v, 4);
    v += __shfl_down_sync(0xffffffff, v, 2);
    v += __shfl_down_sync(0xffffffff, v, 1);
    return v;
}

__global__ __launch_bounds__(128) void gather_linear_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ out) {

    int tid = threadIdx.x;
    int lane = tid & 31;
    int warp = tid >> 5;
    int b = blockIdx.x;
    int k = blockIdx.y * WARPS_PER_BLOCK + warp;

    if (k >= K) return;

    float sum = 0.0f;
    const float* wk = w + k * IN;

    // The virtual concatenation maps input column idx = r*32+h to x[r,b,h].
    #pragma unroll
    for (int idx = lane; idx < IN; idx += 32) {
        int r = idx >> 5;
        int h = idx & 31;
        float xv = x[(r * B + b) * H + h];
        sum = fmaf(xv, wk[idx], sum);
    }

    sum = warp_sum(sum);
    if (lane == 0) {
        out[b * K + k] = sum + bias[k];
    }
}

torch::Tensor gather_linear_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    auto out = torch::empty({B, K}, x.options());
    dim3 block(128);
    dim3 grid(B, (K + WARPS_PER_BLOCK - 1) / WARPS_PER_BLOCK);
    gather_linear_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), out.data_ptr<float>());
    return out;
}
'''

Virtual all-gather on the last dim: stack R tensors of shape (B, H) into (B, R*H), then Linear.
"""

import torch.nn as nn

from kernelbench.distributed_collectives import default_device

B = 12
H = 32

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor virtual_gather_linear_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Fixed problem: x [4,12,32], weight [56,128], bias [56] -> y [12,56]
// One 128-thread block computes one output element.  This avoids materializing
// the virtual all-gather and replaces cat + cuBLAS + bias with one tiny kernel.
__global__ __launch_bounds__(128) void vgl_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ y)
{
    const int out = blockIdx.x;       // 0 .. 12*56-1
    const int k = out % 56;
    const int b = out / 56;
    const int t = threadIdx.x;        // 0 .. 127

    const int r = t >> 5;             // /32
    const int h = t & 31;             // %32
    // x is laid out as [R][B][H]; the virtually gathered feature index is r*32+h.
    float prod = x[r * (12*32) + b * 32 + h] * w[k * 128 + t];

    __shared__ float smem[128];
    smem[t] = prod;
    __syncthreads();

    // Tree reduction.  Deterministic and accurate enough for fp32 tolerance.
    if (t < 64) smem[t] += smem[t + 64];
    __syncthreads();
    if (t < 32) {
        float v = smem[t] + smem[t + 32];
        // warp-synchronous reduction for the final 32 values
        v += __shfl_down_sync(0xffffffff, v, 16);
        v += __shfl_down_sync(0xffffffff, v, 8);
        v += __shfl_down_sync(0xffffffff, v, 4);
        v += __shfl_down_sync(0xffffffff, v, 2);
        v += __shfl_down_sync(0xffffffff, v, 1);
        if (t == 0) y[out] = v + bias[k];
    }
}

torch::Tensor virtual_gather_linear_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    auto y = torch::empty({12, 56}, x.options());
    const dim3 grid(12 * 56);
    const dim3 block(128);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    vgl_kernel<<<grid, block, 0, stream>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), y.data_ptr<float>());
    return y;
}
"""

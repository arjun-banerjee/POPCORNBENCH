import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor gelu_sum_forward(torch::Tensor x);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float gelu_exact(float x) {
    const float inv_sqrt2 = 0.70710678118654752440f;
    return 0.5f * x * (1.0f + erff(x * inv_sqrt2));
}

__global__ void gelu_sum_kernel(const float* __restrict__ x, float* __restrict__ out) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < 560) {
        // x is laid out as [4, 14, 40]; each rank slice has 14*40 = 560 elements.
        float s = gelu_exact(x[idx]);
        s += gelu_exact(x[idx + 560]);
        s += gelu_exact(x[idx + 1120]);
        s += gelu_exact(x[idx + 1680]);
        out[idx] = s;
    }
}

torch::Tensor gelu_sum_forward(torch::Tensor x) {
    auto out = torch::empty({14, 40}, x.options());

    const int blocks = 3;

    gelu_sum_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), out.data_ptr<float>());
    return out;
}

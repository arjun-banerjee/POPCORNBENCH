import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Forward algorithm for Hidden Markov Models.  Computes the log-
    likelihood log p(observations) by iterating the forward recursion
    in log-space for numerical stability.  The dominant inner loop in
    HMM training (Baum-Welch) and Viterbi decoding.
    """

    def __init__(self, num_states, num_obs):
        super().__init__()
        self.num_states = num_states
        self.num_obs = num_obs

        self.log_init = nn.Parameter(torch.zeros(num_states))
        self.log_trans = nn.Parameter(torch.zeros(num_states, num_states))
        self.log_emit = nn.Parameter(torch.zeros(num_states, num_obs))

    def _normalize_log(self, x, dim=-1):
        return x - torch.logsumexp(x, dim=dim, keepdim=True)

    def forward(self, observations: torch.Tensor) -> torch.Tensor:
        """
        Args:
            observations: (B, T) – integer observation indices (as long tensor)
        Returns:
            log_lik: (B,) – log p(observations) under the HMM
        """
        B, T = observations.shape
        log_pi = self._normalize_log(self.log_init)
        log_A = self._normalize_log(self.log_trans, dim=-1)
        log_B = self._normalize_log(self.log_emit, dim=-1)

        # Initial step
        emit_0 = log_B[:, observations[:, 0]].t()  # (B, S)
        log_alpha = log_pi.unsqueeze(0) + emit_0

        for t in range(1, T):
            # Transition: log_alpha[b, j] = logsumexp_i(log_alpha[b, i] + log_A[i, j])
            log_alpha = log_alpha.unsqueeze(-1) + log_A.unsqueeze(0)  # (B, S, S)
            log_alpha = torch.logsumexp(log_alpha, dim=1)  # (B, S)
            emit_t = log_B[:, observations[:, t]].t()
            log_alpha = log_alpha + emit_t

        log_lik = torch.logsumexp(log_alpha, dim=-1)
        return log_lik

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

__global__ void hmm_forward_kernel(const long long* __restrict__ obs,
                                   const float* __restrict__ log_init,
                                   const float* __restrict__ log_trans,
                                   const float* __restrict__ log_emit,
                                   float* __restrict__ out,
                                   int B, int T, int S, int O) {
    int b = blockIdx.x;
    int tid = threadIdx.x;

    extern __shared__ float sh[];
    float* trans = sh;                 // S*S normalized transition entries
    float* emit_den = trans + S * S;   // S emission log normalizers
    float* alpha = emit_den + S;       // S current alphas
    float* next = alpha + S;           // S next alphas
    float* pi_den_p = next + S;        // one scalar

    // Normalize initial probabilities (small S, serial is cheapest here).
    if (tid == 0) {
        float m = -CUDART_INF_F;
        for (int i = 0; i < S; ++i) {
            float v = log_init[i];
            m = v > m ? v : m;
        }
        float s = 0.0f;
        for (int i = 0; i < S; ++i) s += expf(log_init[i] - m);
        *pi_den_p = m + logf(s);
    }

    // Each thread handles one (or more) source-state rows: normalize transition
    // rows and emission rows once for this sequence.
    for (int r = tid; r < S; r += blockDim.x) {
        float mA = -CUDART_INF_F;
        const float* tr = log_trans + r * S;
        for (int c = 0; c < S; ++c) {
            float v = tr[c];
            mA = v > mA ? v : mA;
        }
        float sumA = 0.0f;
        for (int c = 0; c < S; ++c) sumA += expf(tr[c] - mA);
        float denA = mA + logf(sumA);
        for (int c = 0; c < S; ++c) trans[r * S + c] = tr[c] - denA;

        float mE = -CUDART_INF_F;
        const float* er = log_emit + r * O;
        for (int k = 0; k < O; ++k) {
            float v = er[k];
            mE = v > mE ? v : mE;
        }
        float sumE = 0.0f;
        for (int k = 0; k < O; ++k) sumE += expf(er[k] - mE);
        emit_den[r] = mE + logf(sumE);
    }
    __syncthreads();

    const long long* ob = obs + (long long)b * T;
    int o0 = (int)ob[0];
    float pi_den = *pi_den_p;

    // Initial alpha: normalized pi plus normalized emission of first symbol.
    for (int j = tid; j < S; j += blockDim.x) {
        alpha[j] = (log_init[j] - pi_den) + (log_emit[j * O + o0] - emit_den[j]);
    }
    __syncthreads();

    // Forward recursion in log-space.
    for (int t = 1; t < T; ++t) {
        int ot = (int)ob[t];
        for (int j = tid; j < S; j += blockDim.x) {
            float m = -CUDART_INF_F;
            #pragma unroll
            for (int i = 0; i < 16; ++i) {
                if (i < S) {
                    float v = alpha[i] + trans[i * S + j];
                    m = v > m ? v : m;
                }
            }
            float sum = 0.0f;
            #pragma unroll
            for (int i = 0; i < 16; ++i) {
                if (i < S) sum += expf(alpha[i] + trans[i * S + j] - m);
            }
            next[j] = m + logf(sum) + (log_emit[j * O + ot] - emit_den[j]);
        }
        __syncthreads();
        for (int j = tid; j < S; j += blockDim.x) alpha[j] = next[j];
        __syncthreads();
    }

    // Final logsumexp over states.
    if (tid == 0) {
        float m = -CUDART_INF_F;
        for (int j = 0; j < S; ++j) {
            float v = alpha[j];
            m = v > m ? v : m;
        }
        float sum = 0.0f;
        for (int j = 0; j < S; ++j) sum += expf(alpha[j] - m);
        out[b] = m + logf(sum);
    }
}

torch::Tensor hmm_forward_cuda(torch::Tensor observations,
                               torch::Tensor log_init,
                               torch::Tensor log_trans,
                               torch::Tensor log_emit) {
    TORCH_CHECK(observations.is_cuda(), "observations must be a CUDA tensor");
    TORCH_CHECK(log_init.is_cuda() && log_trans.is_cuda() && log_emit.is_cuda(), "parameters must be CUDA tensors");
    TORCH_CHECK(log_init.scalar_type() == torch::kFloat32, "only float32 parameters are supported");
    TORCH_CHECK(observations.scalar_type() == torch::kInt64, "observations must be int64/long");

    const int B = (int)observations.size(0);
    const int T = (int)observations.size(1);
    const int S = (int)log_init.size(0);
    const int O = (int)log_emit.size(1);

    auto out = torch::empty({B}, log_init.options());
    if (B == 0) return out;

    const int threads = 32;
    size_t shmem = (size_t)(S * S + 3 * S + 1) * sizeof(float);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    hmm_forward_kernel<<<B, threads, shmem, stream>>>(
        reinterpret_cast<const long long*>(observations.data_ptr<int64_t>()),
        log_init.data_ptr<float>(),
        log_trans.data_ptr<float>(),
        log_emit.data_ptr<float>(),
        out.data_ptr<float>(),
        B, T, S, O);
    return out;
}

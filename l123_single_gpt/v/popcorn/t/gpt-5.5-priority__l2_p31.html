import math

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Fused MLA-style attention from compressed KV latents.

    The compressed latent representation is expanded into keys and values on the
    fly inside the attention computation rather than materializing a KV cache
    first. This isolates the main compute pattern behind fused MLA attention.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        kv_latent: torch.Tensor,
        w_up_k: torch.Tensor,
        w_up_v: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        rank = kv_latent.shape[-1]
        out = torch.zeros_like(q)
        scale = 1.0 / math.sqrt(head_dim)

        for b in range(batch_size):
            for h in range(num_heads):
                for t in range(seq_len):
                    logits = []
                    values = []
                    for s in range(t + 1):
                        latent = kv_latent[b, s]
                        k = torch.matmul(latent, w_up_k[:, h])
                        v = torch.matmul(latent, w_up_v[:, h])
                        logits.append((q[b, t, h] * k).sum() * scale)
                        values.append(v)
                    attn = torch.softmax(torch.stack(logits, dim=0), dim=0)
                    value_stack = torch.stack(values, dim=0)
                    out[b, t, h] = (attn.unsqueeze(-1) * value_stack).sum(dim=0)

        return out

#include <cuda.h>

// One block computes one (batch, time, head).  The problem sizes in the
// benchmark are tiny (T=32,H=4,D=16,R=24), so keeping the whole causal row in
// shared memory avoids all intermediate K/V materialization and all framework
// launch overhead from the reference.
__global__ void mla_kernel(const float* __restrict__ q,
                           const float* __restrict__ kv,
                           const float* __restrict__ wk,
                           const float* __restrict__ wv,
                           float* __restrict__ out,
                           int B, int T, int H, int D, int R) {
    int t = blockIdx.x;
    int h = blockIdx.y;
    int b = blockIdx.z;
    int tid = threadIdx.x; // 0..31

    __shared__ float logits[32];
    __shared__ float weights[32];

    const float scale = rsqrtf((float)D);

    // Compute one attention logit per thread for s <= t.
    if (tid < T) {
        float logit = -INFINITY;
        int s = tid;
        if (s <= t) {
            float acc = 0.0f;
            #pragma unroll
            for (int d = 0; d < 16; ++d) {
                if (d < D) {
                    float kval = 0.0f;
                    #pragma unroll
                    for (int r = 0; r < 24; ++r) {
                        if (r < R) {
                            float l = kv[(b*T + s)*R + r];
                            float ww = wk[(r*H + h)*D + d];
                            kval = fmaf(l, ww, kval);
                        }
                    }
                    float qv = q[((b*T + t)*H + h)*D + d];
                    acc = fmaf(qv, kval, acc);
                }
            }
            logit = acc * scale;
        }
        logits[tid] = logit;
    }
    __syncthreads();

    // Serial softmax over at most 32 elements.  This is cheap and avoids a more
    // complicated reduction for such a short row.
    if (tid == 0) {
        float m = logits[0];
        #pragma unroll
        for (int s = 1; s < 32; ++s) {
            if (s <= t && s < T) m = fmaxf(m, logits[s]);
        }
        float denom = 0.0f;
        #pragma unroll
        for (int s = 0; s < 32; ++s) {
            float w = 0.0f;
            if (s <= t && s < T) {
                w = expf(logits[s] - m);
                denom += w;
            }
            weights[s] = w;
        }
        float inv = 1.0f / denom;
        #pragma unroll
        for (int s = 0; s < 32; ++s) {
            weights[s] *= inv;
        }
    }
    __syncthreads();

    // Threads 0..D-1 compute output dimensions independently.
    if (tid < D) {
        int d = tid;
        float acc_out = 0.0f;
        #pragma unroll
        for (int s = 0; s < 32; ++s) {
            if (s <= t && s < T) {
                float vval = 0.0f;
                #pragma unroll
                for (int r = 0; r < 24; ++r) {
                    if (r < R) {
                        float l = kv[(b*T + s)*R + r];
                        float ww = wv[(r*H + h)*D + d];
                        vval = fmaf(l, ww, vval);
                    }
                }
                acc_out = fmaf(weights[s], vval, acc_out);
            }
        }
        out[((b*T + t)*H + h)*D + d] = acc_out;
    }
}

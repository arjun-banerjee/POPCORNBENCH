import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Scalar-gated delta-rule linear attention recurrence.

    A recurrent state matrix is decayed by a scalar gate per head, updated by a
    rank-1 key/value outer product, then read out by a query vector.
    """

    def __init__(self):
        super().__init__()

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        gate: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        batch_size, seq_len, num_heads, head_dim = q.shape
        state = torch.zeros(batch_size, num_heads, head_dim, head_dim, dtype=q.dtype, device=q.device)
        outputs = []

        for t in range(seq_len):
            gate_t = gate[:, t].unsqueeze(-1).unsqueeze(-1)
            beta_t = beta[:, t].unsqueeze(-1).unsqueeze(-1)
            kv_outer = k[:, t].unsqueeze(-1) * v[:, t].unsqueeze(-2)
            state = gate_t * state + beta_t * kv_outer
            y_t = (q[:, t].unsqueeze(-1) * state).sum(dim=-2)
            outputs.append(y_t)

        return torch.stack(outputs, dim=1)

#include <torch/extension.h>

torch::Tensor delta_forward(torch::Tensor q,
                            torch::Tensor k,

                            torch::Tensor gate,
                            torch::Tensor beta);

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__inline__ __device__ float warp_sum(float val) {
    unsigned mask = 0xffffffffu;
    val += __shfl_down_sync(mask, val, 16);
    val += __shfl_down_sync(mask, val, 8);
    val += __shfl_down_sync(mask, val, 4);
    val += __shfl_down_sync(mask, val, 2);
    val += __shfl_down_sync(mask, val, 1);
    return val;
}

// One warp computes one output channel/column for one (batch, head).
// Each active lane owns one row of the 24-element state column and carries it
// in a register through the full time recurrence.
__global__ void delta_rule_kernel(const float* __restrict__ q,
                                  const float* __restrict__ k,
                                  const float* __restrict__ v,
                                  const float* __restrict__ gate,
                                  const float* __restrict__ beta,
                                  float* __restrict__ out,
                                  int B, int T, int H, int D) {
    const int j = blockIdx.x;       // output/value dimension
    const int h = blockIdx.y;
    const int b = blockIdx.z;
    const int lane = threadIdx.x;   // 0..31, D is 24 for this workload

    float st = 0.0f;
    const unsigned mask = 0xffffffffu;

    for (int t = 0; t < T; ++t) {
        const int bh_base = (b * T + t) * H + h;
        const int base = bh_base * D;

        float g = 0.0f, be = 0.0f, vj = 0.0f;
        if (lane == 0) {
            g = gate[bh_base];
            be = beta[bh_base];
            vj = v[base + j];
        }
        g = __shfl_sync(mask, g, 0);
        be = __shfl_sync(mask, be, 0);
        vj = __shfl_sync(mask, vj, 0);

        float contrib = 0.0f;
        if (lane < D) {
            const float ki = k[base + lane];
            const float qi = q[base + lane];
            // Match eager PyTorch's sequence of fp32 pointwise operations closely.
            const float decayed = __fmul_rn(g, st);
            const float kv = __fmmul_rn(ki, vj);
            const float upd = __fmul_rn(be, kv);
            st = __fadd_rn(decayed, upd);
            contrib = __fmul_rn(qi, st);
        }

        float y = warp_sum(contrib);
        if (lane == 0) {
            out[base + j] = y;
        }
    }
}

torch::Tensor delta_forward(torch::Tensor q,
                            torch::Tensor k,
                            torch::Tensor v,
                            torch::Tensor gate,
                            torch::Tensor beta) {
    const int B = (int)q.size(0);
    const int T = (int)q.size(1);
    const int H = (int)q.size(2);
    const int D = (int)q.size(3);
    auto out = torch::empty_like(q);

    dim3 block(32, 1, 1);
    dim3 grid(D, H, B);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    delta_rule_kernel<<<grid, block, 0, stream>>>(
        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(),
        gate.data_ptr<float>(), beta.data_ptr<float>(), out.data_ptr<float>(),
        B, T, H, D);
    return out;
}
'''

import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Bayesian linear regression with a conjugate Gaussian prior.
    Given features X and targets y, computes the posterior distribution
    over weights: p(w | X, y) = N(mu_post, Sigma_post) in closed form
    via the matrix inversion lemma / Cholesky decomposition.
    """

    def __init__(self, input_dim, prior_precision=1.0, noise_precision=1.0):
        super().__init__()
        self.input_dim = input_dim
        self.prior_precision = prior_precision
        self.noise_precision = noise_precision

    def forward(
        self, X: torch.Tensor, y: torch.Tensor, X_test: torch.Tensor
    ) -> tuple:
        """
        Args:
            X:      (B, N, D) – training features
            y:      (B, N, 1) – training targets
            X_test: (B, M, D) – test features
        Returns:
            (pred_mean, pred_var):
                pred_mean: (B, M, 1)
                pred_var:  (B, M)
        """
        D = X.shape[-1]
        eye = torch.eye(D, device=X.device, dtype=X.dtype).unsqueeze(0)

        # Posterior precision: Lambda_post = alpha * I + beta * X^T X
        Lambda_post = self.prior_precision * eye + self.noise_precision * (X.transpose(-1, -2) @ X)
        L = torch.linalg.cholesky(Lambda_post)

        # Posterior mean: mu_post = Lambda_post^{-1} * beta * X^T y
        rhs = self.noise_precision * (X.transpose(-1, -2) @ y)
        mu_post = torch.cholesky_solve(rhs, L)

        # Predictive distribution
        pred_mean = X_test @ mu_post
        Sigma_post = torch.cholesky_inverse(L)
        pred_var = (X_test @ Sigma_post * X_test).sum(-1) + 1.0 / self.noise_precision

        return pred_mean, pred_var

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>
#include <vector>

std::vector<torch::Tensor> blr_forward_cuda(torch::Tensor X, torch::Tensor y, torch::Tensor X_test, double prior_precision, double noise_precision);

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>
#include <cmath>

__global__ void blr_kernel(const float* __restrict__ X,
                           const float* __restrict__ y,
                           const float* __restrict__ Xt,
                           float* __restrict__ out_mean,
                           float* __restrict__ out_var,
                           int B, int N, int D, int M,
                           float alpha, float beta) {
    // Optimized for the benchmark dimensions (D=16), but works for D<=16.
    int b = blockIdx.x;
    int tid = threadIdx.x;
    __shared__ float A[16*16];
    __shared__ float rhs[16];
    __shared__ float L[16*16];
    __shared__ float invA[16*16];
    __shared__ float mu[16];
    __shared__ float tmp[16*16];

    const float* Xb = X + ((long long)b) * N * D;
    const float* yb = y + ((long long)b) * N;
    const float* Xtb = Xt + ((long long)b) * M * D;

    if (tid < 256) {
        int i = tid >> 4;
        int j = tid & 15;
        if (i < D && j < D) {
            float s = 0.0f;
            #pragma unroll 4
            for (int n = 0; n < N; ++n) {
                s = fmaf(Xb[n*D + i], Xb[n*D + j], s);
            }
            A[i*16 + j] = beta * s + ((i == j) ? alpha : 0.0f);
        }
    }
    if (tid < 16) {
        int i = tid;
        if (i < D) {
            float s = 0.0f;
            #pragma unroll 4
            for (int n = 0; n < N; ++n) {
                s = fmaf(Xb[n*D + i], yb[n], s);
            }
            rhs[i] = beta * s;
        }
    }
    __syncthreads();

    if (tid == 0) {
        // Cholesky decomposition A = L L^T, lower triangular.
        for (int i = 0; i < D; ++i) {
            for (int j = 0; j <= i; ++j) {
                float sum = A[i*16 + j];
                for (int k = 0; k < j; ++k) sum -= L[i*16 + k] * L[j*16 + k];
                if (i == j) {
                    L[i*16 + j] = sqrtf(fmaxf(sum, 1.0e-20f));
                } else {
                    L[i*16 + j] = sum / L[j*16 + j];
                }
            }
            for (int j = i + 1; j < D; ++j) L[i*16 + j] = 0.0f;
        }

        // Solve A * mu = rhs using the Cholesky factors.
        float z[16];
        for (int i = 0; i < D; ++i) {
            float s = rhs[i];
            for (int k = 0; k < i; ++k) s -= L[i*16 + k] * z[k];
            z[i] = s / L[i*16 + i];
        }
        for (int i = D - 1; i >= 0; --i) {
            float s = z[i];
            for (int k = i + 1; k < D; ++k) s -= L[k*16 + i] * mu[k];
            mu[i] = s / L[i*16 + i];
        }

        // Compute inv(L) into tmp (lower triangular) by triangular inverse.
        for (int i = 0; i < D*16; ++i) tmp[i] = 0.0f;
        for (int col = 0; col < D; ++col) {
            for (int i = 0; i < D; ++i) {
                float s = (i == col) ? 1.0f : 0.0f;
                for (int k = col; k < i; ++k) s -= L[i*16 + k] * tmp[k*16 + col];
                if (i >= col) tmp[i*16 + col] = s / L[i*16 + i];
            }
        }
        // inv(A) = inv(L)^T * inv(L).  Store full symmetric matrix.
        for (int i = 0; i < D; ++i) {
            for (int j = 0; j < D; ++j) {
                float s = 0.0f;
                int start = (i > j) ? i : j;
                for (int k = start; k < D; ++k) s += tmp[k*16 + i] * tmp[k*16 + j];
                invA[i*16 + j] = s;
            }
        }
    }
    __syncthreads();

    // Predictive mean and variance. One thread per test point, striding for general M.
    for (int m = tid; m < M; m += blockDim.x) {
        const float* x = Xtb + m * D;
        float mean = 0.0f;
        for (int i = 0; i < D; ++i) mean = fmaf(x[i], mu[i], mean);
        out_mean[((long long)b) * M + m] = mean;

        float var = 0.0f;
        for (int i = 0; i < D; ++i) {
            float row = 0.0f;
            for (int j = 0; j < D; ++j) row = fmaf(invA[i*16 + j], x[j], row);
            var = fmaf(x[i], row, var);
        }
        out_var[((long long)b) * M + m] = var + 1.0f / beta;
    }
}

std::vector<torch::Tensor> blr_forward_cuda(torch::Tensor X, torch::Tensor y, torch::Tensor X_test, double prior_precision, double noise_precision) {
    const int B = (int)X.size(0);
    const int N = (int)X.size(1);
    const int D = (int)X.size(2);
    const int M = (int)X_test.size(1);
    auto mean = torch::empty({B, M, 1}, X.options());
    auto var = torch::empty({B, M}, X.options());
    const int threads = 256;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    blr_kernel<<<B, threads, 0, stream>>>(
        X.data_ptr<float>(), y.data_ptr<float>(), X_test.data_ptr<float>(),
        mean.data_ptr<float>(), var.data_ptr<float>(), B, N, D, M,
        (float)prior_precision, (float)noise_precision);
    return {mean, var};
}

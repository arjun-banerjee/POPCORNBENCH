import torch
import torch.nn as nn
import torch.nn.functional as F
import math


class Model(nn.Module):
    """
    Row-wise gated self-attention over Multiple Sequence Alignment (MSA)
    representations.  Each row of the MSA is an independent sequence; for
    each row, standard multi-head attention is performed across residue
    positions, with an additive bias from the pair representation.
    """

    def __init__(self, msa_dim, pair_dim, num_heads, dropout=0.0):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim = msa_dim // num_heads
        assert msa_dim % num_heads == 0

        self.layer_norm = nn.LayerNorm(msa_dim)
        self.query = nn.Linear(msa_dim, msa_dim, bias=False)
        self.key = nn.Linear(msa_dim, msa_dim, bias=False)
        self.value = nn.Linear(msa_dim, msa_dim, bias=False)
        self.gate = nn.Linear(msa_dim, msa_dim)
        self.pair_bias = nn.Linear(pair_dim, num_heads, bias=False)
        self.out_proj = nn.Linear(msa_dim, msa_dim)
        self.dropout = nn.Dropout(dropout)

    def forward(
        self,
        msa_repr: torch.Tensor,
        pair_repr: torch.Tensor,
        msa_mask: torch.Tensor,
    ) -> torch.Tensor:
        """
        Args:
            msa_repr:  (B, S, N, msa_dim)
            pair_repr: (B, N, N, pair_dim)
            msa_mask:  (B, S, N)
        Returns:
            updated MSA (B, S, N, msa_dim)
        """
        B, S, N, D = msa_repr.shape
        h = self.num_heads
        d = self.head_dim

        x = self.layer_norm(msa_repr)
        q = self.query(x).view(B, S, N, h, d).permute(0, 1, 3, 2, 4)
        k = self.key(x).view(B, S, N, h, d).permute(0, 1, 3, 2, 4)
        v = self.value(x).view(B, S, N, h, d).permute(0, 1, 3, 2, 4)
        g = torch.sigmoid(self.gate(x)).view(B, S, N, h, d).permute(0, 1, 3, 2, 4)

        # Pair bias: shared across MSA rows
        bias = self.pair_bias(pair_repr).permute(0, 3, 1, 2)  # (B, h, N, N)

        attn = torch.einsum("bshid,bshjd->bshij", q, k) / math.sqrt(d)
        attn = attn + bias.unsqueeze(1)

        mask = msa_mask.view(B, S, 1, 1, N).expand_as(attn)
        attn = attn.masked_fill(mask == 0, float("-inf"))
        attn = F.softmax(attn, dim=-1)
        attn = self.dropout(attn)

        out = torch.einsum("bshij,bshjd->bshid", attn, v)
        out = g * out
        out = out.permute(0, 1, 3, 2, 4).reshape(B, S, N, D)
        return self.out_proj(out)

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#define D 64
#define H 4
#define HD 16
#define PD 64

__global__ void proj_kernel(
    const float* __restrict__ msa,
    const float* __restrict__ ln_w,
    const float* __restrict__ ln_b,
    const float* __restrict__ q_w,
    const float* __restrict__ k_w,
    const float* __restrict__ v_w,
    const float* __restrict__ gate_w,
    const float* __restrict__ gate_b,
    float* __restrict__ q,
    float* __restrict__ k,
    float* __restrict__ v,
    float* __restrict__ g,
    int total_tokens) {
  int token = blockIdx.x;
  int tid = threadIdx.x;
  if (token >= total_tokens) return;

  __shared__ float red[128];
  __shared__ float mean_s;
  __shared__ float inv_s;
  __shared__ float xn[D];

  const float* x = msa + token * D;
  float sum = 0.0f;
  float sum2 = 0.0f;
  for (int c = tid; c < D; c += blockDim.x) {
    float val = x[c];
    sum += val;
    sum2 += val * val;
  }
  red[tid] = sum;
  __syncthreads();
  for (int off = 64; off > 0; off >>= 1) {
    if (tid < off) red[tid] += red[tid + off];
    __syncthreads();
  }
  float total_sum = red[0];

  red[tid] = sum2;
  __syncthreads();
  for (int off = 64; off > 0; off >>= 1) {
    if (tid < off) red[tid] += red[tid + off];
    __syncthreads();
  }
  if (tid == 0) {
    float mean = total_sum * (1.0f / 64.0f);
    float var = red[0] * (1.0f / 64.0f) - mean * mean;
    if (var < 0.0f) var = 0.0f;
    mean_s = mean;
    inv_s = rsqrtf(var + 1.0e-5f);
  }
  __syncthreads();

  for (int c = tid; c < D; c += blockDim.x) {
    xn[c] = (x[c] - mean_s) * inv_s * ln_w[c] + ln_b[c];
  }
  __syncthreads();

  if (tid < D) {
    float sq = 0.0f, sk = 0.0f, sv = 0.0f, sg = gate_b[tid];
    #pragma unroll
    for (int c = 0; c < D; ++c) {
      float xv = xn[c];
      sq += xv * q_w[tid * D + c];
      sk += xv * k_w[tid * D + c];
      sv += xv * v_w[tid * D + c];
      sg += xv * gate_w[tid * D + c];
    }
    int h = tid >> 4;
    int dd = tid & 15;
    int idx = ((token * H + h) * HD + dd);
    q[idx] = sq;
    k[idx] = sk;
    v[idx] = sv;
    g[idx] = 1.0f / (1.0f + expf(-sg));
  }
}

__global__ void attn_out_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    const float* __restrict__ g,
    const float* __restrict__ pair,
    const float* __restrict__ mask,
    const float* __restrict__ pair_w,
    const float* __restrict__ out_w,
    const float* __restrict__ out_b,
    float* __restrict__ out,
    int B, int S, int N) {
  int token = blockIdx.x;  // ((b*S+s)*N+i)
  int tid = threadIdx.x;
  int i = token % N;
  int tmp0 = token / N;
  int s = tmp0 % S;
  int b = tmp0 / S;

  __shared__ float scores[H * 32];
  __shared__ float temp[D];

  // Compute attention logits for this residue i, all heads and key positions.
  for (int idx = tid; idx < H * N; idx += blockDim.x) {
    int h = idx / N;
    int j = idx - h * N;
    float sc = -INFINITY;
    if (mask[(b * S + s) * N + j] != 0.0f) {
      int qi = ((token * H + h) * HD);
      int kj_token = (b * S + s) * N + j;
      int kj = ((kj_token * H + h) * HD);
      float dot = 0.0f;
      #pragma unroll
      for (int d = 0; d < HD; ++d) dot += q[qi + d] * k[kj + d];

      int pbase = ((b * N + i) * N + j) * PD;
      float pb = 0.0f;
      #pragma unroll
      for (int p = 0; p < PD; ++p) pb += pair[pbase + p] * pair_w[h * PD + p];
      sc = dot * 0.25f + pb;
    }
    scores[idx] = sc;
  }
  __syncthreads();

  // Softmax independently per head.  One thread per head is enough for N=32.
  if (tid < H) {
    int base = tid * N;
    float m = scores[base];
    #pragma unroll
    for (int j = 1; j < 32; ++j) {
      if (j < N) {
        float val = scores[base + j];
        m = val > m ? val : m;
      }
    }
    float ss = 0.0f;
    #pragma unroll
    for (int j = 0; j < 32; ++j) {
      if (j < N) {
        float e = expf(scores[base + j] - m);
        scores[base + j] = e;
        ss += e;
      }
    }
    float inv = 1.0f / ss;
    #pragma unroll
    for (int j = 0; j < 32; ++j) {
      if (j < N) scores[base + j] *= inv;
    }
  }
  __syncthreads();

  // Weighted value, followed by the sigmoid gate.
  for (int c = tid; c < D; c += blockDim.x) {
    int h = c >> 4;
    int dd = c & 15;
    float acc = 0.0f;
    #pragma unroll
    for (int j = 0; j < 32; ++j) {
      if (j < N) {
        int vtoken = (b * S + s) * N + j;
        int vidx = ((vtoken * H + h) * HD + dd);
        acc += scores[h * N + j] * v[vidx];
      }
    }
    int gidx = ((token * H + h) * HD + dd);
    temp[c] = acc * g[gidx];
  }
  __syncthreads();

  // Final output projection: out = temp @ W_out^T + bias.
  for (int o = tid; o < D; o += blockDim.x) {
    float acc = out_b[o];
    #pragma unroll
    for (int c = 0; c < D; ++c) acc += temp[c] * out_w[o * D + c];
    out[token * D + o] = acc;
  }
}

torch::Tensor row_attn_cuda(
    torch::Tensor msa,
    torch::Tensor pair,
    torch::Tensor mask,
    torch::Tensor ln_w,
    torch::Tensor ln_b,
    torch::Tensor q_w,
    torch::Tensor k_w,
    torch::Tensor v_w,
    torch::Tensor gate_w,
    torch::Tensor gate_b,
    torch::Tensor pair_w,
    torch::Tensor out_w,
    torch::Tensor out_b) {
  int B = (int)msa.size(0);
  int S = (int)msa.size(1);
  int N = (int)msa.size(2);
  int total = B * S * N;

  auto opts = msa.options();
  auto q = torch::empty({B, S, N, H, HD}, opts);
  auto k = torch::empty({B, S, N, H, HD}, opts);
  auto v = torch::empty({B, S, N, H, HD}, opts);
  auto g = torch::empty({B, S, N, H, HD}, opts);
  auto out = torch::empty_like(msa);

  const int threads = 128;
  proj_kernel<<<total, threads>>>(
      msa.data_ptr<float>(), ln_w.data_ptr<float>(), ln_b.data_ptr<float>(),
      q_w.data_ptr<float>(), k_w.data_ptr<float>(), v_w.data_ptr<float>(),
      gate_w.data_ptr<float>(), gate_b.data_ptr<float>(),
      q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), g.data_ptr<float>(), total);

  attn_out_kernel<<<total, threads>>>(
      q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(), g.data_ptr<float>(),
      pair.data_ptr<float>(), mask.data_ptr<float>(), pair_w.data_ptr<float>(),
      out_w.data_ptr<float>(), out_b.data_ptr<float>(), out.data_ptr<float>(), B, S, N);

  return out;
}
"""

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Performs a depthwise 1D convolution where each input channel is convolved with its own filter.
    Fundamental primitive in Hyena, Mamba, and genomic sequence models (Enformer, HyenaDNA).

    Args:
        channels (int): Number of input/output channels.
        kernel_size (int): Size of the convolution kernel.
        stride (int, optional): Stride of the convolution. Defaults to 1.
        padding (int, optional): Padding applied to the input. Defaults to 0.
        bias (bool, optional): If True, adds a learnable bias. Defaults to False.
    """
    def __init__(self, channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.conv1d = nn.Conv1d(channels, channels, kernel_size, stride=stride, padding=padding, groups=channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.conv1d(x)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>

__global__ void dwconv1d_k3s1p1_vec4_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ y,
    int rows, int C, int L, int has_bias)
{
    int vecs_per_row = L >> 2;
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t total = (int64_t)rows * vecs_per_row;
    if (idx >= total) return;

    int row = (int)(idx / vecs_per_row);
    int vec = (int)(idx - (int64_t)row * vecs_per_row);
    int t = vec << 2;
    int c = row % C;
    int base = row * L + t;

    float w0 = w[c * 3 + 0];
    float w1 = w[c * 3 + 1];
    float w2 = w[c * 3 + 2];
    float bv = has_bias ? bias[c] : 0.0f;

    const float* xp = x + base;
    float xm1 = (t == 0) ? 0.0f : xp[-1];
    float x0 = xp[0];
    float x1 = xp[1];
    float x2 = xp[2];
    float x3 = xp[3];
    float x4 = (t + 4 >= L) ? 0.0f : xp[4];

    float o0 = fmaf(w2, x1, fmaf(w1, x0, w0 * xm1)) + bv;
    float o1 = fmaf(w2, x2, fmaf(w1, x1, w0 * x0)) + bv;
    float o2 = fmaf(w2, x3, fmaf(w1, x2, w0 * x1)) + bv;
    float o3 = fmaf(w2, x4, fmaf(w1, x3, w0 * x2)) + bv;

    *reinterpret_cast<float4*>(y + base) = make_float4(o0, o1, o2, o3);
}

__global__ void dwconv1d_generic_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ y,
    int B, int C, int L, int K, int Lout, int stride, int padding, int has_bias)
{
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t total = (int64_t)B * C * Lout;
    if (idx >= total) return;

    int t = (int)(idx % Lout);
    int tmp = (int)(idx / Lout);
    int c = tmp % C;
    int b = tmp / C;

    float acc = has_bias ? bias[c] : 0.0f;
    int in0 = t * stride - padding;
    const float* xrow = x + ((int64_t)b * C + c) * L;
    const float* wrow = w + (int64_t)c * K;
    #pragma unroll
    for (int k = 0; k < 16; ++k) {
        if (k >= K) break;
        int in_t = in0 + k;
        if ((unsigned)in_t < (unsigned)L) {
            acc = fmaf(wrow[k], xrow[in_t], acc);
        }
    }
    for (int k = 16; k < K; ++k) {
        int in_t = in0 + k;
        if ((unsigned)in_t < (unsigned)L) {
            acc = fmaf(wrow[k], xrow[in_t], acc);
        }
    }
    y[idx] = acc;
}

class Model(nn.Module):
    """
    Fused 1D convolution + GroupNorm for temporal sequences (N, C, T).

    Typical in time-series heads, audio / motor trajectories, and sequence models
    that use strided or padded Conv1d followed by per-channel normalization without
    collapsing the time dimension (GroupNorm over `out_channels` on (N, C_out, T')).

    Reference fusion target: `Conv1d` → `GroupNorm` (no activation), matching a
    common stem or residual branch in temporal backbones.
    """

    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        num_groups: int,
        kernel_size: int = 3,
        stride: int = 1,
        padding: int = 1,
        dilation: int = 1,
        eps: float = 1e-5,
        affine: bool = True,
    ):
        super().__init__()
        assert out_channels % num_groups == 0
        self.conv = nn.Conv1d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=False,
        )
        self.norm = nn.GroupNorm(
            num_groups=num_groups,
            num_channels=out_channels,
            eps=eps,
            affine=affine,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (N, C_in, T) temporal feature map
        Returns:
            (N, C_out, T') with T' determined by conv stride / padding / kernel
        """
        return self.norm(self.conv(x))

// Specialized fast path for the benchmark's Conv1d parameters:
// kernel=3, stride=1, padding=1, dilation=1, fp32, contiguous NCT layout.
// One block computes a 16 (time) x 16 (output-channel) tile for one batch item.
__global__ void conv1d_k3s1p1_tile_kernel(const float* __restrict__ x,
                                           const float* __restrict__ w,
                                           float* __restrict__ y,
                                           int N, int Cin, int Cout, int T, int Tout) {
    const int tx = threadIdx.x;          // 0..15 time within tile
    const int ty = threadIdx.y;          // 0..15 output channel within tile
    const int tid = ty * 16 + tx;
    const int t0 = blockIdx.x * 16;
    const int co = blockIdx.y * 16 + ty;
    const int n = blockIdx.z;
    const int t = t0 + tx;

    __shared__ float sx[18];             // x[n, ci, t0-1 : t0+16]
    __shared__ float sw[16][3];          // w[co_tile, ci, k]

    float acc = 0.0f;
    for (int ci = 0; ci < Cin; ++ci) {
        if (tid < 18) {
            const int tt = t0 + tid - 1;
            float v = 0.0f;
            if ((unsigned)tt < (unsigned)T) {
                v = x[(n * Cin + ci) * T + tt];
            }
            sx[tid] = v;
        }
        if (tx < 3) {
            const int c = co;
            if (c < Cout) {
                sw[ty][tx] = w[(c * Cin + ci) * 3 + tx];
            }
        }
        __syncthreads();
        if (co < Cout && t < Tout) {
            acc = fmaf(sw[ty][0], sx[tx], acc);
            acc = fmaf(sw[ty][1], sx[tx + 1], acc);
            acc = fmaf(sw[ty][2], sx[tx + 2], acc);
        }
        __syncthreads();
    }
    if (co < Cout && t < Tout) {
        y[(n * Cout + co) * Tout + t] = acc;
    }
}

// Generic direct Conv1d fallback for unusual constructor arguments.  It is mainly
// for correctness outside the benchmark shape; the specialized path above is used
// for the provided case.
__global__ void conv1d_generic_kernel(const float* __restrict__ x,
                                      const float* __restrict__ w,
                                      float* __restrict__ y,
                                      int N, int Cin, int Cout, int T, int K,
                                      int Tout, int stride, int padding, int dilation) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * Tout;
    for (; idx < total; idx += blockDim.x * gridDim.x) {
        int t = idx % Tout;
        int co = (idx / Tout) % Cout;
        int n = idx / (Tout * Cout);
        float acc = 0.0f;
        int base = t * stride - padding;
        for (int ci = 0; ci < Cin; ++ci) {
            #pragma unroll
            for (int k = 0; k < 8; ++k) {
                if (k >= K) break;
                int ti = base + k * dilation;
                if ((unsigned)ti < (unsigned)T) {
                    acc = fmaf(w[(co * Cin + ci) * K + k], x[(n * Cin + ci) * T + ti], acc);
                }
            }
            for (int k = 8; k < K; ++k) {
                int ti = base + k * dilation;
                if ((unsigned)ti < (unsigned)T) {
                    acc = fmaf(w[(co * Cin + ci) * K + k], x[(n * Cin + ci) * T + ti], acc);
                }
            }
        }
        y[idx] = acc;
    }
}

// One block per (sample, group).  Reduces over C_per_group * T, then writes the
// normalized and affine-transformed output.
__global__ void groupnorm_forward_kernel(const float* __restrict__ inp,
                                         const float* __restrict__ gamma,
                                         const float* __restrict__ beta,
                                         float* __restrict__ out,
                                         int N, int C, int T, int G,
                                         float eps, bool affine) {
    extern __shared__ float smem[];
    float* ssum = smem;
    float* ssq = smem + blockDim.x;
    const int tid = threadIdx.x;
    const int ng = blockIdx.x;
    const int n = ng / G;
    const int g = ng - n * G;
    const int cpg = C / G;
    const int elems = cpg * T;
    const int cbase = g * cpg;

    float sum = 0.0f;
    float sumsq = 0.0f;
    for (int i = tid; i < elems; i += blockDim.x) {
        int c = cbase + i / T;
        int t = i - (i / T) * T;
        float v = inp[(n * C + c) * T + t];
        sum += v;
        sumsq = fmaf(v, v, sumsq);
    }
    ssum[tid] = sum;
    ssq[tid] = sumsq;
    __syncthreads();

    for (int offset = blockDim.x >> 1; offset > 0; offset >>= 1) {
        if (tid < offset) {
            ssum[tid] += ssum[tid + offset];
            ssq[tid] += ssq[tid + offset];
        }
        __syncthreads();
    }

    const float mean = ssum[0] / (float)elems;
    float var = ssq[0] / (float)elems - mean * mean;
    var = var < 0.0f ? 0.0f : var;
    const float rstd = rsqrtf(var + eps);

    for (int i = tid; i < elems; i += blockDim.x) {
        int c_local = i / T;
        int c = cbase + c_local;
        int t = i - c_local * T;
        float v = inp[(n * C + c) * T + t];
        float z = (v - mean) * rstd;
        if (affine) z = z * gamma[c] + beta[c];
        out[(n * C + c) * T + t] = z;
    }
}

        eps: float = 1e-5,
        affine: bool = True,
    ):
        super().__init__()
        assert out_channels % num_groups == 0
        self.conv = nn.Conv1d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=False,
        )
        self.norm = nn.GroupNorm(
            num_groups=num_groups,
            num_channels=out_channels,
            eps=eps,
            affine=affine,
        )
        self.num_groups = num_groups
        self.eps = eps
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.affine = affine

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.affine:
            gamma = self.norm.weight
            beta = self.norm.bias
        else:
            gamma = torch.empty(0, device=x.device, dtype=x.dtype)
            beta = torch.empty(0, device=x.device, dtype=x.dtype)
        return _ext.fused_conv_gn(
            x,
            self.conv.weight,
            gamma,
            beta,
            self.num_groups,
            float(self.eps),
            int(self.stride),
            int(self.padding),
            int(self.dilation),
        )

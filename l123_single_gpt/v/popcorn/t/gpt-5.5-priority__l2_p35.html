class Model(nn.Module):
    """
    Fused 1D convolution + GroupNorm for temporal sequences (N, C, T).

    Typical in time-series heads, audio / motor trajectories, and sequence models
    that use strided or padded Conv1d followed by per-channel normalization without
    collapsing the time dimension (GroupNorm over `out_channels` on (N, C_out, T')).

    Reference fusion target: `Conv1d` → `GroupNorm` (no activation), matching a
    common stem or residual branch in temporal backbones.
    """

    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        num_groups: int,
        kernel_size: int = 3,
        stride: int = 1,
        padding: int = 1,
        dilation: int = 1,
        eps: float = 1e-5,
        affine: bool = True,
    ):
        super().__init__()
        assert out_channels % num_groups == 0
        self.conv = nn.Conv1d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=False,
        )
        self.norm = nn.GroupNorm(
            num_groups=num_groups,
            num_channels=out_channels,
            eps=eps,
            affine=affine,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: (N, C_in, T) temporal feature map
        Returns:
            (N, C_out, T') with T' determined by conv stride / padding / kernel
        """
        return self.norm(self.conv(x))

static __inline__ __device__ float warp_sum(float v) {
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1)
        v += __shfl_down_sync(0xffffffff, v, off);
    return v;
}

static __inline__ __device__ void block_sum2(float &a, float &b) {
    __shared__ float sa[32];
    __shared__ float sb[32];
    int tid = threadIdx.x;
    int lane = tid & 31;
    int wid = tid >> 5;
    a = warp_sum(a);
    b = warp_sum(b);
    if (lane == 0) { sa[wid] = a; sb[wid] = b; }
    __syncthreads();
    float va = (tid < ((blockDim.x + 31) >> 5)) ? sa[lane] : 0.0f;
    float vb = (tid < ((blockDim.x + 31) >> 5)) ? sb[lane] : 0.0f;
    if (wid == 0) {
        va = warp_sum(va);
        vb = warp_sum(vb);
    }
    __syncthreads();
    a = va; b = vb;
}

__global__ void conv3_gn_stats_cpg8_kernel(const float* __restrict__ x,
                                           const float* __restrict__ w,
                                           float* __restrict__ y,
                                           float* __restrict__ sum,
                                           float* __restrict__ ssq,
                                           int N, int Cin, int T, int Co, int Tout, int G) {
    const int t = blockIdx.x * blockDim.x + threadIdx.x;
    const int g = blockIdx.y;
    const int n = blockIdx.z;
    const int c0 = g * 8;

    float a0=0.f, a1=0.f, a2=0.f, a3=0.f, a4=0.f, a5=0.f, a6=0.f, a7=0.f;
    if (t < Tout) {
        const float* xb = x + ((size_t)n * Cin) * T;
        #pragma unroll 1
        for (int ci = 0; ci < Cin; ++ci) {
            const float xm = (t > 0)     ? xb[(size_t)ci * T + (t - 1)] : 0.0f;
            const float xc =              xb[(size_t)ci * T + t];
            const float xp = (t + 1 < T) ? xb[(size_t)ci * T + (t + 1)] : 0.0f;
            const float* w0 = w + ((size_t)(c0 + 0) * Cin + ci) * 3;
            const float* w1 = w + ((size_t)(c0 + 1) * Cin + ci) * 3;
            const float* w2 = w + ((size_t)(c0 + 2) * Cin + ci) * 3;
            const float* w3 = w + ((size_t)(c0 + 3) * Cin + ci) * 3;
            const float* w4 = w + ((size_t)(c0 + 4) * Cin + ci) * 3;
            const float* w5 = w + ((size_t)(c0 + 5) * Cin + ci) * 3;
            const float* w6 = w + ((size_t)(c0 + 6) * Cin + ci) * 3;
            const float* w7 = w + ((size_t)(c0 + 7) * Cin + ci) * 3;
            a0 = fmaf(xm, w0[0], fmaf(xc, w0[1], fmaf(xp, w0[2], a0)));
            a1 = fmaf(xm, w1[0], fmaf(xc, w1[1], fmaf(xp, w1[2], a1)));
            a2 = fmaf(xm, w2[0], fmaf(xc, w2[1], fmaf(xp, w2[2], a2)));
            a3 = fmaf(xm, w3[0], fmaf(xc, w3[1], fmaf(xp, w3[2], a3)));

        yb[(size_t)0 * Tout] = a0; yb[(size_t)1 * Tout] = a1;
        yb[(size_t)2 * Tout] = a2; yb[(size_t)3 * Tout] = a3;
        yb[(size_t)4 * Tout] = a4; yb[(size_t)5 * Tout] = a5;
        yb[(size_t)6 * Tout] = a6; yb[(size_t)7 * Tout] = a7;

__global__ void conv_generic_stats_kernel(const float* __restrict__ x,
                                          const float* __restrict__ w,
                                          float* __restrict__ y,
                                          float* __restrict__ sum,
                                          float* __restrict__ ssq,
                                          int N, int Cin, int T, int Co, int K,
                                          int stride, int pad, int dil, int Tout,
                                          int G, int cpg) {
    const int t = blockIdx.x * blockDim.x + threadIdx.x;
    const int g = blockIdx.y;
    const int n = blockIdx.z;
    const int cstart = g * cpg;
    float acc[16];
    #pragma unroll
    for (int i=0;i<16;i++) acc[i]=0.f;
    if (t < Tout) {
        const float* xb = x + ((size_t)n * Cin) * T;
        for (int ci=0; ci<Cin; ++ci) {
            for (int kk=0; kk<K; ++kk) {
                int it = t * stride + kk * dil - pad;
                if ((unsigned)it < (unsigned)T) {
                    float xv = xb[(size_t)ci * T + it];
                    for (int c=0; c<cpg; ++c) {
                        acc[c] = fmaf(xv, w[((size_t)(cstart+c) * Cin + ci) * K + kk], acc[c]);
                    }
                }
            }
        }
        float* yb = y + ((size_t)n * Co) * Tout + (size_t)cstart * Tout + t;
        for (int c=0; c<cpg; ++c) yb[(size_t)c * Tout] = acc[c];
    }
    float s=0.f, q=0.f;
    if (t < Tout) {
        for (int c=0; c<cpg; ++c) { s += acc[c]; q += acc[c]*acc[c]; }
    }
    block_sum2(s, q);
    if (threadIdx.x == 0) {
        atomicAdd(sum + (size_t)n * G + g, s);
        atomicAdd(ssq + (size_t)n * G + g, q);
    }
}

__global__ void norm_cpg8_kernel(float* __restrict__ y,
                                 const float* __restrict__ sum,
                                 const float* __restrict__ ssq,
                                 const float* __restrict__ gamma,
                                 const float* __restrict__ beta,
                                 int N, int Co, int Tout, int G, float eps,
                                 int affine) {
    const int t = blockIdx.x * blockDim.x + threadIdx.x;
    const int g = blockIdx.y;
    const int n = blockIdx.z;
    if (t >= Tout) return;
    const int c0 = g * 8;
    const float m_inv = 1.0f / (8.0f * (float)Tout);
    float mean = sum[(size_t)n * G + g] * m_inv;
    float var = ssq[(size_t)n * G + g] * m_inv - mean * mean;
    var = var > 0.f ? var : 0.f;
    float inv = rsqrtf(var + eps);
    float* yb = y + ((size_t)n * Co) * Tout + (size_t)c0 * Tout + t;
    #pragma unroll
    for (int c=0; c<8; ++c) {
        float v = (yb[(size_t)c * Tout] - mean) * inv;
        if (affine) v = v * gamma[c0 + c] + beta[c0 + c];
        yb[(size_t)c * Tout] = v;
    }
}

__global__ void norm_generic_kernel(float* __restrict__ y,
                                    const float* __restrict__ sum,
                                    const float* __restrict__ ssq,
                                    const float* __restrict__ gamma,
                                    const float* __restrict__ beta,
                                    int N, int Co, int Tout, int G, int cpg,
                                    float eps, int affine) {
    const int t = blockIdx.x * blockDim.x + threadIdx.x;
    const int g = blockIdx.y;
    const int n = blockIdx.z;
    if (t >= Tout) return;
    const int c0 = g * cpg;
    const float m_inv = 1.0f / ((float)cpg * (float)Tout);
    float mean = sum[(size_t)n * G + g] * m_inv;
    float var = ssq[(size_t)n * G + g] * m_inv - mean * mean;
    var = var > 0.f ? var : 0.f;
    float inv = rsqrtf(var + eps);
    float* yb = y + ((size_t)n * Co) * Tout + (size_t)c0 * Tout + t;
    for (int c=0; c<cpg; ++c) {
        float v = (yb[(size_t)c * Tout] - mean) * inv;
        if (affine) v = v * gamma[c0 + c] + beta[c0 + c];
        yb[(size_t)c * Tout] = v;
    }
}

torch::Tensor conv_gn_forward(torch::Tensor x, torch::Tensor weight,
                              torch::Tensor gamma, torch::Tensor beta,
                              int64_t num_groups, int64_t stride,
                              int64_t padding, int64_t dilation, double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    const int N = (int)x.size(0);
    const int Cin = (int)x.size(1);
    const int T = (int)x.size(2);
    const int Co = (int)weight.size(0);
    const int K = (int)weight.size(2);
    const int G = (int)num_groups;
    const int cpg = Co / G;
    const int Tout = (T + 2*(int)padding - (int)dilation * (K - 1) - 1) / (int)stride + 1;
    auto y = torch::empty({N, Co, Tout}, x.options());
    auto stats = torch::empty({2, N, G}, x.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cudaMemsetAsync(stats.data_ptr<float>(), 0, (size_t)2 * N * G * sizeof(float), stream);
    float* sum = stats.data_ptr<float>();
    float* ssq = sum + (size_t)N * G;
    dim3 block(128);
    dim3 grid((Tout + 127) / 128, G, N);
    int affine = (gamma.numel() != 0 && beta.numel() != 0) ? 1 : 0;
    if (K == 3 && stride == 1 && padding == 1 && dilation == 1 && cpg == 8 && Tout == T) {
        conv3_gn_stats_cpg8_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), y.data_ptr<float>(),
            sum, ssq, N, Cin, T, Co, Tout, G);
        norm_cpg8_kernel<<<grid, block, 0, stream>>>(
            y.data_ptr<float>(), sum, ssq,
            affine ? gamma.data_ptr<float>() : nullptr,
            affine ? beta.data_ptr<float>() : nullptr,
            N, Co, Tout, G, (float)eps, affine);
    } else {
        TORCH_CHECK(cpg <= 16, "generic path supports at most 16 channels per group");
        conv_generic_stats_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), y.data_ptr<float>(),
            sum, ssq, N, Cin, T, Co, K, (int)stride, (int)padding, (int)dilation, Tout, G, cpg);
        norm_generic_kernel<<<grid, block, 0, stream>>>(
            y.data_ptr<float>(), sum, ssq,
            affine ? gamma.data_ptr<float>() : nullptr,
            affine ? beta.data_ptr<float>() : nullptr,
            N, Co, Tout, G, cpg, (float)eps, affine);
    }
    return y;
}

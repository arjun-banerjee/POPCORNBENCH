import torch
import torch.nn as nn


class Model(nn.Module):
    """
    Compute all-pairs Euclidean distance matrix from a 3-D point cloud.
    A core primitive in molecular dynamics simulations, docking, and
    distance-geometry-based structure prediction.
    """

    def __init__(self):
        super().__init__()

    def forward(self, coords: torch.Tensor) -> torch.Tensor:
        """
        Args:
            coords: (B, N, 3) – 3-D coordinates of N atoms / residues
        Returns:
            dist: (B, N, N) – pairwise Euclidean distance matrix
        """
        diff = coords.unsqueeze(2) - coords.unsqueeze(1)  # (B, N, N, 3)
        dist = torch.sqrt((diff ** 2).sum(dim=-1) + 1e-8)
        return dist

import torch.nn as nn
from torch.utils.cpp_extension import load_inline


_cpp_src = r"""
#include <torch/extension.h>

// Host wrapper implemented in the CUDA translation unit.
torch::Tensor pairdist_forward(torch::Tensor coords);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#ifndef TILE
#define TILE 16
#endif

__global__ void pairdist_kernel(const float* __restrict__ coords,
                                float* __restrict__ out,
                                const int N) {
    __shared__ float si[TILE][3];
    __shared__ float sj[TILE][3];

    const int b = blockIdx.z;
    const int i = blockIdx.y * TILE + threadIdx.y;
    const int j = blockIdx.x * TILE + threadIdx.x;

    const int lane_x = threadIdx.x;
    const int lane_y = threadIdx.y;
    const int base = b * N * 3;

    // Cooperative load of the two coordinate tiles.  Only the first three
    // columns of threads are used, avoiding redundant global loads for the
    // 16x16 output tile.
    if (lane_x < 3) {
        const int ii = blockIdx.y * TILE + lane_y;
        si[lane_y][lane_x] = (ii < N) ? coords[base + ii * 3 + lane_x] : 0.0f;
    }
    if (lane_y < 3) {
        const int jj = blockIdx.x * TILE + lane_x;
        sj[lane_x][lane_y] = (jj < N) ? coords[base + jj * 3 + lane_y] : 0.0f;
    }
    __syncthreads();

    if (i < N && j < N) {
        const float dx = si[lane_y][0] - sj[lane_x][0];
        const float dy = si[lane_y][1] - sj[lane_x][1];
        const float dz = si[lane_y][2] - sj[lane_x][2];
        const float d2 = fmaf(dx, dx, fmaf(dy, dy, dz * dz)) + 1.0e-8f;
        out[(b * N + i) * N + j] = sqrtf(d2);
    }
}

torch::Tensor pairdist_forward(torch::Tensor coords) {
    const int64_t B64 = coords.size(0);
    const int64_t N64 = coords.size(1);
    auto out = torch::empty({B64, N64, N64}, coords.options());

}

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Dilated convolution tower with exponentially increasing dilation rates and residual connections.
    Each block applies LayerNorm -> GELU -> DilatedConv1d, then adds back the original input as a residual.
    This architecture is a core building block of genomic foundation models (Enformer, Basenji, Caduceus)
    for capturing multi-scale sequence dependencies in DNA/RNA.
    """
    def __init__(self, channels, seq_len, kernel_size, max_dilation_power):
        super(Model, self).__init__()
        self.norms = nn.ModuleList()
        self.convs = nn.ModuleList()
        self.activation = nn.GELU()
        for i in range(max_dilation_power + 1):
            dilation = 2 ** i
            self.norms.append(nn.LayerNorm([channels, seq_len]))
            self.convs.append(nn.Conv1d(channels, channels, kernel_size, padding=dilation, dilation=dilation))

    def forward(self, x):
        residual = x
        for norm, conv in zip(self.norms, self.convs):
            x = norm(x)
            x = self.activation(x)
            x = conv(x)
            x = x + residual
        return x

#define CUBLAS_CHECK(expr) do { cublasStatus_t _s = (expr); if (_s != CUBLAS_STATUS_SUCCESS) { TORCH_CHECK(false, "cuBLAS error: ", (int)_s); } } while(0)

__global__ void pack_w_kernel(const float* __restrict__ w, float* __restrict__ pack, int C) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = 3 * C * C;
    for (; idx < total; idx += blockDim.x * gridDim.x) {
        int k = idx / (C*C);
        int rem = idx - k*C*C;
        int ic = rem / C;
        int oc = rem - ic*C;
        pack[idx] = w[(oc*C + ic)*3 + k];
    }
}

__global__ void ln_gelu_kernel(const float* __restrict__ x,
                               const float* __restrict__ gamma,
                               const float* __restrict__ beta,
                               float* __restrict__ out,
                               int B, int C, int L, int layout_nlc) {
    extern __shared__ float sh[];
    float* ss = sh;
    float* sq = sh + blockDim.x;
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int N = C * L;
    const float* xb = x + (long long)b * N;

    float sum = 0.0f;
    float sumsq = 0.0f;
    if (!layout_nlc) {
        for (int i = tid; i < N; i += blockDim.x) {
            float v = xb[i];
            sum += v;
            sumsq += v * v;
        }
    } else {
        for (int i = tid; i < N; i += blockDim.x) {
            float v = xb[i];
            sum += v;
            sumsq += v * v;
        }
    }
    ss[tid] = sum;
    sq[tid] = sumsq;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (tid < s) {
            ss[tid] += ss[tid + s];
            sq[tid] += sq[tid + s];
        }
        __syncthreads();
    }
    float mean = ss[0] / (float)N;
    float var = sq[0] / (float)N - mean * mean;
    var = fmaxf(var, 0.0f);
    float inv = rsqrtf(var + 1.0e-5f);
    float* ob = out + (long long)b * N; // always NLC
    const float inv_sqrt2 = 0.70710678118654752440f;

    if (!layout_nlc) {
        // input is NCL, gamma/beta are NCL; output NLC
        for (int i = tid; i < N; i += blockDim.x) {
            int c = i / L;
            int t = i - c * L;
            float y = (xb[i] - mean) * inv;
            y = y * gamma[i] + beta[i];
            float g = 0.5f * y * (1.0f + erff(y * inv_sqrt2));
            ob[t * C + c] = g;
        }
    } else {
        // input/output are NLC; gamma/beta are NCL
        for (int i = tid; i < N; i += blockDim.x) {
            int t = i / C;
            int c = i - t * C;
            int gi = c * L + t;
            float y = (xb[i] - mean) * inv;
            y = y * gamma[gi] + beta[gi];
            float g = 0.5f * y * (1.0f + erff(y * inv_sqrt2));
            ob[i] = g;
        }
    }
}

__global__ void init_out_kernel(const float* __restrict__ residual_ncl,
                                const float* __restrict__ bias,
                                float* __restrict__ out_nlc,
                                int B, int C, int L) {
    long long N = (long long)B * C * L;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (; idx < N; idx += stride) {
        int c = (idx / L) % C;
        int t = idx % L;
        int b = idx / ((long long)C * L);
        out_nlc[((long long)b * L + t) * C + c] = residual_ncl[idx] + bias[c];
    }
}

__global__ void nlc_to_ncl_kernel(const float* __restrict__ in,
                                  float* __restrict__ out,
                                  int B, int C, int L) {
    long long N = (long long)B * C * L;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (; idx < N; idx += stride) {
        int c = (idx / L) % C;
        int t = idx % L;
        int b = idx / ((long long)C * L);
        out[idx] = in[((long long)b * L + t) * C + c];
    }
}

static inline void rowmajor_gemm_batched(cublasHandle_t handle,
                                         const float* Bmat, const float* Amat, float* Cmat,
                                         int rows, int ch, long long strideA, long long strideC,
                                         float beta) {
    const float alpha = 1.0f;
    // Row-major: C[rows,ch] += A[rows,ch] * B[ch,ch].  Use column-major identity C^T = B^T * A^T.
    CUBLAS_CHECK(cublasSgemmStridedBatched(handle,
                                           CUBLAS_OP_N, CUBLAS_OP_N,
                                           ch, rows, ch,
                                           &alpha,
                                           Bmat, ch, 0,
                                           Amat, ch, strideA,
                                           &beta,
                                           Cmat, ch, strideC,
                                           32));
}

torch::Tensor tower_forward(torch::Tensor x,
                            std::vector<torch::Tensor> norm_w,
                            std::vector<torch::Tensor> norm_b,
                            std::vector<torch::Tensor> conv_w,
                            std::vector<torch::Tensor> conv_b) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only fp32 supported");
    c10::cuda::CUDAGuard device_guard(x.device());
    x = x.contiguous();
    int B = (int)x.size(0);
    int C = (int)x.size(1);
    int L = (int)x.size(2);
    int layers = (int)conv_w.size();
    TORCH_CHECK(C == 256, "optimized kernel expects 256 channels");

    auto opts = x.options();
    auto tmp = torch::empty({B, L, C}, opts);      // normalized+GELU, NLC
    auto cur = torch::empty({B, L, C}, opts);      // conv output, NLC
    auto pack = torch::empty({3, C, C}, opts);     // [k, ic, oc]

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    CUBLAS_CHECK(cublasSetStream(handle, stream));
    CUBLAS_CHECK(cublasSetMathMode(handle, CUBLAS_DEFAULT_MATH));

    const int threads = 256;
    long long total = (long long)B * C * L;
    int grid_big = (int)std::min<long long>((total + threads - 1) / threads, 2147483647LL);
    long long strideMat = (long long)L * C;

    for (int i = 0; i < layers; ++i) {
        const float* inp = (i == 0) ? x.data_ptr<float>() : cur.data_ptr<float>();
        int layout_nlc = (i == 0) ? 0 : 1;
        ln_gelu_kernel<<<B, threads, 2 * threads * sizeof(float), stream>>>(
            inp,
            norm_w[i].contiguous().data_ptr<float>(),
            norm_b[i].contiguous().data_ptr<float>(),
            tmp.data_ptr<float>(), B, C, L, layout_nlc);

        init_out_kernel<<<grid_big, threads, 0, stream>>>(
            x.data_ptr<float>(), conv_b[i].contiguous().data_ptr<float>(), cur.data_ptr<float>(), B, C, L);

        pack_w_kernel<<<(3*C*C + threads - 1) / threads, threads, 0, stream>>>(
            conv_w[i].contiguous().data_ptr<float>(), pack.data_ptr<float>(), C);

        int d = 1 << i;
        float beta = 1.0f;
        // k=0: input t-d contributes to output t, valid output t in [d, L)
        int rows = L - d;
        if (rows > 0) {
            rowmajor_gemm_batched(handle,
                                  pack.data_ptr<float>() + 0 * C * C,
                                  tmp.data_ptr<float>(),
                                  cur.data_ptr<float>() + (long long)d * C,
                                  rows, C, strideMat, strideMat, beta);
        }
        // k=1: center
        rowmajor_gemm_batched(handle,
                              pack.data_ptr<float>() + 1 * C * C,
                              tmp.data_ptr<float>(),
                              cur.data_ptr<float>(),
                              L, C, strideMat, strideMat, beta);
        // k=2: input t+d contributes to output t, valid output t in [0, L-d)
        if (rows > 0) {
            rowmajor_gemm_batched(handle,
                                  pack.data_ptr<float>() + 2 * C * C,
                                  tmp.data_ptr<float>() + (long long)d * C,
                                  cur.data_ptr<float>(),
                                  rows, C, strideMat, strideMat, beta);
        }
    }

    auto out = torch::empty({B, C, L}, opts);
    nlc_to_ncl_kernel<<<grid_big, threads, 0, stream>>>(cur.data_ptr<float>(), out.data_ptr<float>(), B, C, L);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return out;
}

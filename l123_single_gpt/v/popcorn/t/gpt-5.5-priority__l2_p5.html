import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Pre-norm 1D dilated convolution block: LayerNorm, GELU activation, dilated Conv1d, and residual addition.
    Common in genomic sequence models (Enformer, Basenji) for capturing long-range dependencies in DNA/RNA sequences.
    """
    def __init__(self, channels, seq_len, kernel_size, dilation):
        super(Model, self).__init__()
        self.norm = nn.LayerNorm([channels, seq_len])
        self.activation = nn.GELU()
        self.conv = nn.Conv1d(channels, channels, kernel_size, padding=dilation, dilation=dilation)

    def forward(self, x):
        residual = x
        x = self.norm(x)
        x = self.activation(x)
        x = self.conv(x)
        x = x + residual
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

static constexpr float LN_EPS = 1.0e-5f;
static constexpr float INV_SQRT2 = 0.70710678118654752440f;

__global__ void partial_stats_kernel(const float* __restrict__ x,
                                     float* __restrict__ partial_sum,
                                     float* __restrict__ partial_sumsq,
                                     int64_t N, int chunks) {
    int chunk = blockIdx.x;
    int b = blockIdx.y;
    int tid = threadIdx.x;
    int64_t chunk_size = (N + chunks - 1) / chunks;
    int64_t start = (int64_t)chunk * chunk_size;
    int64_t end = start + chunk_size;
    if (end > N) end = N;
    const float* xb = x + (int64_t)b * N;

    float s = 0.0f;
    float ss = 0.0f;
    for (int64_t i = start + tid; i < end; i += blockDim.x) {
        float v = xb[i];
        s += v;
        ss += v * v;
    }

    __shared__ float sh_s[256];
    __shared__ float sh_ss[256];
    sh_s[tid] = s;
    sh_ss[tid] = ss;
    __syncthreads();

    for (int offset = 128; offset > 0; offset >>= 1) {
        if (tid < offset) {
            sh_s[tid] += sh_s[tid + offset];
            sh_ss[tid] += sh_ss[tid + offset];
        }
        __syncthreads();
    }
    if (tid == 0) {
        int idx = b * chunks + chunk;
        partial_sum[idx] = sh_s[0];
        partial_sumsq[idx] = sh_ss[0];
    }
}

__global__ void final_stats_kernel(const float* __restrict__ partial_sum,
                                   const float* __restrict__ partial_sumsq,
                                   float* __restrict__ mean,
                                   float* __restrict__ rstd,
                                   int64_t N, int chunks) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    float s = 0.0f;
    float ss = 0.0f;
    const int base = b * chunks;
    for (int i = tid; i < chunks; i += blockDim.x) {
        s += partial_sum[base + i];
        ss += partial_sumsq[base + i];
    }

    __shared__ float sh_s[256];
    __shared__ float sh_ss[256];
    sh_s[tid] = s;
    sh_ss[tid] = ss;
    __syncthreads();
    for (int offset = 128; offset > 0; offset >>= 1) {
        if (tid < offset) {
            sh_s[tid] += sh_s[tid + offset];
            sh_ss[tid] += sh_ss[tid + offset];
        }
        __syncthreads();
    }
    if (tid == 0) {
        float m = sh_s[0] / (float)N;
        float var = sh_ss[0] / (float)N - m * m;
        var = var > 0.0f ? var : 0.0f;
        mean[b] = m;
        rstd[b] = rsqrtf(var + LN_EPS);
    }
}

__global__ void norm_gelu_kernel(const float* __restrict__ x,
                                 const float* __restrict__ gamma,
                                 const float* __restrict__ beta,
                                 const float* __restrict__ mean,
                                 const float* __restrict__ rstd,
                                 float* __restrict__ y,
                                 int64_t total, int64_t N) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        int b = (int)(idx / N);
        int64_t j = idx - (int64_t)b * N;
        float z = (x[idx] - mean[b]) * rstd[b];
        z = z * gamma[j] + beta[j];
        float g = 0.5f * z * (1.0f + erff(z * INV_SQRT2));
        y[idx] = g;
    }
}

__global__ void add_residual_kernel(float* __restrict__ y,
                                    const float* __restrict__ residual,
                                    int64_t total) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        y[idx] += residual[idx];
    }
}

void layernorm_gelu_launcher(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta,
                             torch::Tensor y, torch::Tensor partial_sum, torch::Tensor partial_sumsq,
                             torch::Tensor mean, torch::Tensor rstd, int64_t chunks64) {
    const int chunks = (int)chunks64;
    const int64_t B = x.size(0);
    const int64_t N = x.numel() / B;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    dim3 block(256);
    dim3 grid_stats(chunks, (unsigned int)B);
    partial_stats_kernel<<<grid_stats, block, 0, stream>>>(
        x.data_ptr<float>(), partial_sum.data_ptr<float>(), partial_sumsq.data_ptr<float>(), N, chunks);
    final_stats_kernel<<<(unsigned int)B, block, 0, stream>>>(
        partial_sum.data_ptr<float>(), partial_sumsq.data_ptr<float>(), mean.data_ptr<float>(), rstd.data_ptr<float>(), N, chunks);
    int64_t total = x.numel();
    int blocks = (int)((total + 255) / 256);
    if (blocks > 65535) blocks = 65535;
    norm_gelu_kernel<<<blocks, block, 0, stream>>>(
        x.data_ptr<float>(), gamma.data_ptr<float>(), beta.data_ptr<float>(), mean.data_ptr<float>(), rstd.data_ptr<float>(),
        y.data_ptr<float>(), total, N);
}

void add_residual_launcher(torch::Tensor y, torch::Tensor residual) {
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    int64_t total = y.numel();

}

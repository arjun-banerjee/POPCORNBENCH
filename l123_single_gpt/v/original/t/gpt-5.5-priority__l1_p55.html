                                      int N, int C, int H, int Wd, int Kout, int KH, int KW,

                                      long total) {

    int oh = t % OH;
    t /= OH;
    int oc = t % Kout;
    int n = t / Kout;

    int kpg = Kout / groups;
    int g = oc / kpg;
    int c0 = g * cpg;
    float sum = has_bias ? __ldg(bias + oc) : 0.0f;
    for (int icg = 0; icg < cpg; ++icg) {
        int ic = c0 + icg;

            }

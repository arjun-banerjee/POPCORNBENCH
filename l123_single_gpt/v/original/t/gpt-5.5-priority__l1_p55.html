// Generic scalar fallback for unusual grouped convolutions or pre-Ampere compilation.

                                     const float* __restrict__ bias,
                                     float* __restrict__ y,

                                     int pad_h, int pad_w,

    int Og = O / groups;

        long long t = idx / Wout;

        int oc = t % O;
        int n = t / O;
        int g = oc / Og;

                    }
                }
            }
        }
        y[idx] = acc;
    }
}

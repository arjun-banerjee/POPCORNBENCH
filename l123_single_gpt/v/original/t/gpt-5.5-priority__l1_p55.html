        kernel_size (int): Size of the square convolution kernel.
        stride (int, optional): Stride of the convolution. Defaults to 1.
        padding (int, optional): Padding applied to the input. Defaults to 0.
        dilation (int, optional): Spacing between kernel elements. Defaults to 1.
        groups (int, optional): Number of blocked connections from input channels to output channels. Defaults to 1.
        bias (bool, optional): If `True`, adds a learnable bias to the output. Defaults to `False`.
    """

        """
        Performs the 2D convolution.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, height, width).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, height_out, width_out).

        return self.conv2d(x)

#ifndef CHECK_CONTIGUOUS

// Four warps per CTA.  Each warp computes a 16 (NHW) x 16 (OC) output tile
// using TF32 tensor cores over the implicit im2col K dimension.
template<int WARPS_PER_BLOCK>
__global__ void conv2d_wmma_tf32_kernel(const float* __restrict__ x,
                                        const float* __restrict__ w,
                                        const float* __restrict__ bias,
                                        float* __restrict__ y,
                                        int N, int C, int H, int W,
                                        int O, int KH, int KW,
                                        int Hout, int Wout,
                                        int stride_h, int stride_w,
                                        int pad_h, int pad_w,
                                        int dil_h, int dil_w,
                                        int has_bias) {
#if __CUDA_ARCH__ >= 800
    constexpr int TM = 16;
    constexpr int TN = 16;
    constexpr int TK = 8;

    __shared__ float As[WARPS_PER_BLOCK * TM * TK];
    __shared__ float Bs[TK * TN];
    __shared__ float Cs[WARPS_PER_BLOCK * TM * TN];

    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp_id = tid >> 5;

    const long long Mtot = (long long)N * Hout * Wout;
    const int oc_base = blockIdx.y * TN;
    const long long m_base = ((long long)blockIdx.x * WARPS_PER_BLOCK + warp_id) * TM;
    const int Ktot = C * KH * KW;

    wmma::fragment<wmma::matrix_a, 16, 16, 8, wmma::precision::tf32, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, 16, 16, 8, wmma::precision::tf32, wmma::row_major> b_frag;
    wmma::fragment<wmma::accumulator, 16, 16, 8, float> c_frag;
    wmma::fill_fragment(c_frag, 0.0f);

    for (int k0 = 0; k0 < Ktot; k0 += TK) {
        // Load the B tile once per CTA: [8 x 16] = K x OC, row-major.
        for (int idx = tid; idx < TK * TN; idx += blockDim.x) {
            int kk = idx / TN;
            int jj = idx - kk * TN;
            int k = k0 + kk;
            int oc = oc_base + jj;
            float v = 0.0f;
            if (k < Ktot && oc < O) {
                int c = k / (KH * KW);
                int r = k - c * KH * KW;
                int kh = r / KW;
                int kw = r - kh * KW;
                v = w[((oc * C + c) * KH + kh) * KW + kw];
            }
            Bs[idx] = v;
        }

        // Each warp loads its own A tile: [16 x 8] = NHW x K, row-major.
        float* Ash = As + warp_id * TM * TK;
        for (int idx = lane; idx < TM * TK; idx += 32) {
            int ii = idx / TK;
            int kk = idx - ii * TK;
            long long m = m_base + ii;
            int k = k0 + kk;
            float v = 0.0f;
            if (m < Mtot && k < Ktot) {
                int ow = (int)(m % Wout);
                long long t = m / Wout;
                int oh = (int)(t % Hout);
                int n = (int)(t / Hout);
                int c = k / (KH * KW);
                int r = k - c * KH * KW;
                int kh = r / KW;
                int kw = r - kh * KW;
                int ih = oh * stride_h - pad_h + kh * dil_h;
                int iw = ow * stride_w - pad_w + kw * dil_w;
                if ((unsigned)ih < (unsigned)H && (unsigned)iw < (unsigned)W) {
                    v = x[((n * C + c) * H + ih) * W + iw];
                }
            }
            Ash[idx] = v;
        }
        __syncthreads();

        wmma::load_matrix_sync(a_frag, Ash, TK);
        wmma::load_matrix_sync(b_frag, Bs, TN);
        wmma::mma_sync(c_frag, a_frag, b_frag, c_frag);
        __syncthreads();
    }

    float* Csh = Cs + warp_id * TM * TN;
    wmma::store_matrix_sync(Csh, c_frag, TN, wmma::mem_row_major);

    for (int idx = lane; idx < TM * TN; idx += 32) {
        int ii = idx / TN;
        int jj = idx - ii * TN;
        long long m = m_base + ii;
        int oc = oc_base + jj;
        if (m < Mtot && oc < O) {
            int ow = (int)(m % Wout);
            long long t = m / Wout;
            int oh = (int)(t % Hout);
            int n = (int)(t / Hout);
            float v = Csh[idx];
            if (has_bias) v += bias[oc];
            y[((n * O + oc) * Hout + oh) * Wout + ow] = v;
        }
    }
#endif
}

// Generic scalar fallback for unusual grouped convolutions or pre-Ampere compilation.
__global__ void conv2d_scalar_kernel(const float* __restrict__ x,
                                     const float* __restrict__ w,
                                     const float* __restrict__ bias,
                                     float* __restrict__ y,
                                     int N, int C, int H, int W,
                                     int O, int KH, int KW,
                                     int Hout, int Wout,
                                     int stride_h, int stride_w,
                                     int pad_h, int pad_w,
                                     int dil_h, int dil_w,
                                     int groups, int has_bias) {
    long long total = (long long)N * O * Hout * Wout;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    int Cg = C / groups;
    int Og = O / groups;
    for (; idx < total; idx += (long long)blockDim.x * gridDim.x) {
        int ow = idx % Wout;
        long long t = idx / Wout;
        int oh = t % Hout;
        t /= Hout;
        int oc = t % O;
        int n = t / O;
        int g = oc / Og;
        float acc = has_bias ? bias[oc] : 0.0f;
        for (int cr = 0; cr < Cg; ++cr) {
            int c = g * Cg + cr;
            for (int kh = 0; kh < KH; ++kh) {
                int ih = oh * stride_h - pad_h + kh * dil_h;
                if ((unsigned)ih >= (unsigned)H) continue;
                for (int kw = 0; kw < KW; ++kw) {
                    int iw = ow * stride_w - pad_w + kw * dil_w;
                    if ((unsigned)iw < (unsigned)W) {
                        acc += x[((n*C + c)*H + ih)*W + iw] * w[((oc*Cg + cr)*KH + kh)*KW + kw];
                    }
                }
            }
        }
        y[idx] = acc;
    }
}

torch::Tensor conv2d_forward_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                                  int64_t stride_h, int64_t stride_w,
                                  int64_t pad_h, int64_t pad_w,
                                  int64_t dil_h, int64_t dil_w,
                                  int64_t groups) {
    CHECK_CUDA(x); CHECK_CUDA(weight);
    CHECK_CONTIGUOUS(x); CHECK_CONTIGUOUS(weight);
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 input is supported");
    TORCH_CHECK(weight.scalar_type() == torch::kFloat32, "only float32 weight is supported");

    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int O = (int)weight.size(0);
    const int Cg = (int)weight.size(1);
    const int KH = (int)weight.size(2);
    const int KW = (int)weight.size(3);
    const int Hout = (H + 2*(int)pad_h - (int)dil_h * (KH - 1) - 1) / (int)stride_h + 1;
    const int Wout = (W + 2*(int)pad_w - (int)dil_w * (KW - 1) - 1) / (int)stride_w + 1;

    auto y = torch::empty({N, O, Hout, Wout}, x.options());
    const int has_bias = (bias.defined() && bias.numel() == O) ? 1 : 0;
    const float* bptr = has_bias ? bias.data_ptr<float>() : nullptr;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    // Fast path for the benchmark case: dense NCHW fp32 convolution.  It also
    // handles arbitrary stride/padding/dilation for groups==1.
    if ((int)groups == 1 && Cg == C) {
        constexpr int WPB = 4;
        long long Mtot = (long long)N * Hout * Wout;
        dim3 block(32 * WPB);
        dim3 grid((unsigned)((Mtot + 16LL * WPB - 1) / (16LL * WPB)), (unsigned)((O + 15) / 16));
        conv2d_wmma_tf32_kernel<WPB><<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), bptr, y.data_ptr<float>(),
            N, C, H, W, O, KH, KW, Hout, Wout,
            (int)stride_h, (int)stride_w, (int)pad_h, (int)pad_w, (int)dil_h, (int)dil_w, has_bias);
    } else {
        int threads = 256;
        long long total = (long long)N * O * Hout * Wout;
        int blocks = (int)min(65535LL, (total + threads - 1) / threads);
        conv2d_scalar_kernel<<<blocks, threads, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), bptr, y.data_ptr<float>(),
            N, C, H, W, O, KH, KW, Hout, Wout,
            (int)stride_h, (int)stride_w, (int)pad_h, (int)pad_w, (int)dil_h, (int)dil_w,
            (int)groups, has_bias);
    }
    return y;
}

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__device__ __forceinline__ float gelu_fast_scalar(float x) {
    // Exact GELU is 0.5*x*(1+erf(x/sqrt(2))).  This is a high-accuracy
    // erf approximation (max abs error ~1.5e-7), easily within fp32 tolerance
    // while avoiding the slower libdevice erff path.
    const float inv_sqrt2 = 0.70710678118654752440f;
    float z = x * inv_sqrt2;
    float s = 1.0f;
    if (z < 0.0f) { s = -1.0f; z = -z; }
    const float p  = 0.3275911f;
    const float a1 = 0.254829592f;
    const float a2 = -0.284496736f;
    const float a3 = 1.421413741f;
    const float a4 = -1.453152027f;
    const float a5 = 1.061405429f;
    float t = 1.0f / (1.0f + p * z);
    float poly = (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t;
    float erfv = s * (1.0f - poly * __expf(-z * z));
    return 0.5f * x * (1.0f + erfv);
}

__global__ __launch_bounds__(256, 4) void gelu_float4_kernel(const float* __restrict__ x,
                                                            float* __restrict__ y,
                                                            int64_t n4) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(x);
    float4* __restrict__ y4 = reinterpret_cast<float4*>(y);
    for (int64_t i = idx; i < n4; i += stride) {
        float4 v = x4[i];
        v.x = gelu_fast_scalar(v.x);
        v.y = gelu_fast_scalar(v.y);
        v.z = gelu_fast_scalar(v.z);
        v.w = gelu_fast_scalar(v.w);
        y4[i] = v;
    }
}

__global__ __launch_bounds__(256, 4) void gelu_scalar_kernel(const float* __restrict__ x,
                                                            float* __restrict__ y,

    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;

}

}

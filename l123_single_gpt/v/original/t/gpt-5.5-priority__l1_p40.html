__global__ void partial_sum_kernel(const float* __restrict__ x,
                                   float* __restrict__ psum,
                                   float* __restrict__ psq,
                                   int64_t N, int chunks){

    const int b = blockIdx.y;
    const int tid = threadIdx.x;

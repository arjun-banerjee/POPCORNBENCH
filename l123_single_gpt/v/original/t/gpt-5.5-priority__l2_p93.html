import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a transposed convolution, adds a value, takes the minimum, applies GELU, and multiplies by a value.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, add_value, multiply_value):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride)
        self.add_value = add_value
        self.multiply_value = multiply_value

    def forward(self, x):
        x = self.conv_transpose(x)
        x = x + self.add_value
        x = torch.min(x, torch.tensor(0.0, device=x.device))
        x = torch.nn.functional.gelu(x)
        x = x * self.multiply_value
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

#define TILE_M 16
#define TILE_N 32
#define TILE_K 32

__device__ __forceinline__ float gelu_exact_f(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.70710678118654752440f));
}

// Specialized for N=128, Cin=64, Cout=128, H=W=64, kernel=4, stride=2, OH=OW=130.
// Computes one output-parity sub-problem per grid.z as a tiled GEMM with fused epilogue.
__global__ void deconv4s2_gelu_kernel(const float* __restrict__ x,
                                      const float* __restrict__ w,
                                      const float* __restrict__ b,
                                      float* __restrict__ y,
                                      float addv, float mulv) {
    __shared__ float As[TILE_M][TILE_K];
    __shared__ float Bs[TILE_K][TILE_N];

    const int tid = threadIdx.x; // 0..255
    const int lr = tid >> 4;     // 0..15
    const int lc = tid & 15;     // 0..15

    const int parity = blockIdx.z;
    const int ph = parity >> 1;
    const int pw = parity & 1;

    const int M = 128 * 65 * 65;
    const int row = blockIdx.x * TILE_M + lr;
    const int co0 = blockIdx.y * TILE_N + lc;
    const int co1 = co0 + 16;

    float acc0 = 0.0f;
    float acc1 = 0.0f;

    int n = 0, rr = 0, cc = 0;
    if (row < M) {
        int tmp = row;
        cc = tmp % 65;
        tmp /= 65;
        rr = tmp % 65;
        n = tmp / 65;
    }

    // K = Cin * 4 spatial contributors for a fixed parity.
    #pragma unroll
    for (int kt = 0; kt < 256; kt += TILE_K) {

        int aidx = tid;

    }

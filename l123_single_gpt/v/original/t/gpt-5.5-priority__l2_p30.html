import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a GEMM, applies Group Normalization, and then HardTanh.
    """
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.hardtanh = nn.Hardtanh(min_val=hardtanh_min, max_val=hardtanh_max)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.gemm(x)
        x = self.group_norm(x)
        x = self.hardtanh(x)
        return x

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void gn_ht_kernel(const float* __restrict__ x,
                             const float* __restrict__ gamma,
                             const float* __restrict__ beta,
                             float* __restrict__ y,
                             int rows, int cols, int group_size,
                             float ht_min, float ht_max, float eps) {
    int bg = blockIdx.x;
    int row = bg / gridDim.y;
    int g = bg - row * gridDim.y;
    int base = row * cols + g * group_size;
    int tid = threadIdx.x;

    float sum = 0.0f;
    float sumsq = 0.0f;
    // Designed for group_size=512, blockDim=256: two elements per thread.
    for (int i = tid; i < group_size; i += blockDim.x) {
        float v = x[base + i];
        sum += v;
        sumsq += v * v;
    }

    // warp reductions
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
        sumsq += __shfl_down_sync(0xffffffff, sumsq, offset);
    }

    __shared__ float sh_sum[8];
    __shared__ float sh_sumsq[8];
    int lane = tid & 31;
    int warp = tid >> 5;
    if (lane == 0) {
        sh_sum[warp] = sum;
        sh_sumsq[warp] = sumsq;
    }
    __syncthreads();

    if (warp == 0) {
        sum = (tid < (blockDim.x >> 5)) ? sh_sum[lane] : 0.0f;
        sumsq = (tid < (blockDim.x >> 5)) ? sh_sumsq[lane] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
            sumsq += __shfl_down_sync(0xffffffff, sumsq, offset);
        }
        if (tid == 0) {
            float mean = sum / (float)group_size;
            float var = sumsq / (float)group_size - mean * mean;
            if (var < 0.0f) var = 0.0f;
            sh_sum[0] = mean;
            sh_sumsq[0] = rsqrtf(var + eps);
        }
    }
    __syncthreads();

    float mean = sh_sum[0];
    float invstd = sh_sumsq[0];
    for (int i = tid; i < group_size; i += blockDim.x) {
        int c = g * group_size + i;
        float v = (x[base + i] - mean) * invstd;
        v = v * gamma[c] + beta[c];
        v = fminf(fmaxf(v, ht_min), ht_max);
        y[base + i] = v;
    }
}

torch::Tensor gn_ht_cuda(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, int64_t num_groups, double ht_min, double ht_max, double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    auto y = torch::empty_like(x);
    int rows = (int)x.size(0);
    int cols = (int)x.size(1);
    int groups = (int)num_groups;
    int group_size = cols / groups;
    dim3 block(256);
    dim3 grid(rows * groups);

    // kernel uses gridDim.y, so launch equivalent 2D when supported

}

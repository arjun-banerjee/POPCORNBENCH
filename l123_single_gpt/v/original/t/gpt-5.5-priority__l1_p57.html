                int iw = num_w / stride;
                if ((unsigned)iw >= (unsigned)W) continue;
                float xv = __ldg(x + ((n * IC + ic) * H + ih) * W + iw);
                float wv = __ldg(w + (((ic * ocpg + (oc - g * ocpg)) * K + kh) * K + kw));
                acc = fmaf(xv, wv, acc);
            }
        }
    }
    if (has_bias) acc += __ldg(bias + oc);
    y[idx] = acc;
}

torch::Tensor conv_transpose2d_forward(torch::Tensor x,
                                      torch::Tensor weight,
                                      torch::Tensor bias,
                                      int64_t stride,
                                      int64_t padding,
                                      int64_t output_padding,
                                      int64_t groups) {
    const at::cuda::CUDAGuard device_guard(x.device());
    const int N = (int)x.size(0);
    const int IC = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int K = (int)weight.size(2);
    const int OC = (int)weight.size(1) * (int)groups;
    const int OH = (H - 1) * (int)stride - 2 * (int)padding + K + (int)output_padding;
    const int OW = (W - 1) * (int)stride - 2 * (int)padding + K + (int)output_padding;
    auto y = torch::empty({N, OC, OH, OW}, x.options());
    const int has_bias = (bias.numel() > 0);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (K == 3 && stride == 1 && padding == 0 && output_padding == 0 && groups == 1) {
        dim3 block(32, 4, 1);
        dim3 grid((N * OH * OW + block.x - 1) / block.x, (OC + 8 * block.y - 1) / (8 * block.y), 1);
        deconv3x3_s1p0_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), has_bias ? bias.data_ptr<float>() : nullptr,
            y.data_ptr<float>(), N, IC, OC, H, W, OH, OW, has_bias);
    } else {
        int total = N * OC * OH * OW;
        int threads = 256;
        int blocks = (total + threads - 1) / threads;
        deconv_generic_kernel<<<blocks, threads, 0, stream>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), has_bias ? bias.data_ptr<float>() : nullptr,
            y.data_ptr<float>(), N, IC, OC, H, W, K, (int)stride, (int)padding, (int)groups, OH, OW, has_bias);
    }
    return y;
}

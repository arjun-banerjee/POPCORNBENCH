#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TH 4
#define TW 32
#define KH 5
#define KW 9
#define IC 32
#define OCT 16
#define INH (TH + KH - 1)
#define INW (TW + KW - 1)

__global__ __launch_bounds__(128, 2)
void conv_32_64_5x9_kernel(const float* __restrict__ x,
                           const float* __restrict__ w,
                           const float* __restrict__ b,
                           float* __restrict__ y,
                           int N, int H, int W, int OH, int OW,
                           int OC, bool has_bias) {
    const int tid = threadIdx.x;
    const int lx = tid & 31;       // 0..31
    const int ly = tid >> 5;       // 0..3
    const int ow0 = blockIdx.x * TW;
    const int oh0 = blockIdx.y * TH;
    const int oc_groups = OC / OCT;
    const int n = blockIdx.z / oc_groups;
    const int oc0 = (blockIdx.z - n * oc_groups) * OCT;

    __shared__ float xs[IC * INH * INW];
    __shared__ float ws[OCT * KH * KW];

    // Load the input patch for this (n, spatial tile).  The final W tile reads
    // a little past the image; those entries are zero and are only used by
    // threads whose outputs are out of bounds.
    const int total_x = IC * INH * INW;
    for (int idx = tid; idx < total_x; idx += 128) {
        int t = idx;
        const int ix = t % INW; t /= INW;
        const int iy = t % INH; t /= INH;
        const int c = t;
        const int ih = oh0 + iy;
        const int iw = ow0 + ix;
        float v = 0.0f;
        if (ih < H && iw < W) {
            v = x[((n * IC + c) * H + ih) * W + iw];
        }
        xs[idx] = v;
    }
    __syncthreads();

    float acc[OCT];
#pragma unroll
    for (int i = 0; i < OCT; ++i) acc[i] = 0.0f;

    const bool valid = (oh0 + ly < OH) && (ow0 + lx < OW);
    const int base_xy = ly * INW + lx;

#pragma unroll
    for (int c = 0; c < IC; ++c) {
        // Load this input-channel's 16x5x9 weights once per block into shared
        // memory.  This avoids the otherwise very expensive warp-uniform global
        // reloads of the same coefficients by all 128 threads.

            const int k = idx - oo * (KH * KW);
            ws[idx] = w[((oc0 + oo) * IC + c) * (KH * KW) + k];

        __syncthreads();

            const float* __restrict__ xp = xs + c * (INH * INW) + base_xy;
#pragma unroll

#pragma unroll

#pragma unroll

    }

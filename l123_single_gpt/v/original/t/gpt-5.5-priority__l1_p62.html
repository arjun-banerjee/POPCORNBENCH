#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

#define BW 64
#define OC_TILE 16
#define SW (BW + 9 - 1)

__global__ void conv_32_64_5x9_s1p0_kernel(const float* __restrict__ x,
                                            const float* __restrict__ w,
                                            const float* __restrict__ b,
                                            float* __restrict__ y,
                                            int N, int H, int W, int OC, int OH, int OW,
                                            int has_bias) {
    extern __shared__ float sh[]; // [32][5][72]
    const int tx = threadIdx.x;       // 0..31, spatial lane
    const int ty = threadIdx.y;       // output channel inside tile
    const int tid = ty * blockDim.x + tx;
    const int n_oh = blockIdx.y;
    const int n = n_oh / OH;
    const int oh = n_oh - n * OH;
    const int ow0 = blockIdx.x * BW;
    const int oc = blockIdx.z * OC_TILE + ty;

    // Load the input stripe needed by this output-row/spatial tile.  The same
    // input stripe is reused by 16 output channels, and each warp computes one
    // output channel for 64 adjacent output pixels (two pixels per lane).
    const int total = 32 * 5 * SW;
    for (int idx = tid; idx < total; idx += 512) {
        int tw = idx % SW;
        int t = idx / SW;
        int kh = t % 5;
        int c = t / 5;
        int ih = oh + kh;
        int iw = ow0 + tw;
        float v = 0.0f;
        if (iw >= 0 && iw < W) {
            v = x[((n * 32 + c) * H + ih) * W + iw];
        }
        sh[idx] = v;
    }
    __syncthreads();

    const int ow_a = ow0 + tx;
    const int ow_b = ow_a + 32;
    if (oc >= OC) return;

    float acc0 = has_bias ? b[oc] : 0.0f;
    float acc1 = acc0;

    for (int c = 0; c < 32; ++c) {
        #pragma unroll
        for (int kh = 0; kh < 5; ++kh) {
            const int sbase0 = (c * 5 + kh) * SW + tx;
            const int wbase = ((oc * 32 + c) * 5 + kh) * 9;
            #pragma unroll
            for (int kw = 0; kw < 9; ++kw) {
                const float ww = w[wbase + kw];
                acc0 = fmaf(sh[sbase0 + kw], ww, acc0);
                acc1 = fmaf(sh[sbase0 + 32 + kw], ww, acc1);
            }
        }
    }

    if (ow_a < OW) y[((n * OC + oc) * OH + oh) * OW + ow_a] = acc0;
    if (ow_b < OW) y[((n * OC + oc) * OH + oh) * OW + ow_b] = acc1;
}

__global__ void conv2d_generic_kernel(const float* __restrict__ x,
                                      const float* __restrict__ w,
                                      const float* __restrict__ b,
                                      float* __restrict__ y,
                                      int total, int N, int C, int H, int W,
                                      int OC, int KH, int KW, int OH, int OW,
                                      int stride_h, int stride_w, int pad_h, int pad_w,
                                      int dil_h, int dil_w, int groups, int has_bias) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int oc = t % OC;
    int n = t / OC;

    int cpg = C / groups;
    int opg = OC / groups;
    int g = oc / opg;
    int c0 = g * cpg;
    float acc = has_bias ? b[oc] : 0.0f;
    for (int ci = 0; ci < cpg; ++ci) {
        int c = c0 + ci;
        for (int kh = 0; kh < KH; ++kh) {
            int ih = oh * stride_h - pad_h + kh * dil_h;
            if ((unsigned)ih >= (unsigned)H) continue;
            for (int kw = 0; kw < KW; ++kw) {
                int iw = ow * stride_w - pad_w + kw * dil_w;
                if ((unsigned)iw < (unsigned)W) {
                    float xv = x[((n * C + c) * H + ih) * W + iw];
                    float wv = w[((oc * cpg + ci) * KH + kh) * KW + kw];
                    acc = fmaf(xv, wv, acc);
                }
            }
        }
    }
    y[idx] = acc;
}

torch::Tensor conv2d_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                          int stride_h, int stride_w, int pad_h, int pad_w,
                          int dil_h, int dil_w, int groups) {
    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int OC = (int)w.size(0);
    const int KH = (int)w.size(2);
    const int KW = (int)w.size(3);
    const int OH = (H + 2 * pad_h - dil_h * (KH - 1) - 1) / stride_h + 1;
    const int OW = (W + 2 * pad_w - dil_w * (KW - 1) - 1) / stride_w + 1;
    auto y = torch::empty({N, OC, OH, OW}, x.options());
    const int has_bias = (b.numel() != 0);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (x.scalar_type() == at::kFloat && w.scalar_type() == at::kFloat &&
        C == 32 && KH == 5 && KW == 9 && stride_h == 1 && stride_w == 1 &&
        pad_h == 0 && pad_w == 0 && dil_h == 1 && dil_w == 1 && groups == 1) {
        dim3 block(32, OC_TILE, 1);
        dim3 grid((OW + BW - 1) / BW, N * OH, (OC + OC_TILE - 1) / OC_TILE);
        size_t shmem = 32 * 5 * SW * sizeof(float);
        conv_32_64_5x9_s1p0_kernel<<<grid, block, shmem, stream>>>(
            x.data_ptr<float>(), w.data_ptr<float>(), has_bias ? b.data_ptr<float>() : nullptr,
            y.data_ptr<float>(), N, H, W, OC, OH, OW, has_bias);
    } else {
        int total = N * OC * OH * OW;
        int block = 256;
        int grid = (total + block - 1) / block;
        conv2d_generic_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), w.data_ptr<float>(), has_bias ? b.data_ptr<float>() : nullptr,
            y.data_ptr<float>(), total, N, C, H, W, OC, KH, KW, OH, OW,
            stride_h, stride_w, pad_h, pad_w, dil_h, dil_w, groups, has_bias);
    }
    return y;
}

class Model(nn.Module):
    """
    Simple model that performs a GEMM (general matrix multiplication), applies scaling, 
    and then batch normalization.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.bn = nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):

#define CHUNKS 8

__global__ void partial_stats_kernel(const float* __restrict__ y,
                                     const float* __restrict__ scale,
                                     float* __restrict__ partial,
                                     int N, int C) {
    int lane = threadIdx.x & 31;
    int col_base = blockIdx.x * 32;
    int c = col_base + lane;
    int chunk = blockIdx.y;
    if (c >= C) return;

    int start = (N * chunk) / CHUNKS;
    int end   = (N * (chunk + 1)) / CHUNKS;
    float s = scale[c];
    float sum = 0.0f;
    float sum2 = 0.0f;

    for (int n = start; n < end; ++n) {
        float z = y[(long long)n * C + c] * s;
        sum += z;
        sum2 += z * z;
    }
    partial[(long long)chunk * C + c] = sum;
    partial[(long long)(CHUNKS + chunk) * C + c] = sum2;
}

__global__ void finalize_stats_kernel(const float* __restrict__ partial,
                                      float* __restrict__ stats,
                                      float* __restrict__ running_mean,
                                      float* __restrict__ running_var,
                                      int N, int C, float momentum, float eps) {
    int lane = threadIdx.x & 31;
    int c = blockIdx.x * 32 + lane;
    if (c >= C) return;

    float sum = 0.0f;
    float sum2 = 0.0f;
    #pragma unroll
    for (int k = 0; k < CHUNKS; ++k) {
        sum  += partial[(long long)k * C + c];
        sum2 += partial[(long long)(CHUNKS + k) * C + c];
    }
    float invN = 1.0f / (float)N;
    float mean = sum * invN;
    float var = sum2 * invN - mean * mean;
    if (var < 0.0f) var = 0.0f;
    float invstd = rsqrtf(var + eps);

    stats[c] = mean;
    stats[(long long)C + c] = invstd;

    // PyTorch BatchNorm updates running_var with the unbiased batch variance.
    float unbiased = (N > 1) ? (var * ((float)N / (float)(N - 1))) : var;
    running_mean[c] = running_mean[c] * (1.0f - momentum) + mean * momentum;

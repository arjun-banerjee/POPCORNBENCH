import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a GEMM (general matrix multiplication), applies scaling, 
    and then batch normalization.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.bn = nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.gemm(x)
        x = x * self.scale
        x = self.bn(x)
        return x

// Implemented in the CUDA translation unit below.
torch::Tensor bn_post_forward(torch::Tensor y,
                              torch::Tensor scale,
                              torch::Tensor bn_weight,
                              torch::Tensor bn_bias,
                              torch::Tensor running_mean,
                              torch::Tensor running_var,
                              bool training,
                              double eps,
                              double momentum);

torch::Tensor forward_cuda(torch::Tensor input,
                           torch::Tensor weight,
                           torch::Tensor bias,
                           torch::Tensor scale,
                           torch::Tensor bn_weight,
                           torch::Tensor bn_bias,
                           torch::Tensor running_mean,
                           torch::Tensor running_var,
                           bool training,
                           double eps,
                           double momentum) {
    // Same GEMM path as nn.Linear/addmm (cuBLAS), then fuse the scale and BN work.
    auto y = at::addmm(bias, input, weight.t());
    return bn_post_forward(y, scale, bn_weight, bn_bias, running_mean, running_var,
                           training, eps, momentum);
}

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>

#ifndef THREADS
#define THREADS 256
#endif
#define ROW_TILE 16

__global__ void partial_stats_kernel(const float* __restrict__ y,
                                     const float* __restrict__ scale,
                                     float* __restrict__ partial_sum,
                                     float* __restrict__ partial_sumsq,
                                     int N, int C, int row_blocks) {
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    int rb = blockIdx.y;
    if (j >= C) return;
    int r0 = rb * ROW_TILE;
    int r1 = min(r0 + ROW_TILE, N);
    float s = scale[j];
    float sum = 0.0f;
    float ss = 0.0f;
    #pragma unroll
    for (int r = r0; r < r0 + ROW_TILE; ++r) {
        if (r < r1) {
            float v = y[(long long)r * C + j] * s;
            sum += v;
            ss += v * v;
        }
    }
    partial_sum[(long long)rb * C + j] = sum;
    partial_sumsq[(long long)rb * C + j] = ss;
}

__global__ void finalize_stats_kernel(const float* __restrict__ partial_sum,
                                      const float* __restrict__ partial_sumsq,
                                      float* __restrict__ mean,
                                      float* __restrict__ invstd,
                                      float* __restrict__ running_mean,
                                      float* __restrict__ running_var,
                                      int N, int C, int row_blocks,
                                      float eps, float momentum) {
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if (j >= C) return;
    float sum = 0.0f;
    float ss = 0.0f;
    for (int rb = 0; rb < row_blocks; ++rb) {
        long long idx = (long long)rb * C + j;
        sum += partial_sum[idx];
        ss += partial_sumsq[idx];
    }
    float invN = 1.0f / (float)N;
    float m = sum * invN;
    float var = ss * invN - m * m;       // biased variance, used for normalization
    if (var < 0.0f) var = 0.0f;
    mean[j] = m;
    invstd[j] = rsqrtf(var + eps);

    // PyTorch BatchNorm updates running_var with the unbiased variance estimate.
    float unb = (N > 1) ? (var * ((float)N / (float)(N - 1))) : var;
    running_mean[j] = running_mean[j] * (1.0f - momentum) + m * momentum;
    running_var[j]  = running_var[j]  * (1.0f - momentum) + unb * momentum;
}

__global__ void train_norm_kernel(float* __restrict__ y,
                                  const float* __restrict__ scale,
                                  const float* __restrict__ gamma,
                                  const float* __restrict__ beta,
                                  const float* __restrict__ mean,
                                  const float* __restrict__ invstd,
                                  long long total, int C) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    int j = (int)(idx % C);
    float v = y[idx] * scale[j];
    y[idx] = (v - mean[j]) * invstd[j] * gamma[j] + beta[j];
}

__global__ void eval_params_kernel(const float* __restrict__ scale,
                                   const float* __restrict__ gamma,
                                   const float* __restrict__ beta,
                                   const float* __restrict__ running_mean,
                                   const float* __restrict__ running_var,
                                   float* __restrict__ alpha,
                                   float* __restrict__ delta,
                                   int C, float eps) {
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if (j >= C) return;
    float ginv = gamma[j] * rsqrtf(running_var[j] + eps);
    alpha[j] = scale[j] * ginv;
    delta[j] = beta[j] - running_mean[j] * ginv;
}

__global__ void eval_apply_kernel(float* __restrict__ y,
                                  const float* __restrict__ alpha,
                                  const float* __restrict__ delta,
                                  long long total, int C) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;
    int j = (int)(idx % C);
    y[idx] = y[idx] * alpha[j] + delta[j];
}

torch::Tensor bn_post_forward(torch::Tensor y,
                              torch::Tensor scale,
                              torch::Tensor bn_weight,
                              torch::Tensor bn_bias,
                              torch::Tensor running_mean,
                              torch::Tensor running_var,
                              bool training,
                              double eps,
                              double momentum) {
    const at::cuda::CUDAGuard device_guard(y.device());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    int N = (int)y.size(0);
    int C = (int)y.size(1);
    long long total = (long long)N * C;
    dim3 block(THREADS);

    if (training) {
        int row_blocks = (N + ROW_TILE - 1) / ROW_TILE;
        auto partial_sum = at::empty({row_blocks, C}, y.options());
        auto partial_sumsq = at::empty({row_blocks, C}, y.options());
        auto mean = at::empty({C}, y.options());
        auto invstd = at::empty({C}, y.options());

        dim3 grid_stats((C + THREADS - 1) / THREADS, row_blocks);
        partial_stats_kernel<<<grid_stats, block, 0, stream>>>(
            y.data_ptr<float>(), scale.data_ptr<float>(),
            partial_sum.data_ptr<float>(), partial_sumsq.data_ptr<float>(),
            N, C, row_blocks);

        dim3 grid_c((C + THREADS - 1) / THREADS);
        finalize_stats_kernel<<<grid_c, block, 0, stream>>>(
            partial_sum.data_ptr<float>(), partial_sumsq.data_ptr<float>(),
            mean.data_ptr<float>(), invstd.data_ptr<float>(),
            running_mean.data_ptr<float>(), running_var.data_ptr<float>(),
            N, C, row_blocks, (float)eps, (float)momentum);

        dim3 grid_all((total + THREADS - 1) / THREADS);
        train_norm_kernel<<<grid_all, block, 0, stream>>>(
            y.data_ptr<float>(), scale.data_ptr<float>(), bn_weight.data_ptr<float>(),
            bn_bias.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(),
            total, C);
    } else {
        auto alpha = at::empty({C}, y.options());
        auto delta = at::empty({C}, y.options());
        dim3 grid_c((C + THREADS - 1) / THREADS);
        eval_params_kernel<<<grid_c, block, 0, stream>>>(
            scale.data_ptr<float>(), bn_weight.data_ptr<float>(), bn_bias.data_ptr<float>(),
            running_mean.data_ptr<float>(), running_var.data_ptr<float>(),
            alpha.data_ptr<float>(), delta.data_ptr<float>(), C, (float)eps);
        dim3 grid_all((total + THREADS - 1) / THREADS);
        eval_apply_kernel<<<grid_all, block, 0, stream>>>(
            y.data_ptr<float>(), alpha.data_ptr<float>(), delta.data_ptr<float>(), total, C);
    }
    return y;
}
'''

    """Drop-in replacement for the reference model using a fused CUDA post-GEMM path."""
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()

        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.bn = nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):

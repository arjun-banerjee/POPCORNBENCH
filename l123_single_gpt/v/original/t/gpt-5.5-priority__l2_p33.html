import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a GEMM (general matrix multiplication), applies scaling, 
    and then batch normalization.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.bn = nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.gemm(x)
        x = x * self.scale
        x = self.bn(x)
        return x

__global__ void partial_stats_kernel(const float* __restrict__ y,
                                     const float* __restrict__ scale,
                                     float* __restrict__ partial_sum,
                                     float* __restrict__ partial_sumsq,
                                     int N, int C, int row_blocks) {

    int rb = blockIdx.y;
    if (j >= C) return;

    int r1 = min(r0 + ROW_TILE, N);

            sum += v;
            ss += v * v;
        }
    }
    partial_sum[(long long)rb * C + j] = sum;
    partial_sumsq[(long long)rb * C + j] = ss;
}

__global__ void finalize_stats_kernel(const float* __restrict__ partial_sum,
                                      const float* __restrict__ partial_sumsq,
                                      float* __restrict__ mean,
                                      float* __restrict__ invstd,

                                      float* __restrict__ running_var,

    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if (j >= C) return;
    float sum = 0.0f;
    float ss = 0.0f;
    for (int rb = 0; rb < row_blocks; ++rb) {

        sum += partial_sum[idx];

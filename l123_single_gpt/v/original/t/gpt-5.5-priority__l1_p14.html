_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE_M 64
#define TILE_N 64
#define TILE_K 32

__global__ __launch_bounds__(256, 2)
void triumm64_kernel(const float* __restrict__ A,
                     const float* __restrict__ B,
                     float* __restrict__ C,
                     int n) {
    const int blockCol = blockIdx.x * TILE_N;
    const int blockRow = blockIdx.y * TILE_M;
    const int tx = threadIdx.x; // 0..15
    const int ty = threadIdx.y; // 0..15

    // Tiles strictly below the diagonal are exactly zero.  Write them here so
    // the returned empty tensor has the same lower triangle as torch.triu.
    if (blockCol < blockRow) {
        #pragma unroll
        for (int rr = 0; rr < 4; ++rr) {
            int row = blockRow + ty + rr * 16;
            if (row < n) {
                #pragma unroll
                for (int cc = 0; cc < 4; ++cc) {
                    int col = blockCol + tx + cc * 16;
                    if (col < n) C[(long long)row * n + col] = 0.0f;
                }
            }
        }
        return;
    }

    __shared__ float As[TILE_M][TILE_K + 1];
    __shared__ float Bs[TILE_K][TILE_N + 1];

    float acc[4][4];
    #pragma unroll
    for (int i = 0; i < 4; ++i) {
        #pragma unroll
        for (int j = 0; j < 4; ++j) acc[i][j] = 0.0f;
    }

    const int tid = ty * 16 + tx;
    int kEnd = blockCol + TILE_N;
    if (kEnd > n) kEnd = n;

    // Because A and B are upper triangular, only k in [blockRow, blockCol+63]
    // can contribute to this 64x64 output tile.  Values inside the boundary
    // tiles that are outside the true per-element range are already stored as
    // zeros in A or B.
    for (int kk = blockRow; kk < kEnd; kk += TILE_K) {
        for (int idx = tid; idx < TILE_M * TILE_K; idx += 256) {
            int r = idx / TILE_K;
            int c = idx - r * TILE_K;
            int ar = blockRow + r;
            int ac = kk + c;
            As[r][c] = (ar < n && ac < n) ? A[(long long)ar * n + ac] : 0.0f;
        }
        for (int idx = tid; idx < TILE_K * TILE_N; idx += 256) {
            int r = idx / TILE_N;
            int c = idx - r * TILE_N;
            int br = kk + r;
            int bc = blockCol + c;
            Bs[r][c] = (br < n && bc < n) ? B[(long long)br * n + bc] : 0.0f;

            acc[0][3] = fmaf(a0, b3, acc[0][3]);

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D transposed convolution, followed by two max pooling layers and a sum operation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool1 = nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = nn.MaxPool3d(kernel_size=3)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool1(x)
        x = self.max_pool2(x)
        x = torch.sum(x, dim=1, keepdim=True) 
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

// Specialized for the problem dimensions:
// x:     [B,32,32,32,32]
// wpack: [32,5,5,5,64]  (packed from ConvTranspose3d weight [32,64,5,5,5])
// bias:  [64]
// out:   [B,1,10,10,10]
// The two max-pools combine to a max over a 6x6x6 cube of the transposed-conv output.
__global__ void fused_deconv_pool_sum_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int B)
{
    extern __shared__ float smem[];
    float* sx = smem;              // 32*5*5*5 = 4000 floats
    float* red = smem + 4000;      // blockDim.x floats

    const int block = blockIdx.x;
    const int n = block / 1000;
    const int rem = block - n * 1000;
    const int oz2 = rem / 100;
    const int rem2 = rem - oz2 * 100;
    const int oy2 = rem2 / 10;
    const int ox2 = rem2 - oy2 * 10;

    // For a final pooled cell t, only input coordinates 3*t-1 ... 3*t+3 can contribute.
    const int base_z = oz2 * 3 - 1;
    const int base_y = oy2 * 3 - 1;
    const int base_x = ox2 * 3 - 1;

    // Cache the small input tile shared by all output channels for this pooled cell.
    for (int idx = threadIdx.x; idx < 4000; idx += blockDim.x) {
        int t = idx;
        const int lx = t % 5; t /= 5;
        const int ly = t % 5; t /= 5;
        const int lz = t % 5; t /= 5;
        const int ic = t;
        const int iz = base_z + lz;
        const int iy = base_y + ly;
        const int ix = base_x + lx;
        float v = 0.0f;
        if ((unsigned)iz < 32u && (unsigned)iy < 32u && (unsigned)ix < 32u) {
            size_t off = (((size_t)n * 32 + ic) * 32 + iz) * 32 * 32 + (size_t)iy * 32 + ix;
            v = x[off];
        }
        sx[idx] = v;
    }
    __syncthreads();

    float mysum = 0.0f;

    // One thread computes the pooled maximum for one output channel.
    if (threadIdx.x < 64) {
        const int oc = threadIdx.x;
        float m = -FLT_MAX;

        #pragma unroll
        for (int dz = 0; dz < 6; ++dz) {
            #pragma unroll
            for (int dy = 0; dy < 6; ++dy) {
                #pragma unroll
                for (int dx = 0; dx < 6; ++dx) {
                    float acc = bias[oc];

                    #pragma unroll
                    for (int ic = 0; ic < 32; ++ic) {
                        #pragma unroll
                        for (int lz = 0; lz < 5; ++lz) {
                            const int kd = dz + 4 - 2 * lz;
                            if ((unsigned)kd < 5u) {
                                #pragma unroll
                                for (int ly = 0; ly < 5; ++ly) {
                                    const int kh = dy + 4 - 2 * ly;
                                    if ((unsigned)kh < 5u) {
                                        #pragma unroll

                                            const int kw = dx + 4 - 2 * lx;
                                            if ((unsigned)kw < 5u) {
                                                const float xv = sx[((ic * 5 + lz) * 5 + ly) * 5 + lx];
                                                const size_t woff = (((((size_t)ic * 5 + kd) * 5 + kh) * 5 + kw) * 64 + oc);
                                                acc = fmaf(xv, w[woff], acc);
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                    m = fmaxf(m, acc);
                }
            }
        }
        mysum = m;
    }

    red[threadIdx.x] = mysum;
    __syncthreads();

    // Sum the 64 channel maxima.
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) red[threadIdx.x] += red[threadIdx.x + stride];
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        out[((size_t)n * 1000) + oz2 * 100 + oy2 * 10 + ox2] = red[0];
    }
}

class ModelNew(nn.Module):
    """Fused replacement for ConvTranspose3d -> MaxPool3d(2) -> MaxPool3d(3) -> channel sum."""
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size, stride=stride, padding=padding
        )
        # Kept for state/module compatibility with the reference (no parameters).
        self.max_pool1 = nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = nn.MaxPool3d(kernel_size=3)
        self._w_packed = None
        self._w_version = None

    def forward(self, x):
        w = self.conv_transpose.weight
        # Pack weights as [IC,KD,KH,KW,OC], so the 64 output-channel threads read coalesced weights.
        ver = w._version
        if (self._w_packed is None or self._w_version != ver or
                self._w_packed.device != w.device or self._w_packed.dtype != w.dtype):
            self._w_packed = w.detach().permute(0, 2, 3, 4, 1).contiguous()
            self._w_version = ver
        return _ext.fused_forward(x, self._w_packed, self.conv_transpose.bias)

import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        """
        :param in_channels: Number of input channels
        :param out_1x1: Number of output channels for the 1x1 convolution
        :param reduce_3x3: Number of output channels for the 1x1 reduction before 3x3 convolution
        :param out_3x3: Number of output channels for the 3x3 convolution
        :param reduce_5x5: Number of output channels for the 1x1 reduction before 5x5 convolution
        :param out_5x5: Number of output channels for the 5x5 convolution
        :param pool_proj: Number of output channels for the pooling projection
        """
        super(Model, self).__init__()
        
        # 1x1 convolution branch
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        
        # 3x3 convolution branch
        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        
        # 5x5 convolution branch
        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        
        # Max pooling branch
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )
    
    def forward(self, x):
        """
        :param x: Input tensor, shape (batch_size, in_channels, height, width)
        :return: Output tensor, shape (batch_size, out_channels, height, width)
        """
        branch1x1 = self.branch1x1(x)
        branch3x3 = self.branch3x3(x)
        branch5x5 = self.branch5x5(x)
        branch_pool = self.branch_pool(x)
        
        outputs = [branch1x1, branch3x3, branch5x5, branch_pool]
        return torch.cat(outputs, 1)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

#define TILE_M 16
#define TILE_O 16
#define TILE_K 32

__global__ void pointwise1x1_tiled_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int C, int OC, int HW,
    int out_total_c, int out_offset) {

  __shared__ float xs[TILE_K][TILE_M];
  __shared__ float ws[TILE_O][TILE_K];

  int tx = threadIdx.x; // spatial lane
  int ty = threadIdx.y; // output-channel lane
  int tid = ty * blockDim.x + tx;

  int m = blockIdx.x * TILE_M + tx;   // flattened N*HW
  int oc = blockIdx.y * TILE_O + ty;
  int n = m / HW;
  int s = m - n * HW;

  float acc = 0.0f;
  if (oc < OC) acc = b[oc];

  for (int k0 = 0; k0 < C; k0 += TILE_K) {
    for (int li = tid; li < TILE_K * TILE_M; li += TILE_M * TILE_O) {
      int kk = li / TILE_M;
      int sx = li - kk * TILE_M;
      int mm = blockIdx.x * TILE_M + sx;
      int nn = mm / HW;
      int ss = mm - nn * HW;
      int c = k0 + kk;
      float v = 0.0f;
      if (mm < N * HW && c < C) v = x[((nn * C + c) * HW) + ss];
      xs[kk][sx] = v;
    }
    for (int li = tid; li < TILE_O * TILE_K; li += TILE_M * TILE_O) {
      int oy = li / TILE_K;
      int kk = li - oy * TILE_K;
      int o = blockIdx.y * TILE_O + oy;
      int c = k0 + kk;
      float v = 0.0f;
      if (o < OC && c < C) v = w[o * C + c];
      ws[oy][kk] = v;
    }
    __syncthreads();

    if (m < N * HW && oc < OC) {
      #pragma unroll
      for (int kk = 0; kk < TILE_K; ++kk) {
        if (k0 + kk < C) acc = fmaf(xs[kk][tx], ws[ty][kk], acc);
      }
    }
    __syncthreads();
  }

  if (m < N * HW && oc < OC) {
    y[((n * out_total_c + (out_offset + oc)) * HW) + s] = acc;
  }
}

__global__ void maxpool3x3_kernel(const float* __restrict__ x, float* __restrict__ y,
                                  int N, int C, int H, int W) {
  int HW = H * W;
  long long total = (long long)N * C * HW;
  long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
  if (idx >= total) return;
  int s = idx % HW;
  int c = (idx / HW) % C;
  int n = idx / (HW * C);
  int h = s / W;
  int w0 = s - h * W;
  float mx = -FLT_MAX;
  for (int dy = -1; dy <= 1; ++dy) {
    int yy = h + dy;
    if ((unsigned)yy >= (unsigned)H) continue;
    for (int dx = -1; dx <= 1; ++dx) {
      int xx = w0 + dx;
      if ((unsigned)xx >= (unsigned)W) continue;
      float v = x[((n * C + c) * HW) + yy * W + xx];
      mx = fmaxf(mx, v);
    }
  }
  y[idx] = mx;
}

__global__ void conv_spatial_kernel(const float* __restrict__ x,
                                    const float* __restrict__ w,
                                    const float* __restrict__ b,
                                    float* __restrict__ y,
                                    int N, int IC, int OC, int H, int Wd,
                                    int K, int out_total_c, int out_offset) {
  int HW = H * Wd;
  long long total = (long long)N * OC * HW;
  long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
  if (idx >= total) return;
  int s = idx % HW;
  int oc = (idx / HW) % OC;
  int n = idx / (HW * OC);
  int h = s / Wd;
  int ww = s - h * Wd;
  int pad = K >> 1;
  float acc = b[oc];

  for (int ic = 0; ic < IC; ++ic) {
    for (int ky = 0; ky < K; ++ky) {
      int yy = h + ky - pad;
      if ((unsigned)yy >= (unsigned)H) continue;
      for (int kx = 0; kx < K; ++kx) {
        int xx = ww + kx - pad;
        if ((unsigned)xx >= (unsigned)Wd) continue;
        float xv = x[((n * IC + ic) * HW) + yy * Wd + xx];
        float wv = w[(((oc * IC + ic) * K + ky) * K) + kx];
        acc = fmaf(xv, wv, acc);
      }
    }
  }
  y[((n * out_total_c + (out_offset + oc)) * HW) + s] = acc;
}

torch::Tensor inception_forward_cuda(
    torch::Tensor x,
    torch::Tensor b1_w, torch::Tensor b1_b,
    torch::Tensor r3_w, torch::Tensor r3_b,
    torch::Tensor c3_w, torch::Tensor c3_b,
    torch::Tensor r5_w, torch::Tensor r5_b,
    torch::Tensor c5_w, torch::Tensor c5_b,
    torch::Tensor p_w, torch::Tensor p_b) {

  const int N = (int)x.size(0);
  const int C = (int)x.size(1);
  const int H = (int)x.size(2);
  const int Wd = (int)x.size(3);
  const int HW = H * Wd;
  const int O1 = (int)b1_w.size(0);
  const int R3 = (int)r3_w.size(0);
  const int O3 = (int)c3_w.size(0);
  const int R5 = (int)r5_w.size(0);
  const int O5 = (int)c5_w.size(0);
  const int OP = (int)p_w.size(0);
  const int OT = O1 + O3 + O5 + OP;

  auto opts = x.options();
  torch::Tensor out = torch::empty({N, OT, H, Wd}, opts);
  torch::Tensor red3 = torch::empty({N, R3, H, Wd}, opts);
  torch::Tensor red5 = torch::empty({N, R5, H, Wd}, opts);
  torch::Tensor pool = torch::empty({N, C, H, Wd}, opts);

  cudaStream_t stream = at::cuda::getCurrentCUDAStream();
  dim3 block2(TILE_M, TILE_O);
  dim3 grid_b1((N * HW + TILE_M - 1) / TILE_M, (O1 + TILE_O - 1) / TILE_O);
  pointwise1x1_tiled_kernel<<<grid_b1, block2, 0, stream>>>(
      x.data_ptr<float>(), b1_w.data_ptr<float>(), b1_b.data_ptr<float>(),
      out.data_ptr<float>(), N, C, O1, HW, OT, 0);

  dim3 grid_r3((N * HW + TILE_M - 1) / TILE_M, (R3 + TILE_O - 1) / TILE_O);
  pointwise1x1_tiled_kernel<<<grid_r3, block2, 0, stream>>>(
      x.data_ptr<float>(), r3_w.data_ptr<float>(), r3_b.data_ptr<float>(),
      red3.data_ptr<float>(), N, C, R3, HW, R3, 0);

  dim3 grid_r5((N * HW + TILE_M - 1) / TILE_M, (R5 + TILE_O - 1) / TILE_O);
  pointwise1x1_tiled_kernel<<<grid_r5, block2, 0, stream>>>(
      x.data_ptr<float>(), r5_w.data_ptr<float>(), r5_b.data_ptr<float>(),
      red5.data_ptr<float>(), N, C, R5, HW, R5, 0);

  int threads = 256;
  long long pool_total = (long long)N * C * HW;
  maxpool3x3_kernel<<<(pool_total + threads - 1) / threads, threads, 0, stream>>>(
      x.data_ptr<float>(), pool.data_ptr<float>(), N, C, H, Wd);

  long long c3_total = (long long)N * O3 * HW;
  conv_spatial_kernel<<<(c3_total + threads - 1) / threads, threads, 0, stream>>>(
      red3.data_ptr<float>(), c3_w.data_ptr<float>(), c3_b.data_ptr<float>(),
      out.data_ptr<float>(), N, R3, O3, H, Wd, 3, OT, O1);

  long long c5_total = (long long)N * O5 * HW;
  conv_spatial_kernel<<<(c5_total + threads - 1) / threads, threads, 0, stream>>>(
      red5.data_ptr<float>(), c5_w.data_ptr<float>(), c5_b.data_ptr<float>(),
      out.data_ptr<float>(), N, R5, O5, H, Wd, 5, OT, O1 + O3);

  dim3 grid_p((N * HW + TILE_M - 1) / TILE_M, (OP + TILE_O - 1) / TILE_O);
  pointwise1x1_tiled_kernel<<<grid_p, block2, 0, stream>>>(
      pool.data_ptr<float>(), p_w.data_ptr<float>(), p_b.data_ptr<float>(),
      out.data_ptr<float>(), N, C, OP, HW, OT, O1 + O3 + O5);

  return out;
}

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

void double_mish_inplace_cuda(torch::Tensor x);

void double_mish_inplace(torch::Tensor x) {
    double_mish_inplace_cuda(x);
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float mish_f(float x) {
    // softplus with the same stable branches customarily used for fp32.
    float sp;
    if (x > 20.0f) {
        sp = x;
    } else if (x < -20.0f) {
        sp = expf(x);
    } else {
        sp = log1pf(expf(x));
    }
    return x * tanhf(sp);
}

__global__ void double_mish_kernel(float* __restrict__ p, int64_t n) {
    int64_t i = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; i < n; i += stride) {
        float v = p[i];
        v = mish_f(v);
        v = mish_f(v);
        p[i] = v;
    }
}

void double_mish_inplace_cuda(torch::Tensor x) {
    const int64_t n = x.numel();
    if (n == 0) return;
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    double_mish_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), n);
}
"""

_ext = load_inline(
    name="double_mish_ext_v1",
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=["double_mish_inplace"],
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],

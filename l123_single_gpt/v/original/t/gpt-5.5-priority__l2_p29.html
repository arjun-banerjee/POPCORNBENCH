import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor double_mish_inplace(torch::Tensor x);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cmath>

// Compute mish(mish(x)) in-place for float32 tensors.  Mish is x*tanh(softplus(x)).
// The branch form mirrors the stable behavior of common softplus implementations
// while avoiding two separate PyTorch pointwise kernel launches.
__device__ __forceinline__ float softplus_f(float x) {
    // For large positive x, log1p(exp(x)) == x to fp32 precision and avoids overflow.
    // For large negative x, exp(x) is tiny and log1p is accurate.
    if (x > 20.0f) return x;
    return log1pf(expf(x));
}

__device__ __forceinline__ float mish_f(float x) {
    return x * tanhf(softplus_f(x));
}

__global__ void double_mish_kernel(float* __restrict__ data, long long n) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (long long i = idx; i < n; i += stride) {
        float v = data[i];
        v = mish_f(v);
        v = mish_f(v);
        data[i] = v;
    }
}

torch::Tensor double_mish_inplace(torch::Tensor x) {
    const long long n = x.numel();
    if (n == 0) return x;
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    double_mish_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), n);
    return x;
}
"""

_ext = load_inline(

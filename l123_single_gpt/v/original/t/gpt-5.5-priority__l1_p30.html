        Args:
            x (torch.Tensor): Input tensor of any shape.

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

void softsign_cuda(torch::Tensor x, torch::Tensor y);

torch::Tensor softsign(torch::Tensor x) {
    auto y = torch::empty_like(x);
    softsign_cuda(x, y);
    return y;
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__device__ __forceinline__ float softsign_f(float v) {
    return v / (1.0f + fabsf(v));
}

__global__ void softsign_scalar_kernel(const float* __restrict__ x, float* __restrict__ y, int64_t n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < n; idx += stride) {
        float v = x[idx];
        y[idx] = softsign_f(v);
    }
}

__global__ void softsign_float4_kernel(const float4* __restrict__ x4, float4* __restrict__ y4, int64_t n4) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < n4; idx += stride) {
        float4 v = x4[idx];
        v.x = softsign_f(v.x);
        v.y = softsign_f(v.y);
        v.z = softsign_f(v.z);
        v.w = softsign_f(v.w);
        y4[idx] = v;
    }
}

void softsign_cuda(torch::Tensor x, torch::Tensor y) {
    const int64_t n = x.numel();
    if (n == 0) return;

    const int threads = 256;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float* xp = x.data_ptr<float>();
    float* yp = y.data_ptr<float>();

    // Vectorized path for the common contiguous, aligned fp32 case.
    uintptr_t xa = reinterpret_cast<uintptr_t>(xp);
    uintptr_t ya = reinterpret_cast<uintptr_t>(yp);
    if (((xa | ya) & 15) == 0 && (n % 4 == 0)) {
        int64_t n4 = n / 4;
        int blocks = (int)((n4 + threads - 1) / threads);
        if (blocks > 65535) blocks = 65535;
        softsign_float4_kernel<<<blocks, threads, 0, stream>>>(
            reinterpret_cast<const float4*>(xp), reinterpret_cast<float4*>(yp), n4);
    } else {
        int blocks = (int)((n + threads - 1) / threads);
        if (blocks > 65535) blocks = 65535;
        softsign_scalar_kernel<<<blocks, threads, 0, stream>>>(xp, yp, n);
    }
}
"""

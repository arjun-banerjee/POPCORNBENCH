import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a convolution, applies HardSwish, and then ReLU.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, height, width).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, height, width).
        """
        x = self.conv(x)
        x = torch.nn.functional.hardswish(x)

        return x

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>
#include <ATen/ATen.h>

at::Tensor hardswish_relu_inplace(at::Tensor x);
"""

_cuda_src = r"""
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda_runtime.h>
#include <algorithm>
#include <cstdint>

__global__ void hardswish_relu_inplace_kernel(float* __restrict__ x, int64_t n) {
    int64_t tid = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    const float one_sixth = 0.16666666666666666f;
    for (int64_t i = tid; i < n; i += stride) {
        float v = x[i];
        float y;
        // relu(hardswish(v)) simplifies to this piecewise form.
        if (v <= 0.0f) {
            y = 0.0f;
        } else if (v < 3.0f) {
            y = v * (v + 3.0f) * one_sixth;
        } else {
            y = v;
        }
        x[i] = y;
    }
}

at::Tensor hardswish_relu_inplace(at::Tensor x) {
    const int64_t n = x.numel();
    if (n == 0) return x;

    constexpr int threads = 256;
    int64_t needed = (n + threads - 1) / threads;
    int blocks = (int)std::min<int64_t>(needed, 4096);

    hardswish_relu_inplace_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), n);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return x;
}
"""

_ext = load_inline(
    name="fused_hardswish_relu_inplace_ext_v1",
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=["hardswish_relu_inplace"],
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Conv2d followed by fused HardSwish+ReLU.  The convolution module is kept so
    the parameter names/state_dict match the reference implementation exactly.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        y = self.conv(x)
        return _ext.hardswish_relu_inplace(y)

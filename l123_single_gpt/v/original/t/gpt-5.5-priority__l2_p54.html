import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a convolution, multiplies by a learnable scalar, applies LeakyReLU, and then GELU.
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.multiplier = nn.Parameter(torch.randn(multiplier_shape)) 
        self.leaky_relu = nn.LeakyReLU()

    def forward(self, x):
        x = self.conv(x)
        x = x * self.multiplier
        x = self.leaky_relu(x)
        x = torch.nn.functional.gelu(x)
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define IC 64
#define OC 64
#define KH 3
#define KW 3
#define HT 4
#define WT 8
#define OCT 16
#define SH (HT + KH - 1)
#define SW (WT + KW - 1)

__device__ __forceinline__ float gelu_exact_f(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.70710678118654752440f));
}

__global__ void conv3x3_fused_kernel(const float* __restrict__ x,
                                     const float* __restrict__ w,
                                     const float* __restrict__ b,
                                     const float* __restrict__ m,
                                     float* __restrict__ y,
                                     int N, int H, int W, int Hout, int Wout) {
    __shared__ float tile[IC * SH * SW];

    const int tx = threadIdx.x;          // 0..WT-1, output column inside tile
    const int ty = threadIdx.y;          // 0..OCT-1, output channel inside tile
    const int tid = ty * WT + tx;

    const int wt_block = blockIdx.x;
    const int ht_block = blockIdx.y;
    const int ocz = blockIdx.z & 3;      // 4 OC tiles for OC=64
    const int n = blockIdx.z >> 2;

    const int h0 = ht_block * HT;
    const int w0 = wt_block * WT;

    // Cooperative load of the input patch for all input channels.
    #pragma unroll 1
    for (int idx = tid; idx < IC * SH * SW; idx += WT * OCT) {
        int c = idx / (SH * SW);
        int rem = idx - c * (SH * SW);
        int rr = rem / SW;
        int cc = rem - rr * SW;
        int ih = h0 + rr;
        int iw = w0 + cc;
        float v = 0.0f;
        if (ih < H && iw < W) {
            v = __ldg(x + ((n * IC + c) * H + ih) * W + iw);
        }
        tile[idx] = v;
    }
    __syncthreads();

    const int oc = ocz * OCT + ty;
    const int ow = w0 + tx;

    float acc0 = __ldg(b + oc);
    float acc1 = acc0;
    float acc2 = acc0;
    float acc3 = acc0;

    #pragma unroll 1
    for (int c = 0; c < IC; ++c) {
        const int s_c_base = c * SH * SW;
        const int w_c_base = (oc * IC + c) * KH * KW;
        #pragma unroll
        for (int ky = 0; ky < KH; ++ky) {
            const int s_row = s_c_base + ky * SW + tx;
            const int w_row = w_c_base + ky * KW;
            float ww0 = __ldg(w + w_row + 0);
            float ww1 = __ldg(w + w_row + 1);
            float ww2 = __ldg(w + w_row + 2);

            float x00 = tile[s_row + 0];
            float x01 = tile[s_row + 1];
            float x02 = tile[s_row + 2];
            acc0 = fmaf(x00, ww0, acc0);
            acc0 = fmaf(x01, ww1, acc0);
            acc0 = fmaf(x02, ww2, acc0);

            float x10 = tile[s_row + SW + 0];
            float x11 = tile[s_row + SW + 1];
            float x12 = tile[s_row + SW + 2];
            acc1 = fmaf(x10, ww0, acc1);
            acc1 = fmaf(x11, ww1, acc1);
            acc1 = fmaf(x12, ww2, acc1);

            float x20 = tile[s_row + 2 * SW + 0];
            float x21 = tile[s_row + 2 * SW + 1];
            float x22 = tile[s_row + 2 * SW + 2];
            acc2 = fmaf(x20, ww0, acc2);
            acc2 = fmaf(x21, ww1, acc2);
            acc2 = fmaf(x22, ww2, acc2);

            float x30 = tile[s_row + 3 * SW + 0];
            float x31 = tile[s_row + 3 * SW + 1];
            float x32 = tile[s_row + 3 * SW + 2];
            acc3 = fmaf(x30, ww0, acc3);
            acc3 = fmaf(x31, ww1, acc3);
            acc3 = fmaf(x32, ww2, acc3);
        }
    }

    if (ow < Wout) {
        const float mult = __ldg(m + oc);
        const long out_base = ((long)n * OC + oc) * Hout * Wout + ow;
        float v0 = acc0 * mult;
        v0 = (v0 >= 0.0f) ? v0 : v0 * 0.01f;
        y[out_base + (long)(h0 + 0) * Wout] = gelu_exact_f(v0);

        if (h0 + 1 < Hout) {
            float v1 = acc1 * mult;
            v1 = (v1 >= 0.0f) ? v1 : v1 * 0.01f;
            y[out_base + (long)(h0 + 1) * Wout] = gelu_exact_f(v1);
        }
        if (h0 + 2 < Hout) {
            float v2 = acc2 * mult;
            v2 = (v2 >= 0.0f) ? v2 : v2 * 0.01f;
            y[out_base + (long)(h0 + 2) * Wout] = gelu_exact_f(v2);
        }
        if (h0 + 3 < Hout) {
            float v3 = acc3 * mult;
            v3 = (v3 >= 0.0f) ? v3 : v3 * 0.01f;
            y[out_base + (long)(h0 + 3) * Wout] = gelu_exact_f(v3);
        }
    }
}

torch::Tensor fused_conv_act_forward(torch::Tensor x,
                                     torch::Tensor weight,
                                     torch::Tensor bias,
                                     torch::Tensor multiplier) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(weight.is_cuda() && bias.is_cuda() && multiplier.is_cuda(), "parameters must be CUDA tensors");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only fp32 is supported");
    TORCH_CHECK(weight.scalar_type() == at::kFloat && bias.scalar_type() == at::kFloat && multiplier.scalar_type() == at::kFloat,
                "only fp32 parameters are supported");
    TORCH_CHECK(x.is_contiguous() && weight.is_contiguous() && bias.is_contiguous() && multiplier.is_contiguous(),
                "all tensors must be contiguous");
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "bad tensor rank");
    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int O = (int)weight.size(0);
    TORCH_CHECK(C == IC && O == OC && weight.size(1) == IC && weight.size(2) == 3 && weight.size(3) == 3,
                "this optimized kernel expects 64x64 3x3 convolution");
    const int Hout = H - 2;
    const int Wout = W - 2;
    auto y = torch::empty({N, O, Hout, Wout}, x.options());

    dim3 block(WT, OCT);
    dim3 grid((Wout + WT - 1) / WT, (Hout + HT - 1) / HT, N * ((O + OCT - 1) / OCT));
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    conv3x3_fused_kernel<<<grid, block, 0, stream>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
        multiplier.data_ptr<float>(), y.data_ptr<float>(), N, H, W, Hout, Wout);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}

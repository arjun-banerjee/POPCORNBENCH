import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies Batch Normalization, and scales the output.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv(x)
        x = self.bn(x)
        x = x * self.scaling_factor
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

static constexpr int THREADS = 256;
static constexpr int PARTS = 32;

__global__ void partial_stats_kernel(const float* __restrict__ y,
                                     double* __restrict__ psum,
                                     double* __restrict__ psq,
                                     int N, int C, int H, int W) {
    int c = blockIdx.x;
    int part = blockIdx.y;
    int tid = threadIdx.x;
    long HW = (long)H * W;
    long M = (long)N * HW;
    double s = 0.0;
    double q = 0.0;
    for (long m = (long)part * blockDim.x + tid; m < M; m += (long)blockDim.x * gridDim.y) {
        long n = m / HW;
        long r = m - n * HW;
        float v = y[(n * C + c) * HW + r];
        s += (double)v;
        q += (double)v * (double)v;
    }
    extern __shared__ double sh[];
    double* ss = sh;
    double* qq = sh + blockDim.x;
    ss[tid] = s;
    qq[tid] = q;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            ss[tid] += ss[tid + stride];
            qq[tid] += qq[tid + stride];
        }
        __syncthreads();
    }
    if (tid == 0) {
        psum[c * PARTS + part] = ss[0];
        psq [c * PARTS + part] = qq[0];
    }
}

__global__ void finalize_train_kernel(const double* __restrict__ psum,
                                      const double* __restrict__ psq,
                                      const float* __restrict__ gamma,
                                      const float* __restrict__ beta,
                                      float* __restrict__ running_mean,
                                      float* __restrict__ running_var,
                                      float* __restrict__ a,
                                      float* __restrict__ b,
                                      int C, long M, double momentum,
                                      double eps, double scaling) {
    int c = blockIdx.x * blockDim.x + threadIdx.x;
    if (c >= C) return;
    double s = 0.0, q = 0.0;
    #pragma unroll
    for (int p = 0; p < PARTS; ++p) {
        s += psum[c * PARTS + p];
        q += psq [c * PARTS + p];
    }
    double mean = s / (double)M;
    double var = q / (double)M - mean * mean;
    if (var < 0.0) var = 0.0;
    double inv = rsqrt(var + eps);
    double g = (double)gamma[c];
    double be = (double)beta[c];
    double sc = scaling;
    a[c] = (float)(g * inv * sc);
    b[c] = (float)((be - mean * g * inv) * sc);

    // Match BatchNorm's running-stat update (unbiased variance for running_var).
    double unb = (M > 1) ? var * ((double)M / (double)(M - 1)) : var;
    running_mean[c] = (float)(((1.0 - momentum) * (double)running_mean[c]) + momentum * mean);
    running_var[c]  = (float)(((1.0 - momentum) * (double)running_var[c])  + momentum * unb);
}

__global__ void coeff_eval_kernel(const float* __restrict__ gamma,
                                  const float* __restrict__ beta,
                                  const float* __restrict__ running_mean,
                                  const float* __restrict__ running_var,
                                  float* __restrict__ a,
                                  float* __restrict__ b,
                                  int C, double eps, double scaling) {
    int c = blockIdx.x * blockDim.x + threadIdx.x;
    if (c >= C) return;
    double inv = rsqrt((double)running_var[c] + eps);
    double g = (double)gamma[c];
    double sc = scaling;
    a[c] = (float)(g * inv * sc);
    b[c] = (float)((double)beta[c] * sc - (double)running_mean[c] * g * inv * sc);
}

__global__ void apply_affine_kernel(const float* __restrict__ y,
                                    const float* __restrict__ a,
                                    const float* __restrict__ b,
                                    float* __restrict__ out,
                                    long total, int C, int H, int W) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    long stride = (long)blockDim.x * gridDim.x;
    long HW = (long)H * W;
    for (; idx < total; idx += stride) {
        int c = (int)((idx / HW) % C);
        out[idx] = y[idx] * a[c] + b[c];
    }
}

                   torch::Tensor running_mean, torch::Tensor running_var,
                   torch::Tensor out, torch::Tensor a, torch::Tensor b,

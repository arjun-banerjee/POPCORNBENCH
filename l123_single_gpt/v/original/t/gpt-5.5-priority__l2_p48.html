import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, scales the output, applies tanh, multiplies by a scaling factor, and applies sigmoid.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.scaling_factor = nn.Parameter(torch.randn(bias_shape))
        self.bias = nn.Parameter(torch.randn(bias_shape)) 

    def forward(self, x):
        x = self.conv(x)
        x = x * self.scaling_factor 
        x = torch.tanh(x)
        x = x * self.bias
        x = torch.sigmoid(x)
        return x

                    const float* wp = w + ((long)oc*C + c)*Ktot + kd*KHW + kh*Kw;
                    for (int kw=0; kw<Kw; ++kw) acc = fmaf(__ldg(xp+kw), __ldg(wp+kw), acc);
                }
            }
        }
        float v = acc * __ldg(scale + oc);
        v = tanhf(v);
        v *= __ldg(post_b + oc);
        y[idx] = 1.0f / (1.0f + expf(-v));
    }
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor scale, torch::Tensor post_bias) {
    check_inputs(x, weight);
    TORCH_CHECK(x.dim() == 5 && weight.dim() == 5, "expected 5D tensors");
    auto xc = x.contiguous();
    auto wc = weight.contiguous();
    auto sc = scale.contiguous();
    auto pc = post_bias.contiguous();
    torch::Tensor bc = conv_bias.defined() ? conv_bias.contiguous() : conv_bias;

    int N = (int)xc.size(0), C = (int)xc.size(1), D = (int)xc.size(2), H = (int)xc.size(3), W = (int)xc.size(4);
    int O = (int)wc.size(0), Kd = (int)wc.size(2), Kh = (int)wc.size(3), Kw = (int)wc.size(4);
    int Do = D - Kd + 1, Ho = H - Kh + 1, Wo = W - Kw + 1;
    auto y = torch::empty({N, O, Do, Ho, Wo}, xc.options());
    long total = y.numel();
    if (total == 0) return y;
    int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    int maxBlocks = 65535;
    if (blocks > maxBlocks) blocks = maxBlocks;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const float* bp = bc.defined() ? bc.data_ptr<float>() : nullptr;
    if (C == 3 && Kd == 3 && Kh == 3 && Kw == 3) {
        conv3d_fused_k3_c3_kernel<<<blocks, threads, 0, stream>>>(
            xc.data_ptr<float>(), wc.data_ptr<float>(), bp, sc.data_ptr<float>(), pc.data_ptr<float>(), y.data_ptr<float>(),
            total, N, O, D, H, W, Do, Ho, Wo);

            total, N, C, O, D, H, W, Kd, Kh, Kw, Do, Ho, Wo);
    }
    return y;
}

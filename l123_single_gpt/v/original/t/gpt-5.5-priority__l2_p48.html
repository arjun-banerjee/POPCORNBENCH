import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, scales the output, applies tanh, multiplies by a scaling factor, and applies sigmoid.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.scaling_factor = nn.Parameter(torch.randn(bias_shape))
        self.bias = nn.Parameter(torch.randn(bias_shape)) 

    def forward(self, x):
        x = self.conv(x)
        x = x * self.scaling_factor 
        x = torch.tanh(x)
        x = x * self.bias
        x = torch.sigmoid(x)
        return x

#include <cuda.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAException.h>

__device__ __forceinline__ float sigmoidf_dev(float x) {
    return 1.0f / (1.0f + expf(-x));
}

// Specialized for the benchmark shape: Cin=3, Cout=16, K=3.
// One CUDA thread computes all 16 output channels for one (N,D,H,W) output point.
__global__ void conv3d_k3_c3_oc16_fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ cb,
    const float* __restrict__ scale,
    const float* __restrict__ pb,
    float* __restrict__ y,
    int N, int D, int H, int W, int OD, int OH, int OW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OD * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    int n = t / OD;

    float a0  = cb[0];  float a1  = cb[1];  float a2  = cb[2];  float a3  = cb[3];
    float a4  = cb[4];  float a5  = cb[5];  float a6  = cb[6];  float a7  = cb[7];
    float a8  = cb[8];  float a9  = cb[9];  float a10 = cb[10]; float a11 = cb[11];
    float a12 = cb[12]; float a13 = cb[13]; float a14 = cb[14]; float a15 = cb[15];

    const int DHW = D * H * W;
    const int HW = H * W;
    const int KSPAN = 27;       // KD*KH*KW
    const int WSPAN = 81;       // Cin*KD*KH*KW per output channel

    #pragma unroll
    for (int ci = 0; ci < 3; ++ci) {
        int x_c_base = (n * 3 + ci) * DHW;
        int k_c_base = ci * KSPAN;
        #pragma unroll
        for (int kd = 0; kd < 3; ++kd) {
            int x_d_base = x_c_base + (od + kd) * HW;
            int k_d_base = k_c_base + kd * 9;
            #pragma unroll
            for (int kh = 0; kh < 3; ++kh) {
                int x_base = x_d_base + (oh + kh) * W + ow;
                int k_base = k_d_base + kh * 3;
                #pragma unroll
                for (int kw = 0; kw < 3; ++kw) {
                    float xv = x[x_base + kw];
                    int kk = k_base + kw;
                    a0  = fmaf(xv, w[0  * WSPAN + kk], a0);
                    a1  = fmaf(xv, w[1  * WSPAN + kk], a1);
                    a2  = fmaf(xv, w[2  * WSPAN + kk], a2);
                    a3  = fmaf(xv, w[3  * WSPAN + kk], a3);
                    a4  = fmaf(xv, w[4  * WSPAN + kk], a4);
                    a5  = fmaf(xv, w[5  * WSPAN + kk], a5);
                    a6  = fmaf(xv, w[6  * WSPAN + kk], a6);
                    a7  = fmaf(xv, w[7  * WSPAN + kk], a7);
                    a8  = fmaf(xv, w[8  * WSPAN + kk], a8);
                    a9  = fmaf(xv, w[9  * WSPAN + kk], a9);
                    a10 = fmaf(xv, w[10 * WSPAN + kk], a10);
                    a11 = fmaf(xv, w[11 * WSPAN + kk], a11);
                    a12 = fmaf(xv, w[12 * WSPAN + kk], a12);
                    a13 = fmaf(xv, w[13 * WSPAN + kk], a13);
                    a14 = fmaf(xv, w[14 * WSPAN + kk], a14);
                    a15 = fmaf(xv, w[15 * WSPAN + kk], a15);
                }
            }
        }
    }

    const int spatial = OD * OH * OW;
    int out_base = n * 16 * spatial + od * OH * OW + oh * OW + ow;

    y[out_base + 0  * spatial] = sigmoidf_dev(pb[0]  * tanhf(a0  * scale[0]));
    y[out_base + 1  * spatial] = sigmoidf_dev(pb[1]  * tanhf(a1  * scale[1]));
    y[out_base + 2  * spatial] = sigmoidf_dev(pb[2]  * tanhf(a2  * scale[2]));
    y[out_base + 3  * spatial] = sigmoidf_dev(pb[3]  * tanhf(a3  * scale[3]));
    y[out_base + 4  * spatial] = sigmoidf_dev(pb[4]  * tanhf(a4  * scale[4]));
    y[out_base + 5  * spatial] = sigmoidf_dev(pb[5]  * tanhf(a5  * scale[5]));
    y[out_base + 6  * spatial] = sigmoidf_dev(pb[6]  * tanhf(a6  * scale[6]));

}

// General correctness fallback for other constructor choices.
__global__ void conv3d_general_fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ cb,
    const float* __restrict__ scale,
    const float* __restrict__ pb,
    float* __restrict__ y,
    int N, int CI, int D, int H, int W,
    int CO, int KD, int KH, int KW,
    int OD, int OH, int OW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * CO * OD * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    t /= OD;
    int co = t % CO;
    int n = t / CO;

    float acc = cb[co];
    int DHW = D * H * W;
    int HW = H * W;
    int KHW = KH * KW;
    int WCO = CI * KD * KH * KW;
    for (int ci = 0; ci < CI; ++ci) {
        for (int kd = 0; kd < KD; ++kd) {
            for (int kh = 0; kh < KH; ++kh) {
                int x_base = ((n * CI + ci) * D + od + kd) * HW + (oh + kh) * W + ow;
                int w_base = co * WCO + ((ci * KD + kd) * KH + kh) * KW;
                for (int kw = 0; kw < KW; ++kw) {
                    acc = fmaf(x[x_base + kw], w[w_base + kw], acc);
                }
            }
        }
    }
    y[idx] = sigmoidf_dev(pb[co] * tanhf(acc * scale[co]));
}

torch::Tensor fused_conv3d_act_cuda(torch::Tensor x,
                                    torch::Tensor weight,
                                    torch::Tensor conv_bias,
                                    torch::Tensor scale,
                                    torch::Tensor post_bias) {
    x = x.contiguous();
    weight = weight.contiguous();
    conv_bias = conv_bias.contiguous();
    scale = scale.contiguous();
    post_bias = post_bias.contiguous();

    const int64_t N64 = x.size(0), CI64 = x.size(1), D64 = x.size(2), H64 = x.size(3), W64 = x.size(4);
    const int64_t CO64 = weight.size(0), KD64 = weight.size(2), KH64 = weight.size(3), KW64 = weight.size(4);
    const int64_t OD64 = D64 - KD64 + 1, OH64 = H64 - KH64 + 1, OW64 = W64 - KW64 + 1;

    auto y = torch::empty({N64, CO64, OD64, OH64, OW64}, x.options());

    int N = (int)N64, CI = (int)CI64, D = (int)D64, H = (int)H64, W = (int)W64;
    int CO = (int)CO64, KD = (int)KD64, KH = (int)KH64, KW = (int)KW64;
    int OD = (int)OD64, OH = (int)OH64, OW = (int)OW64;

    const int threads = 128;
    if (CI == 3 && CO == 16 && KD == 3 && KH == 3 && KW == 3) {
        int total_pos = N * OD * OH * OW;
        int blocks = (total_pos + threads - 1) / threads;
        conv3d_k3_c3_oc16_fused_kernel<<<blocks, threads>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), conv_bias.data_ptr<float>(),
            scale.data_ptr<float>(), post_bias.data_ptr<float>(), y.data_ptr<float>(),
            N, D, H, W, OD, OH, OW);
    } else {
        int total = N * CO * OD * OH * OW;
        int blocks = (total + threads - 1) / threads;
        conv3d_general_fused_kernel<<<blocks, threads>>>(
            x.data_ptr<float>(), weight.data_ptr<float>(), conv_bias.data_ptr<float>(),
            scale.data_ptr<float>(), post_bias.data_ptr<float>(), y.data_ptr<float>(),
            N, CI, D, H, W, CO, KD, KH, KW, OD, OH, OW);
    }
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}

    Model that performs a GEMM, BatchNorm, GELU, and ReLU in sequence.
    """

        super(Model, self).__init__()

__device__ __forceinline__ float gelu_exact(float x) {
    // PyTorch default GELU is the exact erf formulation.
    return 0.5f * x * (1.0f + erff(x * 0.7071067811865475244f));
}

__global__ void bn_gelu_relu_kernel(const float* __restrict__ x,
                                    const float* __restrict__ gamma,
                                    const float* __restrict__ beta,
                                    const float* __restrict__ mean,
                                    const float* __restrict__ var,
                                    float* __restrict__ y,
                                    int64_t n,
                                    int64_t c,
                                    float eps) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < n; idx += stride) {
        int col = (int)(idx % c);
        float v = x[idx];
        float z = (v - mean[col]) * rsqrtf(var[col] + eps) * gamma[col] + beta[col];
        float g = gelu_exact(z);
        y[idx] = g > 0.0f ? g : 0.0f;
    }
}

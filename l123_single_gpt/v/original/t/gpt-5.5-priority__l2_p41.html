import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a GEMM, BatchNorm, GELU, and ReLU in sequence.
    """
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.batch_norm = nn.BatchNorm1d(out_features)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).

        """
        x = self.gemm(x)
        x = self.batch_norm(x)
        x = torch.nn.functional.gelu(x)
        x = torch.relu(x)

#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#define COL_TILE 32
#define ROW_TILE 16
#define Y_THREADS 8

__global__ void partial_sum_kernel(const float* __restrict__ x,
                                   float* __restrict__ partial,
                                   float* __restrict__ partial2,
                                   int N, int C, int tiles) {
    int col0 = blockIdx.x * COL_TILE;
    int tile = blockIdx.y;
    int tx = threadIdx.x;  // 0..31, feature within tile
    int ty = threadIdx.y;  // 0..7, row reducer
    int c = col0 + tx;
    int row0 = tile * ROW_TILE;

    float s = 0.0f;
    float q = 0.0f;
    if (c < C) {
        #pragma unroll
        for (int r = ty; r < ROW_TILE; r += Y_THREADS) {
            int n = row0 + r;
            if (n < N) {
                float v = x[(long long)n * C + c];
                s += v;
                q += v * v;
            }
        }
    }

    __shared__ float sh[Y_THREADS][COL_TILE];
    __shared__ float sh2[Y_THREADS][COL_TILE];
    sh[ty][tx] = s;
    sh2[ty][tx] = q;
    __syncthreads();

    if (ty == 0 && c < C) {
        float ss = 0.0f;
        float qq = 0.0f;
        #pragma unroll
        for (int k = 0; k < Y_THREADS; ++k) {
            ss += sh[k][tx];
            qq += sh2[k][tx];
        }
        partial[(long long)tile * C + c] = ss;
        partial2[(long long)tile * C + c] = qq;
    }
}

__global__ void finalize_stats_kernel(const float* __restrict__ partial,
                                      const float* __restrict__ partial2,
                                      float* __restrict__ mean,
                                      float* __restrict__ invstd,
                                      float* __restrict__ running_mean,
                                      float* __restrict__ running_var,
                                      int N, int C, int tiles,
                                      float eps, float momentum) {
    int c = blockIdx.x;
    int tid = threadIdx.x;
    float s = 0.0f;
    float q = 0.0f;
    for (int t = tid; t < tiles; t += blockDim.x) {
        float a = partial[(long long)t * C + c];
        float b = partial2[(long long)t * C + c];
        s += a;
        q += b;
    }

    __shared__ float ss[256];
    __shared__ float qq[256];
    ss[tid] = s;
    qq[tid] = q;
    __syncthreads();

    for (int stride = 128; stride > 0; stride >>= 1) {
        if (tid < stride) {
            ss[tid] += ss[tid + stride];
            qq[tid] += qq[tid + stride];
        }
        __syncthreads();
    }

    if (tid == 0) {
        float n_inv = 1.0f / (float)N;
        float m = ss[0] * n_inv;
        float var = qq[0] * n_inv - m * m;
        var = var < 0.0f ? 0.0f : var;
        float inv = rsqrtf(var + eps);
        mean[c] = m;
        invstd[c] = inv;
        // PyTorch BatchNorm updates running_var with the unbiased variance.
        float unb = (N > 1) ? (var * ((float)N / (float)(N - 1))) : var;
        running_mean[c] = running_mean[c] * (1.0f - momentum) + m * momentum;
        running_var[c] = running_var[c] * (1.0f - momentum) + unb * momentum;
    }
}

__device__ __forceinline__ float gelu_relu_exact(float z) {
    if (z <= 0.0f) return 0.0f;
    const float inv_sqrt2 = 0.70710678118654752440f;
    return 0.5f * z * (1.0f + erff(z * inv_sqrt2));
}

                                      const float* __restrict__ mean,
                                      const float* __restrict__ invstd,
                                      const float* __restrict__ gamma,
                                      const float* __restrict__ beta,

        z = z * gamma[c] + beta[c];

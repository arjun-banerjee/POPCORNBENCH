import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a GEMM, BatchNorm, GELU, and ReLU in sequence.
    """
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.batch_norm = nn.BatchNorm1d(out_features)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.gemm(x)
        x = self.batch_norm(x)
        x = torch.nn.functional.gelu(x)
        x = torch.relu(x)
        return x

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

__device__ __forceinline__ float gelu_exact(float x) {
    // PyTorch default GELU is the exact erf formulation.
    return 0.5f * x * (1.0f + erff(x * 0.7071067811865475244f));
}

__global__ void bn_gelu_relu_kernel(const float* __restrict__ x,
                                    const float* __restrict__ gamma,
                                    const float* __restrict__ beta,
                                    const float* __restrict__ mean,
                                    const float* __restrict__ var,
                                    float* __restrict__ y,
                                    int64_t n,
                                    int64_t c,
                                    float eps) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < n; idx += stride) {
        int col = (int)(idx % c);
        float v = x[idx];
        float z = (v - mean[col]) * rsqrtf(var[col] + eps) * gamma[col] + beta[col];
        float g = gelu_exact(z);
        y[idx] = g > 0.0f ? g : 0.0f;
    }
}

__global__ void gelu_relu_kernel(const float* __restrict__ x, float* __restrict__ y, int64_t n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; idx < n; idx += stride) {
        float g = gelu_exact(x[idx]);
        y[idx] = g > 0.0f ? g : 0.0f;
    }
}

torch::Tensor bn_gelu_relu_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                                   torch::Tensor running_mean, torch::Tensor running_var, double eps) {
    auto y = torch::empty_like(x);
    const int64_t n = x.numel();
    const int64_t c = x.size(1);
    if (n == 0) return y;
    int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 65535) blocks = 65535;
    bn_gelu_relu_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
        running_mean.data_ptr<float>(), running_var.data_ptr<float>(), y.data_ptr<float>(),
        n, c, (float)eps);
    return y;
}

torch::Tensor gelu_relu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    const int64_t n = x.numel();
    if (n == 0) return y;

    gelu_relu_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}
"""

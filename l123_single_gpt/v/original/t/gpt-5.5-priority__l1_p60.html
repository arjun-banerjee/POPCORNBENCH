__global__ __launch_bounds__(256)
void conv3d_3_64_3x5x7_d2_kernel(const float* __restrict__ x,
                                  const float* __restrict__ w,
                                  const float* __restrict__ b,
                                  float* __restrict__ y,
                                  int N, int D, int H, int W,
                                  int Do, int Ho, int Wo,
                                  int has_bias) {
    extern __shared__ float sw[];
    const int tx = threadIdx.x;        // 0..63, output W
    const int ty = threadIdx.y;        // 0..3, output H
    const int tid = ty * 64 + tx;

    int gz = blockIdx.z;
    const int od_pair = gz % ((Do + 1) >> 1);
    gz /= ((Do + 1) >> 1);
    const int oc = gz & 63;            // OC == 64
    const int n = gz >> 6;

    for (int i = tid; i < 315; i += 256) {
        sw[i] = w[oc * 315 + i];

    float sum0 = has_bias ? b[oc] : 0.0f;
    float sum1 = has_bias ? b[oc] : 0.0f;

    const int in_n_base = n * 3 * D * HW;

    #pragma unroll
    for (int c = 0; c < 3; ++c) {

    if isinstance(v, tuple):
        return v
    return (v, v, v)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.conv3d = nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        st = self.conv3d.stride
        pa = self.conv3d.padding
        di = self.conv3d.dilation
        b = self.conv3d.bias
        if b is None:
            b = torch.empty(0, device=x.device, dtype=x.dtype)
        return _ext.conv3d_forward(x.contiguous(), self.conv3d.weight.contiguous(), b,
                                   int(st[0]), int(st[1]), int(st[2]),
                                   int(pa[0]), int(pa[1]), int(pa[2]),
                                   int(di[0]), int(di[1]), int(di[2]))

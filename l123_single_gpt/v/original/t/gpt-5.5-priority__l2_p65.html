import torch
import torch.nn as nn

class Model(nn.Module):
    """
    This model performs a convolution, average pooling, applies sigmoid, and sums the result.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.avg_pool = nn.AvgPool2d(pool_kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = self.avg_pool(x)
        x = torch.sigmoid(x)
        x = torch.sum(x, dim=[1,2,3]) # Sum over all spatial dimensions
        return x

// Build the 6x6 weights equivalent to avgpool4(conv3(x)) and clear output.
__global__ void transform_kernel(const float* __restrict__ w,
                                 float* __restrict__ coeff,
                                 float* __restrict__ out,
                                 int N, int Cout, int total) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) out[idx] = 0.0f;
    if (idx >= total) return;

    int t = idx;
    int dx = t % 6; t /= 6;
    int dy = t % 6; t /= 6;
    int ic = t % 8; t /= 8;
    int oc = t;

    float v = 0.0f;
#pragma unroll
    for (int ky = 0; ky < 3; ++ky) {
        int r = dy - ky;
        if ((unsigned)r < 4u) {
#pragma unroll
            for (int kx = 0; kx < 3; ++kx) {
                int s = dx - kx;
                if ((unsigned)s < 4u) {
                    v += w[((oc * 8 + ic) * 3 + ky) * 3 + kx];
                }
            }
        }
    }
    coeff[idx] = v * 0.0625f;
}

__global__ void fused_kernel(const float* __restrict__ x,
                             const float* __restrict__ coeff,
                             const float* __restrict__ bias,
                             float* __restrict__ out,
                             int N, int Cout, int H, int W, int PH, int PW,
                             int tiles_per_oc) {
    __shared__ float sc[8 * 36];
    __shared__ float sr[256];

    int bid = blockIdx.x;
    int tile = bid % tiles_per_oc;
    int tmp = bid / tiles_per_oc;
    int oc = tmp % Cout;
    int n = tmp / Cout;
    int tid = threadIdx.x;

    // Load the transformed weights for this output channel once per block.
    for (int i = tid; i < 8 * 36; i += blockDim.x) {
        sc[i] = coeff[oc * (8 * 36) + i];
    }
    __syncthreads();

    int S = PH * PW;
    int sp = tile * blockDim.x + tid;
    float sum = 0.0f;

    if (sp < S) {
        int ph = sp / PW;
        int pw = sp - ph * PW;
        int baseh = ph << 2;
        int basew = pw << 2;
        int HW = H * W;
        int base = n * 8 * HW + baseh * W + basew;
        float acc = bias[oc];

#pragma unroll
        for (int ic = 0; ic < 8; ++ic) {
            int ib = base + ic * HW;
            int cb = ic * 36;
#pragma unroll
            for (int dy = 0; dy < 6; ++dy) {
                int row = ib + dy * W;
                int co = cb + dy * 6;
#pragma unroll
                for (int dx = 0; dx < 6; ++dx) {
                    acc = fmaf(sc[co + dx], __ldg(x + row + dx), acc);
                }
            }
        }
        float e = __expf(-acc);
        sum = 1.0f / (1.0f + e);
    }

    sr[tid] = sum;
    __syncthreads();

#pragma unroll
    for (int stride = 128; stride > 0; stride >>= 1) {
        if (tid < stride) sr[tid] += sr[tid + stride];
        __syncthreads();
    }
    if (tid == 0) atomicAdd(out + n, sr[0]);
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    const int N = (int)x.size(0);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int Cout = (int)weight.size(0);

    const int convH = H - 2;
    const int convW = W - 2;
    const int PH = (convH - 4) / 4 + 1;
    const int PW = (convW - 4) / 4 + 1;

    auto out = torch::empty({N}, x.options());
    auto coeff = torch::empty({Cout, 8, 6, 6}, x.options());

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const int threads = 256;
    int total_coeff = Cout * 8 * 36;
    int init_total = total_coeff > N ? total_coeff : N;
    transform_kernel<<<(init_total + threads - 1) / threads, threads, 0, stream>>>(
        weight.data_ptr<float>(), coeff.data_ptr<float>(), out.data_ptr<float>(), N, Cout, total_coeff);

    int tiles = (PH * PW + threads - 1) / threads;
    int blocks = N * Cout * tiles;
    fused_kernel<<<blocks, threads, 0, stream>>>(
        x.data_ptr<float>(), coeff.data_ptr<float>(), bias.data_ptr<float>(), out.data_ptr<float>(),
        N, Cout, H, W, PH, PW, tiles);

    return out;
}

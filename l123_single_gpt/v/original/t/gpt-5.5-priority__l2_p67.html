import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies GELU, and then performs global average pooling.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, height, width)
        Returns:
            Output tensor of shape (batch_size, out_channels)
        """
        x = self.conv(x)
        x = torch.nn.functional.gelu(x)
        x = torch.nn.functional.adaptive_avg_pool2d(x, 1)
        x = x.squeeze(-1).squeeze(-1)
        return x

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#ifndef CHECK_CUDA
#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")
#endif
#define CHECK_FLOAT(x) TORCH_CHECK(x.scalar_type() == at::kFloat, #x " must be float32")
#define CHECK_CONTIGUOUS(x) TORCH_CHECK(x.is_contiguous(), #x " must be contiguous")

// One block computes one batch item and a group of 8 output channels.
// The spatial reduction is fused with conv and exact GELU, avoiding the huge
// intermediate conv/GELU tensors.
template<int OC_G>
__global__ void conv_gelu_gap_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int Cin, int H, int W, int Cout, int K, int Ho, int Wo)
{
    const int n = blockIdx.x;
    const int oc0 = blockIdx.y * OC_G;
    const int tid = threadIdx.x;
    const int HWo = Ho * Wo;

    float sum[OC_G];
    #pragma unroll
    for (int g = 0; g < OC_G; ++g) sum[g] = 0.0f;

    // grid-stride over output pixels within this (n, output-channel group)
    for (int idx = tid; idx < HWo; idx += blockDim.x) {
        const int oh = idx / Wo;
        const int ow = idx - oh * Wo;
        float v[OC_G];
        #pragma unroll
        for (int g = 0; g < OC_G; ++g) {
            int oc = oc0 + g;
            v[g] = (oc < Cout) ? b[oc] : 0.0f;
        }

        // K is 3 for the benchmark; keep loops generic for constructor compatibility.
        for (int ic = 0; ic < Cin; ++ic) {
            const float* xp = x + ((n * Cin + ic) * H + oh) * W + ow;
            for (int kh = 0; kh < K; ++kh) {
                const float* xrow = xp + kh * W;
                for (int kw = 0; kw < K; ++kw) {
                    float xv = xrow[kw];
                    int wk_base0 = ((oc0 * Cin + ic) * K + kh) * K + kw;
                    #pragma unroll
                    for (int g = 0; g < OC_G; ++g) {
                        int oc = oc0 + g;
                        if (oc < Cout) {
                            v[g] = fmaf(xv, w[wk_base0 + g * Cin * K * K], v[g]);
                        }
                    }
                }
            }
        }
        // exact GELU: 0.5*x*(1+erf(x/sqrt(2)))
        #pragma unroll
        for (int g = 0; g < OC_G; ++g) {
            int oc = oc0 + g;
            if (oc < Cout) {
                float z = v[g];
                sum[g] += 0.5f * z * (1.0f + erff(z * 0.70710678118654752440f));
            }
        }
    }

    __shared__ float smem[256 * OC_G];
    #pragma unroll
    for (int g = 0; g < OC_G; ++g) smem[g * blockDim.x + tid] = sum[g];
    __syncthreads();

    // blockDim.x is 256
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            #pragma unroll
            for (int g = 0; g < OC_G; ++g) {
                smem[g * blockDim.x + tid] += smem[g * blockDim.x + tid + stride];
            }
        }
        __syncthreads();
    }

    if (tid == 0) {
        const float scale = 1.0f / (float)HWo;
        #pragma unroll
        for (int g = 0; g < OC_G; ++g) {
            int oc = oc0 + g;
            if (oc < Cout) y[n * Cout + oc] = smem[g * blockDim.x] * scale;
        }
    }
}

torch::Tensor conv_gelu_gap_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    CHECK_CUDA(x); CHECK_CUDA(weight); CHECK_CUDA(bias);
    CHECK_FLOAT(x); CHECK_FLOAT(weight); CHECK_FLOAT(bias);
    CHECK_CONTIGUOUS(x); CHECK_CONTIGUOUS(weight); CHECK_CONTIGUOUS(bias);
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "bad tensor dims");

    const int N = (int)x.size(0);
    const int Cin = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int Cout = (int)weight.size(0);
    const int K = (int)weight.size(2);
    const int Ho = H - K + 1;
    const int Wo = W - K + 1;
    auto y = torch::empty({N, Cout}, x.options());

    constexpr int OC_G = 8;
    dim3 block(256);
    dim3 grid(N, (Cout + OC_G - 1) / OC_G);
    conv_gelu_gap_kernel<OC_G><<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), y.data_ptr<float>(),
        N, Cin, H, W, Cout, K, Ho, Wo);
    return y;
}

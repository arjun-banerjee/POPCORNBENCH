    """

        """
        Performs the transposed 1D convolution.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, length).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, length_out).
        """
        return self.conv1d_transpose(x)

    if (idx >= total) return;
    int l = (int)(idx % Lout);
    long long t = idx / Lout;
    int oc = (int)(t % Cout);
    int b = (int)(t / Cout);
    float acc = has_bias ? __ldg(bias + oc) : 0.0f;
    for (int ic = 0; ic < Cin; ++ic) {
        for (int k = 0; k < K; ++k) {
            int num = l + padding - k * dilation;
            if (num % stride == 0) {
                int il = num / stride;
                if ((unsigned)il < (unsigned)L) {
                    acc = fmaf(__ldg(x + (b * Cin + ic) * L + il),
                               __ldg(w + (ic * Cout + oc) * K + k), acc);
                }
            }
        }
    }
    y[idx] = acc;
}

torch::Tensor conv_transpose1d_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                                       int64_t stride, int64_t padding, int64_t dilation) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(weight.is_cuda(), "weight must be CUDA");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 supported");
    int B = (int)x.size(0);
    int Cin = (int)x.size(1);
    int L = (int)x.size(2);
    int Cout = (int)weight.size(1);
    int K = (int)weight.size(2);
    int Lout = (int)((L - 1) * stride - 2 * padding + dilation * (K - 1) + 1);
    auto y = torch::empty({B, Cout, Lout}, x.options());
    int has_bias = (bias.numel() != 0);
    const float* bp = has_bias ? bias.data_ptr<float>() : nullptr;

    if (Cin == 32 && Cout == 64 && K == 5 && stride == 1 && padding == 0 && dilation == 3) {
        const int threads = 256;
        long long total = (long long)B * 8 * Lout;
        int blocks = (int)((total + threads - 1) / threads);
        deconv1d_fixed_vec8_kernel<<<blocks, threads>>>(x.data_ptr<float>(), weight.data_ptr<float>(), bp,
                                                        y.data_ptr<float>(), B, L, Lout, has_bias);
    } else {
        const int threads = 256;
        long long total = (long long)B * Cout * Lout;
        int blocks = (int)((total + threads - 1) / threads);

                                                     y.data_ptr<float>(), B, Cin, L, Cout, K, Lout,

    return y;
}

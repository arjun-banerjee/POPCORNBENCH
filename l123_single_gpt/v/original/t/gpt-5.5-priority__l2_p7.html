import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies ReLU, LeakyReLU, GELU, Sigmoid activations, and bias in sequence.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape)) 

    def forward(self, x):
        x = self.conv(x)
        x = torch.relu(x)
        x = torch.nn.functional.leaky_relu(x, negative_slope=0.01)
        x = torch.nn.functional.gelu(x)
        x = torch.sigmoid(x)
        x = x + self.bias
        return x

#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void post_kernel(float* __restrict__ x,
                            const float* __restrict__ bias,
                            long long n_elem,
                            int channels,
                            int spatial) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    const float inv_sqrt2 = 0.70710678118654752440f;

    for (; idx < n_elem; idx += stride) {
        float v = x[idx];
        // ReLU followed by leaky_relu(negative_slope=0.01) is just ReLU.
        v = v > 0.0f ? v : 0.0f;
        // Exact GELU, matching torch.nn.functional.gelu default (approximate='none').
        float g = 0.5f * v * (1.0f + erff(v * inv_sqrt2));
        float s = 1.0f / (1.0f + expf(-g));
        int c = (int)((idx / spatial) % channels);
        x[idx] = s + bias[c];
    }
}

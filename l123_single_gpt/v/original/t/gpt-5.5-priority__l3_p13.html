        act[sr * CIN_FIXED + ci] = av;
    }

    // Load a 32x32 weight tile, w is [Cout, 32, 1, 1].
    for (int k = tid; k < CO_TILE * CIN_FIXED; k += SP_TILE * CO_TILE) {
        int lco = k / CIN_FIXED;
        int ci = k - lco * CIN_FIXED;
        int co = co_base + lco;
        wt[lco * CIN_FIXED + ci] = (co < Cout) ? w[co * CIN_FIXED + ci] : 0.0f;
    }
    __syncthreads();

    int s = spatial_base + tx;
    int co = co_base + ty;
    if (s < S && co < Cout) {
        float sum = 0.0f;
        #pragma unroll
        for (int ci = 0; ci < CIN_FIXED; ++ci) {
            sum = fmaf(act[tx * CIN_FIXED + ci], wt[ty * CIN_FIXED + ci], sum);
        }
        int n = s / OHW;
        int r = s - n * OHW;
        out[((long long)n * Cout + co) * (long long)OHW + r] = sum;
    }
}

torch::Tensor trans_forward_cuda(torch::Tensor x,
                                 torch::Tensor bn_weight,
                                 torch::Tensor bn_bias,
                                 torch::Tensor running_mean,
                                 torch::Tensor running_var,
                                 torch::Tensor conv_weight,
                                 bool training,
                                 double momentum,
                                 double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 supported");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous NCHW");
    TORCH_CHECK(conv_weight.is_contiguous(), "conv weight must be contiguous");
    int N = (int)x.size(0);
    int C = (int)x.size(1);
    int H = (int)x.size(2);
    int W = (int)x.size(3);
    int Cout = (int)conv_weight.size(0);
    TORCH_CHECK(C == CIN_FIXED, "this optimized kernel expects 32 input channels");
    TORCH_CHECK((H % 2) == 0 && (W % 2) == 0, "H and W must be even");

    c10::cuda::CUDAGuard device_guard(x.device());
    auto out = torch::empty({N, Cout, H / 2, W / 2}, x.options());
    auto scale = torch::empty({C}, x.options());
    auto shift = torch::empty({C}, x.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (training) {
        auto dopt = x.options().dtype(torch::kFloat64);
        auto psum = torch::empty({C, BPC}, dopt);
        auto psq = torch::empty({C, BPC}, dopt);
        dim3 rg(C, BPC);
        reduce_sum_sumsq_kernel<<<rg, 256, 0, stream>>>(x.data_ptr<float>(), psum.data_ptr<double>(), psq.data_ptr<double>(), N, C, H, W);
        train_scale_shift_kernel<<<C, 256, 0, stream>>>(psum.data_ptr<double>(), psq.data_ptr<double>(),
            bn_weight.data_ptr<float>(), bn_bias.data_ptr<float>(), running_mean.data_ptr<float>(), running_var.data_ptr<float>(),
            scale.data_ptr<float>(), shift.data_ptr<float>(), N, H, W, (float)momentum, (float)eps, C);
    } else {
        eval_scale_shift_kernel<<<1, 64, 0, stream>>>(bn_weight.data_ptr<float>(), bn_bias.data_ptr<float>(),
            running_mean.data_ptr<float>(), running_var.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), (float)eps, C);
    }

    int OH = H / 2, OW = W / 2;
    int S = N * OH * OW;
    dim3 block(SP_TILE, CO_TILE);
    dim3 grid((S + SP_TILE - 1) / SP_TILE, (Cout + CO_TILE - 1) / CO_TILE);
    fused_pool_conv_kernel<<<grid, block, 0, stream>>>(x.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(),
        conv_weight.data_ptr<float>(), out.data_ptr<float>(), N, H, W, Cout);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return out;
}

class ModelNew(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super(ModelNew, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2),
        )

    def forward(self, x):
        bn = self.transition[0]
        conv = self.transition[2]
        if self.training and bn.track_running_stats:
            bn.num_batches_tracked.add_(1)
        mom = 0.1 if bn.momentum is None else float(bn.momentum)
        return _ext.trans_forward(x, bn.weight, bn.bias, bn.running_mean, bn.running_var,
                                  conv.weight, bool(self.training), mom, float(bn.eps))

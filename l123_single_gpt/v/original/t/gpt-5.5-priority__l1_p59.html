        stride (int, optional): Stride of the convolution. Defaults to 1.
        padding (int, optional): Padding applied to the input. Defaults to 0.
        dilation (int, optional): Spacing between kernel elements. Defaults to 1.
        groups (int, optional): Number of blocked connections from input channels to output channels. Defaults to 1.
        bias (bool, optional): If `True`, adds a learnable bias to the output. Defaults to `False`.

        """
        Performs the 3D convolution.

        Args:

__global__ void conv3d_k3c3_oc8_kernel(const float* __restrict__ x,
                                       const float* __restrict__ w,
                                       const float* __restrict__ b,
                                       float* __restrict__ y,
                                       int total, int N, int H, int W, int D,
                                       int O, int Hout, int Wout, int inner,
                                       bool has_bias) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int stride_grid = blockDim.x * gridDim.x;
    for (int idx = tid; idx < total; idx += stride_grid) {
        int spatial = idx % inner;
        int t0 = idx / inner;
        int og = t0 & 7;          // O/8 == 8 in the target case (O=64)
        int n = t0 >> 3;
        int oc0 = og << 3;

        int d = spatial % D;
        int t = spatial / D;
        int ow = t % Wout;
        int oh = t / Wout;

        float a0 = has_bias ? b[oc0 + 0] : 0.0f;
        float a1 = has_bias ? b[oc0 + 1] : 0.0f;
        float a2 = has_bias ? b[oc0 + 2] : 0.0f;
        float a3 = has_bias ? b[oc0 + 3] : 0.0f;

__global__ void conv3d_generic_kernel(const float* __restrict__ x,
                                      const float* __restrict__ w,
                                      const float* __restrict__ b,
                                      float* __restrict__ y,
                                      int total, int N, int C, int H, int W, int D,
                                      int O, int Cg, int kH, int kW, int kD,
                                      int Hout, int Wout, int Dout,
                                      int stride, int pad, int dil, int groups,
                                      bool has_bias) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int grid_stride = blockDim.x * gridDim.x;
    for (int idx = tid; idx < total; idx += grid_stride) {
        int od = idx % Dout;
        int q = idx / Dout;
        int ow = q % Wout; q /= Wout;
        int oh = q % Hout; q /= Hout;
        int oc = q % O;
        int n = q / O;
        int g = oc / (O / groups);
        int cbase = g * Cg;
        float acc = has_bias ? b[oc] : 0.0f;
        for (int ci = 0; ci < Cg; ++ci) {
            int ic = cbase + ci;
            for (int kh = 0; kh < kH; ++kh) {
                int ih = oh * stride - pad + kh * dil;
                if ((unsigned)ih >= (unsigned)H) continue;
                for (int kw = 0; kw < kW; ++kw) {
                    int iw = ow * stride - pad + kw * dil;
                    if ((unsigned)iw >= (unsigned)W) continue;
                    for (int kd = 0; kd < kD; ++kd) {
                        int id = od * stride - pad + kd * dil;
                        if ((unsigned)id >= (unsigned)D) continue;
                        float xv = x[((((n * C + ic) * H + ih) * W + iw) * D + id)];
                        float wv = w[(((((oc * Cg + ci) * kH + kh) * kW + kw) * kD + kd))];
                        acc = fmaf(xv, wv, acc);
                    }
                }
            }
        }
        y[idx] = acc;
    }
}

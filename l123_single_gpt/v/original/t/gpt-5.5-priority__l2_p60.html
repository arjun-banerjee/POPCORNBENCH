import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D transposed convolution, applies Swish activation, 
    group normalization, and then HardSwish activation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, eps, bias=True):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.group_norm = nn.GroupNorm(num_groups=groups, num_channels=out_channels, eps=eps)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = torch.sigmoid(x) * x  # Swish activation
        x = self.group_norm(x)
        x = torch.nn.functional.hardswish(x)  # HardSwish activation
        return x

__global__ void group_reduce_kernel(
    const float* __restrict__ y,
    float* __restrict__ mean,
    float* __restrict__ invstd,
    int N, int C, int G, int S, float eps)
{
    extern __shared__ float sh[];
    float* shsum = sh;
    float* shsq  = sh + blockDim.x;

    int ng = blockIdx.x;
    int g = ng % G;
    int n = ng / G;
    int cpg = C / G;
    int L = cpg * S;
    int base = (n * C + g * cpg) * S;

    float sum = 0.0f;
    float sq = 0.0f;
    for (int i = threadIdx.x; i < L; i += blockDim.x) {
        float v = y[base + i];
        sum += v;
        sq = fmaf(v, v, sq);
    }
    shsum[threadIdx.x] = sum;
    shsq[threadIdx.x] = sq;
    __syncthreads();

    for (int off = blockDim.x >> 1; off > 0; off >>= 1) {
        if (threadIdx.x < off) {
            shsum[threadIdx.x] += shsum[threadIdx.x + off];
            shsq[threadIdx.x]  += shsq[threadIdx.x + off];
        }
        __syncthreads();

        float var = shsq[0] / (float)L - m * m;
        if (var < 0.0f) var = 0.0f;
        mean[ng] = m;
        invstd[ng] = rsqrtf(var + eps);
    }
}

__global__ void groupnorm_hardswish_kernel(
    float* __restrict__ y,
    const float* __restrict__ mean,
    const float* __restrict__ invstd,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    int N, int C, int G, int S)
{
    int s = blockIdx.x * blockDim.x + threadIdx.x;
    if (s >= S) return;
    int nc = blockIdx.y;

    int n = nc / C;

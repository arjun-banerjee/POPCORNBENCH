import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cublas_v2.h>

static inline void cublas_check(cublasStatus_t status) {
    TORCH_CHECK(status == CUBLAS_STATUS_SUCCESS, "cuBLAS error: ", (int)status);
}

torch::Tensor matmul3d_forward(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "A and B must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == at::kFloat && B.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(A.dim() == 3 && B.dim() == 2, "expected A (N,M,K), B (K,L)");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "A and B must be contiguous");

    const int64_t N = A.size(0);
    const int64_t M = A.size(1);
    const int64_t K = A.size(2);
    TORCH_CHECK(B.size(0) == K, "dimension mismatch");
    const int64_t L = B.size(1);
    const int64_t P = N * M;
    TORCH_CHECK(P <= INT_MAX && K <= INT_MAX && L <= INT_MAX, "matrix too large for cublasSgemm int dimensions");

    auto C = torch::empty({N, M, L}, A.options());
    c10::cuda::CUDAGuard device_guard(A.device());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();

    // A, B, C are row-major.  View their storage as column-major transposes:
    //   C(row-major P x L) = A(P x K) * B(K x L)
    // is computed as
    //   C^T(column-major L x P) = B^T(L x K) * A^T(K x P).
    const float alpha = 1.0f;
    const float beta  = 0.0f;
    cublas_check(cublasSgemm(handle,
                             CUBLAS_OP_N, CUBLAS_OP_N,
                             (int)L, (int)P, (int)K,
                             &alpha,
                             B.data_ptr<float>(), (int)L,
                             A.data_ptr<float>(), (int)K,
                             &beta,
                             C.data_ptr<float>(), (int)L));
    return C;
}
'''

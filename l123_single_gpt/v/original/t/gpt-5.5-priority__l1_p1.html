import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <ATen/cuda/CUDABlas.h>

// Direct cuBLAS SGEMM wrapper for row-major C = A @ B.
// PyTorch tensors are row-major contiguous; interpreting their storage as
// column-major gives transposed matrices, so compute C^T = B^T * A^T.
torch::Tensor matmul_forward(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == at::kFloat && B.scalar_type() == at::kFloat, "only fp32 supported");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be 2D");
    TORCH_CHECK(A.size(0) == A.size(1) && B.size(0) == B.size(1) && A.size(1) == B.size(0), "square compatible matrices required");
    // Benchmark inputs are contiguous. Make robust without falling back to torch matmul.
    if (!A.is_contiguous()) A = A.contiguous();
    if (!B.is_contiguous()) B = B.contiguous();

    const int64_t n64 = A.size(0);
    TORCH_CHECK(n64 <= INT_MAX, "matrix too large");
    const int n = static_cast<int>(n64);

    auto C = torch::empty({n64, n64}, A.options());

    const float alpha = 1.0f;
    const float beta = 0.0f;

    auto handle = at::cuda::getCurrentCUDABlasHandle();
    // cublas uses column-major: C_col = B_col * A_col corresponds to C_row = A_row * B_row.
    cublasStatus_t stat = cublasSgemm(
        handle,
        CUBLAS_OP_N, CUBLAS_OP_N,
        n, n, n,
        &alpha,
        B.data_ptr<float>(), n,
        A.data_ptr<float>(), n,
        &beta,
        C.data_ptr<float>(), n);
    TORCH_CHECK(stat == CUBLAS_STATUS_SUCCESS, "cublasSgemm failed");
    return C;
}
'''

_ext = load_inline(
    name='direct_cublas_matmul_ext_v1',
    cpp_sources=[_cpp_src],
    functions=['matmul_forward'],
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3'],
    verbose=False,
)

    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:

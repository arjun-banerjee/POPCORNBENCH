import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs a GEMM, BiasAdd, Hardtanh, Mish, and GroupNorm operations in sequence.
    """
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.hardtanh = nn.Hardtanh()
        self.mish = nn.Mish()
        self.groupnorm = nn.GroupNorm(num_groups=num_groups, num_channels=out_features)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.gemm(x)
        x = x + self.bias
        x = self.hardtanh(x)
        x = self.mish(x)
        x = self.groupnorm(x)
        return x

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <ATen/ops/linear.h>

void postprocess_launch(torch::Tensor y, torch::Tensor bias, torch::Tensor gn_weight, torch::Tensor gn_bias, int64_t num_groups);

torch::Tensor fused_forward(torch::Tensor x,
                           torch::Tensor weight,
                           torch::Tensor lin_bias,
                           torch::Tensor bias,
                           torch::Tensor gn_weight,
                           torch::Tensor gn_bias,
                           int64_t num_groups) {

    // fuse all remaining elementwise ops and per-(N,group) normalization below.

    postprocess_launch(y, bias, gn_weight, gn_bias, num_groups);
    return y;
}

_cuda_src = r'''
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

static __device__ __forceinline__ float warp_sum(float v) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, offset);
    }
    return v;
}

__global__ void postprocess_gn32_kernel(float* __restrict__ y,
                                        const float* __restrict__ bias,
                                        const float* __restrict__ gn_weight,
                                        const float* __restrict__ gn_bias,
                                        int N, int C, int G) {
    // One warp computes one group of one batch item.  For the benchmark
    // C=8192,G=256, so group size is exactly 32.
    const int n = blockIdx.y;
    const int g = blockIdx.x;
    const int lane = threadIdx.x;
    const int group_size = C / G;
    const int c = g * group_size + lane;

    float v = 0.0f;
    if (lane < group_size) {
        const int idx = n * C + c;
        v = y[idx] + bias[c];
        // Hardtanh default range [-1, 1]
        v = fminf(1.0f, fmaxf(-1.0f, v));
        // Mish: x * tanh(softplus(x)); x is already in [-1,1].
        v = v * tanhf(log1pf(expf(v)));
    }

    float sum = warp_sum(v);
    float sum2 = warp_sum(v * v);
    float mean = sum * (1.0f / (float)group_size);
    float var = sum2 * (1.0f / (float)group_size) - mean * mean;
    var = fmaxf(var, 0.0f);
    float inv_std = rsqrtf(var + 1.0e-5f);

    if (lane < group_size) {
        y[n * C + c] = (v - mean) * inv_std * gn_weight[c] + gn_bias[c];
    }
}

__global__ void postprocess_gn_generic_kernel(float* __restrict__ y,
                                              const float* __restrict__ bias,
                                              const float* __restrict__ gn_weight,
                                              const float* __restrict__ gn_bias,
                                              int N, int C, int G) {
    const int n = blockIdx.y;
    const int g = blockIdx.x;
    const int tid = threadIdx.x;
    const int group_size = C / G;
    const int base = n * C + g * group_size;

    float sum = 0.0f;
    float sum2 = 0.0f;
    for (int i = tid; i < group_size; i += blockDim.x) {
        int c = g * group_size + i;
        float v = y[base + i] + bias[c];
        v = fminf(1.0f, fmaxf(-1.0f, v));
        v = v * tanhf(log1pf(expf(v)));
        sum += v;
        sum2 += v * v;
    }

    // reduce within block (blockDim <= 256 expected)
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
        sum2 += __shfl_down_sync(0xffffffff, sum2, offset);
    }
    int warp = tid >> 5;
    int lane = tid & 31;
    if (lane == 0) { smem[warp] = sum; smem[warp + 32] = sum2; }
    __syncthreads();
    if (warp == 0) {
        sum = (lane < ((blockDim.x + 31) >> 5)) ? smem[lane] : 0.0f;
        sum2 = (lane < ((blockDim.x + 31) >> 5)) ? smem[lane + 32] : 0.0f;

}

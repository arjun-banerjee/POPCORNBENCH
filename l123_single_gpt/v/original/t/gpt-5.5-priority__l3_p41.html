class Model(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=3, bias=True, batch_first=False):
        """
        :param input_size: The number of expected features in the input x
        :param hidden_size: The number of features in the hidden state h
        :param num_layers: Number of recurrent layers (default: 1)
        :param bias: If False, then the layer does not use bias weights b_ih and b_hh (default: True)
        :param batch_first: If True, then the input and output tensors are provided as (batch, seq, feature) (default: False)
        """
        super(Model, self).__init__()
        
        self.gru = nn.GRU(input_size, hidden_size, num_layers, bias, batch_first, dropout=0, bidirectional=True)
        self.h0 = torch.randn((num_layers * 2, batch_size, hidden_size))
    
    def forward(self, x,h0):
        """
        :param x: The input tensor, shape (seq_len, batch_size, input_size) if batch_first=False, otherwise (batch_size, seq_len, input_size)
        :param h_0: The initial hidden state for the input sequence, shape (num_layers * num_directions, batch_size, hidden_size) (default: None)
        :return: output, h_n
            - output: The output features (h_t) from the last layer of the GRU, for each t, shape (seq_len, batch_size, num_directions * hidden_size) if batch_first=False, otherwise (batch_size, seq_len, num_directions * hidden_size)
            - h_n: The hidden state for t = seq_len, shape (num_layers * num_directions, batch_size, hidden_size)
        """
        output, h_n = self.gru(x, h0)
        return output

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

__device__ __forceinline__ float sigmoidf_dev(float x) {
    return 1.0f / (1.0f + expf(-x));
}

// One CTA owns one batch element and one direction of one layer.  This is a
// persistent sequential kernel over time; thread j computes hidden component j.
__global__ void gru_layer_dir_kernel(
    const float* __restrict__ inp,
    const float* __restrict__ h0,
    const float* __restrict__ wihT,   // [I, 3H], transposed/contiguous
    const float* __restrict__ whhT,   // [H, 3H], transposed/contiguous
    const float* __restrict__ bih,    // [3H]
    const float* __restrict__ bhh,    // [3H]
    float* __restrict__ out,          // [T, B, 2H] seq-major
    int T, int B, int I, int H,
    int layer_dir_index,
    int dir,
    bool inp_batch_first)
{
    extern __shared__ float hsh[];
    const int b = blockIdx.x;
    const int j = threadIdx.x;
    if (j >= H || b >= B) return;

    float h = h0[(layer_dir_index * B + b) * H + j];
    hsh[j] = h;
    __syncthreads();

    const int threeH = 3 * H;
    for (int s = 0; s < T; ++s) {
        const int t = (dir == 0) ? s : (T - 1 - s);

        float xr = bih[j];
        float xz = bih[H + j];
        float xn = bih[2 * H + j];

        // input contribution.  For layer 0 inp can be batch-major; internal
        // layer buffers are always seq-major.
        const float* xbase;
        if (inp_batch_first) xbase = inp + (b * T + t) * I;
        else                 xbase = inp + (t * B + b) * I;

        for (int k = 0; k < I; ++k) {
            float xv = xbase[k];
            const float* wk = wihT + k * threeH;
            xr += xv * wk[j];
            xz += xv * wk[H + j];
            xn += xv * wk[2 * H + j];
        }

        float hr = bhh[j];
        float hz = bhh[H + j];
        float hn = bhh[2 * H + j];
        for (int k = 0; k < H; ++k) {
            float hv = hsh[k];
            const float* wk = whhT + k * threeH;
            hr += hv * wk[j];
            hz += hv * wk[H + j];
            hn += hv * wk[2 * H + j];
        }

        float r = sigmoidf_dev(xr + hr);
        float z = sigmoidf_dev(xz + hz);
        float n = tanhf(xn + r * hn);
        float hnew = (1.0f - z) * n + z * h;

        out[(t * B + b) * (2 * H) + dir * H + j] = hnew;
        h = hnew;
        hsh[j] = hnew;
        __syncthreads();
    }
}

torch::Tensor gru_forward(torch::Tensor x, torch::Tensor h0, std::vector<torch::Tensor> weights, int64_t num_layers, bool batch_first) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(h0.is_cuda(), "h0 must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");

    const int64_t T64 = batch_first ? x.size(1) : x.size(0);
    const int64_t B64 = batch_first ? x.size(0) : x.size(1);
    const int64_t input0_64 = x.size(2);
    const int64_t H64 = h0.size(2);
    const int T = (int)T64, B = (int)B64, input0 = (int)input0_64, H = (int)H64;
    TORCH_CHECK(H <= 1024, "hidden too large for this kernel");
    TORCH_CHECK((int64_t)weights.size() >= num_layers * 8, "bad weight list");

    torch::Tensor prev = x;
    torch::Tensor curr;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    dim3 block(H);
    dim3 grid(B);
    size_t shmem = (size_t)H * sizeof(float);

    for (int l = 0; l < (int)num_layers; ++l) {
        curr = torch::empty({T64, B64, 2 * H64}, x.options());
        int I = (l == 0) ? input0 : (2 * H);
        bool inp_bf = (l == 0) ? batch_first : false;
        for (int dir = 0; dir < 2; ++dir) {
            int base = l * 8 + dir * 4;
            int layer_dir = l * 2 + dir;
            gru_layer_dir_kernel<<<grid, block, shmem, stream>>>(
                prev.data_ptr<float>(), h0.data_ptr<float>(),
                weights[base + 0].data_ptr<float>(), weights[base + 1].data_ptr<float>(),
                weights[base + 2].data_ptr<float>(), weights[base + 3].data_ptr<float>(),
                curr.data_ptr<float>(), T, B, I, H, layer_dir, dir, inp_bf);
            C10_CUDA_KERNEL_LAUNCH_CHECK();
        }
        prev = curr;
    }
    if (batch_first) {
        return curr.permute({1,0,2}).contiguous();
    }
    return curr;
}

class ModelNew(nn.Module):

        super(ModelNew, self).__init__()
        self.gru = nn.GRU(input_size, hidden_size, num_layers, bias, batch_first, dropout=0, bidirectional=True)
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.bias = bias
        self.batch_first = batch_first
        self._cache_versions = None
        self._cached_weights = None
        # Reference has this non-parameter attribute; keep it when batch_size is defined.
        try:
            bs = globals().get('batch_size', 10)
            self.h0 = torch.randn((num_layers * 2, bs, hidden_size))
        except Exception:
            pass

    def _make_weight_cache(self):
        ws = []
        versions = []
        for l in range(self.num_layers):
            for suffix in ("", "_reverse"):
                wih = getattr(self.gru, f"weight_ih_l{l}{suffix}")
                whh = getattr(self.gru, f"weight_hh_l{l}{suffix}")
                versions.append(wih._version); versions.append(whh._version)
                ws.append(wih.t().contiguous())
                ws.append(whh.t().contiguous())
                if self.bias:
                    bih = getattr(self.gru, f"bias_ih_l{l}{suffix}")
                    bhh = getattr(self.gru, f"bias_hh_l{l}{suffix}")
                    versions.append(bih._version); versions.append(bhh._version)
                    ws.append(bih.contiguous())
                    ws.append(bhh.contiguous())
                else:
                    # The benchmark uses bias=True; this path is only for API completeness.
                    device = wih.device
                    z = torch.zeros(3 * self.hidden_size, device=device, dtype=wih.dtype)
                    ws.append(z); ws.append(z)
        self._cache_versions = tuple(versions)
        self._cached_weights = ws

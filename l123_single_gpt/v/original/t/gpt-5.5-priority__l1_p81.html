__global__ void fill_bias_kernel(float* __restrict__ out, const float* __restrict__ bias,
                                 long long total, int Cout, int HWout) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long step = (long long)blockDim.x * gridDim.x;
    for (; idx < total; idx += step) {
        int oc = (idx / HWout) % Cout;
        out[idx] = bias[oc];
    }
}

// Specialized for the benchmark shape: Cin=32, Cout=64, K=3, stride=5, pad=1, dilation=2.
// One block computes all 64*3*3 non-zero values generated by one input pixel.
__global__ void deconv2d_s5p1d2_k3_c32_c64_kernel(const float* __restrict__ x,
                                                   const float* __restrict__ w,
                                                   float* __restrict__ out,
                                                   int N, int H, int W,
                                                   int Hout, int Wout,
                                                   bool has_bias) {
    int sp = blockIdx.x;
    int iw = sp % W;
    int t = sp / W;
    int ih = t % H;
    int n  = t / H;

    __shared__ float sx[32];
    if (threadIdx.x < 32) {
        sx[threadIdx.x] = x[((n * 32 + threadIdx.x) * H + ih) * W + iw];
    }
    __syncthreads();

    // idx = oc*9 + kh*3 + kw, matching contiguous weight layout for each input channel.
    for (int idx = threadIdx.x; idx < 576; idx += blockDim.x) {

                                        int Hout, int Wout,
                                        bool has_bias, bool use_atomic) {

    int iw = sp % W;

    int n = tmp / H;

    extern __shared__ float sx[];
    for (int c = threadIdx.x; c < Cin; c += blockDim.x) {
        sx[c] = x[((n * Cin + c) * H + ih) * W + iw];

torch::Tensor conv_transpose_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias,
                                     int64_t stride64, int64_t padding64, int64_t dilation64) {
    const int N = (int)x.size(0);
    const int Cin = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int Cout = (int)weight.size(1);

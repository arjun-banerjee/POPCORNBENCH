    switch (status) {
        case CUBLAS_STATUS_SUCCESS: return "CUBLAS_STATUS_SUCCESS";
        case CUBLAS_STATUS_NOT_INITIALIZED: return "CUBLAS_STATUS_NOT_INITIALIZED";

at::Tensor matmul_at_b(at::Tensor A, at::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "A and B must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == at::kFloat && B.scalar_type() == at::kFloat, "only fp32 tensors are supported");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "A and B must be 2D");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "A and B must be contiguous");
    TORCH_CHECK(A.device() == B.device(), "A and B must be on the same device");

    const int64_t K64 = A.size(0);
    const int64_t M64 = A.size(1);
    TORCH_CHECK(B.size(0) == K64, "inner dimensions must match");
    const int64_t N64 = B.size(1);
    TORCH_CHECK(K64 <= std::numeric_limits<int>::max() && M64 <= std::numeric_limits<int>::max() && N64 <= std::numeric_limits<int>::max(), "matrix too large for cuBLAS int dimensions");

    const int K = static_cast<int>(K64);
    const int M = static_cast<int>(M64);
    const int N = static_cast<int>(N64);

    c10::cuda::CUDAGuard device_guard(A.device());
    at::Tensor C = at::empty({M64, N64}, A.options());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    cublasStatus_t st;
    st = cublasSetStream(handle, at::cuda::getCurrentCUDAStream());
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSetStream failed: ", cublas_status_string(st));
    st = cublasSetPointerMode(handle, CUBLAS_POINTER_MODE_HOST);
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSetPointerMode failed: ", cublas_status_string(st));

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // PyTorch tensors are row-major.  Treating the same memory as column-major,
    // B is an (N x K) column-major matrix and A is an (M x K) column-major matrix.
    // The row-major output C(M x N) is the same buffer as a column-major (N x M)
    // matrix, so compute: C_col = B_col * A_col^T.
    st = cublasGemmEx(
        handle,
        CUBLAS_OP_N, CUBLAS_OP_T,
        N, M, K,
        &alpha,
        static_cast<const void*>(B.data_ptr<float>()), CUDA_R_32F, N,
        static_cast<const void*>(A.data_ptr<float>()), CUDA_R_32F, M,
        &beta,
        static_cast<void*>(C.data_ptr<float>()), CUDA_R_32F, N,
        CUBLAS_COMPUTE_32F_PEDANTIC,
        CUBLAS_GEMM_DEFAULT);

    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasGemmEx failed: ", cublas_status_string(st));
    return C;
}
"""

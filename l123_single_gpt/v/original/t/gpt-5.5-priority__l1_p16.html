import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor matmul_at_b(torch::Tensor A, torch::Tensor B);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cublas_v2.h>

static inline void check_cublas_status(cublasStatus_t status) {
    TORCH_CHECK(status == CUBLAS_STATUS_SUCCESS, "cuBLAS error: ", (int)status);
}

torch::Tensor matmul_at_b(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == torch::kFloat32 && B.scalar_type() == torch::kFloat32,
                "only float32 tensors are supported");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be 2D");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "inputs must be contiguous");

    const int64_t K64 = A.size(0);
    const int64_t M64 = A.size(1);
    TORCH_CHECK(B.size(0) == K64, "inner dimensions must match");
    const int64_t N64 = B.size(1);
    TORCH_CHECK(K64 <= INT_MAX && M64 <= INT_MAX && N64 <= INT_MAX, "matrix too large");

    const int K = (int)K64;
    const int M = (int)M64;
    const int N = (int)N64;

    auto C = torch::empty({M64, N64}, A.options());
    if (M == 0 || N == 0) return C;

    c10::cuda::CUDAGuard device_guard(A.device());
    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // Row-major C(M,N) is the same memory as column-major Ct(N,M).
    // A(K,M) row-major is column-major At(M,K), B(K,N) row-major is Bt(N,K).
    // Hence Ct = Bt * At^T.
    check_cublas_status(cublasSgemm(handle,
                                    CUBLAS_OP_N, CUBLAS_OP_T,
                                    N, M, K,
                                    &alpha,
                                    B.data_ptr<float>(), N,
                                    A.data_ptr<float>(), M,
                                    &beta,
                                    C.data_ptr<float>(), N));
    return C;
}
"""

_ext = load_inline(
    name="at_b_cublas_inline_ext",
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=["matmul_at_b"],
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3"],
    extra_ldflags=["-lcublas"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return _ext.matmul_at_b(A, B)

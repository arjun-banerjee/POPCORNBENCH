#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

#ifndef WARP_SIZE
#define WARP_SIZE 32
#endif

template<int BLOCK>
__device__ __forceinline__ float block_reduce_max(float v) {
    __shared__ float smem[BLOCK / WARP_SIZE];
    int lane = threadIdx.x & (WARP_SIZE - 1);
    int wid  = threadIdx.x >> 5;
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) v = fmaxf(v, __shfl_down_sync(0xffffffff, v, off));
    if (lane == 0) smem[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (BLOCK / WARP_SIZE)) ? smem[lane] : -FLT_MAX;
    if (wid == 0) {
        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) v = fmaxf(v, __shfl_down_sync(0xffffffff, v, off));
    }
    return __shfl_sync(0xffffffff, v, 0);
}

template<int BLOCK>
__device__ __forceinline__ float block_reduce_sum(float v) {
    __shared__ float smem[BLOCK / WARP_SIZE];
    int lane = threadIdx.x & (WARP_SIZE - 1);
    int wid  = threadIdx.x >> 5;
    #pragma unroll

template<int BLOCK>
__global__ void logsoftmax_dim1_kernel(const float* __restrict__ x, float* __restrict__ y,
                                       int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;
    const float* __restrict__ xr = x + (long long)row * cols;
    float* __restrict__ yr = y + (long long)row * cols;

    float vmax = -FLT_MAX;
    // vector-friendly strided loop. cols is large; use all threads in one CTA per row.
    for (int c = threadIdx.x; c < cols; c += BLOCK) {
        vmax = fmaxf(vmax, xr[c]);
    }
    vmax = block_reduce_max<BLOCK>(vmax);

    for (int c = threadIdx.x; c < cols; c += BLOCK) {
        sum += expf(xr[c] - vmax);

    sum = block_reduce_sum<BLOCK>(sum);
    float lsum = logf(sum) + vmax;

    for (int c = threadIdx.x; c < cols; c += BLOCK) {
        yr[c] = xr[c] - lsum;
    }
}

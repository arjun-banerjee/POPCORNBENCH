import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cublas_v2.h>
#include <mutex>

static cublasHandle_t g_handle = nullptr;
static std::once_flag g_handle_once;

static void create_handle_once() {
    CUBLAS_CHECK(cublasCreate(&g_handle));
    // Use standard fp32 math for numerical agreement with torch fp32 matmul.
    CUBLAS_CHECK(cublasSetMathMode(g_handle, CUBLAS_DEFAULT_MATH));
}

static cublasHandle_t get_handle() {
    std::call_once(g_handle_once, create_handle_once);
    return g_handle;
}

void init_cublas() {
    c10::cuda::CUDAGuard device_guard(0);
    (void)get_handle();
}

torch::Tensor tri_matmul(torch::Tensor A, torch::Tensor B) {
    const int64_t n64 = A.size(0);
    const int n = static_cast<int>(n64);

    c10::cuda::CUDAGuard device_guard(A.device());
    auto C = torch::empty_like(A);

    cublasHandle_t handle = get_handle();
    CUBLAS_CHECK(cublasSetStream(handle, at::cuda::getCurrentCUDAStream(A.get_device())));

    const float alpha = 1.0f;

    // PyTorch tensors are row-major.  Interpreting the same storage as
    // column-major transposes each matrix.  For C_row = A_row * B_row,
    // we compute C_col = B_col * A_col = B_row^T * A_row^T.
    // A_col is upper triangular because A_row is lower triangular.
    CUBLAS_CHECK(cublasStrmm(
        handle,
        CUBLAS_SIDE_RIGHT,
        CUBLAS_FILL_MODE_UPPER,
        CUBLAS_OP_N,
        CUBLAS_DIAG_NON_UNIT,
        n, n,
        &alpha,
        A.data_ptr<float>(), n,
        B.data_ptr<float>(), n,
        C.data_ptr<float>(), n));

    return C;
}
'''

_ext = load_inline(
    name='lower_triangular_trmm_ext_v1',

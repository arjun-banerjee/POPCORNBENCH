import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D transposed convolution, followed by a sum, 
    a residual add, a multiplication, and another residual add.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        original_x = x.clone().detach()
        x = x + self.bias
        x = x + original_x
        x = x * original_x
        x = x + original_x
        return x

import torch.nn.functional as F

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>

__global__ void fused_residual_bias_kernel(float* __restrict__ y,
                                           const float* __restrict__ bias,
                                           long long total,
                                           int C,
                                           long long spatial) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        long long c = (idx / spatial) % C;
        float v = y[idx];
        // Reference after transposed convolution y:
        // ((y + custom_bias) + y) * y + y = 2*y*y + (custom_bias + 1)*y
        y[idx] = fmaf(2.0f * v, v, (bias[c] + 1.0f) * v);
    }
}

torch::Tensor fused_residual_bias_forward(torch::Tensor y, torch::Tensor bias) {
    const c10::cuda::CUDAGuard device_guard(y.device());
    y = y.contiguous();
    bias = bias.contiguous();

    """
    Same module interface as the reference.  The expensive ConvTranspose3d is
    executed by PyTorch/cuDNN, while the clone/add/add/mul/add tail is replaced
    by one custom CUDA epilogue kernel executed in-place on the convolution
    output.

            in_channels,

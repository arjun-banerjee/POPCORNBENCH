_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <c10/cuda/CUDAException.h>
#include <cuda_runtime.h>
#include <algorithm>
#include <cstdint>

__device__ __forceinline__ float warp_sum(float v) {
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, off);
    }
    return v;
}

__device__ __forceinline__ float block_sum(float v) {
    __shared__ float smem[32];
    int lane = threadIdx.x & 31;
    int wid  = threadIdx.x >> 5;
    v = warp_sum(v);
    if (lane == 0) smem[wid] = v;
    __syncthreads();
    v = (threadIdx.x < (blockDim.x >> 5)) ? smem[lane] : 0.0f;
    if (wid == 0) v = warp_sum(v);
    return v;
}

__device__ __forceinline__ float smooth_l1_val(float d) {
    float a = fabsf(d);
    return (a < 1.0f) ? (0.5f * d * d) : (a - 0.5f);
}

__device__ __forceinline__ void accum4(const float4 a, const float4 b, float &s) {
    float d0 = a.x - b.x;
    float d1 = a.y - b.y;
    float d2 = a.z - b.z;
    float d3 = a.w - b.w;
    s += smooth_l1_val(d0);
    s += smooth_l1_val(d1);
    s += smooth_l1_val(d2);
    s += smooth_l1_val(d3);
}

__global__ void smooth_l1_vec_kernel(const float* __restrict__ p,
                                     const float* __restrict__ t,
                                     float* __restrict__ out,
                                     int64_t n,
                                     float inv_n) {
    const int tid = threadIdx.x;
    const int64_t n4 = n >> 2;
    const float4* __restrict__ p4 = reinterpret_cast<const float4*>(p);
    const float4* __restrict__ t4 = reinterpret_cast<const float4*>(t);
    const int64_t stride4 = (int64_t)blockDim.x * gridDim.x;
    int64_t i = (int64_t)blockIdx.x * blockDim.x + tid;

    float s = 0.0f;

    // Unroll the grid-stride loop a little for better memory-level parallelism.
    for (; i + 3 * stride4 < n4; i += 4 * stride4) {
        accum4(p4[i],               t4[i],               s);
        accum4(p4[i + stride4],     t4[i + stride4],     s);
        accum4(p4[i + 2*stride4],   t4[i + 2*stride4],   s);
        accum4(p4[i + 3*stride4],   t4[i + 3*stride4],   s);
    }
    for (; i < n4; i += stride4) {
        accum4(p4[i], t4[i], s);
    }

    // Tail, for non-multiple-of-4 tensors.
    const int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (int64_t j = (n4 << 2) + (int64_t)blockIdx.x * blockDim.x + tid; j < n; j += stride) {
        s += smooth_l1_val(p[j] - t[j]);
    }

    s = block_sum(s);
    if (tid == 0) atomicAdd(out, s * inv_n);
}

__global__ void smooth_l1_scalar_kernel(const float* __restrict__ p,
                                        const float* __restrict__ t,
                                        float* __restrict__ out,
                                        int64_t n,
                                        float inv_n) {
    float s = 0.0f;
    const int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (int64_t i = (int64_t)blockIdx.x * blockDim.x + threadIdx.x; i < n; i += stride) {
        s += smooth_l1_val(p[i] - t[i]);
    }
    s = block_sum(s);
    if (threadIdx.x == 0) atomicAdd(out, s * inv_n);
}

    c10::cuda::CUDAGuard device_guard(predictions.device());

    auto p_contig = predictions.contiguous();
    auto t_contig = targets.contiguous();
    const int64_t n = p_contig.numel();

    auto out = torch::empty({}, p_contig.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    AT_CUDA_CHECK(cudaMemsetAsync(out.data_ptr<float>(), 0, sizeof(float), stream));

    if (n == 0) {
        return out;

    int dev = 0;
    cudaDeviceProp prop;
    AT_CUDA_CHECK(cudaGetDevice(&dev));
    AT_CUDA_CHECK(cudaGetDeviceProperties(&prop, dev));

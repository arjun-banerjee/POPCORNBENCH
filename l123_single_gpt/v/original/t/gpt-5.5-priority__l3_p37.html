import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        """
        Initialize the LSTM model.

        :param input_size: The number of expected features in the input `x`
        :param hidden_size: The number of features in the hidden state `h`
        :param num_layers: Number of recurrent layers
        :param output_size: The number of output features
        :param dropout: If non-zero, introduces a Dropout layer on the outputs of each LSTM layer except the last layer, with dropout probability equal to `dropout`
        """
        super(Model, self).__init__()
        # Initialize hidden state with random values
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = nn.Linear(hidden_size, output_size)
    
    def forward(self, x, h0, c0):
        """
        Forward pass through the LSTM model.

        :param x: The input tensor, shape (batch_size, sequence_length, input_size)
        :return: The output tensor, shape (batch_size, sequence_length, output_size)
        """
        
        # Forward propagate LSTM
        out, state = self.lstm(x, (h0, c0))  # out: tensor of shape (batch_size, seq_length, hidden_size)
        
        # Decode the hidden state of the last time step
        out = self.fc(out[:, -1, :])  # out: tensor of shape (batch_size, output_size)
        
        return state[1]

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ __forceinline__ float sigmoidf_lstm(float x) {
    return 1.0f / (1.0f + expf(-x));
}

__device__ __forceinline__ void grid_sync(unsigned int* counter, unsigned int* sense, int total_blocks) {
    __syncthreads();
    if (threadIdx.x == 0) {
        unsigned int my = atomicAdd(sense, 0u);
        __threadfence();
        unsigned int ticket = atomicAdd(counter, 1u);
        if (ticket == (unsigned int)(total_blocks - 1)) {
            *counter = 0u;
            __threadfence();
            atomicAdd(sense, 1u);
        } else {
            while (atomicAdd(sense, 0u) == my) { }
        }
    }
    __syncthreads();
}

__device__ __forceinline__ const float* sel_wih(int l,
    const float* w0,const float* w1,const float* w2,const float* w3,const float* w4,const float* w5) {
    return l==0?w0:(l==1?w1:(l==2?w2:(l==3?w3:(l==4?w4:w5))));
}
__device__ __forceinline__ const float* sel_whh(int l,
    const float* w0,const float* w1,const float* w2,const float* w3,const float* w4,const float* w5) {
    return l==0?w0:(l==1?w1:(l==2?w2:(l==3?w3:(l==4?w4:w5))));
}
__device__ __forceinline__ const float* sel_b(int l,
    const float* b0,const float* b1,const float* b2,const float* b3,const float* b4,const float* b5) {
    return l==0?b0:(l==1?b1:(l==2?b2:(l==3?b3:(l==4?b4:b5))));
}

__device__ __forceinline__ float warp_sum(float v) {
    v += __shfl_down_sync(0xffffffff, v, 16);
    v += __shfl_down_sync(0xffffffff, v, 8);
    v += __shfl_down_sync(0xffffffff, v, 4);
    v += __shfl_down_sync(0xffffffff, v, 2);
    v += __shfl_down_sync(0xffffffff, v, 1);
    return v;
}

__global__ void lstm_kernel(const float* __restrict__ x,
    const float* __restrict__ h0, const float* __restrict__ c0,
    float* __restrict__ hprev, float* __restrict__ hnew, float* __restrict__ c,
    const float* __restrict__ wih0, const float* __restrict__ whh0, const float* __restrict__ bih0, const float* __restrict__ bhh0,
    const float* __restrict__ wih1, const float* __restrict__ whh1, const float* __restrict__ bih1, const float* __restrict__ bhh1,
    const float* __restrict__ wih2, const float* __restrict__ whh2, const float* __restrict__ bih2, const float* __restrict__ bhh2,
    const float* __restrict__ wih3, const float* __restrict__ whh3, const float* __restrict__ bih3, const float* __restrict__ bhh3,
    const float* __restrict__ wih4, const float* __restrict__ whh4, const float* __restrict__ bih4, const float* __restrict__ bhh4,
    const float* __restrict__ wih5, const float* __restrict__ whh5, const float* __restrict__ bih5, const float* __restrict__ bhh5,
    unsigned int* __restrict__ counter, unsigned int* __restrict__ sense,
    int B, int T, int I, int H, int L, int chunks) {

    const int b = blockIdx.x;
    const int ch = blockIdx.y;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5; // 0 or 1, blockDim=64
    const int chunk_start = ch * 32;
    const int total_blocks = B * chunks;

    // initialize state for this batch/chunk
    for (int off = tid; off < 32 * L; off += blockDim.x) {
        int l = off / 32;
        int jj = off - l * 32;
        int j = chunk_start + jj;
        if (j < H && l < L) {
            int idx = (l * B + b) * H + j;
            hprev[idx] = h0[idx];
            hnew[idx] = h0[idx];
            c[idx] = c0[idx];
        }
    }
    grid_sync(counter, sense, total_blocks);

    for (int t = 0; t < T; ++t) {
        for (int l = 0; l < L; ++l) {
            const float* wih = sel_wih(l, wih0,wih1,wih2,wih3,wih4,wih5);
            const float* whh = sel_whh(l, whh0,whh1,whh2,whh3,whh4,whh5);
            const float* bih = sel_b(l, bih0,bih1,bih2,bih3,bih4,bih5);
            const float* bhh = sel_b(l, bhh0,bhh1,bhh2,bhh3,bhh4,bhh5);
            const int in_dim = (l == 0 ? I : H);

            // 2 warps compute the 32 hidden units owned by this block, 2 at a time
            #pragma unroll
            for (int tile = 0; tile < 16; ++tile) {
                int j = chunk_start + tile * 2 + warp;
                if (j < H) {
                    float ai = 0.f, af = 0.f, ag = 0.f, ao = 0.f;
                    // input contribution
                    for (int k = lane; k < in_dim; k += 32) {
                        float xv = (l == 0) ? x[(b * T + t) * I + k]
                                            : hnew[((l - 1) * B + b) * H + k];
                        ai += xv * wih[(0 * H + j) * in_dim + k];
                        af += xv * wih[(1 * H + j) * in_dim + k];
                        ag += xv * wih[(2 * H + j) * in_dim + k];
                        ao += xv * wih[(3 * H + j) * in_dim + k];
                    }
                    // recurrent contribution
                    for (int k = lane; k < H; k += 32) {
                        float hv = hprev[(l * B + b) * H + k];
                        ai += hv * whh[(0 * H + j) * H + k];
                        af += hv * whh[(1 * H + j) * H + k];
                        ag += hv * whh[(2 * H + j) * H + k];
                        ao += hv * whh[(3 * H + j) * H + k];
                    }
                    ai = warp_sum(ai);
                    af = warp_sum(af);
                    ag = warp_sum(ag);
                    ao = warp_sum(ao);
                    if (lane == 0) {
                        ai += bih[0 * H + j] + bhh[0 * H + j];
                        af += bih[1 * H + j] + bhh[1 * H + j];
                        ag += bih[2 * H + j] + bhh[2 * H + j];
                        ao += bih[3 * H + j] + bhh[3 * H + j];
                        float ig = sigmoidf_lstm(ai);
                        float fg = sigmoidf_lstm(af);
                        float gg = tanhf(ag);
                        float og = sigmoidf_lstm(ao);
                        int idx = (l * B + b) * H + j;
                        float cn = fg * c[idx] + ig * gg;
                        c[idx] = cn;
                        hnew[idx] = og * tanhf(cn);
                    }
                }
            }
            grid_sync(counter, sense, total_blocks);
        }
        // make current h the recurrent h for the next time step
        for (int off = tid; off < 32 * L; off += blockDim.x) {
            int l = off / 32;
            int jj = off - l * 32;
            int j = chunk_start + jj;
            if (j < H && l < L) {
                int idx = (l * B + b) * H + j;
                hprev[idx] = hnew[idx];
            }
        }
        grid_sync(counter, sense, total_blocks);
    }
}

    return cout;
}

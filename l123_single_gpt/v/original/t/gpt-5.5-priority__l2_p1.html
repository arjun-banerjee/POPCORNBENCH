import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies ReLU, and adds a bias term.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape)) 

    def forward(self, x):
        x = self.conv(x)
        x = torch.relu(x)
        x = x + self.bias
        return x

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


_cpp_src = r"""
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <vector>

void relu_bias_inplace_cuda(torch::Tensor y, torch::Tensor bias);

torch::Tensor conv_relu_bias_forward(torch::Tensor x,
                                     torch::Tensor weight,
                                     torch::Tensor conv_bias,
                                     torch::Tensor add_bias) {
    // Same convolution parameters as nn.Conv2d(..., kernel_size=3) in the task:
    // stride=1, padding=0, dilation=1, groups=1.  Use ATen/cuDNN for the heavy
    // convolution, then fuse ReLU and the channel bias add in one CUDA pass.
    auto y = at::conv2d(x, weight, conv_bias, {1, 1}, {0, 0}, {1, 1}, 1);
    relu_bias_inplace_cuda(y, add_bias);
    return y;
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void relu_bias_kernel(float* __restrict__ y,
                                 const float* __restrict__ bias,
                                 long total,
                                 int C,
                                 int HW) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    long stride = (long)blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        int c = (int)((idx / HW) % C);
        float v = y[idx];
        v = v > 0.0f ? v : 0.0f;
        y[idx] = v + bias[c];
    }
}

void relu_bias_inplace_cuda(torch::Tensor y, torch::Tensor bias) {
    const long total = y.numel();
    if (total == 0) return;
    const int C = (int)y.size(1);
    const int HW = (int)(y.size(2) * y.size(3));
    const int threads = 256;
    int blocks = (int)((total + threads - 1) / threads);
    if (blocks > 4096) blocks = 4096;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    relu_bias_kernel<<<blocks, threads, 0, stream>>>(
        y.data_ptr<float>(), bias.data_ptr<float>(), total, C, HW);
}
"""

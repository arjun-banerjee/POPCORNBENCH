import torch

class Model(nn.Module):
    """
    Simple model that performs sum reduction over a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): Dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies sum reduction over the specified dimension.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor after sum reduction, shape (..., 1, ...).
        """
        return torch.sum(x, dim=self.dim, keepdim=True)

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Sum a contiguous 3D tensor of shape [B, N, K] over N, keepdim=True.
// One CUDA thread computes one output column for one batch.  For each reduction
// index, threads in a warp access consecutive K positions, yielding coalesced
// global loads despite the large strided reduction for each individual output.
__global__ void sum_dim1_kernel(const float* __restrict__ x,
                                float* __restrict__ y,
                                int B, int N, int K) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int b = blockIdx.y;
    if (col >= K || b >= B) return;

    const float* base = x + ((long long)b * N * K + col);
    float s0 = 0.0f;

    int j = 0;
    // Unroll by 8 to reduce loop overhead and expose memory-level parallelism.
    for (; j + 7 < N; j += 8) {
        s0 += base[(long long)(j + 0) * K];
        s0 += base[(long long)(j + 1) * K];
        s0 += base[(long long)(j + 2) * K];
        s0 += base[(long long)(j + 3) * K];
        s0 += base[(long long)(j + 4) * K];
        s0 += base[(long long)(j + 5) * K];
        s0 += base[(long long)(j + 6) * K];
        s0 += base[(long long)(j + 7) * K];
    }
    for (; j < N; ++j) {
        s0 += base[(long long)j * K];
    }
    y[(long long)b * K + col] = s0;
}

// A slightly more specialized kernel for the benchmark shape [128,4096,4095].
// Kept separate so the compiler can fully remove dynamic loop bounds.
__global__ void sum_dim1_4096_4095_kernel(const float* __restrict__ x,
                                          float* __restrict__ y) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int b = blockIdx.y;
    if (col >= 4095) return;

    const float* base = x + ((long long)b * 4096LL * 4095LL + col);
    float s = 0.0f;
    #pragma unroll 8
    for (int j = 0; j < 4096; ++j) {
        s += base[(long long)j * 4095LL];
    }
    y[(long long)b * 4095LL + col] = s;
}

torch::Tensor sum_dim1_forward(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(x.dim() == 3, "expected a 3D tensor");
    auto xc = x.contiguous();
    const int B = (int)xc.size(0);
    const int N = (int)xc.size(1);
    const int K = (int)xc.size(2);
    auto y = torch::empty({B, 1, K}, xc.options());

    const int threads = 256;
    dim3 block(threads);
    dim3 grid((K + threads - 1) / threads, B);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    if (N == 4096 && K == 4095) {
        sum_dim1_4096_4095_kernel<<<grid, block, 0, stream>>>(xc.data_ptr<float>(), y.data_ptr<float>());
    } else {
        sum_dim1_kernel<<<grid, block, 0, stream>>>(xc.data_ptr<float>(), y.data_ptr<float>(), B, N, K);
    }
    return y;
}

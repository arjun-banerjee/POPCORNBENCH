import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs mean reduction over a specific dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): The dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Reduces the input tensor along the specified dimension by taking the mean.

        Args:
            x (torch.Tensor): Input tensor of arbitrary shape.

        Returns:
            torch.Tensor: Output tensor with reduced dimension. The shape of the output is the same as the input except for the reduced dimension which is removed.
        """
        return torch.mean(x, dim=self.dim)

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor mean_dim1_cuda(torch::Tensor x);

torch::Tensor mean_dim1(torch::Tensor x) {
    return mean_dim1_cuda(x);
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Specialized for the benchmark shape [128, 4096, 4095], reducing dim=1.
// One thread computes one output element.  For every reduction index, a warp
// reads consecutive columns, giving fully coalesced global loads.
__global__ __launch_bounds__(256, 2)
void mean_dim1_4096_4095_kernel(const float* __restrict__ x,
                                float* __restrict__ out) {
    const int col = blockIdx.x * 256 + threadIdx.x;
    const int b = blockIdx.y;
    if (col >= 4095) return;

    const float* p = x + ((long long)b * 4096LL * 4095LL + col);

    float s0 = 0.0f,  s1 = 0.0f,  s2 = 0.0f,  s3 = 0.0f;
    float s4 = 0.0f,  s5 = 0.0f,  s6 = 0.0f,  s7 = 0.0f;
    float s8 = 0.0f,  s9 = 0.0f,  s10 = 0.0f, s11 = 0.0f;
    float s12 = 0.0f, s13 = 0.0f, s14 = 0.0f, s15 = 0.0f;

    #pragma unroll 4
    for (int k = 0; k < 4096; k += 16) {
        s0  += p[0 * 4095];
        s1  += p[1 * 4095];
        s2  += p[2 * 4095];
        s3  += p[3 * 4095];
        s4  += p[4 * 4095];
        s5  += p[5 * 4095];
        s6  += p[6 * 4095];
        s7  += p[7 * 4095];
        s8  += p[8 * 4095];
        s9  += p[9 * 4095];
        s10 += p[10 * 4095];
        s11 += p[11 * 4095];
        s12 += p[12 * 4095];
        s13 += p[13 * 4095];
        s14 += p[14 * 4095];
        s15 += p[15 * 4095];
        p += 16 * 4095;
    }

    float a0 = s0 + s1;
    float a1 = s2 + s3;
    float a2 = s4 + s5;
    float a3 = s6 + s7;
    float a4 = s8 + s9;
    float a5 = s10 + s11;
    float a6 = s12 + s13;
    float a7 = s14 + s15;
    float b0 = a0 + a1;
    float b1 = a2 + a3;
    float b2 = a4 + a5;
    float b3 = a6 + a7;
    float sum = (b0 + b1) + (b2 + b3);
    out[(long long)b * 4095LL + col] = sum * 0.000244140625f; // 1/4096
}

__global__ __launch_bounds__(256, 2)
void mean_dim1_generic_kernel(const float* __restrict__ x,
                              float* __restrict__ out,
                              int B, int K, int N) {
    const int col = blockIdx.x * 256 + threadIdx.x;
    const int b = blockIdx.y;
    if (b >= B || col >= N) return;

    const float* p = x + ((long long)b * K * N + col);
    float s = 0.0f;
    for (int k = 0; k < K; ++k) {
        s += p[(long long)k * N];
    }
    out[(long long)b * N + col] = s / (float)K;
}

torch::Tensor mean_dim1_cuda(torch::Tensor x) {
    const int64_t B64 = x.size(0);
    const int64_t K64 = x.size(1);
    const int64_t N64 = x.size(2);
    auto out = torch::empty({B64, N64}, x.options());

    const dim3 block(256);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (B64 == 128 && K64 == 4096 && N64 == 4095) {
        const dim3 grid((4095 + 255) / 256, 128);
        mean_dim1_4096_4095_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), out.data_ptr<float>());
    } else {
        const int B = (int)B64;
        const int K = (int)K64;
        const int N = (int)N64;
        const dim3 grid((N + 255) / 256, B);
        mean_dim1_generic_kernel<<<grid, block, 0, stream>>>(
            x.data_ptr<float>(), out.data_ptr<float>(), B, K, N);
    }
    return out;
}
"""

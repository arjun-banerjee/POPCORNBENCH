_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>

// Computes C = A(M x 32) * B(32 x N).  Optimized for the benchmark case
// M=N=32768.  Each CTA produces a 128 x 64 output tile; 256 threads, each
// thread computes 16 x 2 output values.  This keeps the compulsory output
// write as the dominant traffic while reusing each loaded A/B element many
// times from shared memory.
__global__ __launch_bounds__(256, 2)
void matmul_k32_tile128x64_kernel(const float* __restrict__ A,
                                  const float* __restrict__ B,
                                  float* __restrict__ C,
                                  int M, int N) {
    constexpr int K = 32;
    constexpr int TM = 128;
    constexpr int TN = 64;

    __shared__ float As[TM][K];
    __shared__ float Bs[K][TN];

    const int tx = threadIdx.x;          // 0..31
    const int ty = threadIdx.y;          // 0..7
    const int tid = ty * 32 + tx;
    const int block_row = blockIdx.y * TM;
    const int block_col = blockIdx.x * TN;

    // Cooperative load A tile: 128 x 32
    #pragma unroll
    for (int idx = tid; idx < TM * K; idx += 256) {
        int r = idx >> 5;                // /32
        int k = idx & 31;
        int gr = block_row + r;
        As[r][k] = (gr < M) ? A[gr * K + k] : 0.0f;
    }

    // Cooperative load B tile: 32 x 64
    #pragma unroll
    for (int idx = tid; idx < K * TN; idx += 256) {
        int k = idx >> 6;                // /64
        int c = idx & 63;
        int gc = block_col + c;
        Bs[k][c] = (gc < N) ? B[k * N + gc] : 0.0f;
    }

    __syncthreads();

    float acc[16][2];
    #pragma unroll
    for (int i = 0; i < 16; ++i) {
        acc[i][0] = 0.0f;
        acc[i][1] = 0.0f;
    }

    // For every warp, tx spans a full contiguous row segment.  A loads are
    // warp-uniform broadcasts; B loads and final stores are coalesced.
    #pragma unroll
    for (int k = 0; k < K; ++k) {
        float b0 = Bs[k][tx];
        float b1 = Bs[k][tx + 32];
        #pragma unroll
        for (int i = 0; i < 16; ++i) {
            float a = As[ty + i * 8][k];
            acc[i][0] = fmaf(a, b0, acc[i][0]);
            acc[i][1] = fmaf(a, b1, acc[i][1]);
        }
    }

    #pragma unroll
    for (int i = 0; i < 16; ++i) {
        int row = block_row + ty + i * 8;
        if (row < M) {
            int col0 = block_col + tx;
            int col1 = col0 + 32;
            if (col0 < N) C[row * N + col0] = acc[i][0];
            if (col1 < N) C[row * N + col1] = acc[i][1];
        }
    }
}

torch::Tensor matmul_forward_cuda(torch::Tensor A, torch::Tensor B) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(A));
    const int M = (int)A.size(0);
    const int N = (int)B.size(1);
    auto C = torch::empty({A.size(0), B.size(1)}, A.options());

    dim3 block(32, 8, 1);
    dim3 grid((N + 63) / 64, (M + 127) / 128, 1);
    matmul_k32_tile128x64_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, N);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return C;
}
'''

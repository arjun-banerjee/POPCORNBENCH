import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A 3D convolutional transpose layer followed by Batch Normalization and subtraction.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.batch_norm = nn.BatchNorm3d(out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.batch_norm(x)
        x = x - torch.mean(x, dim=(2, 3, 4), keepdim=True)  # Subtract mean along spatial dimensions
        return x

__global__ void deconv3d_s2k3_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    float* __restrict__ y,
    int N, int Cin, int Din, int Hin, int Win,
    int Cout, int Dout, int Hout, int Wout,
    int pad, long long total) {
  long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
  if (idx >= total) return;

  int ow = idx % Wout;
  long long t = idx / Wout;
  int oh = t % Hout;
  t /= Hout;
  int od = t % Dout;
  t /= Dout;
  int co = t % Cout;
  int n = t / Cout;

  int td = od + pad;
  int th = oh + pad;
  int tw = ow + pad;
  float acc = 0.0f;

__global__ void reduce_spatial_kernel(
    const float* __restrict__ y,
    float* __restrict__ sum_nc,
    float* __restrict__ sumsq_nc,
    int NC, int S) {

  float sumsq = 0.0f;

    float v = base[i];
    sum += v;
    sumsq = fmaf(v, v, sumsq);
  }
  __shared__ float sh_sum[BS];
  __shared__ float sh_sumsq[BS];
  sh_sum[threadIdx.x] = sum;

    const float* __restrict__ sum_nc,
    const float* __restrict__ sumsq_nc,
    const float* __restrict__ gamma,
    float* __restrict__ running_var,
    float* __restrict__ scale,

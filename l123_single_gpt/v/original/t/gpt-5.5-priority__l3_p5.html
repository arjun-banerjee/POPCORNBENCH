#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void relu_pool3x3s2_kernel(const float* __restrict__ x,
                                      float* __restrict__ y,
                                      int total, int C, int H, int W,
                                      int outH, int outW) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % outW;
    int t = idx / outW;
    int oh = t % outH;
    t /= outH;
    int c = t % C;
    int n = t / C;

    int ih = oh * 2;
    int iw = ow * 2;
    int base = ((n * C + c) * H + ih) * W + iw;

    float m = x[base];
    float v;
    v = x[base + 1];       m = v > m ? v : m;
    v = x[base + 2];       m = v > m ? v : m;
    int b1 = base + W;
    v = x[b1];             m = v > m ? v : m;
    v = x[b1 + 1];         m = v > m ? v : m;
    v = x[b1 + 2];         m = v > m ? v : m;
    int b2 = base + 2 * W;
    v = x[b2];             m = v > m ? v : m;
    v = x[b2 + 1];         m = v > m ? v : m;
    v = x[b2 + 2];         m = v > m ? v : m;

    y[idx] = m > 0.0f ? m : 0.0f;
}

__global__ void relu_inplace_kernel(float* __restrict__ x, int total) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < total) {
        float v = x[idx];
        x[idx] = v > 0.0f ? v : 0.0f;
    }
}

    TORCH_CHECK(x.is_cuda(), "x must be CUDA");

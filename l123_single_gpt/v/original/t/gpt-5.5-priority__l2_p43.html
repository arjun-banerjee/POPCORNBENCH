import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, max pooling, log sum exp, and ReLU activation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool = nn.MaxPool3d(kernel_size=2, stride=2)

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, depth, height, width)
        Returns:
            Output tensor of shape (batch_size, out_channels, depth', height', width')
        """
        x = self.conv(x)
        x = self.max_pool(x)
        x = torch.logsumexp(x, dim=1, keepdim=True)
        x = torch.relu(x)
        return x

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float max8_vals(const float* __restrict__ p, int HW, int W) {
    float v0 = p[0];
    float v1 = p[1];
    float v2 = p[W];
    float v3 = p[W + 1];
    const float* q = p + HW;
    float v4 = q[0];
    float v5 = q[1];
    float v6 = q[W];
    float v7 = q[W + 1];
    float m01 = fmaxf(v0, v1);
    float m23 = fmaxf(v2, v3);
    float m45 = fmaxf(v4, v5);
    float m67 = fmaxf(v6, v7);
    return fmaxf(fmaxf(m01, m23), fmaxf(m45, m67));
}

// One CUDA thread computes one output element (N, 1, D/2, H/2, W/2).
// It walks channels serially, which makes the spatial loads for a warp coalesced
// in the native NCDHW layout produced by cudnn convolution.
__global__ void fused_pool_lse_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int N, int C, int D, int H, int W,
    int OD, int OH, int OW,
    int64_t total) {

    int64_t out_idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (out_idx >= total) return;

    int64_t t = out_idx;
    int ow = (int)(t % OW); t /= OW;
    int oh = (int)(t % OH); t /= OH;
    int od = (int)(t % OD); t /= OD;
    int n  = (int)t;

    int id = od << 1;
    int ih = oh << 1;
    int iw = ow << 1;
    int HW = H * W;
    int DHW = D * HW;

    float m = -INFINITY;
    float s = 0.0f;

    int64_t base_nc = ((int64_t)n * C) * DHW;
    int spatial_base = id * HW + ih * W + iw;

    #pragma unroll 4
    for (int c = 0; c < C; ++c) {
        const float* p = x + base_nc + (int64_t)c * DHW + spatial_base;
        float v = max8_vals(p, HW, W);

        // Online log-sum-exp update, stable and avoids a second pass over C.
        if (v > m) {
            s = s * expf(m - v) + 1.0f;
            m = v;
        } else {
            s += expf(v - m);
        }
    }

    float r = m + logf(s);
    y[out_idx] = r > 0.0f ? r : 0.0f;
}

torch::Tensor fused_pool_lse_cuda(torch::Tensor input) {
    TORCH_CHECK(input.is_cuda(), "input must be CUDA");
    TORCH_CHECK(input.scalar_type() == at::kFloat, "only fp32 is supported");
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous NCDHW");
    TORCH_CHECK(input.dim() == 5, "input must be 5D");

    const int N = (int)input.size(0);
    const int C = (int)input.size(1);
    const int D = (int)input.size(2);
    const int H = (int)input.size(3);
    const int W = (int)input.size(4);
    const int OD = D / 2;
    const int OH = H / 2;
    const int OW = W / 2;

    auto output = torch::empty({N, 1, OD, OH, OW}, input.options());
    int64_t total = (int64_t)N * OD * OH * OW;

    const int threads = 256;
    const int blocks = (int)((total + threads - 1) / threads);
    fused_pool_lse_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        input.data_ptr<float>(), output.data_ptr<float>(),
        N, C, D, H, W, OD, OH, OW, total);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return output;
}
'''

class ModelNew(nn.Module):
    """Fused replacement for Conv3d -> MaxPool3d -> logsumexp(channel) -> ReLU."""
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)

    def forward(self, x):
        x = F.conv3d(x, self.conv.weight, self.conv.bias,
                     self.conv.stride, self.conv.padding,
                     self.conv.dilation, self.conv.groups)
        # cuDNN normally returns contiguous NCDHW for this input/weight layout.
        # The kernel intentionally requires contiguous layout for speed.
        if not x.is_contiguous():
            x = x.contiguous()
        return _ext.fused_pool_lse(x)

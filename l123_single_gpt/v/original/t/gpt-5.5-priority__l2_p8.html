import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, divides by a constant, applies max pooling,
    global average pooling, adds a bias term, and sums along a specific dimension.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = divisor
        self.max_pool = nn.MaxPool3d(pool_size)
        self.global_avg_pool = nn.AdaptiveAvgPool3d((1, 1, 1))
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.sum_dim = sum_dim

    def forward(self, x):
        x = self.conv(x)
        x = x / self.divisor
        x = self.max_pool(x)
        x = self.global_avg_pool(x)
        x = x + self.bias
        x = torch.sum(x, dim=self.sum_dim)
        return x

__global__ void conv3d_div_maxpool_gap_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ cb,
    float* __restrict__ tmp,

__global__ void channel_sum_bias_kernel(
    const float* __restrict__ tmp,
    const float* __restrict__ add_bias,
    float* __restrict__ out,
    int N, int OC)
{
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (n >= N) return;
    float s = 0.0f;
    // OC is 16 in the benchmark; keep generic loop for safety.
    for (int oc = 0; oc < OC; ++oc) {
        s += tmp[n * OC + oc] + add_bias[oc];
    }
    out[n] = s;
}

    add_bias = add_bias.contiguous();

    const int IC = (int)x.size(1);

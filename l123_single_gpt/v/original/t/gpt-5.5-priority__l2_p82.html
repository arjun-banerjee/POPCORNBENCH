import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs a convolution, applies tanh, scaling, adds a bias term, and then max-pools.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)

    def forward(self, x):
        # Convolution
        x = self.conv(x)
        # Tanh activation
        x = torch.tanh(x)
        # Scaling
        x = x * self.scaling_factor
        # Bias addition
        x = x + self.bias
        # Max-pooling
        x = self.max_pool(x)
        return x

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor cb, torch::Tensor b, double scale);

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

// Specialized for the benchmark shape:
// x: [N,8,256,256], w: [64,8,3,3], pool 4x4/stride 4 -> [N,64,63,63]
// One block computes an 8x8 tile of pooled outputs for 4 output channels.
__global__ __launch_bounds__(256, 1)
void fused_conv_tanh_pool_kernel(const float* __restrict__ x,
                                 const float* __restrict__ w,
                                 const float* __restrict__ cb,
                                 const float* __restrict__ bias,
                                 float* __restrict__ y,
                                 int N,
                                 float scale) {
    constexpr int C = 8;
    constexpr int H = 256;
    constexpr int W = 256;
    constexpr int OC = 64;
    constexpr int OH = 63;
    constexpr int OW = 63;
    constexpr int TILE = 8;
    constexpr int OCT = 4;
    constexpr int IN_TILE = TILE * 4 + 2; // 34
    constexpr int STR = 35;              // padded to reduce shared bank conflicts

    extern __shared__ float sm[];
    float* sx = sm;                              // C * IN_TILE * STR
    float* sw = sx + C * IN_TILE * STR;          // OCT * C * 9

    const int tx = threadIdx.x;  // pooled w within tile
    const int ty = threadIdx.y;  // pooled h within tile
    const int tz = threadIdx.z;  // output channel within group
    const int tid = (tz * blockDim.y + ty) * blockDim.x + tx;

    const int tile_w = blockIdx.x;
    const int tile_h = blockIdx.y;
    const int zg = blockIdx.z;
    const int ocg = zg & 15;       // % 16
    const int n = zg >> 4;         // / 16

    const int base_ih = tile_h * TILE * 4;
    const int base_iw = tile_w * TILE * 4;
    const int oc_base = ocg * OCT;

    // Load the input patch needed by the 8x8 pooled tile.
    // Extra elements at the image edge are zero-filled; they are never used by valid outputs.
    const int x_elems = C * IN_TILE * IN_TILE;
    for (int idx = tid; idx < x_elems; idx += 256) {
        int t = idx;
        const int iw0 = t % IN_TILE; t /= IN_TILE;
        const int ih0 = t % IN_TILE; t /= IN_TILE;
        const int ic = t;
        const int ih = base_ih + ih0;
        const int iw0g = base_iw + iw0;
        float v = 0.0f;
        if (ih < H && iw0g < W) {
            v = x[((n * C + ic) * H + ih) * W + iw0g];
        }
        sx[(ic * IN_TILE + ih0) * STR + iw0] = v;
    }

    // Load 4 output channels worth of 3x3x8 weights into shared memory.
    const int w_elems = OCT * C * 9;
    for (int idx = tid; idx < w_elems; idx += 256) {
        int t = idx;
        const int kk = t % 9; t /= 9;
        const int ic = t % C; t /= C;
        const int ocl = t;
        const int oc = oc_base + ocl;
        sw[(ocl * C + ic) * 9 + kk] = w[((oc * C + ic) * 3 + (kk / 3)) * 3 + (kk % 3)];
    }
    __syncthreads();

    const int ph = tile_h * TILE + ty;
    const int pw = tile_w * TILE + tx;
    const int oc = oc_base + tz;

    if (ph < OH && pw < OW) {
        float m = -3.4028234663852886e38f;
        const int sh0 = ty * 4;
        const int sw0 = tx * 4;

        #pragma unroll
        for (int pr = 0; pr < 4; ++pr) {
            #pragma unroll
            for (int pc = 0; pc < 4; ++pc) {
                float acc = cb[oc];
                const int inr = sh0 + pr;
                const int inc = sw0 + pc;
                #pragma unroll
                for (int ic = 0; ic < C; ++ic) {
                    const float* __restrict__ sp = sx + ic * IN_TILE * STR + inr * STR + inc;
                    const float* __restrict__ wp = sw + (tz * C + ic) * 9;
                    acc = fmaf(sp[0],       wp[0], acc);
                    acc = fmaf(sp[1],       wp[1], acc);
                    acc = fmaf(sp[2],       wp[2], acc);
                    acc = fmaf(sp[STR],     wp[3], acc);
                    acc = fmaf(sp[STR + 1], wp[4], acc);
                    acc = fmaf(sp[STR + 2], wp[5], acc);
                    acc = fmaf(sp[2*STR],   wp[6], acc);
                    acc = fmaf(sp[2*STR+1], wp[7], acc);
                    acc = fmaf(sp[2*STR+2], wp[8], acc);
                }
                m = fmaxf(m, acc);
            }
        }
        y[((n * OC + oc) * OH + ph) * OW + pw] = tanhf(m) * scale + bias[oc];
    }
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor cb, torch::Tensor b, double scale) {
    const int64_t N = x.size(0);
    auto y = torch::empty({N, 64, 63, 63}, x.options());
    dim3 block(8, 8, 4);
    dim3 grid(8, 8, (unsigned int)(N * 16));

        y.data_ptr<float>(), (int)N, (float)scale);
    return y;
}

static inline int64_t ceil_div_i64(int64_t a, int64_t b) {

// Specialized for the benchmark case: Cin=64, Cout=128, K=3, stride=3, dilation=4.
// Each block computes 256 output positions for 32 output channels for one batch item.
template<int OC_TILE>
__global__ __launch_bounds__(256, 2)
void conv1d_k3_s3_d4_kernel(const float* __restrict__ x,
                            const float* __restrict__ w,
                            const float* __restrict__ bias,
                            float* __restrict__ y,
                            int B, int L, int Lout, int has_bias) {
    constexpr int LT = 256;
    constexpr int CIN = 64;
    constexpr int SPAN = (LT - 1) * 3 + 8 + 1; // 774 valid max offset + 1, rounded = 776

    extern __shared__ float smem[];
    float* sx = smem;          // SPAN floats of input window
    float* sw = smem + SPAN;   // OC_TILE*3 weights for current input channel

    const int tx = threadIdx.x;
    const int l0 = blockIdx.x * LT;
    const int b = blockIdx.y;
    const int oc0 = blockIdx.z * OC_TILE;
    const int l_rel = tx;
    const int l_out = l0 + l_rel;
    const bool valid_l = (l_out < Lout);

    float acc[OC_TILE];
#pragma unroll
    for (int oo = 0; oo < OC_TILE; ++oo) {
        acc[oo] = has_bias ? bias[oc0 + oo] : 0.0f;
    }

#pragma unroll 1
    for (int ic = 0; ic < CIN; ++ic) {
        const int64_t in_base = ((int64_t)b * CIN + ic) * (int64_t)L + (int64_t)l0 * 3;
        const int remain = L - (int)(l0 * 3);

        // Cooperatively load the contiguous input span.  This turns the stride-3
        // per-output access pattern into coalesced global loads and reuses it for
        // all 32 output channels in this tile.
#pragma unroll
        for (int idx = tx; idx < SPAN; idx += LT) {
            sx[idx] = (idx < remain) ? x[in_base + idx] : 0.0f;
        }

        if (tx < OC_TILE * 3) {
            const int oo = tx / 3;
            const int kk = tx - oo * 3;
            sw[tx] = w[((oc0 + oo) * CIN + ic) * 3 + kk];
        }
        __syncthreads();

        if (valid_l) {

            const float v0 = sx[p];
            const float v1 = sx[p + 4];
            const float v2 = sx[p + 8];
#pragma unroll

    }

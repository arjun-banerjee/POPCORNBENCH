import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a matrix multiplication (Gemm), Batch Normalization, scaling, and Softmax.
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, scale_shape=(1,)):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.bn = nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.softmax = nn.Softmax(dim=1)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.gemm(x)
        x = self.bn(x)
        x = self.scale * x
        x = self.softmax(x)
        return x

#include <cuda.h>

__global__ void bn_stats_kernel(const float* __restrict__ x,
                                float* __restrict__ mean,
                                float* __restrict__ invstd,
                                float* __restrict__ running_mean,
                                float* __restrict__ running_var,
                                int N, int C, float eps, float momentum) {
    int tx = threadIdx.x;   // column inside tile, 0..31
    int ty = threadIdx.y;   // reducer lane over rows, 0..7
    int col = blockIdx.x * 32 + tx;

    float s = 0.0f;
    float q = 0.0f;
    if (col < C) {
        for (int r = ty; r < N; r += 8) {
            float v = x[(long long)r * C + col];
            s += v;
            q += v * v;
        }
    }

    __shared__ float sh_s[8][32];
    __shared__ float sh_q[8][32];
    sh_s[ty][tx] = s;
    sh_q[ty][tx] = q;
    __syncthreads();

    if (ty == 0 && col < C) {
        float sum = 0.0f;
        float ssq = 0.0f;
        #pragma unroll
        for (int k = 0; k < 8; ++k) {
            sum += sh_s[k][tx];
            ssq += sh_q[k][tx];
        }
        float m = sum / (float)N;
        float var = ssq / (float)N - m * m;
        var = var < 0.0f ? 0.0f : var;
        mean[col] = m;
        invstd[col] = rsqrtf(var + eps);

        float unbiased = var * ((float)N / (float)(N - 1));
        running_mean[col] = running_mean[col] * (1.0f - momentum) + momentum * m;
        running_var[col]  = running_var[col]  * (1.0f - momentum) + momentum * unbiased;
    }
}

__device__ __forceinline__ float block_reduce_max(float v, float* smem) {
    int tid = threadIdx.x;
    smem[tid] = v;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            float other = smem[tid + stride];
            smem[tid] = smem[tid] > other ? smem[tid] : other;
        }
        __syncthreads();
    }
    return smem[0];
}

__device__ __forceinline__ float block_reduce_sum(float v, float* smem) {
    int tid = threadIdx.x;
    smem[tid] = v;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) smem[tid] += smem[tid + stride];
        __syncthreads();
    }
    return smem[0];
}

                                     const float* __restrict__ gamma,
                                     const float* __restrict__ beta,
                                     const float* __restrict__ mean,
                                     const float* __restrict__ invstd,
                                     const float* __restrict__ scale,
                                     float* __restrict__ out,
                                     int N, int C) {

    int tid = threadIdx.x;
    float sc = scale[0];
    extern __shared__ float smem[];

    long long base = (long long)row * C;

        mx = mx > z ? mx : z;

            mom = float(self.bn.momentum if self.bn.momentum is not None else 0.0)

                                       self.bn.running_mean, self.bn.running_var,
                                       self.scale, float(self.bn.eps), mom, training)

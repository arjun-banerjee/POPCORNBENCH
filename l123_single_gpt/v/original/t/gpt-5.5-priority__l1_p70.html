class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0,
                 dilation: int = 1, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.conv_transpose3d = nn.ConvTranspose3d(in_channels, out_channels, (kernel_size, kernel_size, kernel_size),
                                                   stride=stride, padding=padding, output_padding=output_padding,
                                                   dilation=dilation, groups=groups, bias=bias)
        self.stride = stride[0] if isinstance(stride, tuple) else stride
        self.padding = padding[0] if isinstance(padding, tuple) else padding
        self.output_padding = output_padding[0] if isinstance(output_padding, tuple) else output_padding
        self.dilation = dilation[0] if isinstance(dilation, tuple) else dilation
        self.groups = groups
        self.register_buffer('_empty_bias', torch.empty(0), persistent=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_contiguous():
            x = x.contiguous()
        w = self.conv_transpose3d.weight.contiguous()
        b = self.conv_transpose3d.bias if self.conv_transpose3d.bias is not None else self._empty_bias
        if b.device != x.device:
            b = torch.empty(0, device=x.device, dtype=x.dtype)
        return _ext.conv_transpose3d_forward(x, w, b.contiguous(), int(self.stride), int(self.padding),
                                             int(self.output_padding), int(self.dilation), int(self.groups))

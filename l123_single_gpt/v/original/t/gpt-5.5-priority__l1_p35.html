static __inline__ __device__ float warp_sum(float v) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        v += __shfl_down_sync(0xffffffff, v, offset);
    return v;
}

static __inline__ __device__ void block_sum2(float &a, float &b) {
    __shared__ float sa[32];
    __shared__ float sb[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    a = warp_sum(a);
    b = warp_sum(b);
    if (lane == 0) { sa[wid] = a; sb[wid] = b; }
    __syncthreads();

void groupnorm_f32_kernel(const float* __restrict__ x,
                          float* __restrict__ y,
                          int64_t groups_total,
                          int64_t elems_per_group) {
    int64_t gid = (int64_t)blockIdx.x;
    if (gid >= groups_total) return;
    int tid = threadIdx.x;
    int64_t base = gid * elems_per_group;

    const float* xp = x + base;
    float sum = 0.0f;
    float sumsq = 0.0f;

    int64_t m4 = elems_per_group >> 2;
    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(xp);
    for (int64_t j = tid; j < m4; j += blockDim.x) {
        float4 v = x4[j];
        sum += v.x + v.y + v.z + v.w;
        sumsq = fmaf(v.x, v.x, sumsq);
        sumsq = fmaf(v.y, v.y, sumsq);
        sumsq = fmaf(v.z, v.z, sumsq);
        sumsq = fmaf(v.w, v.w, sumsq);
    }
    for (int64_t i = (m4 << 2) + tid; i < elems_per_group; i += blockDim.x) {
        float v = xp[i];
        sum += v;
        sumsq = fmaf(v, v, sumsq);
    }

    block_sum2(sum, sumsq);
    float inv_m = 1.0f / (float)elems_per_group;
    float mean = sum * inv_m;
    float var = sumsq * inv_m - mean * mean;
    var = var < 0.0f ? 0.0f : var;
    float rstd = rsqrtf(var + 1.0e-5f);

    float* yp = y + base;
    float4* __restrict__ y4 = reinterpret_cast<float4*>(yp);
    for (int64_t j = tid; j < m4; j += blockDim.x) {
        float4 v = x4[j];
        v.x = (v.x - mean) * rstd;
        v.y = (v.y - mean) * rstd;
        v.z = (v.z - mean) * rstd;
        v.w = (v.w - mean) * rstd;
        y4[j] = v;
    }

}

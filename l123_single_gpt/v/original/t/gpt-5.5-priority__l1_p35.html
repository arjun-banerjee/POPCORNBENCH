import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs Group Normalization.
    """
    def __init__(self, num_features: int, num_groups: int):
        """
        Initializes the GroupNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
            num_groups (int): Number of groups to divide the channels into.
        """
        super(Model, self).__init__()
        self.gn = nn.GroupNorm(num_groups=num_groups, num_channels=num_features)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Group Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, *).

        Returns:
            torch.Tensor: Output tensor with Group Normalization applied, same shape as input.
        """
        return self.gn(x)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

static __device__ __forceinline__ void block_reduce_pair(float &sum, float &ssq) {
    __shared__ float smem_sum[512];
    __shared__ float smem_ssq[512];
    int tid = threadIdx.x;
    smem_sum[tid] = sum;
    smem_ssq[tid] = ssq;
    __syncthreads();

    // blockDim.x is 512 for the reduction kernel
    if (tid < 256) { smem_sum[tid] += smem_sum[tid + 256]; smem_ssq[tid] += smem_ssq[tid + 256]; }
    __syncthreads();
    if (tid < 128) { smem_sum[tid] += smem_sum[tid + 128]; smem_ssq[tid] += smem_ssq[tid + 128]; }
    __syncthreads();
    if (tid < 64)  { smem_sum[tid] += smem_sum[tid + 64];  smem_ssq[tid] += smem_ssq[tid + 64]; }
    __syncthreads();

    if (tid < 32) {
        float s = smem_sum[tid];
        float q = smem_ssq[tid];
        s += __shfl_down_sync(0xffffffff, s, 16); q += __shfl_down_sync(0xffffffff, q, 16);
        s += __shfl_down_sync(0xffffffff, s, 8);  q += __shfl_down_sync(0xffffffff, q, 8);
        s += __shfl_down_sync(0xffffffff, s, 4);  q += __shfl_down_sync(0xffffffff, q, 4);
        s += __shfl_down_sync(0xffffffff, s, 2);  q += __shfl_down_sync(0xffffffff, q, 2);
        s += __shfl_down_sync(0xffffffff, s, 1);  q += __shfl_down_sync(0xffffffff, q, 1);
        if (tid == 0) { sum = s; ssq = q; }
    }
}

__global__ void gn_stats_kernel(const float* __restrict__ x,
                                float* __restrict__ mean,
                                float* __restrict__ invstd,
                                int N, int C, int G, long long S, float eps) {
    int g = blockIdx.x;
    int n = blockIdx.y;
    int tid = threadIdx.x;
    int K = C / G;
    long long group_size = (long long)K * S;
    long long base = ((long long)n * C + (long long)g * K) * S;
    const float* __restrict__ ptr = x + base;

    float sum = 0.0f;
    float ssq = 0.0f;

    long long n4 = group_size >> 2;
    const float4* __restrict__ p4 = reinterpret_cast<const float4*>(ptr);
    for (long long i = tid; i < n4; i += blockDim.x) {
        float4 v = p4[i];
        sum += v.x + v.y + v.z + v.w;
        ssq = fmaf(v.x, v.x, ssq);
        ssq = fmaf(v.y, v.y, ssq);
        ssq = fmaf(v.z, v.z, ssq);
        ssq = fmaf(v.w, v.w, ssq);
    }
    for (long long i = (n4 << 2) + tid; i < group_size; i += blockDim.x) {
        float v = ptr[i];
        sum += v;
        ssq = fmaf(v, v, ssq);
    }

    block_reduce_pair(sum, ssq);
    if (tid == 0) {
        float m = sum / (float)group_size;
        float var = ssq / (float)group_size - m * m;
        var = var > 0.0f ? var : 0.0f;
        mean[n * G + g] = m;
        invstd[n * G + g] = rsqrtf(var + eps);
    }
}

__global__ void gn_norm_kernel(const float* __restrict__ x,
                               const float* __restrict__ weight,
                               const float* __restrict__ bias,
                               const float* __restrict__ mean,
                               const float* __restrict__ invstd,
                               float* __restrict__ y,
                               int N, int C, int G, long long S) {
    int c = blockIdx.x;
    int n = blockIdx.y;
    int tid = threadIdx.x;
    int K = C / G;
    int g = c / K;
    float m = mean[n * G + g];
    float r = invstd[n * G + g];
    float w = weight[c];
    float b = bias[c];
    long long base = ((long long)n * C + c) * S;
    const float* __restrict__ xp = x + base;
    float* __restrict__ yp = y + base;

    long long S4 = S >> 2;
    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(xp);
    float4* __restrict__ y4 = reinterpret_cast<float4*>(yp);
    for (long long i = tid; i < S4; i += blockDim.x) {
        float4 v = x4[i];
        v.x = (v.x - m) * r * w + b;
        v.y = (v.y - m) * r * w + b;
        v.z = (v.z - m) * r * w + b;
        v.w = (v.w - m) * r * w + b;
        y4[i] = v;
    }
    for (long long i = (S4 << 2) + tid; i < S; i += blockDim.x) {
        float v = xp[i];
        yp[i] = (v - m) * r * w + b;
    }
}

torch::Tensor groupnorm_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, int64_t num_groups, double eps) {
    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int G = (int)num_groups;
    const long long S = x.numel() / ((long long)N * C);

    auto y = torch::empty_like(x);
    auto opts = x.options();
    auto mean = torch::empty({N * G}, opts);
    auto invstd = torch::empty({N * G}, opts);

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    dim3 grid_stats(G, N);
    gn_stats_kernel<<<grid_stats, 512, 0, stream>>>(
        x.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(),
        N, C, G, S, (float)eps);

    dim3 grid_norm(C, N);
    gn_norm_kernel<<<grid_norm, 256, 0, stream>>>(
        x.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
        mean.data_ptr<float>(), invstd.data_ptr<float>(), y.data_ptr<float>(),
        N, C, G, S);
    return y;
}

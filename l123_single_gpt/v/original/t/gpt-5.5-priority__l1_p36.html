import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs RMS Normalization.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        """
        Initializes the RMSNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
            eps (float, optional): A small value added to the denominator to avoid division by zero. Defaults to 1e-5.
        """
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies RMS Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, *).

        Returns:
            torch.Tensor: Output tensor with RMS Normalization applied, same shape as input.
        """
        # Calculate the RMS along the feature dimension
        rms = torch.sqrt(torch.mean(x ** 2, dim=1, keepdim=True) + self.eps)

        # Normalize the input by dividing by the RMS
        return x / rms

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Optimized for the benchmark case C == 64.  The spatial dimension is tiled so
// loads/stores are coalesced even though the reduction dimension is strided.
template<int TILE, int Y>
__launch_bounds__(TILE*Y, 2)
__global__ void rmsnorm64_kernel(const float* __restrict__ x,
                                 float* __restrict__ out,
                                 int64_t S,
                                 float eps) {
    const int tx = threadIdx.x;     // spatial position inside tile
    const int ty = threadIdx.y;     // reducer lane over channels
    const int64_t s = (int64_t)blockIdx.x * TILE + tx;
    const int n = blockIdx.y;

    __shared__ float part[Y][TILE];
    __shared__ float scale[TILE];

    float sum = 0.0f;
    if (s < S) {
        const int64_t base = ((int64_t)n * 64) * S + s;
        #pragma unroll
        for (int c = ty; c < 64; c += Y) {
            float v = x[base + (int64_t)c * S];
            sum = fmaf(v, v, sum);
        }
    }
    part[ty][tx] = sum;
    __syncthreads();

    if (ty == 0) {
        float t = 0.0f;
        #pragma unroll
        for (int j = 0; j < Y; ++j) t += part[j][tx];
        if (s < S) scale[tx] = rsqrtf(t * (1.0f / 64.0f) + eps);
    }
    __syncthreads();

    if (s < S) {
        const float sc = scale[tx];
        const int64_t base = ((int64_t)n * 64) * S + s;
        #pragma unroll
        for (int c = ty; c < 64; c += Y) {
            int64_t idx = base + (int64_t)c * S;
            out[idx] = x[idx] * sc;
        }
    }
}

// Generic fallback for non-64 feature counts (keeps the module correct if used
// outside the stated benchmark shape).
template<int TILE, int Y>
__global__ void rmsnorm_generic_kernel(const float* __restrict__ x,
                                       float* __restrict__ out,
                                       int C,
                                       int64_t S,
                                       float eps) {
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int64_t s = (int64_t)blockIdx.x * TILE + tx;
    const int n = blockIdx.y;

    __shared__ float part[Y][TILE];
    __shared__ float scale[TILE];

    float sum = 0.0f;
    if (s < S) {
        const int64_t base = ((int64_t)n * C) * S + s;
        for (int c = ty; c < C; c += Y) {
            float v = x[base + (int64_t)c * S];
            sum = fmaf(v, v, sum);
        }
    }
    part[ty][tx] = sum;
    __syncthreads();

    if (ty == 0) {
        float t = 0.0f;
        #pragma unroll
        for (int j = 0; j < Y; ++j) t += part[j][tx];
        if (s < S) scale[tx] = rsqrtf(t / (float)C + eps);
    }
    __syncthreads();

    if (s < S) {
        const float sc = scale[tx];
        const int64_t base = ((int64_t)n * C) * S + s;
        for (int c = ty; c < C; c += Y) {
            int64_t idx = base + (int64_t)c * S;
            out[idx] = x[idx] * sc;
        }
    }
}

torch::Tensor rmsnorm_forward(torch::Tensor x, double eps) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(x.dim() >= 2, "input must have at least 2 dimensions");
    TORCH_CHECK(x.is_contiguous(), "input must be contiguous");

    auto out = torch::empty_like(x);
    const int64_t B64 = x.size(0);
    const int64_t C64 = x.size(1);
    TORCH_CHECK(B64 <= INT_MAX && C64 <= INT_MAX, "unsupported tensor shape");
    const int B = (int)B64;
    const int C = (int)C64;

    int64_t S = 1;
    for (int i = 2; i < x.dim(); ++i) S *= x.size(i);

    constexpr int TILE = 128;
    constexpr int Y = 4;
    dim3 block(TILE, Y);
    dim3 grid((unsigned int)((S + TILE - 1) / TILE), (unsigned int)B);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float* xp = x.data_ptr<float>();
    float* op = out.data_ptr<float>();
    if (C == 64) {
        rmsnorm64_kernel<TILE, Y><<<grid, block, 0, stream>>>(xp, op, S, (float)eps);
    } else {
        rmsnorm_generic_kernel<TILE, Y><<<grid, block, 0, stream>>>(xp, op, C, S, (float)eps);
    }
    return out;
}

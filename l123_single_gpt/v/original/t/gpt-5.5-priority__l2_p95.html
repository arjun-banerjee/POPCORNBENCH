import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a matrix multiplication, adds a value, applies Swish, Tanh, GELU, and Hardtanh activation functions.
    """
    def __init__(self, in_features, out_features, add_value_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.add_value = nn.Parameter(torch.randn(add_value_shape)) 

    def forward(self, x):
        x = self.matmul(x)
        x = x + self.add_value
        x = torch.sigmoid(x) * x # Swish
        x = torch.tanh(x)
        x = torch.nn.functional.gelu(x) # GELU
        x = torch.nn.functional.hardtanh(x, min_val=-1, max_val=1) # Hardtanh
        return x

def get_inputs():

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

void activation_launch(torch::Tensor y);

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, torch::Tensor add_value) {
    // Combine the Linear bias and the extra add_value vector, then use addmm for
    // the GEMM. The remaining elementwise activation chain is fused in CUDA.
    auto combined_bias = at::add(bias, add_value);
    auto y = at::addmm(combined_bias, x, weight.t());
    activation_launch(y);
    return y;
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__global__ void activation_kernel(float* __restrict__ p, int64_t n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    const float inv_sqrt2 = 0.70710678118654752440f;
    for (int64_t i = idx; i < n; i += stride) {
        float x = p[i];
        // Swish: x * sigmoid(x)
        float sig = 1.0f / (1.0f + __expf(-x));
        x = x * sig;
        // tanh
        x = tanhf(x);
        // exact GELU form used by torch.nn.functional.gelu default
        x = 0.5f * x * (1.0f + erff(x * inv_sqrt2));
        // hardtanh(-1, 1) is a no-op here: after tanh, GELU is in [-~0.17, ~0.84].
        p[i] = x;
    }
}

void activation_launch(torch::Tensor y) {
    const auto n = y.numel();
    if (n == 0) return;
    c10::cuda::CUDAGuard device_guard(y.device());
    const int threads = 256;
    int blocks = (int)((n + threads - 1) / threads);
    if (blocks > 65535) blocks = 65535;
    activation_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(y.data_ptr<float>(), n);
}
"""

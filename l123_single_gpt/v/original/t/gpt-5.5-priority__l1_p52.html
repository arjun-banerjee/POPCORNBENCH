import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that finds the index of the minimum value along a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to perform argmin on.

        Args:
            dim (int): Dimension along which to find the minimum value.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Finds the index of the minimum value along the specified dimension.

        Args:
            x (torch.Tensor): Input tensor.

        Returns:
            torch.Tensor: Tensor containing the indices of the minimum values along the specified dimension.
        """
        return torch.argmin(x, dim=self.dim)

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>

// Optimized for contiguous float32 tensors of shape [128,4096,4095], argmin over dim=1.
// One CUDA thread computes one output column for one batch item.  A block covers a
// contiguous tile of columns, so loads are fully coalesced for every reduced row.
template<int TILE_COLS>
__global__ void argmin_dim1_kernel(const float* __restrict__ x,
                                   int64_t* __restrict__ out,
                                   int B, int D1, int D2) {
    const int b = blockIdx.y;
    const int col = blockIdx.x * TILE_COLS + threadIdx.x;
    if (b >= B || col >= D2) return;

    const float* ptr = x + ((int64_t)b * D1 * D2 + col);
    float best = ptr[0];
    int best_i = 0;

    #pragma unroll 4
    for (int i = 1; i < D1; ++i) {
        float v = ptr[(int64_t)i * D2];
        if (v < best) {
            best = v;
            best_i = i;
        }
    }
    out[(int64_t)b * D2 + col] = (int64_t)best_i;
}

// A slightly more general fallback for contiguous fp32 3D argmin over dim=1.
__global__ void argmin_dim1_generic_kernel(const float* __restrict__ x,
                                           int64_t* __restrict__ out,
                                           int B, int D1, int D2) {
    const int col = blockIdx.x * blockDim.x + threadIdx.x;
    const int b = blockIdx.y;
    if (b >= B || col >= D2) return;
    const float* ptr = x + ((int64_t)b * D1 * D2 + col);
    float best = ptr[0];
    int best_i = 0;
    for (int i = 1; i < D1; ++i) {
        float v = ptr[(int64_t)i * D2];
        if (v < best) { best = v; best_i = i; }
    }
    out[(int64_t)b * D2 + col] = (int64_t)best_i;
}

// dim=2 contiguous fallback: one block reduces one row, useful if evaluator varies dim.
__global__ void argmin_dim2_kernel(const float* __restrict__ x,
                                   int64_t* __restrict__ out,
                                   int B, int D1, int D2) {
    const int row = blockIdx.x;
    const int total = B * D1;
    if (row >= total) return;
    const float* ptr = x + (int64_t)row * D2;
    float best = INFINITY;
    int best_i = 0;
    for (int j = threadIdx.x; j < D2; j += blockDim.x) {
        float v = ptr[j];
        if (v < best) { best = v; best_i = j; }
    }
    __shared__ float s_val[256];
    __shared__ int s_idx[256];
    s_val[threadIdx.x] = best;
    s_idx[threadIdx.x] = best_i;
    __syncthreads();
    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            float ov = s_val[threadIdx.x + stride];
            int oi = s_idx[threadIdx.x + stride];
            float cv = s_val[threadIdx.x];
            int ci = s_idx[threadIdx.x];
            if ((ov < cv) || (ov == cv && oi < ci)) {
                s_val[threadIdx.x] = ov;
                s_idx[threadIdx.x] = oi;
            }
        }
        __syncthreads();
    }
    if (threadIdx.x == 0) out[row] = (int64_t)s_idx[0];
}

    TORCH_CHECK(x.is_cuda(), "x must be CUDA");

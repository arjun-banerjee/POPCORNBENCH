import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs L1 normalization.
    """
    def __init__(self):
        """
        Initializes the L1 normalization layer.
        """
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies L1 normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor with L1 normalization applied, same shape as input.
        """
        return x / torch.mean(torch.abs(x), dim=1, keepdim=True)

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>

void l1_norm_cuda(torch::Tensor x, torch::Tensor out);

torch::Tensor l1_norm(torch::Tensor x) {
    auto out = torch::empty_like(x);
    l1_norm_cuda(x, out);
    return out;
}
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

#ifndef __CUDA_ARCH__
#define __CUDA_ARCH__ 0
#endif

__global__ __launch_bounds__(256, 2)
void l1_norm_kernel_f32(const float* __restrict__ x, float* __restrict__ out, int rows, int cols) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    const float* __restrict__ xr = x + (int64_t)row * cols;
    float* __restrict__ yr = out + (int64_t)row * cols;

    float sum = 0.0f;
    // Unroll the strided load loop to reduce loop overhead.  For the benchmark
    // cols=65535 and blockDim=256, so each thread handles about 256 elements.
    int i = tid;
    int stride = blockDim.x;
    for (; i + 3 * stride < cols; i += 4 * stride) {
        float v0 = xr[i];
        float v1 = xr[i + stride];
        float v2 = xr[i + 2 * stride];
        float v3 = xr[i + 3 * stride];
        sum += fabsf(v0) + fabsf(v1) + fabsf(v2) + fabsf(v3);
    }
    for (; i < cols; i += stride) {
        sum += fabsf(xr[i]);
    }

    // Warp reductions, then one warp reduces the warp sums.
    sum += __shfl_down_sync(0xffffffff, sum, 16);
    sum += __shfl_down_sync(0xffffffff, sum, 8);
    sum += __shfl_down_sync(0xffffffff, sum, 4);
    sum += __shfl_down_sync(0xffffffff, sum, 2);
    sum += __shfl_down_sync(0xffffffff, sum, 1);

    __shared__ float warp_sums[8];
    int lane = tid & 31;
    int warp = tid >> 5;
    if (lane == 0) warp_sums[warp] = sum;
    __syncthreads();

    float total = 0.0f;
    if (tid < 8) total = warp_sums[tid];
    if (warp == 0) {
        total += __shfl_down_sync(0xffffffff, total, 4);
        total += __shfl_down_sync(0xffffffff, total, 2);
        total += __shfl_down_sync(0xffffffff, total, 1);
    }
    __shared__ float scale_s;
    if (tid == 0) {
        // x / mean(abs(x)) == x * cols / sum(abs(x)).  PyTorch will produce
        // inf/nan in the zero-sum case; division here mirrors that behavior.
        scale_s = ((float)cols) / total;
    }
    __syncthreads();
    float scale = scale_s;

    i = tid;
    for (; i + 3 * stride < cols; i += 4 * stride) {
        float v0 = xr[i];
        float v1 = xr[i + stride];
        float v2 = xr[i + 2 * stride];
        float v3 = xr[i + 3 * stride];
        yr[i] = v0 * scale;
        yr[i + stride] = v1 * scale;
        yr[i + 2 * stride] = v2 * scale;
        yr[i + 3 * stride] = v3 * scale;
    }
    for (; i < cols; i += stride) {
        yr[i] = xr[i] * scale;
    }
}

void l1_norm_cuda(torch::Tensor x, torch::Tensor out) {
    const int64_t rows64 = x.size(0);
    const int64_t cols64 = x.size(1);
    const int rows = (int)rows64;
    const int cols = (int)cols64;
    const dim3 grid(rows);
    const dim3 block(256);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    l1_norm_kernel_f32<<<grid, block, 0, stream>>>(x.data_ptr<float>(), out.data_ptr<float>(), rows, cols);
}
'''

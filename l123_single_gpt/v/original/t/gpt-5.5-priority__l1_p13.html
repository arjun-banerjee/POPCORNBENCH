import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAStream.h>
#include <cublas_v2.h>
#include <mutex>

static cublasHandle_t g_handle = nullptr;
static std::once_flag g_once;

static void init_cublas_handle() {
    cublasCreate(&g_handle);
}

torch::Tensor symm_matmul_forward(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == torch::kFloat32 && B.scalar_type() == torch::kFloat32, "only float32 supported");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be matrices");
    TORCH_CHECK(A.size(0) == A.size(1) && B.size(0) == B.size(1) && A.size(0) == B.size(0), "inputs must be same-size square matrices");
    // Benchmark inputs are contiguous; make this explicit for safety without falling back to torch matmul.
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "inputs must be contiguous");
    const int64_t n64 = A.size(0);
    TORCH_CHECK(n64 <= INT_MAX, "matrix too large");
    const int n = static_cast<int>(n64);
    auto C = torch::empty({n64, n64}, A.options());

    std::call_once(g_once, init_cublas_handle);
    cublasSetStream(g_handle, at::cuda::getCurrentCUDAStream().stream());

    const float alpha = 1.0f;
    const float beta = 0.0f;
    // PyTorch tensors are row-major. Interpreting their storage as column-major gives A^T and B^T.
    // Compute C_col = B_col * A_col, which is exactly (A_row * B_row)^T in the same memory buffer.
    cublasStatus_t st = cublasSgemm(g_handle,
                                    CUBLAS_OP_N, CUBLAS_OP_N,
                                    n, n, n,
                                    &alpha,
                                    B.data_ptr<float>(), n,
                                    A.data_ptr<float>(), n,
                                    &beta,
                                    C.data_ptr<float>(), n);
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSgemm failed");
    return C;
}
'''

_ext = load_inline(
    name='symm_matmul_cublas_ext',
    cpp_sources=_cpp_src,
    cuda_sources='',
    functions=['symm_matmul_forward'],
    extra_cflags=['-O3'],
    extra_ldflags=['-lcublas'],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    def forward(self, A, B):
        return _ext.symm_matmul_forward(A, B)

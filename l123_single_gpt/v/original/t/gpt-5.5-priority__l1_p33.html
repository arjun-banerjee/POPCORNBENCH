static __device__ __forceinline__ void warp_reduce_pair(float &a, float &b) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        a += __shfl_down_sync(0xffffffff, a, offset);
        b += __shfl_down_sync(0xffffffff, b, offset);
    }
}

__global__ void reduce_bn_kernel(const float* __restrict__ x,
                                 float* __restrict__ partial,
                                 long long N, int C, long long HW,
                                 int chunks_per_n, long long chunk_size,
                                 int chunks_per_c) {
    int bid = blockIdx.x;
    int c = bid / chunks_per_c;
    int k = bid - c * chunks_per_c;
    long long n = k / chunks_per_n;
    int seg = k - (int)(n * chunks_per_n);
    long long start = (long long)seg * chunk_size;
    long long len = chunk_size;
    if (start + len > HW) len = HW - start;
    const float* ptr = x + ((n * C + c) * HW + start);

    float sum = 0.0f;
    float sum2 = 0.0f;
    for (long long i = threadIdx.x; i < len; i += blockDim.x) {
        float v = ptr[i];
        sum += v;
        sum2 = fmaf(v, v, sum2);
    }

    warp_reduce_pair(sum, sum2);

    __shared__ float smem_sum[32];
    __shared__ float smem_sum2[32];
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    if (lane == 0) {
        smem_sum[warp] = sum;
        smem_sum2[warp] = sum2;
    }
    __syncthreads();

    if (warp == 0) {
        sum = (lane < (blockDim.x >> 5)) ? smem_sum[lane] : 0.0f;
        sum2 = (lane < (blockDim.x >> 5)) ? smem_sum2[lane] : 0.0f;
        warp_reduce_pair(sum, sum2);
        if (lane == 0) {
            long long out = ((long long)c * chunks_per_c + k) * 2LL;

        }
    }

__global__ void finalize_bn_kernel(const float* __restrict__ partial,
                                   float* __restrict__ stats,
                                   float* __restrict__ running_mean,
                                   float* __restrict__ running_var,
                                   int C, int chunks_per_c, long long M,
                                   float eps, float momentum) {
    int c = blockIdx.x;
    float sum = 0.0f;
    float sum2 = 0.0f;
    for (int k = threadIdx.x; k < chunks_per_c; k += blockDim.x) {
        long long idx = ((long long)c * chunks_per_c + k) * 2LL;
        sum += partial[idx];
        sum2 += partial[idx + 1];
    }

    warp_reduce_pair(sum, sum2);

    __shared__ float smem_sum[32];
    __shared__ float smem_sum2[32];
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    if (lane == 0) {
        smem_sum[warp] = sum;
        smem_sum2[warp] = sum2;
    }
    __syncthreads();

    if (warp == 0) {
        sum = (lane < (blockDim.x >> 5)) ? smem_sum[lane] : 0.0f;
        sum2 = (lane < (blockDim.x >> 5)) ? smem_sum2[lane] : 0.0f;
        warp_reduce_pair(sum, sum2);
        if (lane == 0) {
            float invM = 1.0f / (float)M;
            float mean = sum * invM;
            float var = sum2 * invM - mean * mean;
            var = fmaxf(var, 0.0f);
            float invstd = rsqrtf(var + eps);
            stats[2 * c] = mean;
            stats[2 * c + 1] = invstd;

            float old_v = running_var[c];

            running_var[c] = old_v * (1.0f - momentum) + unbiased * momentum;
        }
    }

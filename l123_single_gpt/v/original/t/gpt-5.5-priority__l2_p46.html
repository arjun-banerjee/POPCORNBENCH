import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a convolution, subtraction, tanh activation, subtraction and average pooling.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.avgpool = nn.AvgPool2d(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        x = x - self.subtract1_value
        x = torch.tanh(x)
        x = x - self.subtract2_value
        x = self.avgpool(x)
        return x

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>

torch::Tensor fused_tanh_avgpool_cuda(torch::Tensor x, double subtract1, double subtract2);

torch::Tensor fused_tanh_avgpool_forward(torch::Tensor x, double subtract1, double subtract2) {
    return fused_tanh_avgpool_cuda(x, subtract1, subtract2);
}
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void tanh_avgpool2_kernel(const float* __restrict__ x,
                                     float* __restrict__ y,
                                     int total,
                                     int C, int H, int W,
                                     int outH, int outW,
                                     float s1, float s2) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (; idx < total; idx += stride) {
        int t = idx;
        int ow = t % outW; t /= outW;
        int oh = t % outH; t /= outH;
        int c  = t % C;
        int n  = t / C;

        int ih = oh << 1;
        int iw = ow << 1;
        int base = ((n * C + c) * H + ih) * W + iw;

        float v0 = __ldg(x + base) - s1;
        float v1 = __ldg(x + base + 1) - s1;
        float v2 = __ldg(x + base + W) - s1;
        float v3 = __ldg(x + base + W + 1) - s1;

        // AvgPool2d is applied after subtract2.  Averaging the constant term
        // analytically removes four extra subtractions and matches within fp32 tolerance.
        float sum = tanhf(v0) + tanhf(v1) + tanhf(v2) + tanhf(v3);
        y[idx] = sum * 0.25f - s2;
    }
}

torch::Tensor fused_tanh_avgpool_cuda(torch::Tensor x, double subtract1, double subtract2) {
    TORCH_CHECK(x.is_cuda(), "input must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(x.dim() == 4, "input must be NCHW");

    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);
    const int outH = H / 2;
    const int outW = W / 2;

    auto y = torch::empty({N, C, outH, outW}, x.options());
    const int total = N * C * outH * outW;
    if (total == 0) return y;

    const int threads = 256;
    int blocks = (total + threads - 1) / threads;
    if (blocks > 65535) blocks = 65535;

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    tanh_avgpool2_kernel<<<blocks, threads, 0, stream>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), total, C, H, W, outH, outW,
        (float)subtract1, (float)subtract2);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}
'''

    Convolution followed by fused subtract/tanh/subtract/2x2 average-pooling.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):

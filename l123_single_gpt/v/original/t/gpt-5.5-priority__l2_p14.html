import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a matrix multiplication, division, summation, and scaling.
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(Model, self).__init__()
        self.weight = nn.Parameter(torch.randn(hidden_size, input_size))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, input_size).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, hidden_size).
        """
        x = torch.matmul(x, self.weight.T)  # Gemm
        x = x / 2  # Divide
        x = torch.sum(x, dim=1, keepdim=True) # Sum
        x = x * self.scaling_factor  # Scaling
        return x

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define COL_TILE 32
#define TY 8
#define ROW_CHUNK 128

__global__ void colsum_partial_kernel(const float* __restrict__ w,
                                      float* __restrict__ partial,
                                      int H, int I, int nblocks_h) {
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int col = blockIdx.x * COL_TILE + tx;
    int row0 = blockIdx.y * ROW_CHUNK;

    float s = 0.0f;
    if (col < I) {
        int row_end = row0 + ROW_CHUNK;
        if (row_end > H) row_end = H;
        for (int r = row0 + ty; r < row_end; r += TY) {
            s += w[(long long)r * I + col];
        }
    }

    __shared__ float sm[TY][COL_TILE];
    sm[ty][tx] = s;
    __syncthreads();

    if (ty == 0 && col < I) {
        float t = 0.0f;
        #pragma unroll
        for (int k = 0; k < TY; ++k) t += sm[k][tx];
        partial[(long long)blockIdx.y * I + col] = t;
    }
}

__global__ void colsum_final_kernel(const float* __restrict__ partial,
                                    float* __restrict__ colsum,
                                    int I, int nblocks_h) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (col >= I) return;
    float s = 0.0f;
    for (int b = 0; b < nblocks_h; ++b) {
        s += partial[(long long)b * I + col];
    }
    colsum[col] = s;
}

__global__ void dot_scale_kernel(const float* __restrict__ x,
                                 const float* __restrict__ colsum,
                                 float* __restrict__ out,
                                 int B, int I, float scale) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    float s = 0.0f;
    const float* xb = x + (long long)b * I;
    for (int i = tid; i < I; i += blockDim.x) {
        s = fmaf(xb[i], colsum[i], s);
    }

    __shared__ float sm[256];
    sm[tid] = s;
    __syncthreads();

    // blockDim is 256
    if (tid < 128) sm[tid] += sm[tid + 128];
    __syncthreads();
    if (tid < 64) sm[tid] += sm[tid + 64];
    __syncthreads();
    if (tid < 32) {
        volatile float* vsm = sm;
        vsm[tid] += vsm[tid + 32];
        vsm[tid] += vsm[tid + 16];
        vsm[tid] += vsm[tid + 8];
        vsm[tid] += vsm[tid + 4];
        vsm[tid] += vsm[tid + 2];
        vsm[tid] += vsm[tid + 1];
    }
    if (tid == 0) out[b] = sm[0] * scale;
}

torch::Tensor compute_colsum(torch::Tensor weight) {
    const int H = (int)weight.size(0);
    const int I = (int)weight.size(1);
    const int nbh = (H + ROW_CHUNK - 1) / ROW_CHUNK;
    auto partial = torch::empty({nbh, I}, weight.options());
    auto colsum = torch::empty({I}, weight.options());

    dim3 block(COL_TILE, TY);
    dim3 grid((I + COL_TILE - 1) / COL_TILE, nbh);
    colsum_partial_kernel<<<grid, block>>>(weight.data_ptr<float>(), partial.data_ptr<float>(), H, I, nbh);

    int threads = 256;
    int blocks = (I + threads - 1) / threads;
    colsum_final_kernel<<<blocks, threads>>>(partial.data_ptr<float>(), colsum.data_ptr<float>(), I, nbh);
    return colsum;
}

torch::Tensor dot_scale(torch::Tensor x, torch::Tensor colsum, double scale_d) {
    const int B = (int)x.size(0);
    const int I = (int)x.size(1);

}

__global__ void dw_vertical_generic_kernel(const float* __restrict__ x,
                                           const float* __restrict__ w,
                                           const float* __restrict__ bias,
                                           float* __restrict__ out,
                                           int N, int C, int H, int W,
                                           int Hout, int Wout, int K,
                                           int stride, int padding, int dilation,
                                           bool has_bias,
                                           int64_t total) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % Wout;
    int64_t t = idx / Wout;
    int oh = t % Hout;
    t /= Hout;
    int c = t % C;
    int n = t / C;

    int ih0 = oh * stride - padding;
    int iw = ow * stride - padding;
    float sum = 0.0f;
    if ((unsigned)iw < (unsigned)W) {
        for (int k = 0; k < K; ++k) {
            int ih = ih0 + k * dilation;
            if ((unsigned)ih < (unsigned)H) {
                sum += x[((int64_t)n * C + c) * H * W + (int64_t)ih * W + iw] * w[c * K + k];
            }
        }
    }
    if (has_bias) sum += bias[c];
    out[idx] = sum;
}

        self._stride_i = stride if isinstance(stride, int) else stride[0]
        self._padding_i = padding if isinstance(padding, int) else padding[0]
        self._dilation_i = dilation if isinstance(dilation, int) else dilation[0]
        self.register_buffer("_empty_bias", torch.empty(0), persistent=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.conv2d.bias
        if b is None:
            b = self._empty_bias
            if b.device != x.device or b.dtype != x.dtype:
                b = x.new_empty((0,))
        return _ext.vertical_conv_forward(x, self.conv2d.weight, b, self._stride_i, self._padding_i, self._dilation_i)

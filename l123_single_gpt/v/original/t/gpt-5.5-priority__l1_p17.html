import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cublas_v2.h>

static inline void cublas_check(cublasStatus_t status) {
    TORCH_CHECK(status == CUBLAS_STATUS_SUCCESS, "cuBLAS error: ", (int)status);
}

torch::Tensor matmul_forward(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == torch::kFloat32 && B.scalar_type() == torch::kFloat32, "inputs must be float32");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be 2D");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "inputs must be contiguous");
    const int64_t M64 = A.size(0);
    const int64_t K64 = A.size(1);
    const int64_t N64 = B.size(0);
    TORCH_CHECK(B.size(1) == K64, "inner dimensions must match");
    TORCH_CHECK(M64 <= INT_MAX && N64 <= INT_MAX && K64 <= INT_MAX, "dimensions too large");

    c10::cuda::CUDAGuard guard(A.device());
    auto C = torch::empty({M64, N64}, A.options());

    const int M = (int)M64;
    const int N = (int)N64;
    const int K = (int)K64;
    const float alpha = 1.0f;
    const float beta = 0.0f;

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
    // Row-major C = A * B^T is computed as column-major C^T = B * A^T.
#if defined(CUBLAS_COMPUTE_32F_FAST_TF32)
    cublas_check(cublasGemmEx(
        handle,
        CUBLAS_OP_T, CUBLAS_OP_N,
        N, M, K,
        &alpha,
        B.data_ptr<float>(), CUDA_R_32F, K,
        A.data_ptr<float>(), CUDA_R_32F, K,
        &beta,
        C.data_ptr<float>(), CUDA_R_32F, N,
        CUBLAS_COMPUTE_32F_FAST_TF32,
        CUBLAS_GEMM_DEFAULT_TENSOR_OP));
#else
    cublas_check(cublasSgemm(
        handle,
        CUBLAS_OP_T, CUBLAS_OP_N,
        N, M, K,
        &alpha,
        B.data_ptr<float>(), K,
        A.data_ptr<float>(), K,
        &beta,
        C.data_ptr<float>(), N));
#endif
    return C;
}
'''

_ext = load_inline(
    name='modelnew_cublas_abt_ext',
    cpp_sources=_cpp_src,
    functions=['matmul_forward'],
    with_cuda=True,
    extra_cflags=['-O3'],
    extra_ldflags=['-lcublas'],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return _ext.matmul_forward(A, B)

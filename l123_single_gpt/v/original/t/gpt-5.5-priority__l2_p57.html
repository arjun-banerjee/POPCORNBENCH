import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies ReLU, and applies HardSwish activation.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = torch.relu(x)
        x = x * torch.clamp((x + 3) / 6, 0, 1)
        return x

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE_N 64
#define TILE_M 16

__global__ void conv_relu_hswish_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int C, int H, int W,
    int M, int KH, int KW,
    int OH, int OW,
    int Ktot, int total_pos)
{
    extern __shared__ float smem[];
    float* __restrict__ sW = smem;                         // TILE_M x Ktot
    float* __restrict__ sX = smem + TILE_M * Ktot;          // Ktot x TILE_N

    const int tx = threadIdx.x; // 0..15
    const int ty = threadIdx.y; // 0..15
    const int tid = ty * blockDim.x + tx;
    const int oc_base = blockIdx.y * TILE_M;
    const int p_base  = blockIdx.x * TILE_N;

    // Load the weight tile. Weight is contiguous as [M, C, KH, KW], i.e. [M, Ktot].
    for (int idx = tid; idx < TILE_M * Ktot; idx += 256) {
        int mm = idx / Ktot;
        int kk = idx - mm * Ktot;
        int oc = oc_base + mm;
        sW[idx] = (oc < M) ? w[oc * Ktot + kk] : 0.0f;
    }

    // Load the im2col tile for TILE_N adjacent output pixels.
    for (int idx = tid; idx < Ktot * TILE_N; idx += 256) {
        int kk = idx / TILE_N;
        int col = idx - kk * TILE_N;
        int p = p_base + col;
        float val = 0.0f;
        if (p < total_pos) {
            int ow = p % OW;
            int t = p / OW;
            int oh = t % OH;
            int n  = t / OH;

            int kw = kk % KW;
            int t2 = kk / KW;
            int kh = t2 % KH;
            int ic = t2 / KH;

            val = x[((n * C + ic) * H + (oh + kh)) * W + (ow + kw)];
        }
        sX[idx] = val;
    }

    __syncthreads();

    const int oc = oc_base + ty;
    float acc0 = 0.0f, acc1 = 0.0f, acc2 = 0.0f, acc3 = 0.0f;
    if (oc < M) {
        float bv = b[oc];
        acc0 = bv; acc1 = bv; acc2 = bv; acc3 = bv;

        #pragma unroll 4
        for (int kk = 0; kk < Ktot; ++kk) {
            float ww = sW[ty * Ktot + kk];
            const float* sx = sX + kk * TILE_N + tx;
            acc0 = fmaf(ww, sx[0],  acc0);
            acc1 = fmaf(ww, sx[16], acc1);
            acc2 = fmaf(ww, sx[32], acc2);
            acc3 = fmaf(ww, sx[48], acc3);
        }
    }

    // Store four columns per thread, applying ReLU and HardSwish fused.
    int cols[4] = {tx, tx + 16, tx + 32, tx + 48};
    float vals[4] = {acc0, acc1, acc2, acc3};

    if (oc < M) {
        #pragma unroll
        for (int i = 0; i < 4; ++i) {
            int p = p_base + cols[i];
            if (p < total_pos) {
                float v = vals[i];
                v = v > 0.0f ? v : 0.0f;
                float h = fminf(fmaxf((v + 3.0f) * 0.16666666666666666f, 0.0f), 1.0f);

                int oh = t % OH;

        }
    }
}

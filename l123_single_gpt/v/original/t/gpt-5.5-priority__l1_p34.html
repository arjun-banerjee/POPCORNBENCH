import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs Instance Normalization.
    """
    def __init__(self, num_features: int):
        """
        Initializes the InstanceNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
        """
        super(Model, self).__init__()
        self.inorm = nn.InstanceNorm2d(num_features=num_features)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Instance Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, height, width).

        Returns:
            torch.Tensor: Output tensor with Instance Normalization applied, same shape as input.
        """
        return self.inorm(x)

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''

    return instance_norm_forward_cuda(x, eps);
}

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

__inline__ __device__ float warp_reduce_sum(float v) {
    v += __shfl_down_sync(0xffffffff, v, 16);
    v += __shfl_down_sync(0xffffffff, v, 8);
    v += __shfl_down_sync(0xffffffff, v, 4);
    v += __shfl_down_sync(0xffffffff, v, 2);
    v += __shfl_down_sync(0xffffffff, v, 1);
    return v;
}

template<int BLOCK>
__global__ void instancenorm2d_fwd_kernel(const float* __restrict__ x,
                                          float* __restrict__ y,
                                          int64_t rows,
                                          int64_t hw,
                                          float eps) {
    const int row = (int)blockIdx.x;
    if ((int64_t)row >= rows) return;

    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int warp = tid >> 5;

    const float* __restrict__ xp = x + (int64_t)row * hw;
    float* __restrict__ yp = y + (int64_t)row * hw;

    float sum = 0.0f;
    float sumsq = 0.0f;

    // Fast vector path for the benchmark shape (H*W divisible by 4 and aligned).
    if (((hw & 3LL) == 0) && ((((uintptr_t)xp) & 15ULL) == 0) && ((((uintptr_t)yp) & 15ULL) == 0)) {
        const float4* __restrict__ x4 = reinterpret_cast<const float4*>(xp);
        const int64_t n4 = hw >> 2;
        for (int64_t i = tid; i < n4; i += BLOCK) {
            float4 v = x4[i];
            sum += (v.x + v.y) + (v.z + v.w);
            sumsq = fmaf(v.x, v.x, sumsq);
            sumsq = fmaf(v.y, v.y, sumsq);
            sumsq = fmaf(v.z, v.z, sumsq);
            sumsq = fmaf(v.w, v.w, sumsq);
        }
    } else {
        for (int64_t i = tid; i < hw; i += BLOCK) {
            float v = xp[i];
            sum += v;
            sumsq = fmaf(v, v, sumsq);
        }
    }

    __shared__ float sh_sum[32];
    __shared__ float sh_sumsq[32];
    __shared__ float sh_param[2];

    sum = warp_reduce_sum(sum);
    sumsq = warp_reduce_sum(sumsq);
    if (lane == 0) {
        sh_sum[warp] = sum;
        sh_sumsq[warp] = sumsq;
    }
    __syncthreads();

    if (warp == 0) {
        float bsum = (lane < (BLOCK >> 5)) ? sh_sum[lane] : 0.0f;
        float bsumsq = (lane < (BLOCK >> 5)) ? sh_sumsq[lane] : 0.0f;
        bsum = warp_reduce_sum(bsum);
        bsumsq = warp_reduce_sum(bsumsq);
        if (lane == 0) {
            float inv_hw = 1.0f / (float)hw;
            float mean = bsum * inv_hw;
            float var = bsumsq * inv_hw - mean * mean;
            var = var < 0.0f ? 0.0f : var;
            sh_param[0] = mean;
            sh_param[1] = rsqrtf(var + eps);
        }
    }
    __syncthreads();

    const float mean = sh_param[0];
    const float invstd = sh_param[1];

    if (((hw & 3LL) == 0) && ((((uintptr_t)xp) & 15ULL) == 0) && ((((uintptr_t)yp) & 15ULL) == 0)) {
        const float4* __restrict__ x4 = reinterpret_cast<const float4*>(xp);
        float4* __restrict__ y4 = reinterpret_cast<float4*>(yp);
        const int64_t n4 = hw >> 2;
        for (int64_t i = tid; i < n4; i += BLOCK) {
            float4 v = x4[i];
            v.x = (v.x - mean) * invstd;
            v.y = (v.y - mean) * invstd;
            v.z = (v.z - mean) * invstd;
            v.w = (v.w - mean) * invstd;
            y4[i] = v;
        }
    } else {
        for (int64_t i = tid; i < hw; i += BLOCK) {
            yp[i] = (xp[i] - mean) * invstd;
        }
    }
}

torch::Tensor instance_norm_forward_cuda(torch::Tensor x, double eps) {
    TORCH_CHECK(x.is_cuda(), "input must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 is supported");
    TORCH_CHECK(x.dim() == 4, "input must be NCHW");

    if (!x.is_contiguous()) {
        x = x.contiguous();
    }

    auto y = torch::empty_like(x);
    const int64_t n = x.size(0);
    const int64_t c = x.size(1);
    const int64_t h = x.size(2);
    const int64_t w = x.size(3);
    const int64_t rows = n * c;
    const int64_t hw = h * w;

    if (rows == 0 || hw == 0) return y;

    constexpr int BLOCK = 256;
    dim3 block(BLOCK);
    dim3 grid((unsigned int)rows);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    instancenorm2d_fwd_kernel<BLOCK><<<grid, block, 0, stream>>>(
        x.data_ptr<float>(), y.data_ptr<float>(), rows, hw, (float)eps);
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}

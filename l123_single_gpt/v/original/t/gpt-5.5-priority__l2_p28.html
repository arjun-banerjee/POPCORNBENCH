import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a batch matrix multiplication, instance normalization, summation, residual addition, and multiplication.
    """
    def __init__(self, in_features, out_features, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.bmm = nn.Linear(in_features, out_features)
        self.instance_norm = nn.InstanceNorm2d(out_features, eps=eps, momentum=momentum)

    def forward(self, x, y):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
            y (torch.Tensor): Input tensor of shape (batch_size, out_features).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.bmm(x)
        x = self.instance_norm(x.unsqueeze(1).unsqueeze(1)).squeeze(1).squeeze(1)
        x = x + y
        x = x * y
        return x

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline


_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor norm_residual_forward(torch::Tensor z, torch::Tensor y, double eps);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>

// One CTA handles one row.  For the benchmark shape C=8192, this gives a
// compact, bandwidth-efficient replacement for InstanceNorm2d+squeeze+add+mul.
template<int BLOCK>
__global__ void row_norm_residual_kernel(float* __restrict__ z,
                                         const float* __restrict__ y,
                                         int rows, int cols, float eps) {
    int r = blockIdx.x;
    if (r >= rows) return;

    const int base = r * cols;
    float sum = 0.0f;
    float sum2 = 0.0f;

    for (int c = threadIdx.x; c < cols; c += BLOCK) {
        float v = z[base + c];
        sum += v;
        sum2 += v * v;
    }

    __shared__ float sh_sum[BLOCK];
    __shared__ float sh_sum2[BLOCK];
    sh_sum[threadIdx.x] = sum;
    sh_sum2[threadIdx.x] = sum2;
    __syncthreads();

    #pragma unroll
    for (int stride = BLOCK / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh_sum[threadIdx.x] += sh_sum[threadIdx.x + stride];
            sh_sum2[threadIdx.x] += sh_sum2[threadIdx.x + stride];
        }
        __syncthreads();
    }

    float mean = sh_sum[0] / (float)cols;
    float var = sh_sum2[0] / (float)cols - mean * mean;
    // Guard only against tiny negative roundoff in the E[x^2]-E[x]^2 formula.
    var = var < 0.0f ? 0.0f : var;
    float inv_std = rsqrtf(var + eps);

    for (int c = threadIdx.x; c < cols; c += BLOCK) {
        float yy = y[base + c];
        float n = (z[base + c] - mean) * inv_std;
        z[base + c] = (n + yy) * yy;
    }
}

torch::Tensor norm_residual_forward(torch::Tensor z, torch::Tensor y, double eps) {
    TORCH_CHECK(z.is_cuda(), "z must be a CUDA tensor");
    TORCH_CHECK(y.is_cuda(), "y must be a CUDA tensor");
    TORCH_CHECK(z.scalar_type() == at::kFloat, "z must be float32");
    TORCH_CHECK(y.scalar_type() == at::kFloat, "y must be float32");
    TORCH_CHECK(z.dim() == 2 && y.dim() == 2, "expected 2D tensors");
    TORCH_CHECK(z.size(0) == y.size(0) && z.size(1) == y.size(1), "shape mismatch");
    TORCH_CHECK(z.is_contiguous(), "z must be contiguous");
    TORCH_CHECK(y.is_contiguous(), "y must be contiguous");

    const int cols = (int)z.size(1);

    C10_CUDA_KERNEL_LAUNCH_CHECK();

}

class ModelNew(nn.Module):
    """
    Equivalent to the reference model.  The expensive GEMM is left to PyTorch's
    cuBLAS-backed linear implementation, while the normalization, residual add,
    and multiply are fused into a single CUDA kernel.
    """
    def __init__(self, in_features, out_features, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.bmm = nn.Linear(in_features, out_features)
        # Kept for state/module compatibility with the reference.  It has no
        # affine parameters/buffers by default, but preserves constructor shape.
        self.instance_norm = nn.InstanceNorm2d(out_features, eps=eps, momentum=momentum)
        self.eps = float(eps)

    def forward(self, x, y):
        z = F.linear(x, self.bmm.weight, self.bmm.bias)
        if not y.is_contiguous():
            y = y.contiguous()
        return _ext.norm_residual_forward(z, y, self.eps)

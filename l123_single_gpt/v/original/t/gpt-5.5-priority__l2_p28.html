import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a batch matrix multiplication, instance normalization, summation, residual addition, and multiplication.
    """
    def __init__(self, in_features, out_features, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.bmm = nn.Linear(in_features, out_features)
        self.instance_norm = nn.InstanceNorm2d(out_features, eps=eps, momentum=momentum)

    def forward(self, x, y):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
            y (torch.Tensor): Input tensor of shape (batch_size, out_features).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.bmm(x)
        x = self.instance_norm(x.unsqueeze(1).unsqueeze(1)).squeeze(1).squeeze(1)
        x = x + y
        x = x * y
        return x

// One block per row. Computes biased instance norm across columns and fuses
// residual add + multiply: out = ( (x-mean)*rsqrt(var+eps) + y ) * y.
__global__ void row_instnorm_resmul_kernel(const float* __restrict__ x,
                                           const float* __restrict__ y,
                                           float* __restrict__ out,
                                           int rows, int cols, float eps) {
    int row = blockIdx.x;
    if (row >= rows) return;
    int tid = threadIdx.x;
    const float* xr = x + (long long)row * cols;
    const float* yr = y + (long long)row * cols;
    float* orow = out + (long long)row * cols;

    float sum = 0.0f;
    float sumsq = 0.0f;
    // Vectorized loads when cols is divisible by 4 and aligned (true for task tensors).
    int n4 = cols >> 2;
    const float4* x4 = reinterpret_cast<const float4*>(xr);
    for (int i = tid; i < n4; i += blockDim.x) {
        float4 v = x4[i];
        sum += v.x + v.y + v.z + v.w;
        sumsq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
    }
    for (int i = (n4 << 2) + tid; i < cols; i += blockDim.x) {
        float v = xr[i];
        sum += v;
        sumsq += v*v;
    }

    __shared__ float sh_sum[256];
    __shared__ float sh_sumsq[256];
    sh_sum[tid] = sum;
    sh_sumsq[tid] = sumsq;
    __syncthreads();

    // blockDim is 256
    for (int s = 128; s > 0; s >>= 1) {
        if (tid < s) {
            sh_sum[tid] += sh_sum[tid + s];
            sh_sumsq[tid] += sh_sumsq[tid + s];
        }
        __syncthreads();
    }

    float mean = sh_sum[0] / (float)cols;
    float var = sh_sumsq[0] / (float)cols - mean * mean;
    // Avoid tiny negative variance from roundoff.
    if (var < 0.0f) var = 0.0f;
    float invstd = rsqrtf(var + eps);

    float4* o4 = reinterpret_cast<float4*>(orow);
    const float4* y4 = reinterpret_cast<const float4*>(yr);
    for (int i = tid; i < n4; i += blockDim.x) {
        float4 xv = x4[i];
        float4 yv = y4[i];
        float4 ov;
        ov.x = ((xv.x - mean) * invstd + yv.x) * yv.x;
        ov.y = ((xv.y - mean) * invstd + yv.y) * yv.y;
        ov.z = ((xv.z - mean) * invstd + yv.z) * yv.z;
        ov.w = ((xv.w - mean) * invstd + yv.w) * yv.w;
        o4[i] = ov;
    }
    for (int i = (n4 << 2) + tid; i < cols; i += blockDim.x) {
        float yy = yr[i];
        orow[i] = ((xr[i] - mean) * invstd + yy) * yy;
    }
}

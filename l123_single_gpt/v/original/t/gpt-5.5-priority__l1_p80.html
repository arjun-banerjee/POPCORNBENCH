                                  const float* __restrict__ bias,

                                      const float* __restrict__ w,
                                      const float* __restrict__ bias,
                                      float* __restrict__ out,
                                      int N, int C, int H, int W, int OC,
                                      int KH, int KW, int Hout, int Wout,
                                      int stride, int pad_h, int pad_w, int dil_h, int dil_w,
                                      int has_bias, long total) {

    if (idx >= total) return;
    int ow = idx % Wout;
    long t = idx / Wout;
    int oh = t % Hout;
    t /= Hout;

    int n = t / OC;

    for (int c = 0; c < C; ++c) {

}

        self.conv2d = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride,
                                padding=padding, dilation=dilation, bias=bias)
        self.stride = int(stride)
        self.padding = (int(padding[0]), int(padding[1]))
        self.dilation = (int(dilation[0]), int(dilation[1]))
        self.register_buffer('_empty_bias', torch.empty(0), persistent=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.conv2d.bias if self.conv2d.bias is not None else self._empty_bias
        return _ext.conv2d_forward_cuda(x, self.conv2d.weight, b,
                                        self.stride, self.padding[0], self.padding[1],
                                        self.dilation[0], self.dilation[1])

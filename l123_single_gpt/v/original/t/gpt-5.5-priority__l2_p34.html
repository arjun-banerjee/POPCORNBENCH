import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D transposed convolution, layer normalization, GELU activation, and scaling.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True, eps=1e-5, scaling_factor=1.0):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.layer_norm = nn.LayerNorm(out_channels, eps=eps)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, D, H, W).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, D', H', W').
        """
        x = self.conv_transpose(x)
        x = self.layer_norm(x)
        x = torch.nn.functional.gelu(x)
        x = x * self.scaling_factor
        return x

#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

// One warp computes one output row (fixed n, co, d, h, all 64 w).  This keeps the
// layer-normalization reduction in registers and avoids writing the intermediate
// convolution result to memory.  The benchmark has N=32,Cin=32,Cout=64,D=16,H=W=32,
// k=4,stride=2,pad=1, hence output N,64,32,64,64.

template<int BD>
__global__ __launch_bounds__(256, 2)
void fused_kernel(const float* __restrict__ x,
                  const float* __restrict__ w,
                  const float* __restrict__ cb,
                  const float* __restrict__ lnw,
                  const float* __restrict__ lnb,
                  float* __restrict__ out,
                  const float eps, const float scale,
                  const int total_rows) {
    const int lane = threadIdx.x & 31;
    const int warp = threadIdx.x >> 5;
    const int warps_per_block = blockDim.x >> 5;
    int row = blockIdx.x * warps_per_block + warp;
    if (row >= total_rows) return;

    // row packs ((((n*Cout + co)*Do + od)*Ho + oh)); W row is contiguous.
    const int oh = row & 63;
    int t = row >> 6;
    const int od = t & 31;
    t >>= 5;
    const int co = t & 63;
    const int n = t >> 6;

    float v0 = 0.f, v1 = 0.f; // lane computes ow=lane and lane+32

    #pragma unroll
    for (int pass = 0; pass < 2; ++pass) {
        const int ow = lane + pass * 32;
        float acc = cb ? cb[co] : 0.0f;
        // valid kernel taps for stride-2 transposed convolution
        const int kd0 = (od + 1) & 1;
        const int kh0 = (oh + 1) & 1;
        const int kw0 = (ow + 1) & 1;
        #pragma unroll
        for (int kdz = 0; kdz < 2; ++kdz) {
            int kd = kd0 + (kdz << 1);
            int id = (od - kd + 1) >> 1;
            if ((unsigned)id >= 16u) continue;
            #pragma unroll
            for (int khz = 0; khz < 2; ++khz) {
                int kh = kh0 + (khz << 1);
                int ih = (oh - kh + 1) >> 1;
                if ((unsigned)ih >= 32u) continue;
                #pragma unroll
                for (int kwz = 0; kwz < 2; ++kwz) {
                    int kw = kw0 + (kwz << 1);
                    int iw = (ow - kw + 1) >> 1;
                    if ((unsigned)iw >= 32u) continue;
                    const float* xp = x + (((n * 32) * 16 + id) * 32 + ih) * 32 + iw;
                    const float* wp = w + ((((0 * 64 + co) * 4 + kd) * 4 + kh) * 4 + kw);
                    #pragma unroll 4
                    for (int ci = 0; ci < 32; ++ci) {
                        acc = fmaf(xp[ci * 16 * 32 * 32], wp[ci * 64 * 4 * 4 * 4], acc);
                    }
                }
            }
        }
        if (pass == 0) v0 = acc; else v1 = acc;
    }

    float sum = v0 + v1;
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) sum += __shfl_down_sync(0xffffffff, sum, off);
    sum = __shfl_sync(0xffffffff, sum, 0);
    const float mean = sum * (1.0f / 64.0f);

    float d0 = v0 - mean;
    float d1 = v1 - mean;
    float ss = d0*d0 + d1*d1;
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) ss += __shfl_down_sync(0xffffffff, ss, off);
    ss = __shfl_sync(0xffffffff, ss, 0);
    const float inv = rsqrtf(ss * (1.0f / 64.0f) + eps);

    const long long base = ((long long)row) * 64LL;
    float z0 = d0 * inv * lnw[lane] + lnb[lane];
    float z1 = d1 * inv * lnw[lane + 32] + lnb[lane + 32];
    // exact-ish GELU (PyTorch default): 0.5*x*(1+erf(x/sqrt(2)))
    z0 = 0.5f * z0 * (1.0f + erff(z0 * 0.70710678118654752440f)) * scale;
    z1 = 0.5f * z1 * (1.0f + erff(z1 * 0.70710678118654752440f)) * scale;
    out[base + lane] = z0;
    out[base + lane + 32] = z1;
}

void deconv_ln_gelu_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor cb,
                         torch::Tensor lnw, torch::Tensor lnb, torch::Tensor out,
                         double eps, double scale) {
    const int N = (int)x.size(0);
    const int Cout = (int)w.size(1);
    const int Do = (int)out.size(2);
    const int Ho = (int)out.size(3);

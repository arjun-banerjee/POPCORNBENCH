                                    int64_t stride64, int64_t padding64, int64_t dilation64) {
    TORCH_CHECK(x.is_cuda() && weight.is_cuda(), "tensors must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat, "float32 only");
    x = x.contiguous();
    weight = weight.contiguous();
    int N = (int)x.size(0), Cin = (int)x.size(1), D = (int)x.size(2), H = (int)x.size(3), W = (int)x.size(4);
    int Cout = (int)weight.size(1), K = (int)weight.size(2);
    int stride = (int)stride64, padding = (int)padding64, dilation = (int)dilation64;
    int outD = (D - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    int outH = (H - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    int outW = (W - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    auto y = at::empty({N, Cout, outD, outH, outW}, x.options());
    bool has_bias = (bias.defined() && bias.numel() > 0);
    if (has_bias) bias = bias.contiguous();
    const float* bptr = has_bias ? bias.data_ptr<float>() : nullptr;
    float* yptr = y.data_ptr<float>();
    auto stream = at::cuda::getCurrentCUDAStream();
    const int threads = 256;

    if (stride == 2 && padding == 1 && dilation == 2 && K == 3) {
        long total = y.numel();
        if (has_bias) {
            int blocks_init = (int)std::min<long>((total + threads - 1) / threads, 65535L);
            init_bias_kernel<<<blocks_init, threads, 0, stream>>>(yptr, bptr, total, Cout, outD, outH, outW);
        } else {
            C10_CUDA_CHECK(cudaMemsetAsync(yptr, 0, total * sizeof(float), stream));
        }
        long nz = (long)N * Cout * D * H * W;
        int blocks = (int)((nz + threads - 1) / threads);
        deconv3d_s2p1d2_k3_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), weight.data_ptr<float>(), bptr, yptr,
                                                                  nz, N, Cin, Cout, D, H, W, outD, outH, outW, has_bias ? 1 : 0);
    } else {
        long total = y.numel();
        int blocks = (int)((total + threads - 1) / threads);
        deconv3d_generic_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), weight.data_ptr<float>(), bptr, yptr,
                                                               total, N, Cin, Cout, D, H, W, K, stride, padding, dilation,
                                                               outD, outH, outW, has_bias ? 1 : 0);
    }
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return y;
}

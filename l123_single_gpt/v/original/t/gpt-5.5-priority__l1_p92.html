import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs an exclusive cumulative sum (does not include the current element).

    Parameters:
        dim (int): The dimension along which to perform the exclusive cumulative sum.
    """

    def __init__(self, dim):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x):
        cumsum = torch.cumsum(x.narrow(dim=self.dim, start=0, length=x.size(self.dim)-1), dim=self.dim)
        return torch.cat((torch.zeros_like(x.select(self.dim, 0).unsqueeze(self.dim)), cumsum), dim=self.dim)

// One CUDA block computes one row of length 32768.  Each thread scans a
// contiguous tile, and CUB scans the per-thread tile sums.
template<int BLOCK_THREADS, int ITEMS>
__global__ void exclusive_cumsum_32768_kernel(const float* __restrict__ x,
                                              float* __restrict__ y,
                                              int rows) {
    using BlockScan = cub::BlockScan<float, BLOCK_THREADS>;
    __shared__ typename BlockScan::TempStorage temp_storage;

    int row = blockIdx.x;
    if (row >= rows) return;

    const float* row_x = x + ((long long)row) * (BLOCK_THREADS * ITEMS);
    float* row_y = y + ((long long)row) * (BLOCK_THREADS * ITEMS);

    int tid = threadIdx.x;
    int base = tid * ITEMS;

    float vals[ITEMS];
    float thread_sum = 0.0f;

    #pragma unroll
    for (int i = 0; i < ITEMS; ++i) {
        float v = row_x[base + i];
        vals[i] = v;
        thread_sum += v;
    }

    float thread_prefix;
    BlockScan(temp_storage).ExclusiveSum(thread_sum, thread_prefix);

    float run = thread_prefix;
    #pragma unroll
    for (int i = 0; i < ITEMS; ++i) {
        row_y[base + i] = run;
        run += vals[i];
    }
}

// Generic 2D fallback for other row lengths.  Assumes contiguous rows.
template<int BLOCK_THREADS>
__global__ void exclusive_cumsum_generic_kernel(const float* __restrict__ x,
                                                float* __restrict__ y,
                                                int rows, int cols) {
    using BlockScan = cub::BlockScan<float, BLOCK_THREADS>;
    __shared__ typename BlockScan::TempStorage temp_storage;

    int row = blockIdx.x;
    if (row >= rows) return;
    const float* row_x = x + ((long long)row) * cols;
    float* row_y = y + ((long long)row) * cols;

    float carry = 0.0f;
    for (int tile = 0; tile < cols; tile += BLOCK_THREADS) {
        int idx = tile + threadIdx.x;
        float v = (idx < cols) ? row_x[idx] : 0.0f;
        float excl;
        BlockScan(temp_storage).ExclusiveSum(v, excl);
        if (idx < cols) y[((long long)row) * cols + idx] = carry + excl;
        __syncthreads();
        if (threadIdx.x == 0) {
            int last = min(tile + BLOCK_THREADS, cols) - 1;
            carry += row_y[last] - carry + row_x[last];
        }
        __syncthreads();
    }
}

torch::Tensor exclusive_cumsum_forward(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");

    auto xc = x.contiguous();
    auto y = torch::empty_like(xc);

    int64_t ndim = xc.dim();
    TORCH_CHECK(ndim >= 1, "x must have at least one dimension");
    int64_t cols64 = xc.size(ndim - 1);
    int64_t rows64 = xc.numel() / cols64;
    TORCH_CHECK(rows64 <= INT_MAX && cols64 <= INT_MAX, "tensor too large for this kernel");
    int rows = (int)rows64;
    int cols = (int)cols64;

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

void selu_forward_cuda(torch::Tensor x, torch::Tensor y);

torch::Tensor selu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    selu_forward_cuda(x, y);
    return y;
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <stdint.h>

static constexpr float SELU_SCALE = 1.0507009873554804934193349852946f;
static constexpr float SELU_ALPHA_SCALE = 1.7580993408473768599402175208123f;

__device__ __forceinline__ float selu_op(float v) {
    // PyTorch SELU: scale * (v > 0 ? v : alpha * (exp(v)-1))
    return (v > 0.0f) ? (SELU_SCALE * v) : (SELU_ALPHA_SCALE * expm1f(v));
}

__global__ void selu_float4_kernel(const float4* __restrict__ x4,
                                   float4* __restrict__ y4,
                                   int64_t n4) {
    int64_t tid = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (int64_t i = tid; i < n4; i += stride) {
        float4 a = x4[i];
        float4 b;
        b.x = selu_op(a.x);
        b.y = selu_op(a.y);
        b.z = selu_op(a.z);
        b.w = selu_op(a.w);
        y4[i] = b;
    }
}

__global__ void selu_float_kernel(const float* __restrict__ x,
                                  float* __restrict__ y,
                                  int64_t n) {
    int64_t tid = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (int64_t i = tid; i < n; i += stride) {
        float v = x[i];
        y[i] = selu_op(v);
    }
}

void selu_forward_cuda(torch::Tensor x, torch::Tensor y) {
    const int64_t n = x.numel();
    if (n == 0) return;

    const int threads = 256;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float* xp = x.data_ptr<float>();
    float* yp = y.data_ptr<float>();

    uintptr_t xa = reinterpret_cast<uintptr_t>(xp);
    uintptr_t ya = reinterpret_cast<uintptr_t>(yp);
    if (((xa | ya) & 15ULL) == 0 && (n % 4 == 0)) {

        if (blocks > 65535) blocks = 65535;

            reinterpret_cast<const float4*>(xp), reinterpret_cast<float4*>(yp), n4);

}

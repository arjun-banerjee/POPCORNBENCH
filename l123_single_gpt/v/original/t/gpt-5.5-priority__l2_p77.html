import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D transposed convolution, scales the output, applies batch normalization, 
    and then performs global average pooling. 
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor
        self.batch_norm = nn.BatchNorm3d(out_channels, eps=eps, momentum=momentum)
        self.global_avg_pool = nn.AdaptiveAvgPool3d((1, 1, 1))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = x * self.scale_factor
        x = self.batch_norm(x)
        x = self.global_avg_pool(x)
        return x

    extern __shared__ float sdata[];
    sdata[threadIdx.x] = acc2;
    __syncthreads();
    for (int s = blockDim.x >> 1; s > 0; s >>= 1) {
        if (threadIdx.x < s) sdata[threadIdx.x] += sdata[threadIdx.x + s];
        __syncthreads();
    }
    if (threadIdx.x == 0) atomicAdd(sumsq + oc, sdata[0]);
}

__global__ void train_out_kernel(const float* __restrict__ conv_sum,
                                 const float* __restrict__ sumsq,
                                 const float* __restrict__ gamma,
                                 const float* __restrict__ beta,
                                 float* __restrict__ out,
                                 int N, int Cout, int Vout,
                                 float scale, float eps) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout;
    if (idx >= total) return;
    int n = idx / Cout;
    int oc = idx - n * Cout;
    float tsum = 0.0f;
    for (int i = 0; i < N; ++i) tsum += conv_sum[i * Cout + oc];
    float invM = 1.0f / (float)(N * Vout);
    float mean = tsum * scale * invM;
    float var = sumsq[oc] * invM - mean * mean;
    if (var < 0.0f) var = 0.0f;
    float avg = conv_sum[n * Cout + oc] * scale / (float)Vout;
    out[idx] = (avg - mean) * rsqrtf(var + eps) * gamma[oc] + beta[oc];
}

                            torch::Tensor bias,

    int D = (int)x.size(2);
    int H = (int)x.size(3);
    int Wd = (int)x.size(4);
    int Cout = (int)weight.size(1);
    int Kd = (int)weight.size(2);
    int Kh = (int)weight.size(3);
    int Kw = (int)weight.size(4);
    int OD = D + Kd - 1;
    int OH = H + Kh - 1;
    int OW = Wd + Kw - 1;
    int Vout = OD * OH * OW;
    auto opts = x.options();
    auto in_sum = torch::empty({N, Cin}, opts);
    auto w_sum = torch::empty({Cin, Cout}, opts);
    auto out = torch::empty({N, Cout, 1, 1, 1}, opts);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    const int T = 256;
    input_sum_kernel<<<N * Cin, T, T * sizeof(float), stream>>>(x.data_ptr<float>(), in_sum.data_ptr<float>(), N, Cin, D, H, Wd);
    weight_sum_kernel<<<Cin * Cout, T, T * sizeof(float), stream>>>(weight.data_ptr<float>(), w_sum.data_ptr<float>(), Cin, Cout, Kd, Kh, Kw);
    float scale = (float)scale_d;
    float eps = (float)eps_d;
    if (!training) {

        auto sumsq = torch::zeros({Cout}, opts);

            out.data_ptr<float>(), N, Cout, Vout, scale, eps);
    }
    return out;
}

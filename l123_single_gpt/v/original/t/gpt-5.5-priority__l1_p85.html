#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ __launch_bounds__(256, 2)
void dwconv3x7_w256_kernel(const float* __restrict__ x,
                           const float* __restrict__ w,
                           const float* __restrict__ b,
                           float* __restrict__ y,
                           int N, int C, int H, int Hout, int Wout,
                           bool has_bias) {
    // One block computes up to four adjacent output rows for one (N,C) plane,
    // and the full output width.  For the benchmark W=256, Wout=250.
    __shared__ float tile[6][256];

    const int c = blockIdx.x;
    const int n = blockIdx.y;
    const int oh0 = blockIdx.z * 4;
    const int tid = threadIdx.x;

    const float* xnc = x + ((n * C + c) * H) * 256;

    #pragma unroll
    for (int r = 0; r < 6; ++r) {
        int ih = oh0 + r;
        tile[r][tid] = (ih < H) ? xnc[ih * 256 + tid] : 0.0f;
    }
    __syncthreads();

    if (tid >= Wout) return;

    const float* wc = w + c * 21;
    float acc0 = has_bias ? b[c] : 0.0f;
    float acc1 = acc0;
    float acc2 = acc0;
    float acc3 = acc0;

    #pragma unroll
    for (int kh = 0; kh < 3; ++kh) {
        #pragma unroll
        for (int kw = 0; kw < 7; ++kw) {
            float ww = __ldg(wc + kh * 7 + kw);
            int col = tid + kw;
            acc0 = fmaf(tile[kh + 0][col], ww, acc0);
            acc1 = fmaf(tile[kh + 1][col], ww, acc1);
            acc2 = fmaf(tile[kh + 2][col], ww, acc2);
            acc3 = fmaf(tile[kh + 3][col], ww, acc3);
        }

    ync[tid] = acc0;

    if (oh0 + 2 < Hout) ync[2 * Wout + tid] = acc2;
    if (oh0 + 3 < Hout) ync[3 * Wout + tid] = acc3;
}

__global__ __launch_bounds__(256, 2)
void dwconv_generic_kernel(const float* __restrict__ x,
                           const float* __restrict__ w,
                           const float* __restrict__ b,
                           float* __restrict__ y,
                           long total,
                           int N, int C, int H, int W,
                           int Kh, int Kw, int Hout, int Wout,
                           int sh, int sw, int ph, int pw, int dh, int dw,
                           bool has_bias) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = (int)(idx % Wout);
    long t = idx / Wout;
    int oh = (int)(t % Hout);
    t /= Hout;
    int c = (int)(t % C);
    int n = (int)(t / C);

    float acc = has_bias ? b[c] : 0.0f;
    const float* xnc = x + ((n * C + c) * H) * W;
    const float* wc = w + c * Kh * Kw;

    for (int kh = 0; kh < Kh; ++kh) {
        int ih = oh * sh - ph + kh * dh;
        if ((unsigned)ih >= (unsigned)H) continue;
        for (int kw = 0; kw < Kw; ++kw) {
            int iw = ow * sw - pw + kw * dw;
            if ((unsigned)iw < (unsigned)W) {
                acc = fmaf(xnc[ih * W + iw], wc[kh * Kw + kw], acc);
            }
        }
    }
    y[idx] = acc;
}

                             int64_t stride_h, int64_t stride_w,
                             int64_t padding_h, int64_t padding_w,

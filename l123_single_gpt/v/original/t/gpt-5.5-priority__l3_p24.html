import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, num_classes=1000):
        """
        EfficientNetB2 architecture implementation.

        :param num_classes: The number of output classes (default is 1000 for ImageNet).
        """
        super(Model, self).__init__()
        
        # Define the EfficientNetB2 architecture components
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, stride=2, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(32)
        self.relu = nn.ReLU(inplace=True)
        
        # Define the MBConv blocks
        self.mbconv1 = self._make_mbconv_block(32, 96, 1, 3)
        self.mbconv2 = self._make_mbconv_block(96, 144, 2, 6)
        self.mbconv3 = self._make_mbconv_block(144, 192, 2, 6)
        self.mbconv4 = self._make_mbconv_block(192, 288, 2, 6)
        self.mbconv5 = self._make_mbconv_block(288, 384, 1, 6)
        
        # Final layers
        self.conv_final = nn.Conv2d(384, 1408, kernel_size=1, stride=1, padding=0, bias=False)
        self.bn_final = nn.BatchNorm2d(1408)
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.fc = nn.Linear(1408, num_classes)
    
    def _make_mbconv_block(self, in_channels, out_channels, stride, expand_ratio):
        """
        Helper function to create a MBConv block.

        :param in_channels: Number of input channels.
        :param out_channels: Number of output channels.
        :param stride: Stride for the depthwise convolution.
        :param expand_ratio: Expansion ratio for the MBConv block.
        :return: A sequential container of layers forming the MBConv block.
        """
        layers = []
        expanded_channels = in_channels * expand_ratio
        
        # Expansion phase
        if expand_ratio != 1:
            layers.append(nn.Conv2d(in_channels, expanded_channels, kernel_size=1, stride=1, padding=0, bias=False))
            layers.append(nn.BatchNorm2d(expanded_channels))
            layers.append(nn.ReLU(inplace=True))
        
        # Depthwise convolution
        layers.append(nn.Conv2d(expanded_channels, expanded_channels, kernel_size=3, stride=stride, padding=1, groups=expanded_channels, bias=False))
        layers.append(nn.BatchNorm2d(expanded_channels))
        layers.append(nn.ReLU(inplace=True))
        
        # Squeeze and Excitation
        layers.append(nn.AdaptiveAvgPool2d((1, 1)))
        layers.append(nn.Conv2d(expanded_channels, expanded_channels // 4, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(nn.ReLU(inplace=True))
        layers.append(nn.Conv2d(expanded_channels // 4, expanded_channels, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(nn.Sigmoid())
        
        # Output phase
        layers.append(nn.Conv2d(expanded_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(nn.BatchNorm2d(out_channels))
        
        return nn.Sequential(*layers)
    
    def forward(self, x):
        """
        Forward pass of the EfficientNetB2 model.

        :param x: The input tensor, shape (batch_size, 3, 224, 224)
        :return: The output tensor, shape (batch_size, num_classes)
        """
        x = self.relu(self.bn1(self.conv1(x)))
        x = self.mbconv1(x)
        x = self.mbconv2(x)
        x = self.mbconv3(x)
        x = self.mbconv4(x)
        x = self.mbconv5(x)
        x = self.relu(self.bn_final(self.conv_final(x)))
        x = self.avgpool(x)
        x = torch.flatten(x, 1)
        x = self.fc(x)
        return x

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

static inline int div_up(int a, int b) { return (a + b - 1) / b; }

__device__ __forceinline__ float do_act(float v, int act) {
    if (act == 1) return v > 0.0f ? v : 0.0f;
    if (act == 2) return 1.0f / (1.0f + expf(-v));
    return v;
}

__global__ void linear_kernel(const float* __restrict__ x,
                              const float* __restrict__ w,
                              const float* __restrict__ bias,
                              float* __restrict__ y,
                              int N, int In, int Out, int act, int has_bias) {
    int t = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Out;
    if (t >= total) return;
    int o = t % Out;
    int n = t / Out;
    const float* xp = x + n * In;
    const float* wp = w + o * In;
    float acc = has_bias ? bias[o] : 0.0f;
    #pragma unroll 4
    for (int i = 0; i < In; ++i) acc += xp[i] * wp[i];
    y[t] = do_act(acc, act);
}

__global__ void linear_bn_kernel(const float* __restrict__ x,
                                 const float* __restrict__ w,
                                 const float* __restrict__ gamma,
                                 const float* __restrict__ beta,
                                 const float* __restrict__ mean,
                                 const float* __restrict__ var,
                                 float* __restrict__ y,
                                 int N, int In, int Out, float eps, int act) {
    int t = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Out;
    if (t >= total) return;
    int o = t % Out;
    int n = t / Out;
    const float* xp = x + n * In;
    const float* wp = w + o * In;
    float acc = 0.0f;
    #pragma unroll 4
    for (int i = 0; i < In; ++i) acc += xp[i] * wp[i];
    acc = (acc - mean[o]) * rsqrtf(var[o] + eps) * gamma[o] + beta[o];
    y[t] = do_act(acc, act);
}

__global__ void depthwise_center_bn_kernel(const float* __restrict__ x,
                                           const float* __restrict__ w,
                                           const float* __restrict__ gamma,
                                           const float* __restrict__ beta,
                                           const float* __restrict__ mean,
                                           const float* __restrict__ var,
                                           float* __restrict__ y,
                                           int N, int C, float eps) {
    int t = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C;
    if (t >= total) return;
    int c = t % C;
    float acc = x[t] * w[c * 9 + 4];
    acc = (acc - mean[c]) * rsqrtf(var[c] + eps) * gamma[c] + beta[c];
    y[t] = acc > 0.0f ? acc : 0.0f;
}

static torch::Tensor lin(torch::Tensor x, torch::Tensor w, int act) {
    int N = (int)x.size(0), In = (int)x.size(1), Out = (int)w.size(0);
    auto y = torch::empty({N, Out}, x.options());
    int threads = 256;
    linear_kernel<<<div_up(N*Out, threads), threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), nullptr, y.data_ptr<float>(),
        N, In, Out, act, 0);
    return y;
}

static torch::Tensor lin_bias(torch::Tensor x, torch::Tensor w, torch::Tensor b) {
    int N = (int)x.size(0), In = (int)x.size(1), Out = (int)w.size(0);
    auto y = torch::empty({N, Out}, x.options());
    int threads = 256;
    linear_kernel<<<div_up(N*Out, threads), threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), b.data_ptr<float>(), y.data_ptr<float>(),
        N, In, Out, 0, 1);
    return y;
}

static torch::Tensor lin_bn(torch::Tensor x, torch::Tensor w, torch::Tensor g, torch::Tensor b,
                            torch::Tensor m, torch::Tensor v, float eps, int act) {
    int N = (int)x.size(0), In = (int)x.size(1), Out = (int)w.size(0);
    auto y = torch::empty({N, Out}, x.options());
    int threads = 256;
    linear_bn_kernel<<<div_up(N*Out, threads), threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), g.data_ptr<float>(), b.data_ptr<float>(),
        m.data_ptr<float>(), v.data_ptr<float>(), y.data_ptr<float>(), N, In, Out, eps, act);
    return y;
}

static torch::Tensor dw_bn(torch::Tensor x, torch::Tensor w, torch::Tensor g, torch::Tensor b,
                           torch::Tensor m, torch::Tensor v, float eps) {
    int N = (int)x.size(0), C = (int)x.size(1);
    auto y = torch::empty_like(x);
    int threads = 256;
    depthwise_center_bn_kernel<<<div_up(N*C, threads), threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), g.data_ptr<float>(), b.data_ptr<float>(),
        m.data_ptr<float>(), v.data_ptr<float>(), y.data_ptr<float>(), N, C, eps);
    return y;
}

torch::Tensor tail_forward(torch::Tensor x, std::vector<torch::Tensor> p, double eps_d) {
    float eps = (float)eps_d;
    int idx = 0;
    torch::Tensor cur = x;

    // finish MBConv1: SE reduce, SE expand/sigmoid, output 1x1 conv + BN
    cur = lin(cur, p[idx++], 1);                     // 96 -> 24, ReLU
    cur = lin(cur, p[idx++], 2);                     // 24 -> 96, Sigmoid
    cur = lin_bn(cur, p[idx++], p[idx++], p[idx++], p[idx++], p[idx++], eps, 0); // 96 -> 96

    // MBConv2..5.  Each block is 1x1 after the first block; the padded 3x3 depthwise
    // convolution therefore contributes only its center coefficient.
    for (int block = 0; block < 4; ++block) {
        cur = lin_bn(cur, p[idx++], p[idx++], p[idx++], p[idx++], p[idx++], eps, 1); // expansion + BN + ReLU
        cur = dw_bn(cur, p[idx++], p[idx++], p[idx++], p[idx++], p[idx++], eps);     // depthwise center + BN + ReLU
        cur = lin(cur, p[idx++], 1);                                                // SE reduce + ReLU
        cur = lin(cur, p[idx++], 2);                                                // SE expand + sigmoid
        cur = lin_bn(cur, p[idx++], p[idx++], p[idx++], p[idx++], p[idx++], eps, 0); // output + BN
    }

    // final 1x1 conv + BN + ReLU, then fully-connected classifier
    cur = lin_bn(cur, p[idx++], p[idx++], p[idx++], p[idx++], p[idx++], eps, 1);
    cur = lin_bias(cur, p[idx++], p[idx++]);
    return cur;
}

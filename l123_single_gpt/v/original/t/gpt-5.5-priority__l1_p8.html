import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor matmul_forward(torch::Tensor A, torch::Tensor B);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define BM 128
#define BN 64
#define BK 16
#define TM 8
#define TN 4

__global__ __launch_bounds__(256, 2)
void sgemm_128x64_kernel(const float* __restrict__ A,
                         const float* __restrict__ B,
                         float* __restrict__ C) {
    // Fixed problem size for this benchmark.
    const int M = 8205;
    const int K = 2949;
    const int N = 5921;

    __shared__ float As[BM][BK + 1];
    __shared__ float Bs[BK][BN + 1];

    const int tx = threadIdx.x;      // 0..15
    const int ty = threadIdx.y;      // 0..15
    const int tid = ty * 16 + tx;

    const int block_row = blockIdx.y * BM;
    const int block_col = blockIdx.x * BN;

    const int row_base = block_row + ty * TM;
    const int col_base = block_col + tx * TN;

    float acc[TM][TN];
#pragma unroll
    for (int i = 0; i < TM; ++i) {
#pragma unroll
        for (int j = 0; j < TN; ++j) acc[i][j] = 0.0f;
    }

    for (int k0 = 0; k0 < K; k0 += BK) {
        // Load A tile: BM*BK = 2048 floats, 8 per thread.
#pragma unroll
        for (int l = 0; l < (BM * BK); l += 256) {
            int idx = tid + l;
            int r = idx / BK;
            int k = idx - r * BK;
            int gr = block_row + r;
            int gk = k0 + k;
            As[r][k] = (gr < M && gk < K) ? A[gr * K + gk] : 0.0f;
        }
        // Load B tile: BK*BN = 1024 floats, 4 per thread.
#pragma unroll
        for (int l = 0; l < (BK * BN); l += 256) {
            int idx = tid + l;
            int k = idx / BN;
            int c = idx - k * BN;
            int gk = k0 + k;
            int gc = block_col + c;
            Bs[k][c] = (gk < K && gc < N) ? B[gk * N + gc] : 0.0f;
        }
        __syncthreads();

#pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            float b0 = Bs[kk][tx * TN + 0];
            float b1 = Bs[kk][tx * TN + 1];
            float b2 = Bs[kk][tx * TN + 2];
            float b3 = Bs[kk][tx * TN + 3];
#pragma unroll
            for (int i = 0; i < TM; ++i) {
                float a = As[ty * TM + i][kk];
                acc[i][0] = fmaf(a, b0, acc[i][0]);
                acc[i][1] = fmaf(a, b1, acc[i][1]);
                acc[i][2] = fmaf(a, b2, acc[i][2]);
                acc[i][3] = fmaf(a, b3, acc[i][3]);
            }
        }
        __syncthreads();
    }

#pragma unroll
    for (int i = 0; i < TM; ++i) {
        int r = row_base + i;
        if (r < M) {
#pragma unroll
            for (int j = 0; j < TN; ++j) {
                int c = col_base + j;
                if (c < N) C[r * N + c] = acc[i][j];
            }
        }
    }
}

torch::Tensor matmul_forward(torch::Tensor A, torch::Tensor B) {
    auto C = torch::empty({A.size(0), B.size(1)}, A.options());
    dim3 block(16, 16);
    dim3 grid((int)((B.size(1) + BN - 1) / BN), (int)((A.size(0) + BM - 1) / BM));
    sgemm_128x64_kernel<<<grid, block>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>());
    return C;
}
'''

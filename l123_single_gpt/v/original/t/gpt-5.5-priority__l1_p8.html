from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <mutex>

static cublasHandle_t g_handle = nullptr;
static std::once_flag g_handle_once;

static void init_handle() {
    cublasCreate(&g_handle);
}

torch::Tensor matmul_cublas(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "inputs must be CUDA tensors");
    TORCH_CHECK(A.scalar_type() == torch::kFloat32 && B.scalar_type() == torch::kFloat32, "inputs must be float32");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "inputs must be 2D");
    TORCH_CHECK(A.size(1) == B.size(0), "matmul dimensions do not match");
    TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "inputs must be contiguous");

    const int64_t M64 = A.size(0);
    const int64_t K64 = A.size(1);
    const int64_t N64 = B.size(1);
    TORCH_CHECK(M64 <= INT_MAX && K64 <= INT_MAX && N64 <= INT_MAX, "dimensions too large");
    const int M = (int)M64;
    const int K = (int)K64;
    const int N = (int)N64;

    c10::cuda::CUDAGuard device_guard(A.device());
    auto C = torch::empty({M64, N64}, A.options());

    std::call_once(g_handle_once, init_handle);
    cublasSetStream(g_handle, at::cuda::getCurrentCUDAStream().stream());

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // Row-major C = A(MxK) * B(KxN).  cuBLAS is column-major, so compute
    // C^T(NxM) = B^T(NxK) * A^T(KxM) without transposing data.
    cublasStatus_t st = cublasSgemm(
        g_handle,
        CUBLAS_OP_N, CUBLAS_OP_N,
        N, M, K,
        &alpha,
        B.data_ptr<float>(), N,
        A.data_ptr<float>(), K,
        &beta,
        C.data_ptr<float>(), N);
    TORCH_CHECK(st == CUBLAS_STATUS_SUCCESS, "cublasSgemm failed");
    return C;
}
'''

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies Softmax, and performs two max pooling operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool1 = nn.MaxPool3d(pool_kernel_size)
        self.pool2 = nn.MaxPool3d(pool_kernel_size)

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, depth, height, width)
        Returns:
            Output tensor of shape (batch_size, out_channels, depth', height', width') where depth', height', width' are the dimensions after pooling.
        """
        x = self.conv(x)
        x = torch.softmax(x, dim=1)
        x = self.pool1(x)
        x = self.pool2(x)
        return x

import torch.nn as nn

_cpp_src = r"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Fused implementation specialized for the benchmark shape:
// input N x 3 x 16 x 32 x 32, conv3d C_out=16,k=3, then softmax(C), then two 2x2x2 maxpools.
// The two maxpools are equivalent to a max over each non-overlapping 4x4x4 region of the softmax output.
__global__ void fused_kernel(const float* __restrict__ x,
                             const float* __restrict__ w,
                             const float* __restrict__ b,
                             float* __restrict__ out,
                             int N, int D2, int H2, int W2) {
    // block per (n, pooled_d, pooled_h, pooled_w), producing all 16 channels
    int bid = blockIdx.x;
    int pw = bid % W2; bid /= W2;
    int ph = bid % H2; bid /= H2;
    int pd = bid % D2; bid /= D2;
    int n = bid;

    const int tid = threadIdx.x;
    const int base_d = pd * 4;
    const int base_h = ph * 4;
    const int base_w = pw * 4;

    extern __shared__ float smem[];
    float* tile = smem;          // 3 * 6 * 6 * 6 = 648 floats
    float* vals = smem + 648;    // 64 * 16 = 1024 floats

    // Load the input tile needed by all 64 convolutions in this pooled cell.
    // Layout: ((ic*6 + td)*6 + th)*6 + tw
    for (int i = tid; i < 648; i += blockDim.x) {
        int t = i;
        int tw = t % 6; t /= 6;
        int th = t % 6; t /= 6;
        int td = t % 6; t /= 6;
        int ic = t;
        int gd = base_d + td;
        int gh = base_h + th;
        int gw = base_w + tw;
        tile[i] = x[((((n * 3 + ic) * 16 + gd) * 32 + gh) * 32 + gw)];
    }
    __syncthreads();

    // Compute convolution values for 64 spatial positions and 16 output channels.
    for (int idx = tid; idx < 1024; idx += blockDim.x) {
        int c = idx & 15;
        int v = idx >> 4;
        int lw = v & 3;
        int lh = (v >> 2) & 3;
        int ld = v >> 4;
        float acc = b[c];
        const float* wc = w + c * 81;
        #pragma unroll
        for (int ic = 0; ic < 3; ++ic) {
            #pragma unroll
            for (int kd = 0; kd < 3; ++kd) {
                #pragma unroll
                for (int kh = 0; kh < 3; ++kh) {
                    #pragma unroll
                    for (int kw = 0; kw < 3; ++kw) {
                        float xv = tile[(((ic * 6 + (ld + kd)) * 6 + (lh + kh)) * 6 + (lw + kw))];
                        float wv = wc[(((ic * 3 + kd) * 3 + kh) * 3 + kw)];
                        acc = fmaf(xv, wv, acc);
                    }
                }
            }
        }
        vals[idx] = acc;
    }
    __syncthreads();

    // Softmax over channels for each of the 64 conv positions.
    if (tid < 64) {
        float m = vals[tid * 16];
        #pragma unroll
        for (int c = 1; c < 16; ++c) {
            float z = vals[tid * 16 + c];
            m = z > m ? z : m;
        }
        float s = 0.0f;
        float e0;
        #pragma unroll
        for (int c = 0; c < 16; ++c) {
            float e = expf(vals[tid * 16 + c] - m);
            vals[tid * 16 + c] = e;
            s += e;
        }
        float invs = 1.0f / s;
        #pragma unroll
        for (int c = 0; c < 16; ++c) {
            vals[tid * 16 + c] *= invs;
        }
    }
    __syncthreads();

    // Max over the 4x4x4 region for each channel, producing the final output.
    if (tid < 16) {
        float mx = vals[tid];
        #pragma unroll
        for (int v = 1; v < 64; ++v) {
            float z = vals[v * 16 + tid];
            mx = z > mx ? z : mx;
        }
        out[((((n * 16 + tid) * D2 + pd) * H2 + ph) * W2 + pw)] = mx;
    }
}

torch::Tensor fused_conv_softmax_pool_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    const int64_t N64 = x.size(0);
    const int Dconv = (int)x.size(2) - 2;
    const int Hconv = (int)x.size(3) - 2;
    const int Wconv = (int)x.size(4) - 2;
    const int D1 = (Dconv - 2) / 2 + 1;
    const int H1 = (Hconv - 2) / 2 + 1;
    const int W1 = (Wconv - 2) / 2 + 1;
    const int D2 = (D1 - 2) / 2 + 1;
    const int H2 = (H1 - 2) / 2 + 1;

    auto out = torch::empty({N64, 16, D2, H2, W2}, x.options());

}

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a gemm, swish, divide, clamp, tanh, and clamp operations.
    """
    def __init__(self, in_features, out_features, bias=True):
        super(Model, self).__init__()
        self.gemm = nn.Linear(in_features, out_features, bias=bias)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        x = self.gemm(x)
        x = x * torch.sigmoid(x)  # Swish activation
        x = x / 2.0
        x = torch.clamp(x, min=-1.0, max=1.0)  # Clamp between -1 and 1
        x = torch.tanh(x)  # Tanh activation
        x = torch.clamp(x, min=-1.0, max=1.0)  # Clamp between -1 and 1
        return x

batch_size = 1024
in_features = 8192
out_features = 8192

def get_inputs():
    return [torch.rand(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features]

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


_cpp_src = r'''
#include <torch/extension.h>

// CUDA launcher implemented in cuda source
void swish_postprocess_cuda(torch::Tensor y);

// Complete forward: use the highly optimized library GEMM used by PyTorch, then
// replace the chain of pointwise kernels with one fused CUDA kernel.
torch::Tensor fused_forward(torch::Tensor x, torch::Tensor weight, c10::optional<torch::Tensor> bias) {
    torch::Tensor y;
    if (bias.has_value() && bias.value().defined()) {
        y = at::linear(x, weight, bias.value());
    } else {
        y = at::linear(x, weight, c10::nullopt);
    }
    swish_postprocess_cuda(y);
    return y;
}
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

__device__ __forceinline__ float sigmoidf_fast(float x) {
    return 1.0f / (1.0f + expf(-x));
}

__global__ void swish_postprocess_kernel(float* __restrict__ y, const long long n) {
    const long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = y[idx];
        // x * sigmoid(x), divide by 2, clamp to [-1,1], tanh, final clamp
        float t = 0.5f * v * sigmoidf_fast(v);
        t = fminf(1.0f, fmaxf(-1.0f, t));
        t = tanhf(t);
        // tanh(clamp([-1,1])) already lies in [-1,1], but keep exact op semantics.
        t = fminf(1.0f, fmaxf(-1.0f, t));
        y[idx] = t;
    }
}

void swish_postprocess_cuda(torch::Tensor y) {
    const long long n = y.numel();
    if (n == 0) return;
    const int threads = 256;
    const int blocks = (int)((n + threads - 1) / threads);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    swish_postprocess_kernel<<<blocks, threads, 0, stream>>>(y.data_ptr<float>(), n);
}
'''

_ext = load_inline(
    name='fused_linear_swish_clamp_tanh_ext',

    cuda_sources=_cuda_src,

class ModelNew(nn.Module):
    """Drop-in replacement for the reference module."""
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features, bias=bias)

    def forward(self, x):
        return _ext.fused_forward(x, self.gemm.weight, self.gemm.bias)

class Model(nn.Module):
    """
    Simple model that performs Max Pooling 2D.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.

        Args:
            kernel_size (int): Size of the pooling window.
            stride (int): Stride of the pooling window.
            padding (int): Padding to be applied before pooling.
            dilation (int): Spacing between kernel elements.
        """
        super(Model, self).__init__()
        self.maxpool = nn.MaxPool2d(kernel_size=kernel_size, stride=stride, padding=padding, dilation=dilation)

        """

            torch.Tensor: Output tensor after Max Pooling 2D, shape (batch_size, channels, pooled_height, pooled_width).
        """

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define BW 32
#define BH 16
#define TX 32
#define TY 8
#define SW (BW + 3)
#define SH (BH + 3)

__device__ __forceinline__ float mymax(float a, float b) {
    return a > b ? a : b;
}

__global__ __launch_bounds__(256, 2)
void maxpool4_kernel(const float* __restrict__ x, float* __restrict__ y,
                     int planes, int H, int W, int outH, int outW) {
    __shared__ float s[SH][SW];
    __shared__ float hmax[SH][BW];

    const int tx = threadIdx.x;   // 0..31
    const int ty = threadIdx.y;   // 0..7
    const int tid = ty * TX + tx;

    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    const int p = blockIdx.z;

    const int ow0 = bx * BW;
    const int oh0 = by * BH;
    const int iw0 = ow0 - 1;
    const int ih0 = oh0 - 1;

    const float* __restrict__ xp = x + ((size_t)p) * H * W;
    float* __restrict__ yp = y + ((size_t)p) * outH * outW;

    const bool full_in = (iw0 >= 0) && (ih0 >= 0) && (iw0 + SW - 1 < W) && (ih0 + SH - 1 < H);
    const bool full_out = (ow0 + BW - 1 < outW) && (oh0 + BH - 1 < outH);

    // Load the input tile, including the 3-pixel pooling halo.  The common case
    // (not touching padding/bottom/right) avoids per-element bounds checks.
    if (full_in) {
        #pragma unroll
        for (int idx = tid; idx < SH * SW; idx += TX * TY) {
            int r = idx / SW;
            int c = idx - r * SW;
            s[r][c] = xp[(ih0 + r) * W + (iw0 + c)];
        }
    } else {
        #pragma unroll
        for (int idx = tid; idx < SH * SW; idx += TX * TY) {
            int r = idx / SW;
            int c = idx - r * SW;
            int ih = ih0 + r;
            int iw = iw0 + c;
            float v = -CUDART_INF_F;
            if ((unsigned)ih < (unsigned)H && (unsigned)iw < (unsigned)W) {
                v = xp[ih * W + iw];
            }
            s[r][c] = v;
        }
    }
    __syncthreads();

    // Horizontal 4-wide max for every row of the shared tile and every output
    // column.  This makes the final vertical pooling only 3 max operations.
    #pragma unroll
    for (int idx = tid; idx < SH * BW; idx += TX * TY) {
        int r = idx >> 5;       // / 32
        int c = idx & 31;       // % 32
        float m0 = mymax(s[r][c], s[r][c + 1]);
        float m1 = mymax(s[r][c + 2], s[r][c + 3]);
        hmax[r][c] = mymax(m0, m1);
    }
    __syncthreads();

    // Each of the 256 threads writes two output rows, for a 32x16 tile.
    int ow = ow0 + tx;

    int ly = ty;
    int oh = oh0 + ly;
    float v0 = mymax(hmax[ly][tx], hmax[ly + 1][tx]);
    float v1 = mymax(hmax[ly + 2][tx], hmax[ly + 3][tx]);
    float outv = mymax(v0, v1);
    if (full_out || ((unsigned)oh < (unsigned)outH && (unsigned)ow < (unsigned)outW)) {
        yp[oh * outW + ow] = outv;
    }

    ly = ty + TY;
    oh = oh0 + ly;
    v0 = mymax(hmax[ly][tx], hmax[ly + 1][tx]);
    v1 = mymax(hmax[ly + 2][tx], hmax[ly + 3][tx]);
    outv = mymax(v0, v1);
    if (full_out || ((unsigned)oh < (unsigned)outH && (unsigned)ow < (unsigned)outW)) {
        yp[oh * outW + ow] = outv;
    }
}

torch::Tensor maxpool4_forward(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    TORCH_CHECK(x.dim() == 4, "expected NCHW tensor");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");

    const int N = (int)x.size(0);
    const int C = (int)x.size(1);
    const int H = (int)x.size(2);
    const int W = (int)x.size(3);

    const int outW = W - 1;

    auto y = torch::empty({N, C, outH, outW}, x.options());

}

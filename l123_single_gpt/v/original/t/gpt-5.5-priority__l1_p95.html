#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>
#include <stdint.h>

__inline__ __device__ float warp_reduce_max(float v) {
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) v = fmaxf(v, __shfl_down_sync(0xffffffff, v, off));
    return v;
}

__inline__ __device__ float warp_reduce_sum(float v) {
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) v += __shfl_down_sync(0xffffffff, v, off);
    return v;
}

__inline__ __device__ float block_reduce_max(float v) {
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    v = warp_reduce_max(v);
    if (lane == 0) shared[wid] = v;

    __shared__ float shared[32];

template<int C, int BLOCK>
__global__ void ce_rows_fixed_kernel(const float* __restrict__ pred,
                                     const int64_t* __restrict__ targets,
                                     float* __restrict__ losses,
                                     int N) {
    int row = blockIdx.x;
    if (row >= N) return;
    const float* __restrict__ p = pred + ((size_t)row) * C;

    float local_max = -FLT_MAX;
    #pragma unroll
    for (int i = threadIdx.x; i < C; i += BLOCK) {
        local_max = fmaxf(local_max, p[i]);
    }
    float m = block_reduce_max(local_max);

    __shared__ float target_logit;
    if (threadIdx.x == 0) {
        target_logit = p[targets[row]];
    }
    __syncthreads();

    float local_sum = 0.0f;
    #pragma unroll
    for (int i = threadIdx.x; i < C; i += BLOCK) {
        local_sum += __expf(p[i] - m);
    }
    float s = block_reduce_sum(local_sum);

    if (threadIdx.x == 0) {
        losses[row] = logf(s) + m - target_logit;
    }
}

template<int BLOCK>
__global__ void ce_rows_general_kernel(const float* __restrict__ pred,
                                       const int64_t* __restrict__ targets,
                                       float* __restrict__ losses,
                                       int N, int C) {
    int row = blockIdx.x;
    if (row >= N) return;
    const float* __restrict__ p = pred + ((size_t)row) * C;

    float local_max = -FLT_MAX;
    for (int i = threadIdx.x; i < C; i += BLOCK) {
        local_max = fmaxf(local_max, p[i]);
    }
    float m = block_reduce_max(local_max);

    __shared__ float target_logit;
    if (threadIdx.x == 0) target_logit = p[targets[row]];
    __syncthreads();

    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < C; i += BLOCK) {
        local_sum += __expf(p[i] - m);
    }
    float s = block_reduce_sum(local_sum);
    if (threadIdx.x == 0) losses[row] = logf(s) + m - target_logit;
}

template<int BLOCK>
__global__ void reduce_mean_kernel(const float* __restrict__ losses,
                                   float* __restrict__ out,
                                   int N, float invN) {
    int tid = threadIdx.x;
    int base = blockIdx.x * BLOCK + tid;
    float v = (base < N) ? losses[base] : 0.0f;
    v = block_reduce_sum(v);
    if (tid == 0) atomicAdd(out, v * invN);
}

    TORCH_CHECK(predictions.is_cuda(), "predictions must be CUDA");
    TORCH_CHECK(targets.is_cuda(), "targets must be CUDA");
    TORCH_CHECK(predictions.scalar_type() == at::kFloat, "only float32 predictions supported");
    TORCH_CHECK(targets.scalar_type() == at::kLong, "targets must be int64");
    TORCH_CHECK(predictions.dim() == 2, "predictions must be 2D");

    const c10::cuda::CUDAGuard device_guard(predictions.device());
    auto pred = predictions.contiguous();
    auto tgt = targets.contiguous();

    int N = (int)pred.size(0);
    int C = (int)pred.size(1);
    auto losses = torch::empty({N}, pred.options());
    auto out = torch::empty({}, pred.options());

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cudaMemsetAsync(out.data_ptr<float>(), 0, sizeof(float), stream);

    constexpr int BLOCK = 256;

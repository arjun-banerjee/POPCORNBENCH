import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor hardtanh_forward_cuda(torch::Tensor x);
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cstdint>

__device__ __forceinline__ float ht_clamp(float v) {
    // Comparison form preserves NaNs like torch.clamp/hardtanh.
    return (v < -1.0f) ? -1.0f : ((v > 1.0f) ? 1.0f : v);
}

__global__ void hardtanh_scalar_kernel(const float* __restrict__ x,
                                       float* __restrict__ y,
                                       int64_t n) {
    int64_t i = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; i < n; i += stride) {
        y[i] = ht_clamp(x[i]);
    }
}

__global__ void hardtanh_vec4_kernel(const float4* __restrict__ x,
                                     float4* __restrict__ y,
                                     int64_t n4) {
    int64_t i = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = (int64_t)blockDim.x * gridDim.x;
    for (; i < n4; i += stride) {
        float4 v = x[i];
        v.x = ht_clamp(v.x);
        v.y = ht_clamp(v.y);
        v.z = ht_clamp(v.z);
        v.w = ht_clamp(v.w);
        y[i] = v;
    }
}

torch::Tensor hardtanh_forward_cuda(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "hardtanh_forward_cuda: expected a CUDA tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "hardtanh_forward_cuda: expected float32 input");

    if (!x.is_contiguous()) {
        x = x.contiguous();
    }

    auto y = torch::empty_like(x);
    const int64_t n = x.numel();
    if (n == 0) return y;

    const int threads = 256;
    int dev = 0;
    cudaGetDevice(&dev);
    cudaDeviceProp prop;
    cudaGetDeviceProperties(&prop, dev);

    auto stream = at::cuda::getCurrentCUDAStream();
    const float* xp = x.data_ptr<float>();
    float* yp = y.data_ptr<float>();

    const uintptr_t xa = reinterpret_cast<uintptr_t>(xp);
    const uintptr_t ya = reinterpret_cast<uintptr_t>(yp);
    const bool use_vec4 = (((xa | ya) & 15u) == 0u) && ((n & 3LL) == 0LL);

    if (use_vec4) {
        const int64_t n4 = n >> 2;
        int64_t max_blocks = (n4 + threads - 1) / threads;
        int blocks = (int)min<int64_t>(max_blocks, (int64_t)prop.multiProcessorCount * 8);
        if (blocks < 1) blocks = 1;
        hardtanh_vec4_kernel<<<blocks, threads, 0, stream>>>(
            reinterpret_cast<const float4*>(xp), reinterpret_cast<float4*>(yp), n4);
    } else {
        int64_t max_blocks = (n + threads - 1) / threads;
        int blocks = (int)min<int64_t>(max_blocks, (int64_t)prop.multiProcessorCount * 8);
        if (blocks < 1) blocks = 1;
        hardtanh_scalar_kernel<<<blocks, threads, 0, stream>>>(xp, yp, n);
    }
    return y;
}
"""

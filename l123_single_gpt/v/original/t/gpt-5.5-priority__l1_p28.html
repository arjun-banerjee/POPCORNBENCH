#include <cuda.h>

__device__ __forceinline__ float hs_f32(float a) {
    // PyTorch hardsigmoid is clamp(a + 3, 0, 6) / 6
    a = fminf(fmaxf(a + 3.0f, 0.0f), 6.0f);
    return a * 0.16666666666666666f;
}

__global__ void hardsigmoid_vec4_kernel(const float* __restrict__ x,
                                        float* __restrict__ y,
                                        long long n4) {
    long long i = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    const float4* __restrict__ x4 = reinterpret_cast<const float4*>(x);
    float4* __restrict__ y4 = reinterpret_cast<float4*>(y);
    for (; i < n4; i += stride) {
        float4 v = x4[i];
        v.x = hs_f32(v.x);
        v.y = hs_f32(v.y);
        v.z = hs_f32(v.z);
        v.w = hs_f32(v.w);
        y4[i] = v;
    }
}

__global__ void hardsigmoid_tail_kernel(const float* __restrict__ x,
                                        float* __restrict__ y,

    long long i = start + (long long)blockIdx.x * blockDim.x + threadIdx.x;

}

}

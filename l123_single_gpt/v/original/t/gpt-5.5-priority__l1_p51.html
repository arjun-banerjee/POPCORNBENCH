import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs Argmax over a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to perform argmax.

        Args:
            dim (int): The dimension to perform argmax over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies argmax over the specified dimension to the input tensor.

        Args:
            x (torch.Tensor): Input tensor.

        Returns:
            torch.Tensor: Output tensor with argmax applied, with the specified dimension removed.
        """
        return torch.argmax(x, dim=self.dim)

from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>

torch::Tensor argmax_dim1_cuda(torch::Tensor x);

torch::Tensor argmax_dim1(torch::Tensor x) {
    return argmax_dim1_cuda(x);
}
"""

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <limits>

// One block computes a tile of columns for one batch item.  Threads are laid
// out with x over contiguous columns and y over a small row split, so all input
// loads are coalesced although the reduction dimension is the middle dimension.
template<int TILE_COLS, int ROW_SPLIT>
__global__ void argmax_dim1_kernel(const float* __restrict__ x,
                                   long long* __restrict__ out,
                                   int B, int M, int K) {
    const int b = blockIdx.y;
    const int col0 = blockIdx.x * TILE_COLS;
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int col = col0 + tx;

    float best = -INFINITY;
    int best_i = 0;

    if (col < K) {
        const float* base = x + ((long long)b * M * K) + col;
        #pragma unroll 4
        for (int r = ty; r < M; r += ROW_SPLIT) {
            float v = base[(long long)r * K];
            // Strict comparison preserves the first index on ties.
            if (v > best) {
                best = v;
                best_i = r;
            }
        }
    }

    __shared__ float s_val[ROW_SPLIT][TILE_COLS];
    __shared__ int   s_idx[ROW_SPLIT][TILE_COLS];
    s_val[ty][tx] = best;
    s_idx[ty][tx] = best_i;
    __syncthreads();

    if (ty == 0 && col < K) {
        float vbest = s_val[0][tx];
        int ibest = s_idx[0][tx];
        #pragma unroll
        for (int j = 1; j < ROW_SPLIT; ++j) {
            float v = s_val[j][tx];
            int ii = s_idx[j][tx];
            if (v > vbest || (v == vbest && ii < ibest)) {
                vbest = v;
                ibest = ii;
            }
        }
        out[(long long)b * K + col] = (long long)ibest;
    }
}

torch::Tensor argmax_dim1_cuda(torch::Tensor x) {
    const int B = (int)x.size(0);
    const int M = (int)x.size(1);
    const int K = (int)x.size(2);
    auto out = torch::empty({B, K}, x.options().dtype(torch::kInt64));

    constexpr int TILE_COLS = 256;
    constexpr int ROW_SPLIT = 4;
    dim3 block(TILE_COLS, ROW_SPLIT, 1);
    dim3 grid((K + TILE_COLS - 1) / TILE_COLS, B, 1);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    argmax_dim1_kernel<TILE_COLS, ROW_SPLIT><<<grid, block, 0, stream>>>(
        (const float*)x.data_ptr<float>(), (long long*)out.data_ptr<int64_t>(), B, M, K);
    return out;
}
"""

import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th


class Model(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(Model, self).__init__()

        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # The `clusters` weights are the `(w,b)` in the paper
        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        # The `clusters2` weights are the visual words `c_k` in the paper
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        """Aggregates feature maps into a fixed size representation.  In the following
        notation, B = batch_size, N = num_features, K = num_clusters, D = feature_size.

        Args:
            x (th.Tensor): B x N x D

        Returns:
            (th.Tensor): B x DK
        """
        max_sample = x.size()[1]
        x = x.view(-1, self.feature_size)  # B x N x D -> BN x D

        if x.device != self.clusters.device:
            msg = f"x.device {x.device} != cluster.device {self.clusters.device}"
            raise ValueError(msg)

        assignment = th.matmul(x, self.clusters)  # (BN x D) x (D x (K+G)) -> BN x (K+G)
        assignment = self.batch_norm(assignment)

        assignment = F.softmax(assignment, dim=1)  # BN x (K+G) -> BN x (K+G)
        # remove ghost assigments
        assignment = assignment[:, :self.cluster_size]
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # -> B x N x K
        a_sum = th.sum(assignment, dim=1, keepdim=True)  # B x N x K -> B x 1 x K
        a = a_sum * self.clusters2

        assignment = assignment.transpose(1, 2)  # B x N x K -> B x K x N

        x = x.view(-1, max_sample, self.feature_size)  # BN x D -> B x N x D
        vlad = th.matmul(assignment, x)  # (B x K x N) x (B x N x D) -> B x K x D
        vlad = vlad.transpose(1, 2)  # -> B x D x K
        vlad = vlad - a

        # L2 intra norm
        vlad = F.normalize(vlad)

        # flattening + L2 norm
        vlad = vlad.reshape(-1, self.cluster_size * self.feature_size)  # -> B x DK
        vlad = F.normalize(vlad)
        return vlad  # B x DK

batch_size = 2048
num_features = 100
num_clusters = 32
feature_size = 512
ghost_clusters = 16

def get_inputs():
  return [torch.rand(batch_size, num_features, feature_size)]

def get_init_inputs():
  return [num_clusters, feature_size, ghost_clusters]

reasoning
We need to write a custom CUDA kernel for a PyTorch model (NetVLAD-like). Let's understand the reference implementation first. Reference: - Input: x of shape (B, N, D) - max_sample = N - x is reshaped to (B*N, D) - assignment = x @ clusters # (BN, D) @ (D, K+G) -> (BN, K+G) - batch_norm on assignment (BatchNorm1d with K+G features) - softmax on assignment across dim=1 - assignment = assignment[:, :K] (remove ghost clusters) - reshape to (B, N, K) - a_sum = sum(assignment, dim=1, keepdim=True) # (B, 1, K) - a = a_sum * clusters2 # clusters2 shape (1, D, K), a_sum (B,1,K) -> (B, D, K) via broadcasting? Let's check. Actually in code: self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size)) a_sum shape: (B, 1, K). a = a_sum * self.clusters2. In PyTorch, a_sum (B,1,K) broadcasts with clusters2 (1,D,K) => (B,D,K). So a has shape (B, D, K). - assignment = assignment.transpose(1,2) # (B, K, N) - x is reshaped to (B, N, D) - vlad = assignment @ x # (B, K, N) @ (B, N, D) -> (B, K, D) - vlad = vlad.transpose(1,2) # (B, D, K) - vlad = vlad - a # (B, D, K) - F.normalize(vlad) # L2 normalize over last dimension? Actually F.normalize default dim=1? Wait PyTorch F.normalize default dim=1? Let's check: torch.nn.functional.normalize(input, p=2.0, dim=1, eps=1e-12, out=None). Default dim=1. So for vlad shape (B, D, K), it normalizes along dim=1 (feature dimension D). So each (K) vector of length D is normalized. - flatten to (B, D*K) - F.normalize(vlad) again, default dim=1, so normalizes the flattened vector of length D*K. We need to write a custom CUDA kernel for this. The shapes are: - B = batch_size = 2048 - N = num_features = 100 - K = num_clusters = 32 - D = feature_size = 512 - G = ghost_clusters = 16 So assignment before ghost removal is (2048*100, 32+16) = (204800, 48). That's a big matrix multiply. But maybe we can fuse some operations into a single kernel or a few kernels to reduce memory bandwidth. However, the main heavy ops are: 1. matmul(x, clusters) -> (BN, D) @ (D, K+G) 2. batch_norm (can be fused) 3. softmax 4. sum over N 5. matmul(assignment, x) -> (B, K, N) @ (B, N, D) 6. normalize We can implement entire forward in one or a few kernels. Given the constraints, we need to produce numerically equivalent outputs (fp32). We can fuse batch norm and softmax and ghost removal into one kernel, then compute a_sum and vlad maybe in another kernel, then normalize. But since we have limited number of steps, we should think about performance. The GPU specs: let's get them.

import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th


class Model(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(Model, self).__init__()

        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # The `clusters` weights are the `(w,b)` in the paper
        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        # The `clusters2` weights are the visual words `c_k` in the paper
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        """Aggregates feature maps into a fixed size representation.  In the following
        notation, B = batch_size, N = num_features, K = num_clusters, D = feature_size.

        Args:
            x (th.Tensor): B x N x D

        Returns:
            (th.Tensor): B x DK
        """
        max_sample = x.size()[1]
        x = x.view(-1, self.feature_size)  # B x N x D -> BN x D

        if x.device != self.clusters.device:
            msg = f"x.device {x.device} != cluster.device {self.clusters.device}"
            raise ValueError(msg)

        assignment = th.matmul(x, self.clusters)  # (BN x D) x (D x (K+G)) -> BN x (K+G)
        assignment = self.batch_norm(assignment)

        assignment = F.softmax(assignment, dim=1)  # BN x (K+G) -> BN x (K+G)
        # remove ghost assigments
        assignment = assignment[:, :self.cluster_size]
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # -> B x N x K
        a_sum = th.sum(assignment, dim=1, keepdim=True)  # B x N x K -> B x 1 x K
        a = a_sum * self.clusters2

        assignment = assignment.transpose(1, 2)  # B x N x K -> B x K x N

        x = x.view(-1, max_sample, self.feature_size)  # BN x D -> B x N x D
        vlad = th.matmul(assignment, x)  # (B x K x N) x (B x N x D) -> B x K x D
        vlad = vlad.transpose(1, 2)  # -> B x D x K
        vlad = vlad - a

        # L2 intra norm
        vlad = F.normalize(vlad)

        # flattening + L2 norm
        vlad = vlad.reshape(-1, self.cluster_size * self.feature_size)  # -> B x DK
        vlad = F.normalize(vlad)
        return vlad  # B x DK

batch_size = 2048

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <float.h>

__global__ void bn_stats_kernel(const float* __restrict__ logits,
                                float* __restrict__ mean,
                                float* __restrict__ var,
                                float* __restrict__ running_mean,
                                float* __restrict__ running_var,
                                int M, int C, float momentum) {
    int c = blockIdx.x;
    __shared__ float ssum[256];
    __shared__ float ssq[256];
    int tid = threadIdx.x;
    float sum = 0.0f;
    float sum2 = 0.0f;
    for (int i = tid; i < M; i += blockDim.x) {
        float v = logits[i * C + c];
        sum += v;
        sum2 += v * v;
    }
    ssum[tid] = sum;
    ssq[tid] = sum2;
    __syncthreads();
    for (int off = 128; off > 0; off >>= 1) {
        if (tid < off) {
            ssum[tid] += ssum[tid + off];
            ssq[tid] += ssq[tid + off];
        }
        __syncthreads();
    }
    if (tid == 0) {
        float m = ssum[0] / (float)M;
        float v = ssq[0] / (float)M - m * m;  // biased variance, as used for BN train output
        if (v < 0.0f) v = 0.0f;
        mean[c] = m;
        var[c] = v;
        if (running_mean != nullptr && running_var != nullptr) {
            float vu = v * ((float)M / (float)(M - 1));
            running_mean[c] = running_mean[c] * (1.0f - momentum) + momentum * m;
            running_var[c]  = running_var[c]  * (1.0f - momentum) + momentum * vu;
        }
    }
}

__inline__ __device__ float warp_sum(float v) {
    unsigned mask = 0xffffffffu;
    v += __shfl_down_sync(mask, v, 16);
    v += __shfl_down_sync(mask, v, 8);
    v += __shfl_down_sync(mask, v, 4);
    v += __shfl_down_sync(mask, v, 2);
    v += __shfl_down_sync(mask, v, 1);
    return v;
}

__inline__ __device__ float warp_max(float v) {
    unsigned mask = 0xffffffffu;
    v = fmaxf(v, __shfl_down_sync(mask, v, 16));
    v = fmaxf(v, __shfl_down_sync(mask, v, 8));
    v = fmaxf(v, __shfl_down_sync(mask, v, 4));
    v = fmaxf(v, __shfl_down_sync(mask, v, 2));
    v = fmaxf(v, __shfl_down_sync(mask, v, 1));
    return v;
}

// One warp computes one softmax row. Optimized for C <= 32 (the benchmark uses 32).
__global__ void softmax_bn_kernel(const float* __restrict__ logits,
                                  float* __restrict__ assign,
                                  const float* __restrict__ mean,
                                  const float* __restrict__ var,
                                  const float* __restrict__ bn_weight,
                                  const float* __restrict__ bn_bias,
                                  int M, int C, int K, float eps) {
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    int row = blockIdx.x * 8 + warp;
    if (row >= M) return;

    float val = -FLT_MAX;
    if (lane < C) {
        float z = (logits[row * C + lane] - mean[lane]) * rsqrtf(var[lane] + eps);
        z = z * bn_weight[lane] + bn_bias[lane];
        val = z;
    }
    float mx = warp_max(val);
    mx = __shfl_sync(0xffffffffu, mx, 0);
    float e = (lane < C) ? __expf(val - mx) : 0.0f;
    float den = warp_sum(e);
    den = __shfl_sync(0xffffffffu, den, 0);
    if (lane < K) {
        assign[row * K + lane] = e / den;
    }
}

__global__ void final_vlad_kernel(const float* __restrict__ prod,      // B,K,D
                                  const float* __restrict__ assign,    // B,N,K
                                  const float* __restrict__ clusters2, // D,K
                                  float* __restrict__ out,             // B,D*K
                                  int B, int N, int D, int K) {
    int bk = blockIdx.x;
    int b = bk / K;
    int k = bk - b * K;
    int tid = threadIdx.x;
    __shared__ float red[256];

    float s = 0.0f;
    for (int n = tid; n < N; n += blockDim.x) {
        s += assign[(b * N + n) * K + k];
    }
    red[tid] = s;
    __syncthreads();
    for (int off = 128; off > 0; off >>= 1) {
        if (tid < off) red[tid] += red[tid + off];
        __syncthreads();
    }
    float asum = red[0];

    float ssq = 0.0f;
    for (int d = tid; d < D; d += blockDim.x) {
        float v = prod[(b * K + k) * D + d] - asum * clusters2[d * K + k];
        ssq += v * v;
    }
    red[tid] = ssq;
    __syncthreads();
    for (int off = 128; off > 0; off >>= 1) {
        if (tid < off) red[tid] += red[tid + off];
        __syncthreads();
    }
    float inv = rsqrtf(red[0]);
    // After intra-normalization, the final flattened norm is sqrt(K) for nonzero clusters.
    inv *= rsqrtf((float)K);

    for (int d = tid; d < D; d += blockDim.x) {
        float v = prod[(b * K + k) * D + d] - asum * clusters2[d * K + k];
        out[b * (D * K) + d * K + k] = v * inv;
    }
}

torch::Tensor netvlad_forward(torch::Tensor x,
                              torch::Tensor clusters,
                              torch::Tensor bn_weight,
                              torch::Tensor bn_bias,
                              torch::Tensor running_mean,
                              torch::Tensor running_var,
                              torch::Tensor clusters2,
                              bool training,
                              double momentum,
                              double eps,
                              int64_t cluster_size,
                              int64_t feature_size) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 is supported");
    const int B = (int)x.size(0);
    const int N = (int)x.size(1);
    const int D = (int)feature_size;
    const int K = (int)cluster_size;
    const int C = (int)clusters.size(1);
    const int M = B * N;

    auto x_contig = x.contiguous();
    auto x2 = x_contig.view({M, D});
    auto logits = at::mm(x2, clusters);

    auto opts = logits.options();
    torch::Tensor mean;
    torch::Tensor var;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (training) {
        mean = torch::empty({C}, opts);
        var = torch::empty({C}, opts);
        bn_stats_kernel<<<C, 256, 0, stream>>>(logits.data_ptr<float>(),
                                               mean.data_ptr<float>(),
                                               var.data_ptr<float>(),
                                               running_mean.data_ptr<float>(),
                                               running_var.data_ptr<float>(),
                                               M, C, (float)momentum);
    } else {
        mean = running_mean;
        var = running_var;
    }

    auto assign = torch::empty({B, N, K}, opts);
    int blocks = (M + 7) / 8;
    softmax_bn_kernel<<<blocks, 256, 0, stream>>>(logits.data_ptr<float>(),
                                                  assign.data_ptr<float>(),
                                                  mean.data_ptr<float>(),
                                                  var.data_ptr<float>(),
                                                  bn_weight.data_ptr<float>(),
                                                  bn_bias.data_ptr<float>(),
                                                  M, C, K, (float)eps);

    auto assign_t = assign.transpose(1, 2);
    auto prod = at::bmm(assign_t, x_contig);  // B,K,D

    auto out = torch::empty({B, D * K}, opts);
    final_vlad_kernel<<<B * K, 256, 0, stream>>>(prod.data_ptr<float>(),
                                                 assign.data_ptr<float>(),
                                                 clusters2.data_ptr<float>(),
                                                 out.data_ptr<float>(),
                                                 B, N, D, K);
    return out;
}

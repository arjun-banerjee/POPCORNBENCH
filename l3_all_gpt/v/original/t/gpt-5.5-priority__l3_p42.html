class Model(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=3, bias=True, batch_first=False):
        """
        :param input_size: The number of expected features in the input x
        :param hidden_size: The number of features in the hidden state h
        :param num_layers: Number of recurrent layers (default: 1)
        :param bias: If False, then the layer does not use bias weights b_ih and b_hh (default: True)
        :param batch_first: If True, then the input and output tensors are provided as (batch, seq, feature) (default: False)
        """

        self.gru = nn.GRU(input_size, hidden_size, num_layers, bias, batch_first, dropout=0, bidirectional=True)
    
    def forward(self, x,h0):
        """
        :param x: The input tensor, shape (seq_len, batch_size, input_size) if batch_first=False, otherwise (batch_size, seq_len, input_size)
        :param h_0: The initial hidden state for the input sequence, shape (num_layers * num_directions, batch_size, hidden_size) (default: None)
        :return: output, h_n

        """

_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAGuard.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

__device__ __forceinline__ float sigmoidf_dev(float x) {
    return 1.0f / (1.0f + expf(-x));
}

__global__ void gru_recur_kernel(const float* __restrict__ gates,
                                 const float* __restrict__ whh,
                                 const float* __restrict__ bhh,
                                 const float* __restrict__ h0,
                                 float* __restrict__ layer_out,
                                 float* __restrict__ hn,
                                 int T, int B, int H, int out_stride,
                                 int layer, int dir, int num_dirs) {
    extern __shared__ float sh[];
    int b = blockIdx.x;
    int j = threadIdx.x;
    if (j >= H || b >= B) return;

    float oldh = h0[((layer * num_dirs + dir) * B + b) * H + j];
    sh[j] = oldh;
    __syncthreads();

    int t0, tend, step;
    if (dir == 0) { t0 = 0; tend = T; step = 1; }
    else { t0 = T - 1; tend = -1; step = -1; }

    float newh = oldh;
    for (int t = t0; t != tend; t += step) {
        oldh = sh[j];
        float rr = 0.0f, zz = 0.0f, nn = 0.0f;
        const float* wr = whh + j * H;
        const float* wz = whh + (H + j) * H;
        const float* wn = whh + (2 * H + j) * H;
        #pragma unroll 4
        for (int k = 0; k < H; ++k) {
            float hv = sh[k];
            rr = fmaf(wr[k], hv, rr);
            zz = fmaf(wz[k], hv, zz);
            nn = fmaf(wn[k], hv, nn);
        }
        const float* g = gates + ((t * B + b) * (3 * H));
        float r = sigmoidf_dev(g[j] + rr + bhh[j]);
        float z = sigmoidf_dev(g[H + j] + zz + bhh[H + j]);
        float n = tanhf(g[2 * H + j] + r * (nn + bhh[2 * H + j]));
        newh = fmaf(z, oldh - n, n);  // (1-z)*n + z*oldh
        layer_out[(t * B + b) * out_stride + dir * H + j] = newh;
        __syncthreads();
        sh[j] = newh;
        __syncthreads();
    }
    hn[((layer * num_dirs + dir) * B + b) * H + j] = newh;
}

torch::Tensor gru_forward(torch::Tensor x, torch::Tensor h0,
                          std::vector<torch::Tensor> wihs,
                          std::vector<torch::Tensor> whhs,
                          std::vector<torch::Tensor> bihs,
                          std::vector<torch::Tensor> bhhs,
                          int64_t num_layers, bool batch_first) {

    torch::Tensor inp;

    const int64_t B64 = inp.size(1);
    const int64_t H64 = h0c.size(2);

    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

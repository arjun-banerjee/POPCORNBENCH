class Model(nn.Module):

        """
        :param input_size: The number of expected features in the input x
        :param hidden_size: The number of features in the hidden state h
        :param num_layers: Number of recurrent layers (default: 1)
        :param bias: If False, then the layer does not use bias weights b_ih and b_hh (default: True)

        """

                wih = getattr(self.gru, f'weight_ih_l{l}{suffix}')
                whh = getattr(self.gru, f'weight_hh_l{l}{suffix}')
                ptrs.append(wih.data_ptr()); ptrs.append(whh.data_ptr())
                wihTs.append(wih.detach().t().contiguous())
                whhTs.append(whh.detach().t().contiguous())
                if self.bias:
                    bih = getattr(self.gru, f'bias_ih_l{l}{suffix}')
                    bhh = getattr(self.gru, f'bias_hh_l{l}{suffix}')
                    ptrs.append(bih.data_ptr()); ptrs.append(bhh.data_ptr())
                    bihs.append(bih.detach().contiguous())
                    bhhs.append(bhh.detach().contiguous())
        if not self.bias:
            dev = next(self.gru.parameters()).device
            empty = torch.empty((0,), device=dev, dtype=next(self.gru.parameters()).dtype)
            bihs = [empty for _ in range(self.num_layers * 2)]
            bhhs = [empty for _ in range(self.num_layers * 2)]
        self._wihTs = wihTs
        self._whhTs = whhTs
        self._bihs = bihs
        self._bhhs = bhhs
        self._cache_ptrs = tuple(ptrs)
        self._cache_ready = True

    def forward(self, x, h0):
        # Fast path for the benchmark configuration (CUDA fp32).  Parameters are cached as
        # transposed contiguous matrices for coalesced reads in the CUDA kernel.
        if (not self._cache_ready) or self._wihTs[0].device != x.device:
            self._build_cache()
        return _ext.gru_forward(x, h0, self._wihTs, self._whhTs, self._bihs, self._bhhs,
                                self.num_layers, self.batch_first, self.bias)

    int idx = blockIdx.x * blockDim.x + threadIdx.x;

torch::Tensor relu_maxpool3x3s2_cl(torch::Tensor x) {
    int N = (int)x.size(0), C = (int)x.size(1), H = (int)x.size(2), W = (int)x.size(3);
    int Ho = (H - 3) / 2 + 1, Wo = (W - 3) / 2 + 1;
    auto y = torch::empty_strided({N, C, Ho, Wo}, {C*Ho*Wo, 1, C*Wo, C}, x.options());
    int total = N*C*Ho*Wo;
    const int threads = 256;
    int blocks = (total + threads - 1) / threads; if (blocks > 65535) blocks = 65535;
    relu_maxpool3x3s2_cl_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), y.data_ptr<float>(), total, C, H, W, Ho, Wo);
    return y;
}

torch::Tensor relu_maxpool3x3s2_cl_to_nchw(torch::Tensor x) {
    int N = (int)x.size(0), C = (int)x.size(1), H = (int)x.size(2), W = (int)x.size(3);

    return y;
}

torch::Tensor relu_inplace(torch::Tensor x) {

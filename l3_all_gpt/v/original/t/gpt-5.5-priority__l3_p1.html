__global__ void bias_relu_kernel(float* __restrict__ y, const float* __restrict__ b, int total, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < total) {
        float v = y[i] + b[i % n];
        y[i] = v > 0.0f ? v : 0.0f;
    }
}

__global__ void bias_kernel(float* __restrict__ y, const float* __restrict__ b, int total, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < total) y[i] += b[i % n];
}

static inline void gemm_fp32(cublasHandle_t h, const float* x, const float* w, float* y, int M, int K, int N) {
    float alpha = 1.0f, beta = 0.0f;
    cublasSgemm(h, CUBLAS_OP_T, CUBLAS_OP_N, N, M, K, &alpha, w, K, x, K, &beta, y, N);
}

static inline void gemm_tf32(cublasHandle_t h, const float* x, const float* w, float* y, int M, int K, int N) {
    float alpha = 1.0f, beta = 0.0f;
    cublasGemmEx(h, CUBLAS_OP_T, CUBLAS_OP_N, N, M, K, &alpha,
                 w, CUDA_R_32F, K, x, CUDA_R_32F, K, &beta, y, CUDA_R_32F, N,
                 CUBLAS_COMPUTE_32F_FAST_TF32, CUBLAS_GEMM_DEFAULT_TENSOR_OP);
}

    int M = (int)x.size(0), K = (int)x.size(1);
    int N1 = (int)w1.size(0), N2 = (int)w2.size(0), N3 = (int)w3.size(0);
    auto h1 = torch::empty({M, N1}, x.options());
    auto h2 = torch::empty({M, N2}, x.options());
    auto y = torch::empty({M, N3}, x.options());
    cublasHandle_t h = at::cuda::getCurrentCUDABlasHandle();
    cublasSetMathMode(h, CUBLAS_DEFAULT_MATH);

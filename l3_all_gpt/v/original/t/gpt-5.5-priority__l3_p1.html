_cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <ATen/cuda/Exceptions.h>
#include <cublas_v2.h>
#include <cuda_runtime.h>

__global__ void bias_relu_kernel(float* __restrict__ y, const float* __restrict__ b, int64_t total, int n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < total) {
        float v = y[idx] + b[idx % n];
        y[idx] = v > 0.0f ? v : 0.0f;
    }
}

__global__ void bias_kernel(float* __restrict__ y, const float* __restrict__ b, int64_t total, int n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < total) {
        y[idx] += b[idx % n];
    }
}

static inline void linear_gemm(cublasHandle_t handle,
                               const torch::Tensor& x,
                               const torch::Tensor& w,
                               torch::Tensor& y,
                               int M, int K, int N) {
    const float alpha = 1.0f;
    const float beta = 0.0f;
    // Row-major y[M,N] = x[M,K] * w[N,K]^T.
    // Interpreted as column-major: y^T[N,M] = op(w_col[K,N])^T * x_col[K,M].
    CUBLAS_CHECK(cublasSgemm(handle,
                             CUBLAS_OP_T, CUBLAS_OP_N,
                             N, M, K,
                             &alpha,
                             w.data_ptr<float>(), K,
                             x.data_ptr<float>(), K,
                             &beta,
                             y.data_ptr<float>(), N));
}

torch::Tensor mlp_forward_cuda(torch::Tensor x,
                               torch::Tensor w1, torch::Tensor b1,
                               torch::Tensor w2, torch::Tensor b2,
                               torch::Tensor w3, torch::Tensor b3) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only fp32 is supported");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");
    TORCH_CHECK(w1.is_contiguous() && w2.is_contiguous() && w3.is_contiguous(), "weights must be contiguous");

    const int M = (int)x.size(0);
    const int K1 = (int)x.size(1);
    const int N1 = (int)w1.size(0);
    const int K2 = (int)w2.size(1);
    const int N2 = (int)w2.size(0);
    const int K3 = (int)w3.size(1);
    const int N3 = (int)w3.size(0);

    auto h1 = torch::empty({M, N1}, x.options());
    auto h2 = torch::empty({M, N2}, x.options());
    auto y  = torch::empty({M, N3}, x.options());

    cublasHandle_t handle = at::cuda::getCurrentCUDABlasHandle();
#if defined(CUBLAS_PEDANTIC_MATH)
    CUBLAS_CHECK(cublasSetMathMode(handle, CUBLAS_PEDANTIC_MATH));
#endif
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    CUBLAS_CHECK(cublasSetStream(handle, stream));

    linear_gemm(handle, x,  w1, h1, M, K1, N1);
    const int threads = 256;
    int64_t total1 = (int64_t)M * N1;
    bias_relu_kernel<<<(total1 + threads - 1) / threads, threads, 0, stream>>>(h1.data_ptr<float>(), b1.data_ptr<float>(), total1, N1);
    C10_CUDA_KERNEL_LAUNCH_CHECK();

    linear_gemm(handle, h1, w2, h2, M, K2, N2);
    int64_t total2 = (int64_t)M * N2;
    bias_relu_kernel<<<(total2 + threads - 1) / threads, threads, 0, stream>>>(h2.data_ptr<float>(), b2.data_ptr<float>(), total2, N2);
    C10_CUDA_KERNEL_LAUNCH_CHECK();

    linear_gemm(handle, h2, w3, y, M, K3, N3);
    int64_t total3 = (int64_t)M * N3;
    bias_kernel<<<(total3 + threads - 1) / threads, threads, 0, stream>>>(y.data_ptr<float>(), b3.data_ptr<float>(), total3, N3);
    C10_CUDA_KERNEL_LAUNCH_CHECK();

    return y;
}

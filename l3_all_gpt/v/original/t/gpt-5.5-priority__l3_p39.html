        """

        :param hidden_size: The number of features in the hidden state h
        :param num_layers: Number of recurrent layers (default: 1)
        :param bias: If False, then the layer does not use bias weights b_ih and b_hh (default: True)

    def _make_pack(self):
        # The CUDA kernel consumes input/hidden weights transposed as [K, 3H] for coalesced loads.
        ws = []
        key = []
        for l in range(6):
            wih = getattr(self.gru, f"weight_ih_l{l}")
            whh = getattr(self.gru, f"weight_hh_l{l}")
            bih = getattr(self.gru, f"bias_ih_l{l}")
            bhh = getattr(self.gru, f"bias_hh_l{l}")
            key += [(wih.data_ptr(), wih._version), (whh.data_ptr(), whh._version),
                    (bih.data_ptr(), bih._version), (bhh.data_ptr(), bhh._version)]
            ws += [wih.detach().t().contiguous(), whh.detach().t().contiguous(), bih.detach().contiguous(), bhh.detach().contiguous()]
        self._pack_key = tuple(key)
        self._packed = ws

        if self.num_layers != 6 or (not self.gru.bias):
            # Not used by the benchmark configuration; keep semantics for accidental construction.
            return self.gru(x, h0)[0]

            wih = getattr(self.gru, f"weight_ih_l{l}")
            whh = getattr(self.gru, f"weight_hh_l{l}")
            bih = getattr(self.gru, f"bias_ih_l{l}")

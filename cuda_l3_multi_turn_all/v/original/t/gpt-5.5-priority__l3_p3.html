import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        """
        :param input_size: The number of input features
        :param hidden_layer_sizes: A list of ints containing the sizes of each hidden layer
        :param output_size: The number of output features
        """
        super(Model, self).__init__()
        
        layers = []
        current_input_size = input_size
        
        for hidden_size in hidden_layer_sizes:
            layers.append(nn.Linear(current_input_size, hidden_size))
            layers.append(nn.ReLU())
            current_input_size = hidden_size
        
        layers.append(nn.Linear(current_input_size, output_size))
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        """
        :param x: The input tensor, shape (batch_size, input_size)
        :return: The output tensor, shape (batch_size, output_size)
        """
        return self.network(x)

        cublasLtMatmulDescSetAttribute(opDesc, CUBLASLT_MATMUL_DESC_EPILOGUE, &epi, sizeof(epi));

        cublasLtMatmulDescSetAttribute(opDesc, CUBLASLT_MATMUL_DESC_BIAS_POINTER, &biasPtr, sizeof(biasPtr));

        // Interpret contiguous row-major tensors as column-major transposes:
        // out(B,N)^T = W(N,K) * cur(B,K)^T .
        cublasLtMatrixLayoutCreate(&aDesc, CUDA_R_32F, K, N, K);
        cublasLtMatrixLayoutCreate(&bDesc, CUDA_R_32F, K, B, K);
        cublasLtMatrixLayoutCreate(&cDesc, CUDA_R_32F, N, B, N);

        cublasStatus_t stat = cublasLtMatmul(g_ltHandle, opDesc,
            &alpha,
            W.data_ptr<float>(), aDesc,
            cur.data_ptr<float>(), bDesc,
            &beta,
            out.data_ptr<float>(), cDesc,
            out.data_ptr<float>(), cDesc,
            nullptr, nullptr, 0, stream);
        TORCH_CHECK(stat == CUBLAS_STATUS_SUCCESS, "cublasLtMatmul failed ", (int)stat);

        cublasLtMatrixLayoutDestroy(aDesc);
        cublasLtMatrixLayoutDestroy(bDesc);
        cublasLtMatrixLayoutDestroy(cDesc);
        cublasLtMatmulDescDestroy(opDesc);
        cur = out;
    }
    return cur;
}

import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, input_size, layer_sizes, output_size):
        """
        :param input_size: The number of input features
        :param layer_sizes: A list of ints containing the sizes of each hidden layer
        :param output_size: The number of output features
        """
        super(Model, self).__init__()
        
        layers = []
        current_input_size = input_size
        
        for layer_size in layer_sizes:
            layers.append(nn.Linear(current_input_size, layer_size))
            layers.append(nn.ReLU())
            current_input_size = layer_size
        
        layers.append(nn.Linear(current_input_size, output_size))
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        """
        :param x: The input tensor, shape (batch_size, input_size)
        :return: The output tensor, shape (batch_size, output_size)
        """
        return self.network(x)

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_src = r'''
#include <torch/extension.h>

void bias_relu_inplace(torch::Tensor y, torch::Tensor b);
void bias_inplace(torch::Tensor y, torch::Tensor b);

torch::Tensor mlp_forward(torch::Tensor x,
                          torch::Tensor w0, torch::Tensor b0,
                          torch::Tensor w1, torch::Tensor b1,
                          torch::Tensor w2, torch::Tensor b2) {
    // GEMMs are delegated to the same high-performance CUDA BLAS path used by
    // torch.mm; the custom kernels below fuse bias addition with ReLU for the
    // two hidden activations and do the final bias add in-place.
    auto h0 = torch::mm(x, w0.t());
    bias_relu_inplace(h0, b0);
    auto h1 = torch::mm(h0, w1.t());
    bias_relu_inplace(h1, b1);
    auto out = torch::mm(h1, w2.t());
    bias_inplace(out, b2);
    return out;
}
'''

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

__global__ void bias_relu_kernel(float* __restrict__ y, const float* __restrict__ b,
                                 int rows, int cols) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (row < rows && col < cols) {
        int idx = row * cols + col;
        float v = y[idx] + b[col];
        y[idx] = v > 0.0f ? v : 0.0f;
    }
}

__global__ void bias_kernel(float* __restrict__ y, const float* __restrict__ b,
                            int rows, int cols) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    int row = blockIdx.y;
    if (row < rows && col < cols) {
        int idx = row * cols + col;
        y[idx] += b[col];
    }
}

void bias_relu_inplace(torch::Tensor y, torch::Tensor b) {
    int rows = (int)y.size(0);
    int cols = (int)y.size(1);
    dim3 block(256);
    dim3 grid((cols + block.x - 1) / block.x, rows);
    bias_relu_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        y.data_ptr<float>(), b.data_ptr<float>(), rows, cols);
}

void bias_inplace(torch::Tensor y, torch::Tensor b) {
    int rows = (int)y.size(0);
    int cols = (int)y.size(1);
    dim3 block(256);
    dim3 grid((cols + block.x - 1) / block.x, rows);
    bias_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        y.data_ptr<float>(), b.data_ptr<float>(), rows, cols);
}
'''

import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        """
        :param num_layers: The number of layers in the dense block
        :param num_input_features: The number of input feature maps
        :param growth_rate: The growth rate for the dense block (new features added per layer)
        """
        super(Model, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        """
        Creates a single layer with BatchNorm, ReLU, Conv2D, and Dropout.
        """
        return nn.Sequential(
            nn.BatchNorm2d(in_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            nn.Dropout(0.0)
        )
    
    def forward(self, x):
        """
        :param x: Input tensor of shape (batch_size, num_input_features, height, width)
        :return: Concatenated output tensor with shape (batch_size, num_output_features, height, width)
        """
        features = [x]
        for layer in self.layers:
            new_feature = layer(x)
            features.append(new_feature)
            x = torch.cat(features, 1)  # Concatenate along channel axis
        return x

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void prep_bn_eval_kernel(const float* __restrict__ gamma, const float* __restrict__ beta,
                               const float* __restrict__ mean, const float* __restrict__ var,
                               float* __restrict__ scale, float* __restrict__ shift,
                               int C, float eps) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < C) {
        float s = gamma[i] * rsqrtf(var[i] + eps);
        scale[i] = s;
        shift[i] = beta[i] - mean[i] * s;
    }
}

__global__ void prep_bn_train_kernel(const float* __restrict__ feat,
                               const float* __restrict__ gamma, const float* __restrict__ beta,
                               float* __restrict__ scale, float* __restrict__ shift,
                               int N, int C, int Ctot, int HW, float eps) {
    int c = blockIdx.x;
    int tid = threadIdx.x;
    __shared__ float ss[256];
    __shared__ float ss2[256];
    float sum = 0.0f, sum2 = 0.0f;
    int M = N * HW;
    for (int m = tid; m < M; m += blockDim.x) {
        int n = m / HW;
        int p = m - n * HW;
        float v = feat[(n*Ctot + c)*HW + p];
        sum += v;
        sum2 = fmaf(v, v, sum2);
    }
    ss[tid] = sum; ss2[tid] = sum2;
    __syncthreads();
    for (int off = 128; off > 0; off >>= 1) {
        if (tid < off) { ss[tid] += ss[tid+off]; ss2[tid] += ss2[tid+off]; }
        __syncthreads();
    }
    if (tid == 0) {
        float invM = 1.0f / (float)M;
        float mean = ss[0] * invM;
        float var = ss2[0] * invM - mean * mean;
        var = var > 0.0f ? var : 0.0f;
        float s = gamma[c] * rsqrtf(var + eps);
        scale[c] = s;
        shift[c] = beta[c] - mean * s;
    }
}

__global__ void prep_weight_t_kernel(const float* __restrict__ w, float* __restrict__ wt, int C, int G) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    int total = C * 9 * G;
    for (int idx = i; idx < total; idx += blockDim.x * gridDim.x) {
        int oc = idx % G;
        int k = idx / G;
        wt[idx] = w[oc * C * 9 + k];
    }
}

template<int OCPT>
__global__ void conv_kernel(const float* __restrict__ feat, float* __restrict__ out,
                             const float* __restrict__ wt,
                             const float* __restrict__ scale, const float* __restrict__ shift,
                             int N, int C, int Ctot, int H, int W, int out_off, int G) {
    int groups = G / OCPT;
    long long total = (long long)N * H * W * groups;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    int HW = H*W;
    for (long long linear = idx; linear < total; linear += stride) {
        int t = (int)linear;
        int g = t % groups; t /= groups;
        int w0 = t % W; t /= W;
        int h0 = t % H; t /= H;
        int n = t;
        int ocb = g * OCPT;
        float acc[OCPT];
        #pragma unroll
        for (int q=0; q<OCPT; ++q) acc[q] = 0.0f;
        for (int ic = 0; ic < C; ++ic) {
            float sc = scale[ic];
            float sh = shift[ic];
            #pragma unroll
            for (int kh = 0; kh < 3; ++kh) {
                int ih = h0 + kh - 1;
                if ((unsigned)ih >= (unsigned)H) continue;
                #pragma unroll
                for (int kw = 0; kw < 3; ++kw) {
                    int iw = w0 + kw - 1;
                    if ((unsigned)iw >= (unsigned)W) continue;
                    float v = feat[(n*Ctot + ic)*HW + ih*W + iw];
                    v = fmaf(v, sc, sh);
                    v = v > 0.0f ? v : 0.0f;
                    int kidx = ((ic*3 + kh)*3 + kw) * G + ocb;
                    #pragma unroll
                    for (int q=0; q<OCPT; ++q) {
                        acc[q] = fmaf(v, wt[kidx + q], acc[q]);
                    }
                }
            }
        }
        #pragma unroll
        for (int q=0; q<OCPT; ++q) {
            out[(n*Ctot + out_off + ocb + q)*HW + h0*W + w0] = acc[q];
        }
    }
}

__global__ void conv_scalar_kernel(const float* __restrict__ feat, float* __restrict__ out,
                             const float* __restrict__ wt,
                             const float* __restrict__ scale, const float* __restrict__ shift,
                             int N, int C, int Ctot, int H, int W, int out_off, int G) {
    long long total = (long long)N * G * H * W;
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long stride = (long long)blockDim.x * gridDim.x;
    int HW = H*W;
    for (long long linear = idx; linear < total; linear += stride) {
        int t = (int)linear;
        int p = t % HW; t /= HW;
        int oc = t % G; int n = t / G;
        int h0 = p / W, w0 = p - h0*W;
        float acc = 0.0f;
        for (int ic=0; ic<C; ++ic) {
            float sc=scale[ic], sh=shift[ic];
            for (int kh=0; kh<3; ++kh) {
                int ih=h0+kh-1; if ((unsigned)ih >= (unsigned)H) continue;
                for (int kw=0; kw<3; ++kw) {
                    int iw=w0+kw-1; if ((unsigned)iw >= (unsigned)W) continue;
                    float v=feat[(n*Ctot+ic)*HW+ih*W+iw];
                    v=fmaf(v,sc,sh); v = v>0.0f ? v : 0.0f;
                    int kidx=((ic*3+kh)*3+kw)*G+oc;
                    acc=fmaf(v, wt[kidx], acc);
                }
            }
        }
        out[(n*Ctot+out_off+oc)*HW+p]=acc;
    }
}

torch::Tensor dense_forward(torch::Tensor x,
    std::vector<torch::Tensor> conv_w,
    std::vector<torch::Tensor> bn_weight,
    std::vector<torch::Tensor> bn_bias,
    std::vector<torch::Tensor> bn_mean,
    std::vector<torch::Tensor> bn_var,
    double eps,
    bool training) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "only float32 supported");
    x = x.contiguous();
    int N = (int)x.size(0), C0 = (int)x.size(1), H = (int)x.size(2), W = (int)x.size(3);
    int L = (int)conv_w.size();
    int G = (int)conv_w[0].size(0);
    int Ctot = C0 + L * G;
    int HW = H * W;
    auto out = torch::empty({N, Ctot, H, W}, x.options());
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    cudaMemcpy2DAsync(out.data_ptr<float>(), (size_t)Ctot*HW*sizeof(float),
                      x.data_ptr<float>(), (size_t)C0*HW*sizeof(float),
                      (size_t)C0*HW*sizeof(float), (size_t)N,
                      cudaMemcpyDeviceToDevice, stream);
    int threads = 256;
    int C = C0;
    for (int l=0; l<L; ++l) {
        auto w = conv_w[l].contiguous();
        auto gamma = bn_weight[l].contiguous();
        auto beta = bn_bias[l].contiguous();
        auto scale = torch::empty({C}, x.options());
        auto shift = torch::empty({C}, x.options());
        if (training) {
            prep_bn_train_kernel<<<C, 256, 0, stream>>>(out.data_ptr<float>(), gamma.data_ptr<float>(), beta.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), N, C, Ctot, HW, (float)eps);
        } else {
            auto mean = bn_mean[l].contiguous();
            auto var = bn_var[l].contiguous();
            prep_bn_eval_kernel<<<(C+255)/256, 256, 0, stream>>>(gamma.data_ptr<float>(), beta.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), C, (float)eps);
        }
        auto wt = torch::empty({C*9, G}, x.options());
        int wt_total = C*9*G;
        int wb = std::min(1024, (wt_total + threads - 1)/threads);
        prep_weight_t_kernel<<<wb, threads, 0, stream>>>(w.data_ptr<float>(), wt.data_ptr<float>(), C, G);
        if (G % 8 == 0) {
            long long tasks = (long long)N*H*W*(G/8);
            int b = (int)std::min(65535LL, (tasks + threads - 1)/threads);
            conv_kernel<8><<<b, threads, 0, stream>>>(out.data_ptr<float>(), out.data_ptr<float>(), wt.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), N, C, Ctot, H, W, C, G);
        } else if (G % 4 == 0) {
            long long tasks = (long long)N*H*W*(G/4);
            int b = (int)std::min(65535LL, (tasks + threads - 1)/threads);
            conv_kernel<4><<<b, threads, 0, stream>>>(out.data_ptr<float>(), out.data_ptr<float>(), wt.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), N, C, Ctot, H, W, C, G);
        } else {
            long long tasks = (long long)N*G*H*W;
            int b = (int)std::min(65535LL, (tasks + threads - 1)/threads);
            conv_scalar_kernel<<<b, threads, 0, stream>>>(out.data_ptr<float>(), out.data_ptr<float>(), wt.data_ptr<float>(), scale.data_ptr<float>(), shift.data_ptr<float>(), N, C, Ctot, H, W, C, G);
        }
        C += G;
    }
    return out;
}
'''

_ext = load_inline(name='denseblock_ext_v2', cpp_sources=cpp_src, cuda_sources=cuda_src, functions=['dense_forward'], extra_cuda_cflags=['-O3', '--use_fast_math'], extra_cflags=['-O3'], verbose=False)

class ModelNew(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            in_features = num_input_features + i * growth_rate
            layers.append(nn.Sequential(
                nn.BatchNorm2d(in_features),
                nn.ReLU(inplace=True),
                nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
                nn.Dropout(0.0)
            ))
        self.layers = nn.ModuleList(layers)
    def forward(self, x):
        conv_w=[]; bw=[]; bb=[]; bm=[]; bv=[]
        for layer in self.layers:
            bn = layer[0]; conv = layer[2]
            conv_w.append(conv.weight)
            bw.append(bn.weight); bb.append(bn.bias); bm.append(bn.running_mean); bv.append(bn.running_var)
        return _ext.dense_forward(x, conv_w, bw, bb, bm, bv, self.layers[0][0].eps, self.training)

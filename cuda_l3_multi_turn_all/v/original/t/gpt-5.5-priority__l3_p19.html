import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, num_classes=1000, input_channels=3, alpha=1.0):
        """
        MobileNetV1 architecture implementation.

        :param num_classes: The number of output classes (default: 1000)
        :param input_channels: The number of input channels (default: 3 for RGB images)
        :param alpha: Width multiplier (default: 1.0)
        """
        super(Model, self).__init__()
        
        def conv_bn(inp, oup, stride):
            return nn.Sequential(
                nn.Conv2d(inp, oup, 3, stride, 1, bias=False),
                nn.BatchNorm2d(oup),
                nn.ReLU(inplace=True)
            )
        
        def conv_dw(inp, oup, stride):
            return nn.Sequential(
                nn.Conv2d(inp, inp, 3, stride, 1, groups=inp, bias=False),
                nn.BatchNorm2d(inp),
                nn.ReLU(inplace=True),
                
                nn.Conv2d(inp, oup, 1, 1, 0, bias=False),
                nn.BatchNorm2d(oup),
                nn.ReLU(inplace=True),
            )
        
        self.model = nn.Sequential(
            conv_bn(input_channels, int(32 * alpha), 2),
            conv_dw(int(32 * alpha), int(64 * alpha), 1),
            conv_dw(int(64 * alpha), int(128 * alpha), 2),
            conv_dw(int(128 * alpha), int(128 * alpha), 1),
            conv_dw(int(128 * alpha), int(256 * alpha), 2),
            conv_dw(int(256 * alpha), int(256 * alpha), 1),
            conv_dw(int(256 * alpha), int(512 * alpha), 2),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(1024 * alpha), 2),
            conv_dw(int(1024 * alpha), int(1024 * alpha), 1),
            nn.AvgPool2d(7),
        )
        self.fc = nn.Linear(int(1024 * alpha), num_classes)
    
    def forward(self, x):
        """
        :param x: The input tensor, shape (batch_size, input_channels, height, width)
        :return: The output tensor, shape (batch_size, num_classes)
        """
        x = self.model(x)
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x

__global__ void bn_relu_kernel3(float* __restrict__ x, const float* __restrict__ gamma, const float* __restrict__ beta,
                               const float* __restrict__ mean, const float* __restrict__ var,
                               int C, int H, int W, float eps) {
    int sw = blockIdx.x * blockDim.x + threadIdx.x;
    int n = blockIdx.y;
    int c = blockIdx.z;
    int spatial = H * W;
    if (sw >= spatial) return;
    float scale = gamma[c] * rsqrtf(var[c] + eps);
    int idx = (n*C + c) * spatial + sw;
    float yy = (x[idx] - mean[c]) * scale + beta[c];
    x[idx] = yy > 0.0f ? yy : 0.0f;
}
torch::Tensor bn_relu_inplace(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, torch::Tensor mean, torch::Tensor var, double eps) {
    int64_t N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3); int spatial=H*W;
    dim3 block(256); dim3 grid((spatial+255)/256, N, C);
    bn_relu_kernel3<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), gamma.data_ptr<float>(), beta.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), (int)C, (int)H, (int)W, (float)eps);
    return x;
}

__global__ void dw_conv_kernel3(const float* __restrict__ x, const float* __restrict__ w, float* __restrict__ y,
                          int C, int H, int W, int OH, int OW, int stride, int tilesH) {
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int by = blockIdx.y;
    int tile_y = by % tilesH;
    int n = by / tilesH;
    int oh = tile_y * blockDim.y + threadIdx.y;
    int c = blockIdx.z;
    if (ow >= OW || oh >= OH) return;
    int ih0 = oh * stride - 1; int iw0 = ow * stride - 1;
    const float* xp = x + ((n*C + c) * H) * W; const float* wp = w + c*9;
    float acc = 0.0f;
    #pragma unroll
    for (int kh=0; kh<3; ++kh) { int ih = ih0 + kh; if ((unsigned)ih < (unsigned)H) {
        #pragma unroll
        for (int kw=0; kw<3; ++kw) { int iw = iw0 + kw; if ((unsigned)iw < (unsigned)W) acc = fmaf(xp[ih*W + iw], wp[kh*3 + kw], acc); }
    }}
    y[((n*C + c)*OH + oh)*OW + ow] = acc;
}

torch::Tensor dw_conv_forward(torch::Tensor x, torch::Tensor w, int64_t stride) {
    int64_t N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3); int64_t OH=(H-1)/stride+1, OW=(W-1)/stride+1;
    auto y=torch::empty({N,C,OH,OW}, x.options());
    dim3 block(32,8); int tilesH=(OH+7)/8; dim3 grid((OW+31)/32, N*tilesH, C);
    dw_conv_kernel3<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), w.data_ptr<float>(), y.data_ptr<float>(), (int)C,(int)H,(int)W,(int)OH,(int)OW,(int)stride, tilesH);
    return y;
}

__global__ void dw_bn_kernel3(const float* __restrict__ x, const float* __restrict__ w,
                          const float* __restrict__ gamma, const float* __restrict__ beta,
                          const float* __restrict__ mean, const float* __restrict__ var, float* __restrict__ y,
                          int C, int H, int W, int OH, int OW, int stride, int tilesH, float eps) {
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int by = blockIdx.y;
    int tile_y = by % tilesH;
    int n = by / tilesH;
    int oh = tile_y * blockDim.y + threadIdx.y;
    int c = blockIdx.z;
    if (ow >= OW || oh >= OH) return;
    int ih0 = oh * stride - 1; int iw0 = ow * stride - 1;
    const float* xp = x + ((n*C + c) * H) * W; const float* wp = w + c*9; float acc = 0.0f;
    #pragma unroll
    for (int kh=0; kh<3; ++kh) { int ih=ih0+kh; if ((unsigned)ih < (unsigned)H) {
        #pragma unroll
        for (int kw=0; kw<3; ++kw) { int iw=iw0+kw; if ((unsigned)iw < (unsigned)W) acc = fmaf(xp[ih*W+iw], wp[kh*3+kw], acc); }
    }}
    float v = (acc - mean[c]) * (gamma[c] * rsqrtf(var[c] + eps)) + beta[c];
    y[((n*C + c)*OH + oh)*OW + ow] = v > 0.0f ? v : 0.0f;
}
torch::Tensor dw_bn_relu_forward(torch::Tensor x, torch::Tensor w, torch::Tensor gamma, torch::Tensor beta, torch::Tensor mean, torch::Tensor var, int64_t stride, double eps) {
    int64_t N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3); int64_t OH=(H-1)/stride+1, OW=(W-1)/stride+1;
    auto y=torch::empty({N,C,OH,OW}, x.options());
    dim3 block(32,8); int tilesH=(OH+7)/8; dim3 grid((OW+31)/32, N*tilesH, C);
    dw_bn_kernel3<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), w.data_ptr<float>(), gamma.data_ptr<float>(), beta.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), y.data_ptr<float>(), (int)C,(int)H,(int)W,(int)OH,(int)OW,(int)stride, tilesH,(float)eps);
    return y;
}

__global__ void avg7_kernel(const float* __restrict__ x, float* __restrict__ y, int total) {
    int idx=blockIdx.x*blockDim.x+threadIdx.x; if(idx>=total) return; const float* xp=x+idx*49; float s=0.0f;
    #pragma unroll
    for(int i=0;i<49;i++) s+=xp[i]; y[idx]=s*(1.0f/49.0f);
}
torch::Tensor avgpool7_forward(torch::Tensor x) {
    int64_t N=x.size(0), C=x.size(1); auto y=torch::empty({N,C}, x.options()); int total=N*C; int threads=256, blocks=(total+threads-1)/threads;
    avg7_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(x.data_ptr<float>(), y.data_ptr<float>(), total); return y;
}

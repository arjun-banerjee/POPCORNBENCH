import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        """
        Initialize the LSTM model.

        :param input_size: The number of expected features in the input `x`
        :param hidden_size: The number of features in the hidden state `h`
        :param num_layers: Number of recurrent layers
        :param output_size: The number of output features
        :param dropout: If non-zero, introduces a Dropout layer on the outputs of each LSTM layer except the last layer
        """
        super(Model, self).__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers,
                            batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = nn.Linear(hidden_size, output_size)

    def forward(self, x, h0=None, c0=None):
        """
        Forward pass through the LSTM model.

        :param x: The input tensor, shape (batch_size, sequence_length, input_size)
        :param h0: Optional initial hidden state (num_layers, batch_size, hidden_size)
        :param c0: Optional initial cell state (num_layers, batch_size, hidden_size)
        :return: The output tensor, shape (batch_size, output_size)
        """
        batch_size = x.size(0)

        if h0 is None:
            h0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device)
        if c0 is None:
            c0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device)

        out, _ = self.lstm(x, (h0, c0))  # out: (batch_size, seq_length, hidden_size)
        out = self.fc(out[:, -1, :])     # out: (batch_size, output_size)

        return out

cuda_src = r'''
#include <torch/extension.h>
#include <ATen/ATen.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void linear_hn_kernel(const float* __restrict__ hn,
                                 const float* __restrict__ w,
                                 const float* __restrict__ b,
                                 float* __restrict__ y,
                                 int B, int H, int O, int L) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * O;
    if (idx >= total) return;

    int o = idx % O;
    int bs = idx / O;
    const float* x = hn + (((L - 1) * B + bs) * H);
    const float* ww = w + o * H;

    float sum = b ? b[o] : 0.0f;
    #pragma unroll 4
    for (int h = 0; h < H; ++h) {
        sum = fmaf(x[h], ww[h], sum);
    }
    y[idx] = sum;
}

torch::Tensor lstm_last_linear(torch::Tensor x, torch::Tensor h0, torch::Tensor c0,
                               std::vector<torch::Tensor> params,
                               torch::Tensor weight, torch::Tensor bias,
                               int64_t num_layers, double dropout, bool train) {

    // h_n[-1] is identical to out[:, -1, :] for this unidirectional LSTM and
    // avoids indexing the large sequence output before the small final linear.
    auto res = at::lstm(x, hx, params, true, num_layers, dropout, train, false, true);
    torch::Tensor hn = std::get<1>(res);

    int B = (int)x.size(0);
    int H = (int)hn.size(2);
    int O = (int)weight.size(0);
    int L = (int)num_layers;

    auto y = torch::empty({B, O}, x.options());
    int total = B * O;
    linear_hn_kernel<<<(total + 127) / 128, 128>>>(
        hn.data_ptr<float>(), weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        y.data_ptr<float>(), B, H, O, L);
    return y;
}

class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super(ModelNew, self).__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.dropout = dropout
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers,
                            batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = nn.Linear(hidden_size, output_size)

    def forward(self, x, h0=None, c0=None):
        batch_size = x.size(0)
        if h0 is None and c0 is None:
            # Same RNG stream/order as two consecutive randn calls, but one allocation/kernel.
            hc = torch.randn(2, self.num_layers, batch_size, self.hidden_size, device=x.device)
            h0 = hc[0]
            c0 = hc[1]
        else:
            if h0 is None:
                h0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device)
            if c0 is None:
                c0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device)

        return _ext.lstm_last_linear(x, h0, c0, self.lstm._flat_weights,
                                     self.fc.weight, self.fc.bias,
                                     self.num_layers, float(self.dropout), bool(self.training))

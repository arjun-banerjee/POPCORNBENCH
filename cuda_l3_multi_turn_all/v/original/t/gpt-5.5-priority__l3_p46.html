import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th


class Model(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(Model, self).__init__()

        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # The `clusters` weights are the `(w,b)` in the paper
        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        # The `clusters2` weights are the visual words `c_k` in the paper
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        """Aggregates feature maps into a fixed size representation.  In the following
        notation, B = batch_size, N = num_features, K = num_clusters, D = feature_size.

        Args:
            x (th.Tensor): B x N x D

        Returns:
            (th.Tensor): B x DK
        """
        max_sample = x.size()[1]
        x = x.view(-1, self.feature_size)  # B x N x D -> BN x D

        if x.device != self.clusters.device:
            msg = f"x.device {x.device} != cluster.device {self.clusters.device}"
            raise ValueError(msg)

        assignment = th.matmul(x, self.clusters)  # (BN x D) x (D x (K+G)) -> BN x (K+G)
        assignment = self.batch_norm(assignment)

        assignment = F.softmax(assignment, dim=1)  # BN x (K+G) -> BN x (K+G)
        # remove ghost assigments
        assignment = assignment[:, :self.cluster_size]
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # -> B x N x K
        a_sum = th.sum(assignment, dim=1, keepdim=True)  # B x N x K -> B x 1 x K
        a = a_sum * self.clusters2

        assignment = assignment.transpose(1, 2)  # B x N x K -> B x K x N

        x = x.view(-1, max_sample, self.feature_size)  # BN x D -> B x N x D
        vlad = th.matmul(assignment, x)  # (B x K x N) x (B x N x D) -> B x K x D
        vlad = vlad.transpose(1, 2)  # -> B x D x K
        vlad = vlad - a

        # L2 intra norm
        vlad = F.normalize(vlad)

        # flattening + L2 norm
        vlad = vlad.reshape(-1, self.cluster_size * self.feature_size)  # -> B x DK
        vlad = F.normalize(vlad)
        return vlad  # B x DK

batch_size = 2048
num_features = 100
num_clusters = 32
feature_size = 512
ghost_clusters = 16

def get_inputs():
  return [torch.rand(batch_size, num_features, feature_size)]

def get_init_inputs():
  return [num_clusters, feature_size, ghost_clusters]

#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void bn_stats_kernel(const float* __restrict__ logits, float* __restrict__ mean, float* __restrict__ invstd,
                                float* __restrict__ running_mean, float* __restrict__ running_var,
                                int M, int C, float eps, float momentum) {
    int c = blockIdx.x;
    extern __shared__ double smem[];
    double* ss = smem;
    double* ss2 = smem + blockDim.x;
    double s = 0.0, s2 = 0.0;
    for (int i = threadIdx.x; i < M; i += blockDim.x) {
        float v = logits[(long long)i*C + c];
        s += (double)v;
        s2 += (double)v * (double)v;
    }
    ss[threadIdx.x] = s;
    ss2[threadIdx.x] = s2;
    __syncthreads();
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            ss[threadIdx.x] += ss[threadIdx.x + stride];
            ss2[threadIdx.x] += ss2[threadIdx.x + stride];
        }
        __syncthreads();
    }
    if (threadIdx.x == 0) {
        double m = ss[0] / (double)M;
        double var = ss2[0] / (double)M - m * m;
        if (var < 0.0) var = 0.0;
        mean[c] = (float)m;
        invstd[c] = rsqrtf((float)var + eps);
        running_mean[c] = (1.0f - momentum) * running_mean[c] + momentum * (float)m;
        float unb = (M > 1) ? (float)(var * ((double)M / (double)(M - 1))) : (float)var;
        running_var[c] = (1.0f - momentum) * running_var[c] + momentum * unb;
    }
}

__global__ void bn_eval_params_kernel(const float* __restrict__ running_mean, const float* __restrict__ running_var,
                                      float* __restrict__ mean, float* __restrict__ invstd, int C, float eps) {
    int c = blockIdx.x * blockDim.x + threadIdx.x;
    if (c < C) {
        mean[c] = running_mean[c];
        invstd[c] = rsqrtf(running_var[c] + eps);
    }
}

__global__ void softmax_assign_warp_kernel(const float* __restrict__ logits, const float* __restrict__ mean,
                                           const float* __restrict__ invstd, const float* __restrict__ gamma,
                                           const float* __restrict__ beta, float* __restrict__ assign,
                                           int M, int C, int K) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    int row = blockIdx.x * 4 + wid;
    if (row >= M) return;

    float mx = -1.0e30f;
    for (int c = lane; c < C; c += 32) {
        float y = (logits[(long long)row * C + c] - mean[c]) * invstd[c] * gamma[c] + beta[c];
        mx = fmaxf(mx, y);
    }
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) mx = fmaxf(mx, __shfl_down_sync(0xffffffff, mx, off));
    mx = __shfl_sync(0xffffffff, mx, 0);

    float sum = 0.0f;
    for (int c = lane; c < C; c += 32) {
        float y = (logits[(long long)row * C + c] - mean[c]) * invstd[c] * gamma[c] + beta[c];
        sum += expf(y - mx);
    }
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) sum += __shfl_down_sync(0xffffffff, sum, off);
    sum = __shfl_sync(0xffffffff, sum, 0);

    if (lane < K) {
        float y = (logits[(long long)row * C + lane] - mean[lane]) * invstd[lane] * gamma[lane] + beta[lane];
        assign[(long long)row * K + lane] = expf(y - mx) / sum;
    }
}

__global__ void vlad4_kernel(const float* __restrict__ x, const float* __restrict__ assign,
                             const float* __restrict__ clusters2, float* __restrict__ out,
                             int B, int N, int D, int K) {
    int bg = blockIdx.x;
    int groups = K / 4;
    int b = bg / groups;
    int kg = (bg - b * groups) * 4;

    extern __shared__ float sh[];
    float* a0 = sh;
    float* a1 = a0 + N;
    float* a2 = a1 + N;
    float* a3 = a2 + N;
    float* r0 = a3 + N;
    float* r1 = r0 + blockDim.x;
    float* r2 = r1 + blockDim.x;
    float* r3 = r2 + blockDim.x;

    for (int n = threadIdx.x; n < N; n += blockDim.x) {
        const float* ap = assign + ((long long)b * N + n) * K + kg;
        a0[n] = ap[0];
        a1[n] = ap[1];
        a2[n] = ap[2];
        a3[n] = ap[3];
    }
    __syncthreads();

    float as0 = 0.f, as1 = 0.f, as2 = 0.f, as3 = 0.f;
    for (int n = 0; n < N; ++n) {
        as0 += a0[n]; as1 += a1[n]; as2 += a2[n]; as3 += a3[n];
    }

    float ln0 = 0.f, ln1 = 0.f, ln2 = 0.f, ln3 = 0.f;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float s0 = 0.f, s1 = 0.f, s2 = 0.f, s3 = 0.f;
        const float* xp = x + ((long long)b * N) * D + d;
        #pragma unroll 4
        for (int n = 0; n < N; ++n) {
            float xv = xp[(long long)n * D];
            s0 += a0[n] * xv;
            s1 += a1[n] * xv;
            s2 += a2[n] * xv;
            s3 += a3[n] * xv;
        }
        float v0 = s0 - as0 * clusters2[(long long)d * K + kg + 0];
        float v1 = s1 - as1 * clusters2[(long long)d * K + kg + 1];
        float v2 = s2 - as2 * clusters2[(long long)d * K + kg + 2];
        float v3 = s3 - as3 * clusters2[(long long)d * K + kg + 3];
        long long o = ((long long)b * D + d) * K + kg;
        out[o + 0] = v0;
        out[o + 1] = v1;
        out[o + 2] = v2;
        out[o + 3] = v3;
        ln0 += v0 * v0; ln1 += v1 * v1; ln2 += v2 * v2; ln3 += v3 * v3;
    }

    r0[threadIdx.x] = ln0; r1[threadIdx.x] = ln1; r2[threadIdx.x] = ln2; r3[threadIdx.x] = ln3;
    __syncthreads();
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            r0[threadIdx.x] += r0[threadIdx.x + stride];
            r1[threadIdx.x] += r1[threadIdx.x + stride];
            r2[threadIdx.x] += r2[threadIdx.x + stride];
            r3[threadIdx.x] += r3[threadIdx.x + stride];
        }
        __syncthreads();
    }

    float sk = rsqrtf((float)K);
    float sc0 = rsqrtf(r0[0] + 1e-20f) * sk;
    float sc1 = rsqrtf(r1[0] + 1e-20f) * sk;
    float sc2 = rsqrtf(r2[0] + 1e-20f) * sk;
    float sc3 = rsqrtf(r3[0] + 1e-20f) * sk;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        long long o = ((long long)b * D + d) * K + kg;
        out[o + 0] *= sc0;
        out[o + 1] *= sc1;
        out[o + 2] *= sc2;
        out[o + 3] *= sc3;
    }
}

torch::Tensor netvlad_forward(torch::Tensor x, torch::Tensor clusters, torch::Tensor bn_weight, torch::Tensor bn_bias,
                              torch::Tensor running_mean, torch::Tensor running_var, torch::Tensor clusters2, bool training) {
    int B = (int)x.size(0), N = (int)x.size(1), D = (int)x.size(2);
    int C = (int)clusters.size(1);
    int K = (int)clusters2.size(2);

    auto x2 = x.reshape({B * N, D});
    auto logits = at::matmul(x2, clusters);
    auto opts = x.options();
    auto mean = torch::empty({C}, opts);
    auto invstd = torch::empty({C}, opts);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (training) {
        bn_stats_kernel<<<C, 256, 2 * 256 * sizeof(double), stream>>>(logits.data_ptr<float>(), mean.data_ptr<float>(),
            invstd.data_ptr<float>(), running_mean.data_ptr<float>(), running_var.data_ptr<float>(), B * N, C, 1e-5f, 0.1f);
    } else {
        bn_eval_params_kernel<<<(C + 255) / 256, 256, 0, stream>>>(running_mean.data_ptr<float>(), running_var.data_ptr<float>(),
            mean.data_ptr<float>(), invstd.data_ptr<float>(), C, 1e-5f);
    }

    auto assign = torch::empty({B * N, K}, opts);
    softmax_assign_warp_kernel<<<(B * N + 3) / 4, 128, 0, stream>>>(logits.data_ptr<float>(), mean.data_ptr<float>(),
        invstd.data_ptr<float>(), bn_weight.data_ptr<float>(), bn_bias.data_ptr<float>(), assign.data_ptr<float>(), B * N, C, K);

    auto out = torch::empty({B, D * K}, opts);
    vlad4_kernel<<<B * (K / 4), 256, (4 * N + 4 * 256) * sizeof(float), stream>>>(x.data_ptr<float>(), assign.data_ptr<float>(),
        clusters2.data_ptr<float>(), out.data_ptr<float>(), B, N, D, K);
    return out;
}

class ModelNew(nn.Module):

        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters
        init_sc = (1 / math.sqrt(feature_size))

        self.out_dim = self.cluster_size * feature_size

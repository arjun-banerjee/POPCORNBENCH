import torch
import torch.nn as nn
import math


class Model(nn.Module):
    """
    One iteration of Expectation-Maximization for a Gaussian Mixture Model.
    E-step: compute soft assignments (responsibilities).
    M-step: update means, covariances, and mixing weights.
    The dominant cost is the batched evaluation of K Gaussian densities
    over N data points.
    """

    def __init__(self, dim, num_components):
        super().__init__()
        self.dim = dim
        self.K = num_components
        self.register_buffer("mus", torch.randn(num_components, dim))
        self.register_buffer("covs", torch.eye(dim).unsqueeze(0).expand(num_components, -1, -1).clone())
        self.register_buffer("log_pi", torch.zeros(num_components))

    def _log_gaussian(self, x, mu, cov):
        """Log N(x | mu, cov) for batched x."""
        D = mu.shape[-1]
        diff = x - mu.unsqueeze(0)  # (N, D)
        L = torch.linalg.cholesky(cov)
        solve = torch.linalg.solve_triangular(L, diff.unsqueeze(-1), upper=False).squeeze(-1)
        maha = (solve ** 2).sum(-1)
        log_det = 2.0 * L.diagonal().log().sum()
        return -0.5 * (D * math.log(2 * math.pi) + log_det + maha)

    def forward(self, x: torch.Tensor) -> tuple:
        """
        Args:
            x: (N, dim) – data points
        Returns:
            (new_mus, new_covs, new_log_pi, log_lik):
                new_mus:    (K, dim)
                new_covs:   (K, dim, dim)
                new_log_pi: (K,)
                log_lik:    scalar – total log-likelihood
        """
        N, D = x.shape
        K = self.K
        log_pi = self.log_pi - torch.logsumexp(self.log_pi, dim=0)

        # E-step
        log_resp = torch.stack([
            self._log_gaussian(x, self.mus[k], self.covs[k]) + log_pi[k]
            for k in range(K)
        ], dim=-1)  # (N, K)
        log_lik = torch.logsumexp(log_resp, dim=-1).sum()
        log_resp = log_resp - torch.logsumexp(log_resp, dim=-1, keepdim=True)
        resp = torch.exp(log_resp)  # (N, K)

        # M-step
        N_k = resp.sum(dim=0).clamp(min=1e-8)  # (K,)
        new_mus = (resp.t() @ x) / N_k.unsqueeze(-1)
        new_covs = torch.zeros_like(self.covs)
        for k in range(K):
            diff = x - new_mus[k].unsqueeze(0)
            weighted = diff * resp[:, k:k+1]
            new_covs[k] = (weighted.t() @ diff) / N_k[k] + 1e-4 * torch.eye(D, device=x.device)

        new_log_pi = N_k.log() - math.log(N)

        return new_mus, new_covs, new_log_pi, log_lik

#define NSTATS (KCONST*(1+DCONST+DCONST*DCONST))

__inline__ __device__ float block_sum(float v) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, off);
    }
    __shared__ float warp_sums[8];
    if (lane == 0) warp_sums[wid] = v;
    __syncthreads();
    v = (threadIdx.x < 8) ? warp_sums[lane] : 0.0f;
    if (wid == 0) {
        #pragma unroll
        for (int off = 16; off > 0; off >>= 1) {
            v += __shfl_down_sync(0xffffffff, v, off);
        }
    }
    return v;
}

__global__ __launch_bounds__(256, 4)
void estep_kernel(const float* __restrict__ x, const float* __restrict__ mus,
                  float* __restrict__ resp, float* __restrict__ ll_part, int N) {
    int tid = threadIdx.x;
    int n = blockIdx.x * blockDim.x + tid;
    float ll = 0.0f;
    if (n < N) {
        float xv[DCONST];
        #pragma unroll
        for (int d = 0; d < DCONST; ++d) xv[d] = x[n * DCONST + d];
        float a[KCONST];
        float m = -3.402823466e38f;
        #pragma unroll
        for (int k = 0; k < KCONST; ++k) {
            float dist = 0.0f;
            #pragma unroll
            for (int d = 0; d < DCONST; ++d) {
                float diff = xv[d] - mus[k * DCONST + d];
                dist += diff * diff;
            }
            a[k] = -0.5f * dist;
            m = fmaxf(m, a[k]);
        }
        float s = 0.0f;
        #pragma unroll
        for (int k = 0; k < KCONST; ++k) {
            a[k] = expf(a[k] - m);
            s += a[k];
        }
        float invs = 1.0f / s;
        #pragma unroll
        for (int k = 0; k < KCONST; ++k) resp[k * N + n] = a[k] * invs;
        const float c = -0.5f * ((float)DCONST) * 1.8378770664093453f - 1.6094379124341003f;
        ll = m + logf(s) + c;
    }
    float sum = block_sum(ll);
    if (tid == 0) ll_part[blockIdx.x] = sum;
}

__global__ __launch_bounds__(256, 4)
void stats_kernel(const float* __restrict__ x, const float* __restrict__ resp,
                  float* __restrict__ stats, int N) {
    int stat = blockIdx.x;
    int k = stat / (1 + DCONST + DCONST * DCONST);
    int rem = stat - k * (1 + DCONST + DCONST * DCONST);
    float acc = 0.0f;
    for (int n = threadIdx.x; n < N; n += blockDim.x) {
        float r = resp[k * N + n];
        if (rem == 0) {
            acc += r;
        } else if (rem <= DCONST) {
            int d = rem - 1;
            acc += r * x[n * DCONST + d];
        } else {
            int idx = rem - 1 - DCONST;
            int i = idx / DCONST;
            int j = idx - i * DCONST;
            acc += r * x[n * DCONST + i] * x[n * DCONST + j];
        }
    }
    float sum = block_sum(acc);
    if (threadIdx.x == 0) stats[stat] = sum;
}

__global__ void finalize_kernel(const float* __restrict__ stats, const float* __restrict__ ll_part,
                                float* __restrict__ new_mus, float* __restrict__ new_covs,
                                float* __restrict__ new_log_pi, float* __restrict__ log_lik,
                                int N, int B) {
    int t = threadIdx.x;
    __shared__ float Nk[KCONST];
    __shared__ float Mu[KCONST * DCONST];
    if (t < KCONST) {
        int base = t * (1 + DCONST + DCONST * DCONST);
        float nk = stats[base];
        if (nk < 1.0e-8f) nk = 1.0e-8f;
        Nk[t] = nk;
        new_log_pi[t] = logf(nk) - logf((float)N);
    }
    if (t == 0) {
        float ll = 0.0f;
        for (int b = 0; b < B; ++b) ll += ll_part[b];
        *log_lik = ll;
    }
    __syncthreads();
    if (t < KCONST * DCONST) {
        int k = t / DCONST;
        int d = t - k * DCONST;
        int base = k * (1 + DCONST + DCONST * DCONST);
        float mu = stats[base + 1 + d] / Nk[k];
        Mu[t] = mu;
        new_mus[t] = mu;
    }
    __syncthreads();
    if (t < KCONST * DCONST * DCONST) {
        int tmp = t;
        int k = tmp / (DCONST * DCONST);
        tmp -= k * (DCONST * DCONST);
        int i = tmp / DCONST;
        int j = tmp - i * DCONST;
        int base = k * (1 + DCONST + DCONST * DCONST) + 1 + DCONST;
        float cov = stats[base + i * DCONST + j] / Nk[k] - Mu[k * DCONST + i] * Mu[k * DCONST + j];
        if (i == j) cov += 1.0e-4f;
        new_covs[t] = cov;
    }
}

std::vector<torch::Tensor> em_forward(torch::Tensor x, torch::Tensor mus, torch::Tensor covs, torch::Tensor log_pi) {
    const int N = x.size(0);
    const int D = x.size(1);
    const int K = mus.size(0);
    auto opts = x.options();
    auto new_mus = torch::empty({K, D}, opts);
    auto new_covs = torch::empty({K, D, D}, opts);
    auto new_log_pi = torch::empty({K}, opts);
    auto log_lik = torch::empty({}, opts);
    const int threads = 256;
    const int B = (N + threads - 1) / threads;
    auto resp = torch::empty({K, N}, opts);
    auto stats = torch::empty({NSTATS}, opts);
    auto ll_part = torch::empty({B}, opts);
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    estep_kernel<<<B, threads, 0, stream>>>(x.data_ptr<float>(), mus.data_ptr<float>(), resp.data_ptr<float>(), ll_part.data_ptr<float>(), N);
    stats_kernel<<<NSTATS, threads, 0, stream>>>(x.data_ptr<float>(), resp.data_ptr<float>(), stats.data_ptr<float>(), N);
    finalize_kernel<<<1, 512, 0, stream>>>(stats.data_ptr<float>(), ll_part.data_ptr<float>(), new_mus.data_ptr<float>(), new_covs.data_ptr<float>(), new_log_pi.data_ptr<float>(), log_lik.data_ptr<float>(), N, B);
    return {new_mus, new_covs, new_log_pi, log_lik};
}

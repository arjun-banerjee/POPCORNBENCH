import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    """
    Full Hyena operator (order 2): projects input into value (v), query-like (q), and key-like (k)
    branches, applies short depthwise convolutions to q and k, generates a long convolution filter
    via a small MLP, convolves v with the filter via FFT, then applies two rounds of elementwise
    gating. This is the core sequence mixing operator in StripedHyena, HyenaDNA, and Evo
    for subquadratic long-range dependency modeling.

    Args:
        d_model (int): Model dimension.
        seq_len (int): Maximum sequence length.
        order (int): Hyena recurrence order (number of gating rounds, typically 2).
        short_filter_size (int): Kernel size for short depthwise convolutions.
    """
    def __init__(self, d_model: int, seq_len: int, order: int, short_filter_size: int):
        super(Model, self).__init__()
        self.d_model = d_model
        self.seq_len = seq_len
        self.order = order
        self.fft_size = 2 * seq_len

        self.in_proj = nn.Linear(d_model, d_model * (order + 1), bias=True)

        self.short_convs = nn.ModuleList([
            nn.Conv1d(d_model, d_model, short_filter_size,
                      padding=short_filter_size // 2, groups=d_model)
            for _ in range(order)
        ])

        self.filter_mlp = nn.Sequential(
            nn.Linear(1, 64),
            nn.SiLU(),
            nn.Linear(64, d_model),
        )
        t = torch.linspace(0, 1, seq_len).unsqueeze(-1)
        self.register_buffer('filter_positions', t)

        self.out_proj = nn.Linear(d_model, d_model, bias=True)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, L, D = x.shape

        projected = self.in_proj(x)
        chunks = projected.chunk(self.order + 1, dim=-1)
        v = chunks[0].transpose(1, 2)
        gates = [c.transpose(1, 2) for c in chunks[1:]]

        for i in range(self.order):
            gates[i] = self.short_convs[i](gates[i])

        h = self.filter_mlp(self.filter_positions).transpose(0, 1)

        v_padded = F.pad(v, (0, self.seq_len))
        h_padded = F.pad(h, (0, self.seq_len))
        v_f = torch.fft.rfft(v_padded, dim=-1)
        h_f = torch.fft.rfft(h_padded, dim=-1)
        y_f = v_f * h_f.unsqueeze(0)
        y = torch.fft.irfft(y_f, n=self.fft_size, dim=-1)[..., :L]

        for gate in gates:
            y = y * gate

        y = y.transpose(1, 2)
        y = self.out_proj(y)
        return y

__global__ void hyena_prep_kernel(const float* __restrict__ p,
                                  const float* __restrict__ w0, const float* __restrict__ b0,
                                  const float* __restrict__ w1, const float* __restrict__ b1,
                                  float* __restrict__ vpad, float* __restrict__ gates,
                                  int B, int L, int D, int S) {
    const long long total_v = (long long)B * D * (2 * S);
    const long long total_g = (long long)B * D * L;
    const long long total = total_v + 2LL * total_g;
    const long long step = (long long)blockDim.x * gridDim.x;
    for (long long t = (long long)blockIdx.x * blockDim.x + threadIdx.x; t < total; t += step) {
        if (t < total_v) {
            int l = (int)(t % (2 * S));
            long long q = t / (2 * S);
            int d = (int)(q % D);
            int b = (int)(q / D);
            float val = 0.0f;
            if (l < L) val = p[((long long)b * L + l) * (3 * D) + d];
            vpad[t] = val;
        } else {
            long long u = t - total_v;
            int which = 0;
            if (u >= total_g) { which = 1; u -= total_g; }
            int l = (int)(u % L);
            long long q = u / L;
            int d = (int)(q % D);
            int b = (int)(q / D);
            const float* w = which ? w1 : w0;
            const float* bias = which ? b1 : b0;
            int src_base = (which ? 2 * D : D) + d;
            float acc = bias[d];
            if (l > 0)     acc += w[(long long)d * 3 + 0] * p[((long long)b * L + (l - 1)) * (3 * D) + src_base];
                           acc += w[(long long)d * 3 + 1] * p[((long long)b * L +  l     ) * (3 * D) + src_base];
            if (l + 1 < L) acc += w[(long long)d * 3 + 2] * p[((long long)b * L + (l + 1)) * (3 * D) + src_base];
            gates[(long long)which * total_g + ((long long)b * D + d) * L + l] = acc;
        }
    }
}

__global__ void hyena_apply_kernel(const float* __restrict__ y,
                                   const float* __restrict__ gates,
                                   float* __restrict__ out,
                                   int B, int D, int L, int S) {
    const long long total = (long long)B * L * D;
    const long long gate_stride = (long long)B * D * L;
    const long long step = (long long)blockDim.x * gridDim.x;
    for (long long t = (long long)blockIdx.x * blockDim.x + threadIdx.x; t < total; t += step) {
        int d = (int)(t % D);
        long long q = t / D;
        int l = (int)(q % L);
        int b = (int)(q / L);
        long long gi = ((long long)b * D + d) * L + l;
        out[t] = y[((long long)b * D + d) * (2 * S) + l] * gates[gi] * gates[gate_stride + gi];
    }
}

std::vector<torch::Tensor> hyena_prep_cuda(torch::Tensor projected,
                                           torch::Tensor w0, torch::Tensor b0,
                                           torch::Tensor w1, torch::Tensor b1,
                                           int64_t seq_len) {
    int B = (int)projected.size(0);
    int L = (int)projected.size(1);
    int D = (int)projected.size(2) / 3;
    int S = (int)seq_len;
    auto vpad = torch::empty({B, D, 2 * S}, projected.options());
    auto gates = torch::empty({2, B, D, L}, projected.options());
    long long total = (long long)B * D * (2 * S) + 2LL * B * D * L;
    int threads = 256;
    int blocks = (int)std::min(65535LL, (total + threads - 1) / threads);
    hyena_prep_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        projected.data_ptr<float>(), w0.data_ptr<float>(), b0.data_ptr<float>(),
        w1.data_ptr<float>(), b1.data_ptr<float>(), vpad.data_ptr<float>(), gates.data_ptr<float>(),
        B, L, D, S);
    return {vpad, gates};
}

torch::Tensor hyena_apply_cuda(torch::Tensor y, torch::Tensor gates) {
    int B = (int)y.size(0);
    int D = (int)y.size(1);
    int S = (int)y.size(2) / 2;
    int L = (int)gates.size(3);
    auto out = torch::empty({B, L, D}, y.options());
    long long total = (long long)B * L * D;
    int threads = 256;
    int blocks = (int)std::min(65535LL, (total + threads - 1) / threads);
    hyena_apply_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        y.data_ptr<float>(), gates.data_ptr<float>(), out.data_ptr<float>(), B, D, L, S);
    return out;
}

        super(ModelNew, self).__init__()
        self.d_model = d_model
        self.seq_len = seq_len
        self.order = order
        self.fft_size = 2 * seq_len

        self.in_proj = nn.Linear(d_model, d_model * (order + 1), bias=True)
        self.short_convs = nn.ModuleList([
            nn.Conv1d(d_model, d_model, short_filter_size,
                      padding=short_filter_size // 2, groups=d_model)
            for _ in range(order)
        ])
        self.filter_mlp = nn.Sequential(
            nn.Linear(1, 64),
            nn.SiLU(),
            nn.Linear(64, d_model),
        )
        t = torch.linspace(0, 1, seq_len).unsqueeze(-1)
        self.register_buffer('filter_positions', t)
        self.out_proj = nn.Linear(d_model, d_model, bias=True)

        self._hf_key = None

               self.filter_mlp[2].weight._version, self.filter_mlp[2].bias._version)
        if self._hf_cache is None or self._hf_key != key or self._hf_cache.device != x.device:

            self.short_convs[1].kernel_size[0] == 3 and x.size(1) == self.seq_len):

            gates[i] = self.short_convs[i](gates[i])
        h = self.filter_mlp(self.filter_positions).transpose(0, 1)

import torch
import torch.nn as nn

# Default ModelNew is PyTorch-only. run_and_check: use check_kernel=False with backend=cuda
# until the candidate file contains a real CUDA kernel (see kernel_static_checker).


class Model(nn.Module):
    """
    **Fused probability-flow ODE integration** (flow-matching style) in one ``forward``.

    Models the ODE ``dx/dt = v_θ(x, t)`` on ``t ∈ [0, 1]`` with a learned velocity
    field. A **single fused pass** unrolls **fixed-step Euler** integration for
    ``num_steps`` substeps—typical for **real-time trajectory / action generation**
    where latency is dominated by repeated ``v_θ`` evaluations.

    This is the continuous-time analogue of a rectified-flow / conditional-flow
    sampler: one ``forward`` returns an approximate **x(1)** from **x(0)** without a
    Python loop across separate module calls (fusion target for the integrator +
    field).
    """

    def __init__(self, dim: int, num_steps: int):
        super().__init__()
        assert dim >= 1 and num_steps >= 1
        self.dim = dim
        self.num_steps = num_steps
        self.register_buffer("dt", torch.tensor(1.0 / float(num_steps)))

        self.velocity = nn.Sequential(
            nn.Linear(dim + 1, dim * 4),
            nn.SiLU(),
            nn.Linear(dim * 4, dim * 4),
            nn.SiLU(),
            nn.Linear(dim * 4, dim),
        )

    def forward(self, x0: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x0: (B, dim) initial state at ``t = 0`` (e.g. noise or anchor trajectory).
        Returns:
            (B, dim) approximate state at ``t = 1`` after fused Euler integration.
        """
        B, d = x0.shape
        assert d == self.dim
        x = x0
        dt = self.dt.to(dtype=x0.dtype, device=x0.device)
        for k in range(self.num_steps):
            t = (float(k) + 0.5) / float(self.num_steps)
            t_b = torch.full((B, 1), t, device=x0.device, dtype=x0.dtype)
            inp = torch.cat([x, t_b], dim=-1)
            v = self.velocity(inp)
            x = x + dt * v
        return x


class ModelNew(nn.Module):
    """KernelBench candidate entry point; default delegates to ``Model``."""

    def __init__(self, dim: int, num_steps: int):
        super().__init__()
        self._impl = Model(dim, num_steps)

    def forward(self, x0: torch.Tensor) -> torch.Tensor:
        return self._impl(x0)

// one-thread-per-neuron implementation.
__global__ __launch_bounds__(256, 2)
void ode_kernel64_warp(const float* __restrict__ x0,
                       const float* __restrict__ w1, const float* __restrict__ b1,
                       const float* __restrict__ w2, const float* __restrict__ b2,
                       const float* __restrict__ w3, const float* __restrict__ b3,
                       float* __restrict__ out, float* __restrict__ temp,
                       int B, int num_steps) {
    cg::grid_group grid = cg::this_grid();
    constexpr int P = 32;
    const int bid = blockIdx.x;
    const int row = bid / P;
    const int part = bid - row * P;
    const int tid = threadIdx.x;
    const int warp = tid >> 5;
    const int lane = tid & 31;

    float* xg = temp;
    float* h1g = xg + B * 64;
    float* h2g = h1g + B * 256;
    __shared__ float sx[256];

    if (tid < 64) xg[row * 64 + tid] = x0[row * 64 + tid];
    grid.sync();

    const float dt = 1.0f / (float)num_steps;
    for (int k = 0; k < num_steps; ++k) {
        if (tid < 64) sx[tid] = xg[row * 64 + tid];
        __syncthreads();
        const float t = ((float)k + 0.5f) * dt;

        // Layer 1: 256 outputs; 8 warps/block, 32 blocks/row.
        {
            const int o = part * 8 + warp;
            float sum = 0.0f;
            for (int j = lane; j < 64; j += 32) {
                sum = fmaf(w1[o * 65 + j], sx[j], sum);
            }
            if (lane == 0) sum = fmaf(w1[o * 65 + 64], t, sum) + b1[o];
            sum = warp_sum(sum);
            if (lane == 0) h1g[row * 256 + o] = silu_f(sum);
        }
        grid.sync();

        if (tid < 256) sx[tid] = h1g[row * 256 + tid];
        __syncthreads();

        // Layer 2: 256 outputs.
        {
            const int o = part * 8 + warp;
            float sum = 0.0f;
            #pragma unroll
            for (int j = lane; j < 256; j += 32) {
                sum = fmaf(w2[o * 256 + j], sx[j], sum);
            }
            if (lane == 0) sum += b2[o];
            sum = warp_sum(sum);
            if (lane == 0) h2g[row * 256 + o] = silu_f(sum);
        }
        grid.sync();

        if (tid < 256) sx[tid] = h2g[row * 256 + tid];
        __syncthreads();

        // Layer 3: 64 outputs; only two warps per partition are needed.
        if (warp < 2) {
            const int o = part * 2 + warp;
            float sum = 0.0f;
            #pragma unroll
            for (int j = lane; j < 256; j += 32) {
                sum = fmaf(w3[o * 256 + j], sx[j], sum);
            }
            if (lane == 0) sum += b3[o];
            sum = warp_sum(sum);
            if (lane == 0) xg[row * 64 + o] += dt * sum;
        }
        grid.sync();
    }

    if (warp < 2 && lane == 0) {
        const int o = part * 2 + warp;
        out[row * 64 + o] = xg[row * 64 + o];
    }
}

// Basic generic CUDA fallback for non-benchmark dimensions.
__global__ __launch_bounds__(256, 2)
void ode_kernel_generic(const float* __restrict__ x0,
                        const float* __restrict__ w1, const float* __restrict__ b1,
                        const float* __restrict__ w2, const float* __restrict__ b2,
                        const float* __restrict__ w3, const float* __restrict__ b3,
                        float* __restrict__ out, int D, int H, int num_steps) {
    extern __shared__ float sm[];
    float* x = sm;
    float* h1 = x + D;
    float* h2 = h1 + H;
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    for (int i = tid; i < D; i += blockDim.x) x[i] = x0[row * D + i];
    __syncthreads();
    const float dt = 1.0f / (float)num_steps;
    for (int k = 0; k < num_steps; ++k) {
        const float t = ((float)k + 0.5f) * dt;
        for (int o = tid; o < H; o += blockDim.x) {
            float acc = b1[o];
            for (int j = 0; j < D; ++j) acc = fmaf(w1[o * (D + 1) + j], x[j], acc);
            acc = fmaf(w1[o * (D + 1) + D], t, acc);
            h1[o] = silu_f(acc);
        }
        __syncthreads();
        for (int o = tid; o < H; o += blockDim.x) {
            float acc = b2[o];
            for (int j = 0; j < H; ++j) acc = fmaf(w2[o * H + j], h1[j], acc);
            h2[o] = silu_f(acc);
        }
        __syncthreads();
        for (int o = tid; o < D; o += blockDim.x) {
            float acc = b3[o];
            for (int j = 0; j < H; ++j) acc = fmaf(w3[o * H + j], h2[j], acc);
            x[o] += dt * acc;
        }
        __syncthreads();
    }
    for (int i = tid; i < D; i += blockDim.x) out[row * D + i] = x[i];
}

torch::Tensor ode_forward(torch::Tensor x0, torch::Tensor w1, torch::Tensor b1,
                          torch::Tensor w2, torch::Tensor b2,
                          torch::Tensor w3, torch::Tensor b3,
                          int64_t num_steps) {
    auto x = x0.contiguous();
    auto out = torch::empty_like(x);
    const int B = (int)x.size(0);
    const int D = (int)x.size(1);
    const int H = D * 4;
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    if (D == 64 && x.scalar_type() == torch::kFloat32) {
        auto temp = torch::empty({B * (64 + 256 + 256)}, x.options());
        float *xp = x.data_ptr<float>(), *w1p = w1.data_ptr<float>(), *b1p = b1.data_ptr<float>();
        float *w2p = w2.data_ptr<float>(), *b2p = b2.data_ptr<float>();
        float *w3p = w3.data_ptr<float>(), *b3p = b3.data_ptr<float>();
        float *outp = out.data_ptr<float>(), *tempp = temp.data_ptr<float>();
        int Bi = B, ns = (int)num_steps;
        void* args[] = {&xp, &w1p, &b1p, &w2p, &b2p, &w3p, &b3p, &outp, &tempp, &Bi, &ns};
        cudaLaunchCooperativeKernel((void*)ode_kernel64_warp, dim3(B * 32), dim3(256), args, 0, stream);
    } else {
        size_t shmem = (size_t)(D + 2 * H) * sizeof(float);
        ode_kernel_generic<<<B, 256, shmem, stream>>>(x.data_ptr<float>(), w1.data_ptr<float>(), b1.data_ptr<float>(),
                                                       w2.data_ptr<float>(), b2.data_ptr<float>(), w3.data_ptr<float>(), b3.data_ptr<float>(),
                                                       out.data_ptr<float>(), D, H, (int)num_steps);
    }
    return out;
}
